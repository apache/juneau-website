034/**
035 * Core class of the Juneau architecture.
036 * <p>
037 *    This class servers multiple purposes:
038 *    <ul class='spaced-list'>
039 *       <li>Provides the ability to wrap beans inside {@link Map} interfaces.
040 *       <li>Serves as a repository for metadata on POJOs, such as associated {@link BeanFilter beanFilters}, {@link PropertyNamer property namers}, etc...
041 *          which are used to tailor how POJOs are serialized and parsed.
042 *       <li>Serves as a common utility class for all {@link Serializer Serializers} and {@link Parser Parsers}
043 *             for serializing and parsing Java beans.
044 *    </ul>
045 *
046 *
047 * <h5 class='topic'>Bean Contexts</h5>
048 * <p>
049 *    Typically, it will be sufficient to use the existing {@link #DEFAULT} contexts for creating
050 *    bean maps.  However, if you want to tweak any of the settings on the context, you must
051 *    either clone the default context or create a new one from scratch (whichever is simpler for you).
052 *    You'll notice that this context class uses a fluent interface for defining settings.
053 * <p>
054 *    Bean contexts are created by {@link ContextFactory context factories}.
055 *    The settings on a bean context are fixed at the point they are created by the factory.
056 *
057 *
058 * <h5 class='topic'>BeanContext configuration properties</h5>
059 *    <code>BeanContexts</code> have several configuration properties that can be used to tweak behavior on how beans are handled.
060 *    These are denoted as the static <jsf>BEAN_*</jsf> fields on this class.
061 * <p>
062 *    Some settings (e.g. {@link BeanContext#BEAN_beansRequireDefaultConstructor}) are used to differentiate between bean and non-bean classes.
063 *    Attempting to create a bean map around one of these objects will throw a {@link BeanRuntimeException}.
064 *    The purpose for this behavior is so that the serializers can identify these non-bean classes and convert them to plain strings using the {@link Object#toString()} method.
065 * <p>
066 *    Some settings (e.g. {@link BeanContext#BEAN_beanFieldVisibility}) are used to determine what kinds of properties are detected on beans.
067 * <p>
068 *    Some settings (e.g. {@link BeanContext#BEAN_beanMapPutReturnsOldValue}) change the runtime behavior of bean maps.
069 * <p>
070 *    Settings are specified using the {@link ContextFactory#setProperty(String, Object)} method and related convenience methods.
071 *
072 * <h6 class='topic'>Examples</h6>
073 * <p class='bcode'>
074 *    <jc>// Construct a context from scratch.</jc>
075 *    BeanContext beanContext = ContextFactory.<jsm>create</jsm>()
076 *       .setProperty(BeanContext.<jsf>BEAN_beansRequireDefaultConstructor</jsf>, <jk>true</jk>)
077 *       .addNotBeanClasses(Foo.<jk>class</jk>)
078 *       .getBeanContext();
079 *
080 *    <jc>// Clone an existing context factory.</jc>
081 *    BeanContext beanContext = ContextFactory.<jsm>create</jsm>(otherConfig)
082 *       .setProperty(BeanContext.<jsf>BEAN_beansRequireDefaultConstructor</jsf>, <jk>true</jk>)
083 *       .addNotBeanClasses(Foo.<jk>class</jk>)
084 *       .getBeanContext();
085 * </p>

271 * <h5 class='topic'>Bean Maps</h5>
272 * <p>
273 *    {@link BeanMap BeanMaps} are wrappers around Java beans that allow properties to be retrieved and
274 *    set using the common {@link Map#put(Object,Object)} and {@link Map#get(Object)} methods.<br>
275 *    <br>
276 *    Bean maps are created in two ways...
277 *    <ol>
278 *       <li> {@link BeanContext#forBean(Object) BeanContext.forBean()} - Wraps an existing bean inside a {@code Map} wrapper.
279 *       <li> {@link BeanContext#newBeanMap(Class) BeanContext.newInstance()} - Create a new bean instance wrapped in a {@code Map} wrapper.
280 *    </ol>
281 *
282 *
283 * <h6 class='topic'>Examples</h6>
284 * <p class='bcode'>
285 *    <jc>// A sample bean class</jc>
286 *    <jk>public class</jk> Person {
287 *       <jk>public</jk> String getName();
288 *       <jk>public void</jk> setName(String name);
289 *       <jk>public int</jk> getAge();
290 *       <jk>public void</jk> setAge(<jk>int</jk> age);
291 *    }
292 *
293 *    <jc>// Wrap an existing bean in a new bean map</jc>
294 *    BeanMap&lt;Person&gt; m1 = BeanContext.<jsf>DEFAULT</jsf>.forBean(<jk>new</jk> Person());
295 *    m1.put(<js>"name"</js>, <js>"John Smith"</js>);
296 *    m1.put(<js>"age"</js>, 45);
297 *
298 *    <jc>// Create a new bean instance wrapped in a new bean map</jc>
299 *    BeanMap&lt;Person&gt; m2 = BeanContext.<jsf>DEFAULT</jsf>.newInstance(Person.<jk>class</jk>);
300 *    m2.put(<js>"name"</js>, <js>"John Smith"</js>);
301 *    m2.put(<js>"age"</js>, 45);
302 *    Person p = m2.getBean();  <jc>// Get the bean instance that was created.</jc>
303 * </p>
304 *
305 *
306 * <h5 class='topic'>Bean Annotations</h5>
307 * <p>
308 *    This package contains annotations that can be applied to
309 *    class definitions to override what properties are detected on a bean.
310 * <h6 class='topic'>Examples</h6>
311 * <p class='bcode'>
312 *    <jc>// Bean class definition where only property 'name' is detected.</jc>
313 *    <ja>&#64;Bean</ja>(properties={<js>"name"</js>})
314 *    <jk>public class</jk> Person {
315 *       <jk>public</jk> String getName();
316 *       <jk>public void</jk> setName(String name);
317 *       <jk>public int</jk> getAge();
318 *       <jk>public void</jk> setAge(<jk>int</jk> age);
319 *    }
320 * <p>
321 *    See {@link Bean @Bean} and {@link BeanProperty @BeanProperty} for more information.
322 *
323 *
324 * <h5 class='topic'>Beans with read-only properties</h5>
325 * <p>
326 *    Bean maps can also be defined on top of beans with read-only properties by adding a
327 *    {@link BeanConstructor @BeanConstructor} annotation to one of the constructors on the
328 *    bean class.  This will allow read-only properties to be set through constructor arguments.
329 * <p>
330 *    When the <code>@BeanConstructor</code> annotation is present, bean instantiation is delayed until the call to {@link BeanMap#getBean()}.
331 *    Until then, bean property values are stored in a local cache until <code>getBean()</code> is called.
332 *    Because of this additional caching step, parsing into read-only beans tends to be slower and use
333 *    more memory than parsing into beans with writable properties.
334 * <p>
335 *    Attempting to call {@link BeanMap#put(String,Object)} on a read-only property after calling {@link BeanMap#getBean()}
336 *    will result in a {@link BeanRuntimeException} being thrown.
337 *    Multiple calls to {@link BeanMap#getBean()} will return the same bean instance.
338 * <p>
339 *    Beans can be defined with a combination of read-only and read-write properties.
340 * <p>
341 *    See {@link BeanConstructor @BeanConstructor} for more information.
342 *
343 *
344 * <h5 class='topic'>BeanFilters and PojoSwaps</h5>
345 * <p>
346 *    {@link BeanFilter BeanFilters} and {@link PojoSwap PojoSwaps} are used to tailor how beans and POJOs are handled.<br>
347 *    <ol class='spaced-list'>
348 *       <li>{@link BeanFilter} - Allows you to tailor handling of bean classes.
349 *          This class can be considered a programmatic equivalent to the {@link Bean} annotation when
350 *          annotating classes are not possible (e.g. you don't have access to the source).
351 *          This includes specifying which properties are visible and the ability to programmatically override the execution of properties.
352 *       <li>{@link PojoSwap} - Allows you to swap out non-serializable objects with serializable replacements.
353 *    </ol>
354 * <p>
355 *    See {@link org.apache.juneau.transform} for more information.
356 *
357 *
358 * <h5 class='topic'>ClassMetas</h5>
359 * <p>
360 *    The {@link ClassMeta} class is a wrapper around {@link Class} object that provides cached information
361 *    about that class (e.g. whether it's a {@link Map} or {@link Collection} or bean).
362 * <p>
363 *    As a general rule, it's best to reuse bean contexts (and therefore serializers and parsers too)
364 *    whenever possible since it takes some time to populate the internal {@code ClassMeta} object cache.
365 *    By reusing bean contexts, the class type metadata only needs to be calculated once which significantly
366 *    improves performance.
367 * <p>
368 *    See {@link ClassMeta} for more information.
369 *
370 * @author Barry M. Caceres
371 * @author James Bognar (james.bognar@salesforce.com)
372 */

1037   /**
1038    * Determines whether the specified class is ignored as a bean class based on the various
1039    *    exclusion parameters specified on this context class.
1040    *
1041    * @param c The class type being tested.
1042    * @return <jk>true</jk> if the specified class matches any of the exclusion parameters.
1043    */
1044   protected boolean isNotABean(Class<?> c) {
1045      if (c.isArray() || c.isPrimitive() || c.isEnum() || c.isAnnotation())
1046         return true;
1047      Package p = c.getPackage();
1048      if (p != null) {
1049         for (String p2 : notBeanPackageNames)
1050            if (p.getName().equals(p2))
1051               return true;
1052         for (String p2 : notBeanPackagePrefixes)
1053            if (p.getName().startsWith(p2))
1054               return true;
1055      }
1056      for (Class exclude : notBeanClasses)
1057         if (isParentClass(exclude, c))
1058            return true;
1059      return false;
1060   }

1311   /**
1312    * Construct a {@code ClassMeta} wrapper around a {@link Class} object.
1313    *
1314    * @param <T> The class type being wrapped.
1315    * @param c The class being wrapped.
1316    *    of type {@link Class} or {@link ClassMeta}.
1317    * @return If the class is not an array, returns a cached {@link ClassMeta} object.
1318    *    Otherwise, returns a new {@link ClassMeta} object every time.<br>
1319    */
1320   public <T> ClassMeta<T> getClassMeta(Class<T> c) {
1321
1322      // If this is an array, then we want it wrapped in an uncached ClassMeta object.
1323      // Note that if it has a pojo swap, we still want to cache it so that
1324      // we can cache something like byte[] with ByteArrayBase64Swap.
1325      if (c.isArray() && findPojoSwap(c) == null)
1326         return new ClassMeta(c, this);
1327
1328      // This can happen if we have transforms defined against String or Object.
1329      if (cmCache == null)
1330         return null;
1331
1332      ClassMeta<T> cm = cmCache.get(c);
1333      if (cm == null) {
1334
1335         synchronized (this) {
1336
1337            // Make sure someone didn't already set it while this thread was blocked.
1338            cm = cmCache.get(c);
1339            if (cm == null) {
1340
1341               // Note:  Bean properties add the possibility that class reference loops exist.
1342               // To handle this possibility, we create a set of pending ClassMetas, and
1343               // call init (which finds the bean properties) after it's been added to the pending set.
1344               for (ClassMeta pcm : pendingClassMetas)
1345                  if (pcm.innerClass == c)
1346                     return pcm;
1347
1348               cm = new ClassMeta<T>(c, this, true);
1349               pendingClassMetas.addLast(cm);
1350               try {
1351                  cm.init();
1352               } finally {
1353                  pendingClassMetas.removeLast();
1354               }
1355               cmCache.put(c, cm);
1356            }
1357         }
1358      }
1359      return cm;
1360   }

1470   ClassMeta getClassMeta(Type o, Map<Class<?>,Class<?>[]> typeVarImpls) {
1471      if (o == null)
1472         return null;
1473
1474      if (o instanceof ClassMeta)
1475         return (ClassMeta)o;
1476
1477      Class c = null;
1478      if (o instanceof Class) {
1479         c = (Class)o;
1480      } else if (o instanceof ParameterizedType) {
1481         // A parameter (e.g. <String>.
1482         c = (Class<?>)((ParameterizedType)o).getRawType();
1483      } else if (o instanceof GenericArrayType) {
1484         // An array parameter (e.g. <byte[]>.
1485         GenericArrayType gat = (GenericArrayType)o;
1486         Type gatct = gat.getGenericComponentType();
1487         if (gatct instanceof Class) {
1488            Class gatctc = (Class)gatct;
1489            c = Array.newInstance(gatctc, 0).getClass();
1490         } else if (gatct instanceof ParameterizedType) {
1491            Class gatctc = (Class<?>)((ParameterizedType)gatct).getRawType();
1492            c = Array.newInstance(gatctc, 0).getClass();
1493         } else {
1494            return null;
1495         }
1496      } else if (o instanceof TypeVariable) {
1497         if (typeVarImpls != null) {
1498            TypeVariable t = (TypeVariable) o;
1499            String varName = t.getName();
1500            int varIndex = -1;
1501            Class gc = (Class)t.getGenericDeclaration();
1502            TypeVariable[] tv = gc.getTypeParameters();
1503            for (int i = 0; i < tv.length; i++) {
1504               if (tv[i].getName().equals(varName)) {
1505                  varIndex = i;
1506               }
1507            }
1508            if (varIndex != -1) {
1509
1510               // If we couldn't find a type variable implementation, that means
1511               // the type was defined at runtime (e.g. Bean b = new Bean<Foo>();)
1512               // in which case the type is lost through erasure.
1513               // Assume java.lang.Object as the type.
1514               if (! typeVarImpls.containsKey(gc))
1515                  return object();
1516
1517               return getClassMeta(typeVarImpls.get(gc)[varIndex]);
1518            }
1519         }
1520         // We don't know the bounded type, so just resolve to Object.
1521         return object();
1522      } else {
1523         // This can happen when trying to resolve the "E getFirst()" method on LinkedList, whose type is a TypeVariable
1524         // These should just resolve to Object.
1525         return object();
1526      }
1527
1528      ClassMeta rawType = getClassMeta(c);
1529
1530      // If this is a Map or Collection, and the parameter types aren't part
1531      // of the class definition itself (e.g. class AddressBook extends List<Person>),
1532      // then we need to figure out the parameters.
1533      if (rawType.isMap() || rawType.isCollection()) {
1534         ClassMeta[] params = findParameters(o, c);
1535         if (params == null)
1536            return rawType;
1537         if (rawType.isMap()) {
1538            if (params.length != 2)
1539               return rawType;
1540            if (params[0].isObject() && params[1].isObject())
1541               return rawType;
1542            return new ClassMeta(rawType.innerClass, this).setKeyType(params[0]).setValueType(params[1]);
1543         }
1544         if (rawType.isCollection()) {
1545            if (params.length != 1)
1546               return rawType;
1547            if (params[0].isObject())
1548               return rawType;
1549            return new ClassMeta(rawType.innerClass, this).setElementType(params[0]);
1550         }
1551      }
1552
1553      return rawType;
1554   }

1571   ClassMeta[] findParameters(Type o, Class c) {
1572      if (o == null)
1573         o = c;
1574
1575      // Loop until we find a ParameterizedType
1576      if (! (o instanceof ParameterizedType)) {
1577         loop: do {
1578            o = c.getGenericSuperclass();
1579            if (o instanceof ParameterizedType)
1580               break loop;
1581            for (Type t : c.getGenericInterfaces()) {
1582               o = t;
1583               if (o instanceof ParameterizedType)
1584                  break loop;
1585            }
1586            c = c.getSuperclass();
1587         } while (c != null);
1588      }
1589
1590      if (o instanceof ParameterizedType) {
1591         ParameterizedType pt = (ParameterizedType)o;
1592         if (! pt.getRawType().equals(Enum.class)) {
1593            List<ClassMeta<?>> l = new LinkedList<ClassMeta<?>>();
1594            for (Type pt2 : pt.getActualTypeArguments()) {
1595               if (pt2 instanceof WildcardType || pt2 instanceof TypeVariable)
1596                  return null;
1597               l.add(getClassMeta(pt2, null));
1598            }
1599            if (l.isEmpty())
1600               return null;
1601            return l.toArray(new ClassMeta[l.size()]);
1602         }
1603      }
1604
1605      return null;
1606   }

1699      int d = 0;
1700      if (s == null || s.isEmpty())
1701         return null;
1702
1703      // Check for Class.getName() on array class types.
1704      if (s.charAt(0) == '[') {
1705         try {
1706            return getClassMeta(findClass(s));

1708            throw new RuntimeException(e);

1710      }
1711
1712      int i1 = 0;
1713      int i2 = 0;
1714      int dim = 0;
1715      List<ClassMeta<?>> p = null;
1716      for (int i = 0; i < s.length(); i++) {
1717         char c = s.charAt(i);
1718         if (c == '<') {
1719            if (d == 0) {
1720               i1 = i;
1721               i2 = i+1;
1722               p = new LinkedList<ClassMeta<?>>();
1723            }
1724            d++;
1725         } else if (c == '>') {
1726            d--;
1727            if (d == 0 && p != null)
1728               p.add(getClassMetaFromString(s.substring(i2, i)));
1729         } else if (c == ',' && d == 1) {
1730            if (p != null)
1731               p.add(getClassMetaFromString(s.substring(i2, i)));
1732            i2 = i+1;
1733         }
1734         if (c == '[') {
1735            if (i1 == 0)
1736               i1 = i;
1737            dim++;
1738         }
1739      }
1740      if (i1 == 0)
1741         i1 = s.length();
1742      try {
1743         String name = s.substring(0, i1).trim();
1744         char x = name.charAt(0);
1745         Class<?> c = null;
1746         if (x >= 'b' && x <= 's') {
1747            if (x == 'b' && name.equals("boolean"))
1748               c = boolean.class;
1749            else if (x == 'b' && name.equals("byte"))
1750               c = byte.class;
1751            else if (x == 'c' && name.equals("char"))
1752               c = char.class;
1753            else if (x == 'd' && name.equals("double"))
1754               c = double.class;
1755            else if (x == 'i' && name.equals("int"))
1756               c = int.class;
1757            else if (x == 'l' && name.equals("long"))
1758               c = long.class;
1759            else if (x == 's' && name.equals("short"))
1760               c = short.class;
1761            else
1762               c = findClass(name);
1763         } else {
