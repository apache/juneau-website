013package org.apache.juneau.html;
014
015import static org.apache.juneau.html.HtmlDocSerializerContext.*;
016
017import java.lang.reflect.*;
018import java.util.*;
019
020import org.apache.juneau.*;
021import org.apache.juneau.internal.*;
022import org.apache.juneau.json.*;
023import org.apache.juneau.serializer.*;
024
025/**
026 * Context object that lives for the duration of a single serialization of {@link HtmlSerializer} and its subclasses.
027 * <p>
028 *    See {@link SerializerContext} for details.
029 * </p>
030 * <p>
031 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
032 * </p>
033 *
034 * @author James Bognar (james.bognar@salesforce.com)
035 */
036public final class HtmlDocSerializerSession extends HtmlSerializerSession {
037
038   private final String title, description, cssUrl;
039   private final String[] cssImports;
040   private final Map<String,String> links;
041   private final boolean nowrap;
042
043   /**
044    * Create a new session using properties specified in the context.
045    *
046    * @param ctx The context creating this session object.
047    *    The context contains all the configuration settings for this object.
048    * @param beanContext The bean context being used.
049    * @param output The output object.  See {@link JsonSerializerSession#getWriter()} for valid class types.
050    * @param op The override properties.
051    *    These override any context properties defined in the context.
052    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
053    */
054   @SuppressWarnings({ "unchecked", "rawtypes" })
055   protected HtmlDocSerializerSession(HtmlDocSerializerContext ctx, BeanContext beanContext, Object output, ObjectMap op, Method javaMethod) {
056      super(ctx, beanContext, output, op, javaMethod);
057      if (op == null || op.isEmpty()) {
058         title = ctx.title;
059         description = ctx.description;
060         links = ctx.links;
061         cssUrl = ctx.cssUrl;
062         cssImports = ctx.cssImports;
063         nowrap = ctx.nowrap;
064      } else {
065         title = op.getString(HTMLDOC_title, ctx.title);
066         description = op.getString(HTMLDOC_description, ctx.description);
067         links = new LinkedHashMap(op.getMap(HTMLDOC_links, ctx.links));
068         cssUrl = op.getString(HTMLDOC_cssUrl, ctx.cssUrl);
069         cssImports = StringUtils.split(op.getString(HTMLDOC_cssImports, null), ',');
070         nowrap = op.getBoolean(HTMLDOC_cssUrl, ctx.nowrap);
071      }
072   }
073
074   /**
075    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_title} setting value in this context.
076    *
077    * @return The {@link HtmlDocSerializerContext#HTMLDOC_title} setting value in this context.
078    */
079   public final String getTitle() {
080      return title;
081   }
082
083   /**
084    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_description} setting value in this context.
085    *
086    * @return The {@link HtmlDocSerializerContext#HTMLDOC_description} setting value in this context.
087    */
088   public final String getDescription() {
089      return description;
090   }
091
092   /**
093    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_links} setting value in this context.
094    *
095    * @return The {@link HtmlDocSerializerContext#HTMLDOC_links} setting value in this context.
096    */
097   public final Map<String,String> getLinks() {
098      return links;
099   }
100
101   /**
102    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_cssUrl} setting value in this context.
103    *
104    * @return The {@link HtmlDocSerializerContext#HTMLDOC_cssUrl} setting value in this context.
105    */
106   public final String getCssUrl() {
107      return cssUrl;
108   }
109
110   /**
111    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_cssImports} setting value in this context.
112    *
113    * @return The {@link HtmlDocSerializerContext#HTMLDOC_cssImports} setting value in this context.
114    */
115   public final String[] getCssImports() {
116      return cssImports;
117   }
118
119   /**
120    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_nowrap} setting value in this context.
121    *
122    * @return The {@link HtmlDocSerializerContext#HTMLDOC_nowrap} setting value in this context.
123    */
124   public final boolean isNoWrap() {
125      return nowrap;
126   }
127
128   @Override /* XmlSerializerSession */
129   public HtmlWriter getWriter() throws Exception {
130      Object output = getOutput();
131      if (output instanceof HtmlWriter)
132         return (HtmlWriter)output;
133      return new HtmlWriter(super.getWriter(), isUseIndentation(), isTrimStrings(), getQuoteChar(), getRelativeUriBase(), getAbsolutePathUriBase());
134   }
135}
