013package org.apache.juneau.jena;
014
015import org.apache.juneau.serializer.*;
016
017/**
018 * Constants used by the {@link RdfSerializer} and {@link RdfParser} classes.
019 *
020 * @author James Bognar (james.bognar@salesforce.com)
021 */
022public final class Constants {
023
024   //--------------------------------------------------------------------------------
025   // Built-in Jena languages.
026   //--------------------------------------------------------------------------------
027
028   /** Jena language support: <js>"RDF/XML"</js>.*/
029   public static final String LANG_RDF_XML = "RDF/XML";
030
031   /** Jena language support: <js>"RDF/XML-ABBREV"</js>.*/
032   public static final String LANG_RDF_XML_ABBREV = "RDF/XML-ABBREV";
033
034   /** Jena language support: <js>"N-TRIPLE"</js>.*/
035   public static final String LANG_NTRIPLE = "N-TRIPLE";
036
037   /** Jena language support: <js>"TURTLE"</js>.*/
038   public static final String LANG_TURTLE = "TURTLE";
039
040   /** Jena language support: <js>"N3"</js>.*/
041   public static final String LANG_N3 = "N3";
042
043
044   //--------------------------------------------------------------------------------
045   // Built-in Juneau properties.
046   //--------------------------------------------------------------------------------
047
048   /**
049    * RDF property identifier <js>"items"</js>.
050    * <p>
051    * For resources that are collections, this property identifies the RDF Sequence
052    *    container for the items in the collection.
053    */
054   public static final String RDF_juneauNs_ITEMS = "items";
055
056   /**
057    * RDF property identifier <js>"root"</js>.
058    * <p>
059    * Property added to root nodes to help identify them as root elements during parsing.
060    * <p>
061    * Added if {@link RdfSerializerContext#RDF_addRootProperty} setting is enabled.
062    */
063   public static final String RDF_juneauNs_ROOT = "root";
064
065   /**
066    * RDF property identifier <js>"class"</js>.
067    * <p>
068    * Property added to bean resources to identify the class type.
069    * <p>
070    * Added if {@link SerializerContext#SERIALIZER_addClassAttrs} setting is enabled.
071    */
072   public static final String RDF_juneauNs_CLASS = "class";
073
074   /**
075    * RDF property identifier <js>"value"</js>.
076    * <p>
077    * Property added to nodes to identify a simple value.
078    */
079   public static final String RDF_juneauNs_VALUE = "value";
080
081   /**
082    * RDF resource that identifies a <jk>null</jk> value.
083    */
084   public static final String RDF_NIL = "http://www.w3.org/1999/02/22-rdf-syntax-ns#nil";
085
086   /**
087    * RDF resource that identifies a <code>Seq</code> value.
088    */
089   public static final String RDF_SEQ = "http://www.w3.org/1999/02/22-rdf-syntax-ns#Seq";
090
091   /**
092    * RDF resource that identifies a <code>Bag</code> value.
093    */
094   public static final String RDF_BAG = "http://www.w3.org/1999/02/22-rdf-syntax-ns#Bag";
095}
