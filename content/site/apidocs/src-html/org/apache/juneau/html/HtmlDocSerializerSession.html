026/**
027 * Context object that lives for the duration of a single serialization of {@link HtmlSerializer} and its subclasses.
028 *
029 * <p>
030 * See {@link SerializerContext} for details.
031 *
032 * <p>
033 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
034 */
035public final class HtmlDocSerializerSession extends HtmlSerializerSession {
036
037   private final String title, description, branding, header, nav, aside, footer, noResultsMessage;
038   private final String[] style, stylesheet, script, links;
039   private final boolean nowrap;
040   private final HtmlDocTemplate template;
041
042   /**
043    * Create a new session using properties specified in the context.
044    *
045    * @param ctx
046    *    The context creating this session object.
047    *    The context contains all the configuration settings for this object.
048    * @param output
049    *    The output object.
050    *    See {@link JsonSerializerSession#getWriter()} for valid class types.
051    * @param op
052    *    The override properties.
053    *    These override any context properties defined in the context.
054    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
055    * @param locale
056    *    The session locale.
057    *    If <jk>null</jk>, then the locale defined on the context is used.
058    * @param timeZone
059    *    The session timezone.
060    *    If <jk>null</jk>, then the timezone defined on the context is used.
061    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
062    * @param uriContext
063    *    The URI context.
064    *    Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
065    */
066   protected HtmlDocSerializerSession(HtmlDocSerializerContext ctx, ObjectMap op, Object output, Method javaMethod,
067         Locale locale, TimeZone timeZone, MediaType mediaType, UriContext uriContext) {
068      super(ctx, op, output, javaMethod, locale, timeZone, mediaType, uriContext);
069      if (op == null || op.isEmpty()) {
070         title = ctx.title;
071         description = ctx.description;
072         branding = ctx.branding;
073         header = ctx.header;
074         nav = ctx.nav;
075         aside = ctx.aside;
076         footer = ctx.footer;
077         links = ctx.links;
078         style = ctx.style;
079         stylesheet = ctx.stylesheet;
080         script = ctx.script;
081         nowrap = ctx.nowrap;
082         noResultsMessage = ctx.noResultsMessage;
083         template = ClassUtils.newInstance(HtmlDocTemplate.class, ctx.template);
084      } else {
085         title = op.getString(HTMLDOC_title, ctx.title);
086         description = op.getString(HTMLDOC_description, ctx.description);
087         branding = op.getString(HTMLDOC_branding, ctx.branding);
088         header = op.getString(HTMLDOC_header, ctx.nav);
089         nav = op.getString(HTMLDOC_nav, ctx.nav);
090         aside = op.getString(HTMLDOC_aside, ctx.aside);
091         footer = op.getString(HTMLDOC_footer, ctx.footer);
092         links = op.getStringArray(HTMLDOC_links, ctx.links);
093         style = op.getStringArray(HTMLDOC_style, ctx.style);
094         stylesheet = op.getStringArray(HTMLDOC_stylesheet, ctx.stylesheet);
095         script = op.getStringArray(HTMLDOC_script, ctx.script);
096         nowrap = op.getBoolean(HTMLDOC_nowrap, ctx.nowrap);
097         noResultsMessage = op.getString(HTMLDOC_noResultsMessage, ctx.noResultsMessage);
098         template = ClassUtils.newInstance(HtmlDocTemplate.class, op.get(HTMLDOC_template, ctx.template));
099      }
100   }
101

262      return noResultsMessage;
263   }
264
265   @Override /* XmlSerializerSession */
266   public HtmlWriter getWriter() throws Exception {
267      Object output = getOutput();
268      if (output instanceof HtmlWriter)
269         return (HtmlWriter)output;
270      return new HtmlWriter(super.getWriter(), isUseWhitespace(), getMaxIndent(), isTrimStrings(), getQuoteChar(), getUriResolver());
271   }
