031/**
032 * Parses text generated by the {@link HtmlSerializer} class back into a POJO model.
033 *
034 *
035 * <h6 class='topic'>Media types</h6>
036 * <p>
037 *    Handles <code>Content-Type</code> types: <code>text/html</code>
038 *
039 *
040 * <h6 class='topic'>Description</h6>
041 * <p>
042 *    See the {@link HtmlSerializer} class for a description of the HTML generated.
043 * <p>
044 *    This class is used primarily for automated testing of the {@link HtmlSerializer} class.
045 *
046 *
047 * <h6 class='topic'>Configurable properties</h6>
048 * <p>
049 *    This class has the following properties associated with it:
050 * <ul>
051 *    <li>{@link HtmlSerializerContext}
052 * </ul>
053 *
054 *
055 * @author James Bognar (james.bognar@salesforce.com)
056 */
057@SuppressWarnings({ "rawtypes", "unchecked" })
058@Consumes("text/html,text/html+stripped")
059public final class HtmlParser extends ReaderParser {

064   /*
065    * Reads anything starting at the current event.
066    * <p>
067    *    Precondition:  Must be pointing at START_ELEMENT or CHARACTERS event.
068    *    Postcondition:  Pointing at next event to be processed.
069    */
070   private <T> T parseAnything(HtmlParserSession session, ClassMeta<T> eType, XMLEventReader r, Object outer, BeanPropertyMeta pMeta) throws Exception {
071
072      BeanContext bc = session.getBeanContext();
073      if (eType == null)
074         eType = (ClassMeta<T>)object();
075      PojoSwap<T,Object> transform = (PojoSwap<T,Object>)eType.getPojoSwap();
076      ClassMeta<?> sType = eType.getSerializedClassMeta();
077      session.setCurrentClass(sType);
078      BeanDictionary bd = (pMeta == null ? bc.getBeanDictionary() : pMeta.getBeanDictionary());
079
080      Object o = null;
081
082      XMLEvent event = r.nextEvent();
083      while (! (event.isStartElement() || (event.isCharacters() && ! event.asCharacters().isWhiteSpace()) || event.isEndDocument()))
084         event = r.nextEvent();
085
086      if (event.isEndDocument())
087         throw new XMLStreamException("Unexpected end of stream in parseAnything for type '"+eType+"'", event.getLocation());
088
089      if (event.isCharacters()) {
090         String text = parseCharacters(event, r);
091         if (sType.isObject())
092            o = text;
093         else if (sType.isCharSequence())
094            o = text;
095         else if (sType.isNumber())
096            o = parseNumber(text, (Class<? extends Number>)eType.getInnerClass());
097         else if (sType.isChar())
098            o = text.charAt(0);
099         else if (sType.isBoolean())
100            o = Boolean.parseBoolean(text);
101         else if (sType.canCreateNewInstanceFromString(outer))
102            o = sType.newInstanceFromString(outer, text);
103         else if (sType.canCreateNewInstanceFromNumber(outer))
104            o = sType.newInstanceFromNumber(outer, parseNumber(text, sType.getNewInstanceFromNumberClass()));
105         else
106            throw new XMLStreamException("Unexpected characters '"+event.asCharacters().getData()+"' for type '"+eType+"'", event.getLocation());
107
108      } else {
109         Tag tag = Tag.forString(event.asStartElement().getName().getLocalPart(), false);
110
111         // The _type attribute can be any of the following:
112         // "object" - A map.
113         // "array" - An array.
114         // "X" - A bean type name as defined through @Bean.typeName().
115
116         String typeName = "object";
117         String text = "";
118
119         if (tag.isOneOf(STRING, NUMBER, BOOLEAN, BR, FF, BS, TB))
120            text = parseCharacters(event, r);
121
122         if (tag == TABLE) {
123            Map<String,String> attrs = getAttributes(event);
124            typeName = attrs.get(bc.getBeanTypePropertyName());
125            if (bd.hasName(typeName))
126               sType = eType = (ClassMeta<T>)bd.getClassMeta(typeName);
127            // Reset to "object" if it was a bean type name.
128            if (! typeName.equals("array"))
129               typeName = "object";
130         }
131
132         boolean isValid = true;
133
134         if (tag == NULL)
135            nextTag(r, xNULL);
136         else if (tag == A)
137            o = parseAnchor(session, event, r, eType);
138         else if (sType.isObject()) {
139            if (tag == STRING)
140               o = text;
141            else if (tag == NUMBER)
142               o = parseNumber(text, null);
143            else if (tag == BOOLEAN)
144               o = Boolean.parseBoolean(text);
145            else if (tag == TABLE) {
146               if (typeName.equals("object")) {
147                  o = parseIntoMap(session, r, (Map)new ObjectMap(bc), sType.getKeyType(), sType.getValueType(), pMeta);
148               } else if (typeName.equals("array")) {
149                  o = parseTableIntoCollection(session, r, (Collection)new ObjectList(bc), sType.getElementType(), pMeta);
150               } else
151                  isValid = false;
152            }
153            else if (tag == UL)
154               o = parseIntoCollection(session, r, new ObjectList(bc), null, pMeta);
155         }
156         else if (tag == STRING && sType.isCharSequence())
157            o = text;
158         else if (tag == STRING && sType.isChar())
159            o = text.charAt(0);
160         else if (tag == STRING && sType.canCreateNewInstanceFromString(outer))
161            o = sType.newInstanceFromString(outer, text);
162         else if (tag == NUMBER && sType.isNumber())
163            o = parseNumber(text, (Class<? extends Number>)sType.getInnerClass());
164         else if (tag == NUMBER && sType.canCreateNewInstanceFromNumber(outer))
165            o = sType.newInstanceFromNumber(outer, parseNumber(text, sType.getNewInstanceFromNumberClass()));
166         else if (tag == BOOLEAN && sType.isBoolean())
167            o = Boolean.parseBoolean(text);
168         else if (tag == TABLE) {
169            if (typeName.equals("object")) {
170               if (sType.isMap()) {
171                  o = parseIntoMap(session, r, (Map)(sType.canCreateNewInstance(outer) ? sType.newInstance(outer) : new ObjectMap(bc)), sType.getKeyType(), sType.getValueType(), pMeta);
172               } else if (sType.canCreateNewInstanceFromObjectMap(outer)) {
173                  ObjectMap m = new ObjectMap(bc);
174                  parseIntoMap(session, r, m, string(), object(), pMeta);
175                  o = sType.newInstanceFromObjectMap(outer, m);
176               } else if (sType.canCreateNewBean(outer)) {
177                  BeanMap m = bc.newBeanMap(outer, sType.getInnerClass());
178                  o = parseIntoBean(session, r, m).getBean();
179               }
180               else
181                  isValid = false;
182            } else if (typeName.equals("array")) {
183               if (sType.isCollection())
184                  o = parseTableIntoCollection(session, r, (Collection)(sType.canCreateNewInstance(outer) ? sType.newInstance(outer) : new ObjectList(bc)), sType.getElementType(), pMeta);
185               else if (sType.isArray())
186                  o = bc.toArray(sType, parseTableIntoCollection(session, r, new ArrayList(), sType.getElementType(), pMeta));
187               else
188                  isValid = false;
189            } else
190               isValid = false;
191         } else if (tag == UL) {
192            if (sType.isCollection())
193               o = parseIntoCollection(session, r, (Collection)(sType.canCreateNewInstance(outer) ? sType.newInstance(outer) : new ObjectList(bc)), sType.getElementType(), pMeta);
194            else if (sType.isArray())
195               o = bc.toArray(sType, parseIntoCollection(session, r, new ArrayList(), sType.getElementType(), pMeta));
196            else
197               isValid = false;
198         } else
199            isValid = false;
200
201         if (! isValid)
202            throw new XMLStreamException("Unexpected tag '"+tag+"' for type '"+eType+"'", event.getLocation());
203      }
204
205
206      if (transform != null && o != null)
207         o = transform.unswap(o, eType, bc);
208
209      if (outer != null)
210         setParent(eType, o, outer);
211
212      return (T)o;
213   }

215   /*
216    * Reads an anchor tag and converts it into a bean.
217    */
218   private <T> T parseAnchor(HtmlParserSession session, XMLEvent e, XMLEventReader r, ClassMeta<T> beanType) throws XMLStreamException {
219      BeanContext bc = session.getBeanContext();
220      String href = e.asStartElement().getAttributeByName(new QName("href")).getValue();
221      String name = parseCharacters(e, r);
222      Class<T> beanClass = beanType.getInnerClass();
223      if (beanClass.isAnnotationPresent(HtmlLink.class)) {
224         HtmlLink h = beanClass.getAnnotation(HtmlLink.class);
225         BeanMap<T> m = bc.newBeanMap(beanClass);
226         m.put(h.hrefProperty(), href);
227         m.put(h.nameProperty(), name);
228         return m.getBean();
229      }
230      return bc.convertToType(href, beanType);
231   }

233   private Map<String,String> getAttributes(XMLEvent e) {
234      Map<String,String> m = new TreeMap<String,String>() ;
235      for (Iterator i = e.asStartElement().getAttributes(); i.hasNext();) {
236         Attribute a = (Attribute)i.next();
237         m.put(a.getName().getLocalPart(), a.getValue());
238      }
239      return m;
240   }

242   /*
243    * Reads contents of <table> element.
244    * Precondition:  Must be pointing at <table> event.
245    * Postcondition:  Pointing at next START_ELEMENT or END_DOCUMENT event.
246    */
247   private <K,V> Map<K,V> parseIntoMap(HtmlParserSession session, XMLEventReader r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType, BeanPropertyMeta pMeta) throws Exception {
248      while (true) {
249         Tag tag = nextTag(r, TR, xTABLE);
250         if (tag == xTABLE)
251            break;
252         tag = nextTag(r, TD, TH);
253         // Skip over the column headers.
254         if (tag == TH) {
255            parseElementText(r, xTH);
256            nextTag(r, TH);
257            parseElementText(r, xTH);
258            nextTag(r, xTR);
259            continue;
260         }
261         K key = parseAnything(session, keyType, r, m, pMeta);
262         nextTag(r, xTD);
263         nextTag(r, TD);
264         V value = parseAnything(session, valueType, r, m, pMeta);
265         setName(valueType, value, key);
266         m.put(key, value);
267         nextTag(r, xTD);
268         nextTag(r, xTR);
269      }
270
271      return m;
272   }

275    * Reads contents of <ul> element.
276    * Precondition:  Must be pointing at event following <ul> event.
277    * Postcondition:  Pointing at next START_ELEMENT or END_DOCUMENT event.
278    */
279   private <E> Collection<E> parseIntoCollection(HtmlParserSession session, XMLEventReader r, Collection<E> l, ClassMeta<E> elementType, BeanPropertyMeta pMeta) throws Exception {
280      while (true) {
281         Tag tag = nextTag(r, LI, xUL);
282         if (tag == xUL)
283            break;
284         l.add(parseAnything(session, elementType, r, l, pMeta));
285         nextTag(r, xLI);
286      }
287      return l;
288   }

291    * Reads contents of <ul> element into an Object array.
292    * Precondition:  Must be pointing at event following <ul> event.

295   private Object[] parseArgs(HtmlParserSession session, XMLEventReader r, ClassMeta<?>[] argTypes) throws Exception {
296      Object[] o = new Object[argTypes.length];
297      int i = 0;
298      while (true) {
299         Tag tag = nextTag(r, LI, xUL);
300         if (tag == xUL)
301            break;
302         o[i] = parseAnything(session, argTypes[i], r, session.getOuter(), null);
303         i++;
304         nextTag(r, xLI);
305      }
306      return o;
307   }

309   /*
310    * Reads contents of <ul> element.
311    * Precondition:  Must be pointing at event following <ul> event.
312    * Postcondition:  Pointing at next START_ELEMENT or END_DOCUMENT event.
313    */
314   private <E> Collection<E> parseTableIntoCollection(HtmlParserSession session, XMLEventReader r, Collection<E> l, ClassMeta<E> elementType, BeanPropertyMeta pMeta) throws Exception {
315
316      BeanContext bc = session.getBeanContext();
317      if (elementType == null)
318         elementType = (ClassMeta<E>)object();
319      BeanDictionary bd = (pMeta == null ? bc.getBeanDictionary() : pMeta.getBeanDictionary());
320
321      Tag tag = nextTag(r, TR);
322      List<String> keys = new ArrayList<String>();
323      while (true) {
324         tag = nextTag(r, TH, xTR);
325         if (tag == xTR)
326            break;
327         keys.add(parseElementText(r, xTH));
328      }
329
330      while (true) {
331         XMLEvent event = r.nextTag();
332         tag = Tag.forEvent(event);
333         if (tag == xTABLE)
334            break;
335         if (elementType.canCreateNewBean(l)) {
336            BeanMap m = bc.newBeanMap(l, elementType.getInnerClass());
337            for (int i = 0; i < keys.size(); i++) {
338               tag = nextTag(r, TD, NULL);
339               if (tag == NULL) {
340                  m = null;
341                  nextTag(r, xNULL);
342                  break;
343               }
344               String key = keys.get(i);
345               BeanMapEntry e = m.getProperty(key);
346               if (e == null) {
347                  //onUnknownProperty(key, m, -1, -1);
348                  parseAnything(session, object(), r, l, null);
349               } else {
350                  BeanPropertyMeta bpm = e.getMeta();
351                  ClassMeta<?> cm = bpm.getClassMeta();
352                  Object value = parseAnything(session, cm, r, m.getBean(false), bpm);
353                  setName(cm, value, key);
354                  bpm.set(m, value);
355               }
356               nextTag(r, xTD);
357            }
358            l.add(m == null ? null : (E)m.getBean());
359         } else {
360            String c = getAttributes(event).get(bc.getBeanTypePropertyName());
361            Map m = (Map)(elementType.isMap() && elementType.canCreateNewInstance(l) ? elementType.newInstance(l) : new ObjectMap(bc));
362            for (int i = 0; i < keys.size(); i++) {
363               tag = nextTag(r, TD, NULL);
364               if (tag == NULL) {
365                  m = null;
366                  nextTag(r, xNULL);
367                  break;
368               }
369               String key = keys.get(i);
370               if (m != null) {
371                  ClassMeta<?> et = elementType.getElementType();
372                  Object value = parseAnything(session, et, r, l, pMeta);
373                  setName(et, value, key);
374                  m.put(key, value);
375               }
376               nextTag(r, xTD);
377            }
378            if (m != null && c != null) {
379               ObjectMap m2 = (m instanceof ObjectMap ? (ObjectMap)m : new ObjectMap(m).setBeanContext(session.getBeanContext()));
380               m2.put(bc.getBeanTypePropertyName(), c);
381               l.add((E)bd.cast(m2));
382            } else {
383               l.add((E)m);
384            }
385         }
386         nextTag(r, xTR);
387      }
388      return l;
389   }

391   /*
392    * Reads contents of <table> element.
393    * Precondition:  Must be pointing at event following <table> event.
394    * Postcondition:  Pointing at next START_ELEMENT or END_DOCUMENT event.
395    */
396   private <T> BeanMap<T> parseIntoBean(HtmlParserSession session, XMLEventReader r, BeanMap<T> m) throws Exception {
397      while (true) {
398         Tag tag = nextTag(r, TR, xTABLE);
399         if (tag == xTABLE)
400            break;
401         tag = nextTag(r, TD, TH);
402         // Skip over the column headers.
403         if (tag == TH) {
404            parseElementText(r, xTH);
405            nextTag(r, TH);
406            parseElementText(r, xTH);
407            nextTag(r, xTR);
408            continue;
409         }
410         String key = parseElementText(r, xTD);
411         nextTag(r, TD);
412         BeanPropertyMeta pMeta = m.getPropertyMeta(key);
413         if (pMeta == null) {
414            if (m.getMeta().isSubTyped()) {
415               Object value = parseAnything(session, object(), r, m.getBean(false), null);
416               m.put(key, value);
417            } else {
418               onUnknownProperty(session, key, m, -1, -1);
419               parseAnything(session, object(), r, null, null);
420            }
421         } else {
422            ClassMeta<?> cm = pMeta.getClassMeta();
423            Object value = parseAnything(session, cm, r, m.getBean(false), pMeta);
424            setName(cm, value, key);
425            pMeta.set(m, value);
426         }
427         nextTag(r, xTD);
428         nextTag(r, xTR);
429      }
430      return m;
431   }

433   /*
434    * Parse until the next event is an end tag.
435    */
436   private String parseCharacters(XMLEvent e, XMLEventReader r) throws XMLStreamException {
437
438      List<String> strings = new LinkedList<String>();
439
440      while (true) {
441         int eventType = e.getEventType();
442         if (eventType == CHARACTERS) {
443            Characters c = e.asCharacters();
444            if (! c.isWhiteSpace())
445               strings.add(c.getData());
446         }
447         else if (eventType == START_ELEMENT) {
448            Tag tag = Tag.forEvent(e);
449            if (tag == BR)
450               strings.add("\n");
451            else if (tag == FF)
452               strings.add("\f");
453            else if (tag == BS)
454               strings.add("\b");
455            else if (tag == TB)
456               strings.add("\t");
457         }
458         // Ignore all other elements.
459
460         XMLEvent eNext = r.peek();
461
462         if (eNext.isStartElement() || eNext.isEndElement()) {
463            Tag tag = Tag.forEvent(eNext);
464            if (! (tag.isOneOf(A, xA, BR, xBR, FF, xFF, BS, xBS, TB, xTB, STRING, xSTRING, NUMBER, xNUMBER, BOOLEAN, xBOOLEAN)))
465               return trim(join(strings));
466         } else if (eNext.isEndDocument()) {
467            return trim(join(strings));
468         }
469
470         e = r.nextEvent();
471      }
472   }

474   private String trim(String s) {
475      int i2 = 0, i3;
476      for (i2 = 0; i2 < s.length(); i2++) {
477         char c = s.charAt(i2);
478         if (c != ' ')
479            break;
480      }
481      for (i3 = s.length(); i3 > i2; i3--) {
482         char c = s.charAt(i3-1);
483         if (c != ' ')
484            break;
485      }
486      return s.substring(i2, i3);
487   }

489   /*
490    * Reads the element text of the current element, accounting for <a> and <br> tags. <br>
491    * Precondition:  Must be pointing at first event AFTER the start tag.
492    * Postcondition:  Pointing at next START_ELEMENT or END_DOCUMENT event.
493    */
494   private String parseElementText(XMLEventReader r, Tag endTag) throws XMLStreamException {
495
496      List<String> strings = new LinkedList<String>();
497
498      XMLEvent e = r.nextEvent();
499      Tag nTag = (e.isEndElement() ? Tag.forEvent(e) : null);
500
501      while (nTag != endTag) {
502         if (e.isCharacters())
503            strings.add(parseCharacters(e, r));
504         e = r.nextEvent();
505
506         if (e.getEventType() == END_ELEMENT)
507            nTag = Tag.forEvent(e);
508
509         if (nTag == endTag)
510            return join(strings);
511      }
512
513      return "";
514   }

516   enum Tag {
517
518      TABLE(1,"<table>"),
519      TR(2,"<tr>"),
520      TH(3,"<th>"),
521      TD(4,"<td>"),

523      LI(6,"<li>"),

530      FF(13,"<ff>"),    // formfeed
531      BS(14,"<bs>"),    // backspace
532      TB(15,"<tb>"),    // tab
533      xTABLE(-1,"</table>"),
534      xTR(-2,"</tr>"),
535      xTH(-3,"</th>"),
536      xTD(-4,"</td>"),
537      xUL(-5,"</ul>"),
538      xLI(-6,"</li>"),
539      xSTRING(-7,"</string>"),
540      xNUMBER(-8,"</number>"),
541      xBOOLEAN(-9,"</boolean>"),
542      xNULL(-10,"</null>"),
543      xA(-11,"</a>"),
544      xBR(-12,"</br>"),
545      xFF(-13,"</ff>"),
546      xBS(-14,"</bs>"),
547      xTB(-15,"</tb>");

549      private Map<Integer,Tag> cache = new HashMap<Integer,Tag>();
550
551      int id;
552      String label;
553
554      Tag(int id, String label) {
555         this.id = id;
556         this.label = label;
557         cache.put(id, this);
558      }

560      static Tag forEvent(XMLEvent event) throws XMLStreamException {
561         if (event.isStartElement())
562            return forString(event.asStartElement().getName().getLocalPart(), false);
563         else if (event.isEndElement())
564            return forString(event.asEndElement().getName().getLocalPart(), true);
565         throw new XMLStreamException("Invalid call to Tag.forEvent on event of type ["+event.getEventType()+"]");
566      }
567
568      private static Tag forString(String tag, boolean end) throws XMLStreamException {
569         char c = tag.charAt(0);
570         Tag t = null;
571         if (c == 'u')
572            t = (end ? xUL : UL);
573         else if (c == 'l')
574            t = (end ? xLI : LI);
575         else if (c == 's')
576            t = (end ? xSTRING : STRING);
577         else if (c == 'b') {
578            c = tag.charAt(1);
579            if (c == 'o')
580               t = (end ? xBOOLEAN : BOOLEAN);
581            else if (c == 'r')
582               t = (end ? xBR : BR);
583            else if (c == 's')
584               t = (end ? xBS : BS);
585         }
586         else if (c == 'a')
587            t = (end ? xA : A);
588         else if (c == 'n') {
589            c = tag.charAt(2);
590            if (c == 'm')
591               t = (end ? xNUMBER : NUMBER);
592            else if (c == 'l')
593               t = (end ? xNULL : NULL);
594         }
595         else if (c == 't') {
596            c = tag.charAt(1);
597            if (c == 'a')
598               t = (end ? xTABLE : TABLE);
599            else if (c == 'r')
600               t = (end ? xTR : TR);
601            else if (c == 'h')
602               t = (end ? xTH : TH);
603            else if (c == 'd')
604               t = (end ? xTD : TD);
605            else if (c == 'b')
606               t = (end ? xTB : TB);
607         }
608         else if (c == 'f')
609            t = (end ? xFF : FF);
610         if (t == null)
611            throw new XMLStreamException("Unknown tag '"+tag+"' encountered");
612         return t;
613      }

629    * Reads the current tag.  Advances past anything that's not a start or end tag.  Throws an exception if
630    *    it's not one of the expected tags.
631    * Precondition:  Must be pointing before the event we want to parse.
632    * Postcondition:  Pointing at the tag just parsed.
633    */
634   private Tag nextTag(XMLEventReader r, Tag...expected) throws XMLStreamException {
635      XMLEvent event = r.nextTag();
636      Tag tag = Tag.forEvent(event);
637      if (expected.length == 0)
638         return tag;
639      for (Tag t : expected)
640         if (t == tag)
641            return tag;
642      throw new XMLStreamException("Unexpected tag: " + tag, event.getLocation());
643   }

645   private String join(List<String> s) {
646      if (s.size() == 0)
647         return "";
648      if (s.size() == 1)
649         return s.get(0);
650      StringBuilder sb = new StringBuilder();
651      for (String ss : s)
652         sb.append(ss);
653      return sb.toString();
654   }
