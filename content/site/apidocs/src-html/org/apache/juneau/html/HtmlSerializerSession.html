028/**
029 * Session object that lives for the duration of a single use of {@link HtmlSerializer}.
030 * <p>
031 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
032 */
033public class HtmlSerializerSession extends XmlSerializerSession {
034
035   private final AnchorText anchorText;
036   private final boolean
037      detectLinksInStrings,
038      lookForLabelParameters,
039      addKeyValueTableHeaders,
040      addBeanTypeProperties;
041   private final Pattern urlPattern = Pattern.compile("http[s]?\\:\\/\\/.*");
042   private final Pattern labelPattern;
043
044
045   @SuppressWarnings("hiding")
046   enum AnchorText {
047      PROPERTY_NAME, TO_STRING, URI, LAST_TOKEN, URI_ANCHOR, CONTEXT_RELATIVE, SERVLET_RELATIVE, PATH_RELATIVE
048   }
049
050   /**
051    * Create a new session using properties specified in the context.
052    *
053    * @param ctx The context creating this session object.
054    *    The context contains all the configuration settings for this object.
055    * @param output The output object.  See {@link JsonSerializerSession#getWriter()} for valid class types.
056    * @param op The override properties.
057    *    These override any context properties defined in the context.
058    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
059    * @param locale The session locale.
060    *    If <jk>null</jk>, then the locale defined on the context is used.
061    * @param timeZone The session timezone.
062    *    If <jk>null</jk>, then the timezone defined on the context is used.
063    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
064    * @param uriContext The URI context.
065    *    Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
066    */
067   protected HtmlSerializerSession(HtmlSerializerContext ctx, ObjectMap op, Object output, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType, UriContext uriContext) {
068      super(ctx, op, output, javaMethod, locale, timeZone, mediaType, uriContext);
069      String labelParameter;
070      if (op == null || op.isEmpty()) {
071         anchorText = Enum.valueOf(AnchorText.class, ctx.uriAnchorText);
072         detectLinksInStrings = ctx.detectLinksInStrings;
073         lookForLabelParameters = ctx.lookForLabelParameters;
074         labelParameter = ctx.labelParameter;
075         addKeyValueTableHeaders = ctx.addKeyValueTableHeaders;
076         addBeanTypeProperties = ctx.addBeanTypeProperties;
077      } else {
078         anchorText = Enum.valueOf(AnchorText.class, op.getString(HTML_uriAnchorText, ctx.uriAnchorText));
079         detectLinksInStrings = op.getBoolean(HTML_detectLinksInStrings, ctx.detectLinksInStrings);
080         lookForLabelParameters = op.getBoolean(HTML_lookForLabelParameters, ctx.lookForLabelParameters);
081         labelParameter = op.getString(HTML_labelParameter, ctx.labelParameter);
082         addKeyValueTableHeaders = op.getBoolean(HTML_addKeyValueTableHeaders, ctx.addKeyValueTableHeaders);
083         addBeanTypeProperties = op.getBoolean(MSGPACK_addBeanTypeProperties, ctx.addBeanTypeProperties);
084      }
085      labelPattern = Pattern.compile("[\\?\\&]" + Pattern.quote(labelParameter) + "=([^\\&]*)");
086   }
087
088   @Override /* XmlSerializerSession */
089   public HtmlWriter getWriter() throws Exception {
090      Object output = getOutput();
091      if (output instanceof HtmlWriter)
092         return (HtmlWriter)output;
093      return new HtmlWriter(super.getWriter(), isUseWhitespace(), getMaxIndent(), isTrimStrings(), getQuoteChar(), getUriResolver());
094   }
095
096   /**
097    * Returns <jk>true</jk> if the specified object is a URL.
098    *
099    * @param cm The ClassMeta of the object being serialized.
100    * @param pMeta The property metadata of the bean property of the object.  Can be <jk>null</jk> if the object isn't from a bean property.
101    * @param o The object.
102    * @return <jk>true</jk> if the specified object is a URL.
103    */
104   public boolean isUri(ClassMeta<?> cm, BeanPropertyMeta pMeta, Object o) {
105      if (cm.isUri())
106         return true;
107      if (pMeta != null && pMeta.isUri())
108         return true;
109      if (detectLinksInStrings && o instanceof CharSequence && urlPattern.matcher(o.toString()).matches())
110         return true;
111      return false;
112   }
113
114   /**
115    * Returns the anchor text to use for the specified URL object.
116    *
117    * @param pMeta The property metadata of the bean property of the object.  Can be <jk>null</jk> if the object isn't from a bean property.
118    * @param o The URL object.
119    * @return The anchor text to use for the specified URL object.
120    */
121   public String getAnchorText(BeanPropertyMeta pMeta, Object o) {
122      String s = o.toString();
123      if (lookForLabelParameters) {
124         Matcher m = labelPattern.matcher(s);
125         if (m.find())
126            return urlDecode(m.group(1));
127      }
128      switch (anchorText) {
129         case LAST_TOKEN:
130            s = resolveUri(s);
131            if (s.indexOf('/') != -1)
132               s = s.substring(s.lastIndexOf('/')+1);
133            if (s.indexOf('?') != -1)
134               s = s.substring(0, s.indexOf('?'));
135            if (s.indexOf('#') != -1)
136               s = s.substring(0, s.indexOf('#'));
137            if (s.isEmpty())
138               s = "/";
139            return urlDecode(s);
140         case URI_ANCHOR:
141            if (s.indexOf('#') != -1)
142               s = s.substring(s.lastIndexOf('#')+1);
143            return urlDecode(s);
144         case PROPERTY_NAME:
145            return pMeta == null ? s : pMeta.getName();
146         case URI:
147            return resolveUri(s);
148         case CONTEXT_RELATIVE:
149            return relativizeUri("context:/", s);
150         case SERVLET_RELATIVE:
151            return relativizeUri("servlet:/", s);
152         case PATH_RELATIVE:
153            return relativizeUri("request:/", s);
154         default /* TO_STRING */:
155            return s;
156      }
157   }
158
159   /**
160    * Returns the {@link HtmlSerializerContext#HTML_addKeyValueTableHeaders} setting value for this session.
161    *
162    * @return The {@link HtmlSerializerContext#HTML_addKeyValueTableHeaders} setting value for this session.
163    */
164   public final boolean isAddKeyValueTableHeaders() {
165      return addKeyValueTableHeaders;
166   }
167
168   /**
169    * Returns the {@link HtmlSerializerContext#HTML_addBeanTypeProperties} setting value for this session.
170    *
171    * @return The {@link HtmlSerializerContext#HTML_addBeanTypeProperties} setting value for this session.
172    */
173   @Override /* SerializerSession */
174   public final boolean isAddBeanTypeProperties() {
175      return addBeanTypeProperties;
176   }
177
178   @Override /* XmlSerializer */
179   public boolean isHtmlMode() {
180      return true;
181   }
