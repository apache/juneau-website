029/**
030 * Parent class for all microservices.
031 * 
032 * <p>
033 * A microservice defines a simple API for starting and stopping simple Java services contained in executable jars.
034 * 
035 * <p>
036 * The general command for invoking these services is...
037 * <p class='bcode'>
038 *    java -jar mymicroservice.jar [mymicroservice.cfg]
039 * </p>
040 * 
041 * <p>
042 * Your microservice class must be specified as the <jk>Main-Class</jk> entry in the manifest file of your microservice 
043 * jar file.
044 *
045 * <h6 class='topic'>Microservice Configuration</h6>
046 *
047 * This class defines the following method for accessing configuration for your microservice:
048 * <ul class='spaced-list'>
049 *    <li>
050 *       {@link #getArgs()} - The command-line arguments passed to the jar file.
051 *    <li>
052 *       {@link #getConfig()} - An external INI-style configuration file.
053 *    <li>
054 *       {@link #getManifest()} - The manifest file for the main jar file.
055 * </ul>
056 *
057 * <h6 class='topic'>Entry point Method</h6>
058 *
059 * Subclasses must implement a static void main method as the entry point for the microservice.
060 * Typically, this method will simply consist of the following...
061 * <p class='bcode'>
062 *    <jk>public static void</jk> main(String[] args) <jk>throws</jk> Exception {
063 *       <jk>new</jk> MyMicroservice(args).start();
064 *    }
065 * </p>
066 *
067 * <h6 class='topic'>Lifecycle Methods</h6>
068 *
069 * Subclasses must implement the following lifecycle methods:
070 * <ul class='spaced-list'>
071 *    <li>
072 *       {@link #start()} - Gets executed during startup.
073 *    <li>
074 *       {@link #stop()} - Gets executed when 'exit' is typed in the console or an external shutdown signal is received.
075 *    <li>
076 *       {@link #kill()} - Can be used to forcibly shut down the service.  Doesn't get called during normal operation.
077 * </ul>
078 *
079 * <h6 class='topic'>Lifecycle Listener Methods</h6>
080 *
081 * Subclasses can optionally implement the following event listener methods:
082 * <ul class='spaced-list'>
083 *    <li>
084 *       {@link #onStart()} - Gets executed before {@link #start()}.
085 *    <li>
086 *       {@link #onStop()} - Gets executed before {@link #stop()}.
087 *    <li>
088 *       {@link #onConfigSave(ConfigFile)} - Gets executed after a config file has been saved.
089 *    <li>
090 *       {@link #onConfigChange(ConfigFile, Set)} - Gets executed after a config file has been modified.
091 * </ul>
092 *
093 * <h6 class='topic'>Other Methods</h6>
094 *
095 * Subclasses can optionally override the following methods to provide customized behavior:
096 * <ul class='spaced-list'>
097 *    <li>
098 *       {@link #createVarResolver()} - Creates the {@link VarResolver} used to resolve variables in the config file 
099 *       returned by {@link #getConfig()}.
100 * </ul>
101 */
102public abstract class Microservice {

120   /**
121    * Specifies the path of the config file for this microservice.
122    * 
123    * <p>
124    * If you do not specify the config file location, we attempt to resolve it through the following methods:
125    * <ol>
126    *    <li>The first argument in the command line arguments passed in through the constructor.
127    *    <li>The value of the <code>Main-ConfigFile</code> entry in the manifest file.
128    *    <li>A config file in the same location and with the same name as the executable jar file.
129    *       (e.g. <js>"java -jar myjar.jar"</js> will look for <js>"myjar.cfg"</js>).
130    * </ol>
131    * 
132    * <p>
133    * If this path does not exist, a {@link FileNotFoundException} will be thrown from the {@link #start()} command.
134    *
135    * @param cfPath The absolute or relative path of the config file.
136    * @param create Create the file if it doesn't exist.
137    * @return This object (for method chaining).
138    * @throws IOException If config file does not exist at the specified location or could not be read or created.
139    */
140   public Microservice setConfig(String cfPath, boolean create) throws IOException {
141      File f = new File(cfPath);
142      if (! f.exists()) {
143         if (! create)
144            throw new FileNotFoundException("Could not locate config file at '"+f.getAbsolutePath()+"'");
145         if (! f.createNewFile())
146            throw new FileNotFoundException("Could not create config file at '"+f.getAbsolutePath()+"'");
147      }
148      this.cfPath = cfPath;
149      return this;
150   }

225   /**
226    * Creates the {@link VarResolver} used to resolve variables in the config file returned by {@link #getConfig()}.
227    * 
228    * <p>
229    * The default implementation resolves the following variables:
230    * <ul>
231    *    <li><code>$S{key}</code>, <code>$S{key,default}</code> - System properties.
232    *    <li><code>$E{key}</code>, <code>$E{key,default}</code> - Environment variables.
233    *    <li><code>$C{key}</code>, <code>$C{key,default}</code> - Config file entries.
234    *    <li><code>$MF{key}</code>, <code>$MF{key,default}</code> - Manifest file entries.
235    *    <li><code>$ARG{key}</code>, <code>$ARG{key,default}</code> - Command-line arguments.
236    *    <li><code>$IF{boolArg,thenValue}</code>, <code>$IF{boolArg,thenValue,elseValue}</code> - If-block logic.
237    *    <li><code>$SW{stringArg,pattern,thenVal...}</code>, 
238    *       <code>$SW{stringArg,pattern,thenVal,elseVal...}</code>  - Switch-block logic.
239    * </ul>
240    * 
241    * <p>
242    * Subclasses can override this method to provide their own variables.
243    *
244    * <h5 class='section'>Example:</h5>
245    * <p class='bcode'>
246    *    <jd>/**
247    *     * Augment default var resolver with a custom $B{...} variable that simply wraps strings inside square brackets.
248    *     * /</jd>
249    *    <ja>@Override</ja> <jc>// Microservice</jc>
250    *    <jk>protected</jk> StringVarResolver createVarResolver() {
251    *       <jk>return super</jk>.createVarResolver()
252    *          .addVar(<js>"B"</js>,
253    *             <jk>new</jk> StringVarWithDefault() {
254    *                <ja>@Override</ja> <jc>// StringVar</jc>
255    *                <jk>public</jk> String resolve(String varVal) {
256    *                   <jk>return</jk> <js>'['</js> + varVal + <js>']'</js>;
257    *                }
258    *             }
259    *          );
260    *    }
261    * </p>
262    * <p class='bcode'>
263    *    <cc># Example config file</cc>
264    *    <cs>[MySection]</cs>
265    *    <ck>myEntry</ck> = $B{foo}
266    * </p>
267    * <p class='bcode'>
268    *    <jc>// Example java code</jc>
269    *    String myentry = getConfig().getString(<js>"MySection/myEntry"</js>); <jc>// == "[foo]"</js>
270    * </p>
271    *
272    * @return A new {@link VarResolver}.
273    */
274   protected VarResolverBuilder createVarResolver() {
275      VarResolverBuilder b = new VarResolverBuilder()
276         .defaultVars()
277         .vars(ConfigFileVar.class, ManifestFileVar.class, ArgsVar.class, SwitchVar.class, IfVar.class)
278         .contextObject(ManifestFileVar.SESSION_manifest, mf)
279         .contextObject(ArgsVar.SESSION_args, args);
280      if (cf != null)
281         b.contextObject(ConfigFileVar.SESSION_config, cf);
282      return b;
283   }

300   /**
301    * Returns the external INI-style configuration file that can be used to configure your microservice.
302    * 
303    * <p>
304    * The config file location is determined in the following order:
305    * <ol class='spaced-list'>
306    *    <li>
307    *       The first argument passed to the microservice jar.
308    *    <li>
309    *       The <code>Main-ConfigFile</code> entry in the microservice jar manifest file.
310    *    <li>
311    *       The name of the microservice jar with a <js>".cfg"</js> suffix (e.g. 
312    *       <js>"mymicroservice.jar"</js>-&gt;<js>"mymicroservice.cfg"</js>).
313    * </ol>
314    * 
315    * <p>
316    * If all methods for locating the config file fail, then this method returns <jk>null</jk>.
317    * 
318    * <p>
319    * Subclasses can set their own config file by calling the {@link #setConfig(ConfigFile)} method.
320    * 
321    * <p>
322    * String variables defined by {@link #createVarResolver()} are automatically resolved when using this method.
323    * 
324    * <p>
325    * This method can be called from the class constructor.
326    *
327    * <h5 class='section'>Example:</h5>
328    * <p class='bcode'>
329    *    <cc>#--------------------------</cc>
330    *    <cc># My section</cc>
331    *    <cc>#--------------------------</cc>
332    *    <cs>[MySection]</cs>
333    *
334    *    <cc># An integer</cc>
335    *    <ck>anInt</ck> = 1
336    *
337    *    <cc># A boolean</cc>
338    *    <ck>aBoolean</ck> = true
339    *
340    *    <cc># An int array</cc>
341    *    <ck>anIntArray</ck> = 1,2,3
342    *
343    *    <cc># A POJO that can be converted from a String</cc>
344    *    <ck>aURL</ck> = http://foo
345    *
346    *    <cc># A POJO that can be converted from JSON</cc>
347    *    <ck>aBean</ck> = {foo:'bar',baz:123}
348    *
349    *    <cc># A system property</cc>
350    *    <ck>locale</ck> = $S{java.locale, en_US}
351    *
352    *    <cc># An environment variable</cc>
353    *    <ck>path</ck> = $E{PATH, unknown}
354    *
355    *    <cc># A manifest file entry</cc>
356    *    <ck>mainClass</ck> = $MF{Main-Class}
357    *
358    *    <cc># Another value in this config file</cc>
359    *    <ck>sameAsAnInt</ck> = $C{MySection/anInt}
360    *
361    *    <cc># A command-line argument in the form "myarg=foo"</cc>
362    *    <ck>myArg</ck> = $ARG{myarg}
363    *
364    *    <cc># The first command-line argument</cc>
365    *    <ck>firstArg</ck> = $ARG{0}
366    *
367    *    <cc># Look for system property, or env var if that doesn't exist, or command-line arg if that doesn't exist.</cc>
368    *    <ck>nested</ck> = $S{mySystemProperty,$E{MY_ENV_VAR,$ARG{0}}}
369    *
370    *    <cc># A POJO with embedded variables</cc>
371    *    <ck>aBean2</ck> = {foo:'$ARG{0}',baz:$C{MySection/anInt}}
372    * </p>
373    * 
374    * <p class='bcode'>
375    *    <jc>// Java code for accessing config entries above.</jc>
376    *    ConfigFile cf = getConfig();
377    *
378    *    <jk>int</jk> anInt = cf.getInt(<js>"MySection/anInt"</js>);
379    *    <jk>boolean</jk> aBoolean = cf.getBoolean(<js>"MySection/aBoolean"</js>);
380    *    <jk>int</jk>[] anIntArray = cf.getObject(<jk>int</jk>[].<jk>class</jk>, <js>"MySection/anIntArray"</js>);
381    *    URL aURL = cf.getObject(URL.<jk>class</jk>, <js>"MySection/aURL"</js>);
382    *    MyBean aBean = cf.getObject(MyBean.<jk>class</jk>, <js>"MySection/aBean"</js>);
383    *    Locale locale = cf.getObject(Locale.<jk>class</jk>, <js>"MySection/locale"</js>);
384    *    String path = cf.getString(<js>"MySection/path"</js>);
385    *    String mainClass = cf.getString(<js>"MySection/mainClass"</js>);
386    *    <jk>int</jk> sameAsAnInt = cf.getInt(<js>"MySection/sameAsAnInt"</js>);
387    *    String myArg = cf.getString(<js>"MySection/myArg"</js>);
388    *    String firstArg = cf.getString(<js>"MySection/firstArg"</js>);
389    * </p>
390    *
391    * @return The config file for this application, or <jk>null</jk> if no config file is configured.
392    */
393   protected static ConfigFile getConfig() {
394      return cf;
395   }

426   /**
427    * Start this application.
428    * 
429    * <p>
430    * Default implementation simply calls {@link #onStart()}.
431    * 
432    * <p>
433    * Overridden methods MUST call this method FIRST so that the {@link #onStart()} method is called.
434    *
435    * @return This object (for method chaining).
436    * @throws Exception
437    */
438   public Microservice start() throws Exception {
439
440      // --------------------------------------------------------------------------------
441      // Try to get the manifest file if it wasn't already set.
442      // --------------------------------------------------------------------------------
443      if (mf == null) {
444         Manifest m = new Manifest();
445
446         // If running within an eclipse workspace, need to get it from the file system.
447         File f = new File("META-INF/MANIFEST.MF");
448         if (f.exists()) {
449            try {
450               m.read(new FileInputStream(f));
451            } catch (IOException e) {
452               System.err.println("Problem detected in MANIFEST.MF.  Contents below:\n" + read(f));
453               throw e;
454            }
455         } else {
456            // Otherwise, read from manifest file in the jar file containing the main class.
457            URLClassLoader cl = (URLClassLoader)getClass().getClassLoader();
458            URL url = cl.findResource("META-INF/MANIFEST.MF");
459            if (url != null) {
460               try {
461                  m.read(url.openStream());
462               } catch (IOException e) {
463                  System.err.println("Problem detected in MANIFEST.MF.  Contents below:\n" + read(url.openStream()));
464                  throw e;
465               }
466            }
467         }
468         mf = new ManifestFile(m);
469      }
470
471      // --------------------------------------------------------------------------------
472      // Resolve the config file if the path was specified.
473      // --------------------------------------------------------------------------------
474      ConfigFileBuilder cfb = new ConfigFileBuilder();
475      if (cfPath != null)
476         cf = cfb.build(cfPath).getResolving(createVarResolver().build());
477
478      // --------------------------------------------------------------------------------
479      // Find config file.
480      // Can either be passed in as first parameter, or we discover it using
481      // the 'sun.java.command' system property.
482      // --------------------------------------------------------------------------------
483      if (cf == null) {
484         if (args.hasArg(0))
485            cfPath = args.getArg(0);
486         else if (mf.containsKey("Main-ConfigFile"))
487            cfPath = mf.getString("Main-ConfigFile");
488         else {
489            String cmd = System.getProperty("sun.java.command", "not_found").split("\\s+")[0];
490            if (cmd.endsWith(".jar"))
491               cfPath = cmd.replace(".jar", ".cfg");
492         }
493
494         if (cfPath == null) {
495            System.err.println("Running class ["+getClass().getSimpleName()+"] without a config file.");
496            cf = cfb.build();
497         } else {
498            System.out.println("Running class ["+getClass().getSimpleName()+"] using config file ["+cfPath+"]");
499            cf = cfb.build(cfPath).getResolving(createVarResolver().build());
500         }
501      }
502
503      if (cfPath != null)
504         System.setProperty("juneau.configFile", cfPath);
505
506      // --------------------------------------------------------------------------------
507      // Set system properties.
508      // --------------------------------------------------------------------------------
509      Set<String> spKeys = cf.getSectionKeys("SystemProperties");
510      if (spKeys != null)
511         for (String key : spKeys)
512            System.setProperty(key, cf.get("SystemProperties", key));
513
514      // --------------------------------------------------------------------------------
515      // Add a config file change listener.
516      // --------------------------------------------------------------------------------
517      cf.addListener(new ConfigFileListener() {
518         @Override /* ConfigFileListener */
519         public void onSave(ConfigFile cf) {
520            onConfigSave(cf);
521         }
522         @Override /* ConfigFileListener */
523         public void onChange(ConfigFile cf, Set<String> changes) {
524            onConfigChange(cf, changes);
525         }
526      });
527
528      // --------------------------------------------------------------------------------
529      // Add exit listeners.
530      // --------------------------------------------------------------------------------
531      new Thread() {
532         @Override /* Thread */
533         public void run() {
534            Console c = System.console();
535            if (c == null)
536               System.out.println("No available console.");
537            else {
538               while (true) {
539                  String l = c.readLine("\nEnter 'exit' to exit.\n");
540                  if (l == null || l.equals("exit")) {
541                     Microservice.this.stop();
542                     break;
543                  }
544               }
545            }
546         }
547      }.start();
548      Runtime.getRuntime().addShutdownHook(
549         new Thread() {
550            @Override /* Thread */
551            public void run() {
552               Microservice.this.stop();
553            }
554         }
555      );
556      onStart();
557      return this;
558   }
