065   /**
066    * Converts the specified class name to a readable form when class name is a special construct like <js>"[[Z"</js>.
067    *
068    * <h5 class='section'>Example:</h5>
069    * <p class='bcode'>
070    *    <jsm>getReadableClassName</jsm>(<js>"java.lang.Object"</js>);  <jc>// Returns "java.lang.Object"</jc>
071    *    <jsm>getReadableClassName</jsm>(<js>"boolean"</js>);  <jc>// Returns "boolean"</jc>
072    *    <jsm>getReadableClassName</jsm>(<js>"[Z"</js>);  <jc>// Returns "boolean[]"</jc>
073    *    <jsm>getReadableClassName</jsm>(<js>"[[Z"</js>);  <jc>// Returns "boolean[][]"</jc>
074    *    <jsm>getReadableClassName</jsm>(<js>"[Ljava.lang.Object;"</js>);  <jc>// Returns "java.lang.Object[]"</jc>
075    *    <jsm>getReadableClassName</jsm>(<jk>null</jk>);  <jc>// Returns null</jc>
076    * </p>
077    *
078    * @param className The class name.
079    * @return A readable class type name, or <jk>null</jk> if parameter is <jk>null</jk>.
080    */
081   public static String getReadableClassName(String className) {
082      if (className == null)
083         return null;
084      if (! StringUtils.startsWith(className, '['))
085         return className;
086      int depth = 0;
087      for (int i = 0; i < className.length(); i++) {
088         if (className.charAt(i) == '[')
089            depth++;
090         else
091            break;
092      }
093      char type = className.charAt(depth);
094      String c;
095      switch (type) {
096         case 'Z': c = "boolean"; break;
097         case 'B': c = "byte"; break;
098         case 'C': c = "char"; break;
099         case 'D': c = "double"; break;
100         case 'F': c = "float"; break;
101         case 'I': c = "int"; break;
102         case 'J': c = "long"; break;
103         case 'S': c = "short"; break;
104         default: c = className.substring(depth+1, className.length()-1);
105      }
106      StringBuilder sb = new StringBuilder(c.length() + 2*depth).append(c);
107      for (int i = 0; i < depth; i++)
108         sb.append("[]");
109      return sb.toString();
110   }

176   /**
177    * Returns the signature of the specified method.
178    *
179    * <p>
180    * For no-arg methods, the signature will be a simple string such as <js>"toString"</js>.
181    * For methods with one or more args, the arguments will be fully-qualified class names (e.g.
182    * <js>"append(java.util.StringBuilder,boolean)"</js>)
183    *
184    * @param m The methods to get the signature on.
185    * @return The methods signature.
186    */
187   public static String getMethodSignature(Method m) {
188      StringBuilder sb = new StringBuilder(m.getName());
189      Class<?>[] pt = m.getParameterTypes();
190      if (pt.length > 0) {
191         sb.append('(');
192         for (int i = 0; i < pt.length; i++) {
193            if (i > 0)
194               sb.append(',');
195            sb.append(getReadableClassName(pt[i]));
196         }
197         sb.append(')');
198      }
199      return sb.toString();
200   }

351    * Returns the specified annotation on the specified method.
352    *
353    * <p>
354    * Similar to {@link Method#getAnnotation(Class)}, but searches up the parent hierarchy for the annotation
355    * defined on parent classes and interfaces.

370   /**
371    * Returns the specified annotation on the specified method.
372    *
373    * <p>
374    * Similar to {@link Method#getAnnotation(Class)}, but searches up the parent hierarchy for the annotation defined
375    * on parent classes and interfaces.
376    *
377    * <p>
378    * Normally, annotations defined on methods of parent classes and interfaces are not inherited by the child methods.
379    * This utility method gets around that limitation by searching the class hierarchy for the "same" method
380    * (i.e. the same name and arguments).
381    *
382    * @param a The annotation to search for.
383    * @param c
384    *    The child class to start searching from.
385    *    Note that it can be a descendant class of the actual declaring class of the method passed in.
386    *    This allows you to find annotations on methods overridden by the method passed in.
387    * @param method The method to search.
388    * @return The annotation, or <jk>null</jk> if it wasn't found.
389    */
390   public static <T extends Annotation> T getMethodAnnotation(Class<T> a, Class<?> c, Method method) {
391      for (Method m : c.getDeclaredMethods()) {
392         if (isSameMethod(method, m)) {
393            T t = m.getAnnotation(a);
394            if (t != null)
395               return t;
396         }
397      }
398      Class<?> pc = c.getSuperclass();
399      if (pc != null) {
400         T t = getMethodAnnotation(a, pc, method);
401         if (t != null)
402            return t;
403      }
404      for (Class<?> ic : c.getInterfaces()) {
405         T t = getMethodAnnotation(a, ic, method);
406         if (t != null)
407            return t;
408      }
409      return null;
410   }

416   /**
417    * Locates the no-arg constructor for the specified class.
418    *
419    * <p>
420    * Constructor must match the visibility requirements specified by parameter 'v'.
421    * If class is abstract, always returns <jk>null</jk>.
422    * Note that this also returns the 1-arg constructor for non-static member classes.
423    *
424    * @param c The class from which to locate the no-arg constructor.
425    * @param v The minimum visibility.
426    * @return The constructor, or <jk>null</jk> if no no-arg constructor exists with the required visibility.
427    */
428   @SuppressWarnings({"rawtypes","unchecked"})
429   public static final <T> Constructor<T> findNoArgConstructor(Class<T> c, Visibility v) {
430      int mod = c.getModifiers();
431      if (Modifier.isAbstract(mod))
432         return null;
433      boolean isMemberClass = c.isMemberClass() && ! isStatic(c);
434      for (Constructor cc : c.getConstructors()) {
435         mod = cc.getModifiers();
436         if (cc.getParameterTypes().length == (isMemberClass ? 1 : 0) && v.isVisible(mod) && isNotDeprecated(cc))
437            return v.transform(cc);
438      }
439      return null;
440   }

442   /**
443    * Finds the real parameter type of the specified class.
444    *
445    * @param c The class containing the parameters (e.g. PojoSwap&lt;T,S&gt;)
446    * @param index The zero-based index of the parameter to resolve.
447    * @param oc The class we're trying to resolve the parameter type for.
448    * @return The resolved real class.
449    */
450   public static Class<?> resolveParameterType(Class<?> c, int index, Class<?> oc) {
451
452      // We need to make up a mapping of type names.
453      Map<Type,Type> typeMap = new HashMap<Type,Type>();
454      while (c != oc.getSuperclass()) {
455         extractTypes(typeMap, oc);
456         oc = oc.getSuperclass();
457      }
458
459      ParameterizedType opt = (ParameterizedType)oc.getGenericSuperclass();
460      Type actualType = opt.getActualTypeArguments()[index];
461
462      if (typeMap.containsKey(actualType))
463         actualType = typeMap.get(actualType);
464
465      if (actualType instanceof Class) {
466         return (Class<?>)actualType;
467
468      } else if (actualType instanceof GenericArrayType) {
469         Class<?> cmpntType = (Class<?>)((GenericArrayType)actualType).getGenericComponentType();
470         return Array.newInstance(cmpntType, 0).getClass();
471
472      } else if (actualType instanceof TypeVariable) {
473         TypeVariable<?> typeVariable = (TypeVariable<?>)actualType;
474         List<Class<?>> nestedOuterTypes = new LinkedList<Class<?>>();
475         for (Class<?> ec = oc.getEnclosingClass(); ec != null; ec = ec.getEnclosingClass()) {
476            try {
477               Class<?> outerClass = oc.getClass();
478               nestedOuterTypes.add(outerClass);
479               Map<Type,Type> outerTypeMap = new HashMap<Type,Type>();
480               extractTypes(outerTypeMap, outerClass);
481               for (Map.Entry<Type,Type> entry : outerTypeMap.entrySet()) {
482                  Type key = entry.getKey(), value = entry.getValue();
483                  if (key instanceof TypeVariable) {
484                     TypeVariable<?> keyType = (TypeVariable<?>)key;
485                     if (keyType.getName().equals(typeVariable.getName()) && isInnerClass(keyType.getGenericDeclaration(), typeVariable.getGenericDeclaration())) {
486                        if (value instanceof Class)
487                           return (Class<?>)value;
488                        typeVariable = (TypeVariable<?>)entry.getValue();
489                     }
490                  }
491               }
492            } catch (Exception e) {
493               throw new RuntimeException(e);
494            }
495         }
496         throw new FormattedRuntimeException("Could not resolve type: {0}", actualType);
497      } else {
498         throw new FormattedRuntimeException("Invalid type found in resolveParameterType: {0}", actualType);
499      }
500   }

502   private static boolean isInnerClass(GenericDeclaration od, GenericDeclaration id) {
503      if (od instanceof Class && id instanceof Class) {
504         Class<?> oc = (Class<?>)od;
505         Class<?> ic = (Class<?>)id;
506         while ((ic = ic.getEnclosingClass()) != null)
507            if (ic == oc)
508               return true;
509      }
510      return false;
511   }

513   private static void extractTypes(Map<Type,Type> typeMap, Class<?> c) {
514      Type gs = c.getGenericSuperclass();
515      if (gs instanceof ParameterizedType) {
516         ParameterizedType pt = (ParameterizedType)gs;
517         Type[] typeParameters = ((Class<?>)pt.getRawType()).getTypeParameters();
518         Type[] actualTypeArguments = pt.getActualTypeArguments();
519         for (int i = 0; i < typeParameters.length; i++) {
520            if (typeMap.containsKey(actualTypeArguments[i]))
521               actualTypeArguments[i] = typeMap.get(actualTypeArguments[i]);
522            typeMap.put(typeParameters[i], actualTypeArguments[i]);
523         }
524      }
525   }

527   /**
528    * Finds a public method with the specified parameters.
529    *
530    * @param c The class to look for the method.
531    * @param name The method name.
532    * @param returnType
533    *    The return type of the method.
534    *    Can be a super type of the actual return type.
535    *    For example, if the actual return type is <code>CharSequence</code>, then <code>Object</code> will match but
536    *    <code>String</code> will not.
537    * @param parameterTypes
538    *    The parameter types of the method.
539    *    Can be subtypes of the actual parameter types.
540    *    For example, if the parameter type is <code>CharSequence</code>, then <code>String</code> will match but
541    *    <code>Object</code> will not.
542    * @return The matched method, or <jk>null</jk> if no match was found.
543    */
544   public static Method findPublicMethod(Class<?> c, String name, Class<?> returnType, Class<?>...parameterTypes) {
545      for (Method m : c.getMethods()) {
546         if (isPublic(m) && m.getName().equals(name)) {
547            Class<?> rt = m.getReturnType();
548            if (isParentClass(returnType, rt)) {
549               Class<?>[] pt = m.getParameterTypes();
550               if (pt.length == parameterTypes.length) {
551                  boolean matches = true;
552                  for (int i = 0; i < pt.length; i++) {
553                     if (! isParentClass(pt[i], parameterTypes[i])) {
554                        matches = false;
555                        break;
556                     }
557                  }
558                  if (matches)
559                     return m;
560               }
561            }
562         }
563      }
564      return null;
565   }

567   /**
568    * Finds a public constructor with the specified parameters without throwing an exception.
569    *
570    * @param c The class to search for a constructor.
571    * @param parameterTypes
572    *    The parameter types in the constructor.
573    *    Can be subtypes of the actual constructor argument types.
574    * @return The matching constructor, or <jk>null</jk> if constructor could not be found.
575    */
576   @SuppressWarnings("unchecked")
577   public static <T> Constructor<T> findPublicConstructor(Class<T> c, Class<?>...parameterTypes) {
578      for (Constructor<?> n : c.getConstructors()) {
579         if (isPublic(n)) {
580            Class<?>[] pt = n.getParameterTypes();
581            if (pt.length == parameterTypes.length) {
582               boolean matches = true;
583               for (int i = 0; i < pt.length; i++) {
584                  if (! isParentClass(pt[i], parameterTypes[i])) {
585                     matches = false;
586                     break;
587                  }
588               }
589               if (matches)
590                  return (Constructor<T>)n;
591            }
592         }
593      }
594      return null;
595   }

790   /**
791    * Creates an instance of the specified class from within the context of another object.
792    *
793    * @param outer
794    *    The outer object.
795    *    Can be <jk>null</jk>.
796    * @param c The class to cast to.
797    * @param c2
798    *    The class to instantiate.
799    *    Can also be an instance of the class.
800    * @param args The arguments to pass to the constructor.
801    * @return The new class instance, or <jk>null</jk> if the class was <jk>null</jk> or is abstract or an interface.
802    * @throws RuntimeException if constructor could not be found or called.
803    */
804   @SuppressWarnings("unchecked")
805   public static <T> T newInstanceFromOuter(Object outer, Class<T> c, Object c2, Object...args) {
806      if (c2 == null)
807         return null;
808      if (c2 instanceof Class) {
809         try {
810            Class<?> c3 = (Class<?>)c2;
811            if (c3.isInterface() || isAbstract(c3))
812               return null;
813            Constructor<?> con = findPublicConstructor(c3, args);
814            if (con != null)
815               return (T)con.newInstance(args);
816            if (outer != null) {
817               Object[] args2 = new AList<Object>().append(outer).appendAll(args).toArray();
818               con = findPublicConstructor(c3, args2);
819               if (con != null)
820                  return (T)con.newInstance(args2);
821            }
822            return (T)c3.getConstructor(getClasses(args)).newInstance(args);
823         } catch (Exception e) {
824            throw new FormattedRuntimeException(e, "Could not instantiate class {0}", c.getName());
825         }
826      } else if (isParentClass(c, c2.getClass())) {
827         return (T)c2;
828      } else {
829         throw new FormattedRuntimeException("Object of type {0} found but was expecting {1}.", c2.getClass(), c.getClass());
830      }
831   }

833   /**
834    * Returns all the fields in the specified class and all parent classes.
835    *
836    * <p>
837    * Fields are ordered in either parent-to-child, or child-to-parent order, then alphabetically.
838    *
839    * @param c The class to get all fields on.
840    * @param parentFirst Order them in parent-class-to-child-class order, otherwise child-class-to-parent-class order.
841    * @return An iterable of all fields in the specified class.
842    */
843   @SuppressWarnings("rawtypes")
844   public static Iterable<Field> getAllFields(final Class c, final boolean parentFirst) {
845      return new Iterable<Field>() {
846         @Override
847         public Iterator<Field> iterator() {
848            return new Iterator<Field>(){
849               final Iterator<Class<?>> classIterator = getParentClasses(c, parentFirst, false);
850               Field[] fields = classIterator.hasNext() ? sort(classIterator.next().getDeclaredFields()) : new Field[0];
851               int fIndex = 0;
852               Field next;
853
854               @Override
855               public boolean hasNext() {
856                  prime();
857                  return next != null;
858               }
859
860               private void prime() {
861                  if (next == null) {
862                     while (fIndex >= fields.length) {
863                        if (classIterator.hasNext()) {
864                           fields = sort(classIterator.next().getDeclaredFields());
865                           fIndex = 0;
866                        } else {
867                           fIndex = -1;
868                        }
869                     }
870                     if (fIndex != -1)
871                        next = fields[fIndex++];
872                  }
873               }
874
875               @Override
876               public Field next() {
877                  prime();
878                  Field f = next;
879                  next = null;
880                  return f;
881               }
882
883               @Override
884               public void remove() {
885               }
886            };
887         }
888      };
889   }

949   private static Comparator<Method> METHOD_COMPARATOR = new Comparator<Method>() {
950
951      @Override
952      public int compare(Method o1, Method o2) {
953         int i = o1.getName().compareTo(o2.getName());
954         if (i == 0) {
955            i = o1.getParameterTypes().length - o2.getParameterTypes().length;
956            if (i == 0) {
957               for (int j = 0; j < o1.getParameterTypes().length && i == 0; j++) {
958                  i = o1.getParameterTypes()[j].getName().compareTo(o2.getParameterTypes()[j].getName());
959               }
960            }
961         }
962         return i;
963      }
964   };

997    * Returns a list of all the parent classes of the specified class including the class itself.
998    *
999    * @param c The class to retrieve the parent classes.
1000    * @param parentFirst In parent-to-child order, otherwise child-to-parent.
1001    * @param includeInterfaces Include interfaces.
1002    * @return An iterator of parent classes in the class hierarchy.
1003    */
1004   public static Iterator<Class<?>> getParentClasses(final Class<?> c, boolean parentFirst, boolean includeInterfaces) {
1005      List<Class<?>> l = getParentClasses(new ArrayList<Class<?>>(), c, parentFirst, includeInterfaces);
1006      return l.iterator();
1007   }
1008
1009   private static List<Class<?>> getParentClasses(List<Class<?>> l, Class<?> c, boolean parentFirst, boolean includeInterfaces) {
1010      if (parentFirst) {
1011         if (includeInterfaces)
1012            for (Class<?> i : c.getInterfaces())
1013               l.add(i);
1014         if (c.getSuperclass() != Object.class && c.getSuperclass() != null)
1015            getParentClasses(l, c.getSuperclass(), parentFirst, includeInterfaces);
1016         l.add(c);
1017      } else {
1018         l.add(c);
1019         if (c.getSuperclass() != Object.class && c.getSuperclass() != null)
1020            getParentClasses(l, c.getSuperclass(), parentFirst, includeInterfaces);
1021         if (includeInterfaces)
1022            for (Class<?> i : c.getInterfaces())
1023               l.add(i);
1024      }
1025      return l;
1026   }

1069      for (int i = 0; i < m.getParameterTypes().length; i++) {
1070         if (i > 0)
1071            sb.append(",");
1072         sb.append(m.getParameterTypes()[i].getSimpleName());
1073      }
1074      sb.append(")");
1075      return sb.toString();
