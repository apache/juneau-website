013package org.apache.juneau.internal;
014
015import java.lang.reflect.*;
016import java.util.*;
017
018/**
019 * Object-related utility methods.
020 */
021public class ObjectUtils {
022
023   /**
024    * Returns <jk>true</jk> if the specified objects are equal.
025    * <p>
026    * Gracefully handles <jk>null</jk>s.
027    *
028    * @param o1 Object #1
029    * @param o2 Object #2
030    * @return <jk>true</jk> if the objects are equal or both <jk>null</jk>.
031    */
032   public static boolean equals(Object o1, Object o2) {
033      if (o1 == null && o2 == null)
034         return true;
035      if (o1 == null || o2 == null)
036         return false;
037      return o1.equals(o2);
038   }
039
040   /**
041    * Returns <jk>true</jk> if the specified object is empty.
042    * <p>
043    * Return <jk>true</jk> if the value is any of the following:
044    * <ul>
045    *    <li><jk>null</jk>
046    *    <li>An empty Collection
047    *    <li>An empty Map
048    *    <li>An empty array
049    *    <li>An empty CharSequence
050    *    <li>An empty String when serialized to a string using {@link Object#toString()}.
051    * </ul>
052    *
053    * @param o The object to test.
054    * @return <jk>true</jk> if the specified object is empty.
055    */
056   @SuppressWarnings("rawtypes")
057   public static boolean isEmpty(Object o) {
058      if (o == null)
059         return true;
060      if (o instanceof Collection)
061         return ((Collection)o).isEmpty();
062      if (o instanceof Map)
063         return ((Map)o).isEmpty();
064      if (o.getClass().isArray())
065         return (Array.getLength(o) == 0);
066      return o.toString().isEmpty();
067   }
068}
