059    * Constructor.
060    * 
061    * @param context The resource context.
062    */
063   public RestInfoProviderDefault(RestContext context) {
064      this.context = context;
065
066      Builder b = new Builder(context);
067      this.siteName = b.siteName;
068      this.title = b.title;
069      this.description = b.description;
070      this.termsOfService = b.termsOfService;
071      this.contact = b.contact;
072      this.license = b.license;
073      this.version = b.version;
074      this.tags = b.tags;
075      this.externalDocs = b.externalDocs;
076   }

078   private static final class Builder {
079      String
080         siteName,
081         title,
082         description,
083         termsOfService,
084         contact,
085         license,
086         version,
087         tags,
088         externalDocs;
089
090      Builder(RestContext context) {
091
092         LinkedHashMap<Class<?>,RestResource> restResourceAnnotationsParentFirst = findAnnotationsMapParentFirst(RestResource.class, context.getResource().getClass());
093
094         for (RestResource r : restResourceAnnotationsParentFirst.values()) {
095            if (! r.siteName().isEmpty())
096               siteName = r.siteName();
097            if (! r.title().isEmpty())
098               title = r.title();
099            if (! r.description().isEmpty())
100               description = r.description();
101            ResourceSwagger sr = r.swagger();
102            if (! sr.termsOfService().isEmpty())
103               termsOfService = sr.termsOfService();
104            if (! sr.contact().isEmpty())
105               contact = sr.contact();
106            if (! sr.license().isEmpty())
107               license = sr.license();
108            if (! sr.version().isEmpty())
109               version = sr.version();
110            if (! sr.tags().isEmpty())
111               tags = sr.tags();
112            if (! sr.externalDocs().isEmpty())
113               externalDocs = sr.externalDocs();
114         }
115      }

118   /**
119    * Returns the localized Swagger from the file system.
120    * 
121    * <p>
122    * Looks for a file called <js>"{ServletClass}_{locale}.json"</js> in the same package as this servlet and returns
123    * it as a parsed {@link Swagger} object.
124    * 
125    * <p>
126    * Returned objects are cached per-locale for later quick-lookup.
127    * 
128    * @param req The incoming HTTP request.
129    * @return The parsed swagger object, or <jk>null</jk> if none could be found.
130    * @throws Exception 
131    *    If swagger file was not valid JSON.
132    */
133   public Swagger getSwaggerFromFile(RestRequest req) throws Exception {
134      Locale locale = req.getLocale();
135      Swagger s = swaggers.get(locale);
136      if (s == null) {
137         s = context.getClasspathResource(Swagger.class, MediaType.JSON, getClass().getSimpleName() + ".json", locale);
138         swaggers.putIfAbsent(locale, s == null ? Swagger.NULL : s);
139      }
140      return s == Swagger.NULL ? null : s;
141   }

143   /**
144    * Returns the localized swagger for this REST resource.
145    * 
146    * <p>
147    * Subclasses can override this method to customize the Swagger.
148    * 
149    * @param req The incoming HTTP request.
150    * @return 
151    *    A new Swagger instance.
152    *    <br>Never <jk>null</jk>.
153    * @throws Exception
154    */
155   @Override /* RestInfoProvider */
156   public Swagger getSwagger(RestRequest req) throws Exception {
157      
158      // If a file is defined, use that.
159      Swagger s = getSwaggerFromFile(req);
160      if (s != null)
161         return s;
162
163      s = swagger(
164         info(getTitle(req), getVersion(req))
165            .contact(getContact(req))
166            .license(getLicense(req))
167            .description(getDescription(req))
168            .termsOfService(getTermsOfService(req))
169         )
170         .consumes(getConsumes(req))
171         .produces(getProduces(req))
172         .tags(getTags(req))
173         .externalDocs(getExternalDocs(req));
174
175      for (RestJavaMethod sm : context.getCallMethods().values()) {
176         if (sm.isRequestAllowed(req)) {
177            Method m = sm.method;
178            Operation o = operation()
179               .operationId(getMethodOperationId(m, req))
180               .description(getMethodDescription(m, req))
181               .tags(getMethodTags(m, req))
182               .summary(getMethodSummary(m, req))
183               .externalDocs(getMethodExternalDocs(m, req))
184               .parameters(getMethodParameters(m, req))
185               .responses(getMethodResponses(m, req));
186
187            if (isDeprecated(m, req))
188               o.deprecated(true);
189            
190            o.consumes(getMethodConsumes(m, req));
191            o.produces(getMethodProduces(m, req));
192
193            s.path(
194               sm.getPathPattern(),
195               sm.getHttpMethod().toLowerCase(),
196               o
197            );
198         }
199      }
200      
201      return s;
202   }

464    * Returns the localized parameter info for the specified java method.
465    * 
466    * <p>
467    * Subclasses can override this method to provide their own parameter info.
468    * 
469    * <p>
470    * The default implementation returns the value from the following locations (whichever matches first):
471    * <ol class='spaced-list'>
472    *    <li>Operation information from swagger file.
473    *    <li>{@link MethodSwagger#parameters() @MethodSwagger.parameters()} annotation.
474    *       <h5 class='figure'>Examples:</h5>
475    *       <p class='bcode'>
476    *    <cc>// Direct value</cc>
477    *    <ja>@RestMethod</ja>(
478    *       swagger=<ja>@MethodSwagger</ja>(
479    *          parameters={
480    *             <ja>@Parameter</ja>(in=<js>"path"</js>, name=<js>"a"</js>, description=<js>"The 'a' attribute"</js>)
481    *          }
482    *       )
483    *    )
484    *    <jk>public</jk> Object myMethod() {...}
485    *    
486    *    <cc>// Pulled from some other location</cc>
487    *    <ja>@RestMethod</ja>(
488    *       swagger=<ja>@MethodSwagger</ja>(
489    *          parameters={
490    *             <ja>@Parameter</ja>(in=<js>"path"</js>, name=<js>"a"</js>, description=<js>"$L{myLocalizedParamADescription}"</js>)
491    *          }
492    *       )
493    *    )
494    *    <jk>public</jk> Object myMethod() {...}
495    *       </p>
496    *    <li>Localized strings from resource bundle identified by {@link RestResource#messages() @RestResource.messages()}
497    *       on the resource class, then any parent classes.
498    *       <ol>
499    *          <li><ck>[ClassName].[javaMethodName].parameters</ck>
500    *          <li><ck>[javaMethodName].parameters</ck>
501    *       </ol>
502    *       <br>Value is a JSON representation of a <code>{@link ParameterInfo}[]</code> object.
503    *       <br>Value can contain any SVL variables defined on the {@link MethodSwagger#parameters() @MethodSwagger.parameters()} annotation.
504    *       <h5 class='figure'>Examples:</h5>
505    *       <p class='bcode'>
506    *    <cc>// Direct value</cc>
507    *    <ck>MyClass.myMethod.parameters</ck> = <cv>[{name:"a",in:"path",description:"The ''a'' attribute"}]</cv>
508    *    
509    *    <cc>// Pulled from some other location</cc>
510    *    <ck>MyClass.myMethod.parameters</ck> = <cv>$C{MyStrings/MyClass.myMethod.parameters}</cv>
511    *       </p>
512    *    <li>Information gathered directly from the parameters on the Java method.
513    * </ol>
514    * 
515    * @param method The Java method annotated with {@link RestMethod @RestMethod}.
516    * @param req The current request.
517    * @return The localized parameter info of the method, or <jk>null</jk> if none was found.
518    * @throws Exception 
519    */
520   public List<ParameterInfo> getMethodParameters(Method method, RestRequest req) throws Exception {
521      
522      Operation o = getSwaggerOperationFromFile(method, req);
523      if (o != null && o.getParameters() != null)
524         return o.getParameters();
525
526      VarResolverSession vr = req.getVarResolverSession();
527      JsonParser jp = JsonParser.DEFAULT;
528      Map<String,ParameterInfo> m = new TreeMap<>();
529
530      // First parse @RestMethod.parameters() annotation.
531      for (org.apache.juneau.rest.annotation.Parameter v : method.getAnnotation(RestMethod.class).swagger().parameters()) {
532         String in = vr.resolve(v.in());
533         ParameterInfo p = parameterInfo(in, vr.resolve(v.name()));
534
535         if (! v.description().isEmpty())
536            p.description(vr.resolve(v.description()));
537         if (v.required())
538            p.required(v.required());
539
540         if ("body".equals(in)) {
541            if (! v.schema().isEmpty())
542               p.schema(jp.parse(vr.resolve(v.schema()), SchemaInfo.class));
543         } else {
544            if (v.allowEmptyValue())
545               p.allowEmptyValue(v.allowEmptyValue());
546            if (! v.collectionFormat().isEmpty())
547               p.collectionFormat(vr.resolve(v.collectionFormat()));
548            if (! v._default().isEmpty())
549               p._default(vr.resolve(v._default()));
550            if (! v.format().isEmpty())
551               p.format(vr.resolve(v.format()));
552            if (! v.items().isEmpty())
553               p.items(jp.parse(vr.resolve(v.items()), Items.class));
554            p.type(vr.resolve(v.type()));
555         }
556         m.put(p.getIn() + '.' + p.getName(), p);
557      }
558
559      // Next, look in resource bundle.
560      String s = context.getMessages().findFirstString(req.getLocale(), method.getName() + ".parameters");
561      if (s != null) {
562         for (ParameterInfo pi : jp.parse(vr.resolve(s), ParameterInfo[].class)) {
563            String key = pi.getIn() + '.' + pi.getName();
564            ParameterInfo p = m.get(key);
565            if (p == null)
566               m.put(key, pi);
567            else 
568               p.copyFrom(pi);
569         }
570      }
571
572      // Finally, look for parameters defined on method.
573      for (RestParam mp : context.getRestParams(method)) {
574         RestParamType in = mp.getParamType();
575         if (in != RestParamType.OTHER) {
576            String k2 = in.toString() + '.' + (in == RestParamType.BODY ? null : mp.getName());
577            ParameterInfo p = m.get(k2);
578            if (p == null) {
579               p = parameterInfoStrict(in.toString(), mp.getName());
580               m.put(k2, p);
581            }
582         }
583      }
584      
585      return m.isEmpty() ? null : new ArrayList<>(m.values());
586   }
