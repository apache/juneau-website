035/**
036 * Used to denote that a class is a REST resource and to associate metadata on it.
037 * <p>
038 * Usually used on a subclass of {@link RestServlet}, but can be used to annotate any class that you want to expose as a REST resource.
039 *
040 * Refer to <a class='doclink' href='../package-summary.html#TOC'>org.apache.juneau.rest</a> doc for information on using this class.
041 */
042@Documented
043@Target(TYPE)
044@Retention(RUNTIME)
045@Inherited
046public @interface RestResource {
047
048   /**
049    * Identifies the location of the resource bundle for this class.
050    * <p>
051    * This annotation is used to provide localized messages for the following methods:
052    * <ul>
053    *    <li>{@link RestRequest#getMessage(String, Object...)}
054    *    <li>{@link RestContext#getMessages()}
055    * </ul>
056    * <p>
057    * Refer to the {@link MessageBundle} class for a description of the message key formats
058    *    used in the properties file.
059    * <p>
060    * The value can be a relative path like <js>"nls/Messages"</js>, indicating to look for the
061    *    resource bundle <js>"com.foo.sample.nls.Messages"</js> if the resource class
062    *    is in <js>"com.foo.sample"</js>, or it can be an absolute path, like <js>"com.foo.sample.nls.Messages"</js>
063    */
064   String messages() default "";
065
066   /**
067    * Class-level guards.
068    * <p>
069    * Associates one or more {@link RestGuard RestGuards} with all REST methods defined
070    *    in this class.
071    * These guards get called immediately before execution of any REST method in this class.
072    * <p>
073    * Typically, guards will be used for permissions checking on the user making the request,
074    *    but it can also be used for other purposes like pre-call validation of a request.
075    * <p>
076    * The programmatic equivalent to this annotation are the {@link RestConfig#addGuards(Class...)}/{@link RestConfig#addGuards(RestGuard...)} methods.
077    */
078   Class<? extends RestGuard>[] guards() default {};
079
080   /**
081    * Class-level converters.
082    * <p>
083    * Associates one or more {@link RestConverter converters} with a resource class.
084    * These converters get called immediately after execution of the REST method in the same
085    *    order specified in the annotation.
086    * <p>
087    * Can be used for performing post-processing on the response object before serialization.
088    * <p>
089    * Default converter implementations are provided in the <a class='doclink' href='../converters/package-summary.html#TOC'>org.apache.juneau.rest.converters</a> package.
090    * <p>
091    * The programmatic equivalent to this annotation are the {@link RestConfig#addConverters(Class...)}/{@link RestConfig#addConverters(RestConverter...)} methods.
092    */
093   Class<? extends RestConverter>[] converters() default {};
094
095   /**
096    * Class-level bean filters.
097    * <p>
098    * Shortcut to add bean filters to the bean contexts of the objects returned by the following methods:
099    * <ul>
100    *    <li>{@link RestContext#getBeanContext()}
101    *    <li>{@link RestContext#getSerializers()}
102    *    <li>{@link RestContext#getParsers()}
103    * </ul>
104    * <p>
105    * If the specified class is an instance of {@link BeanFilterBuilder}, then a filter built from that builder is added.
106    * Any other classes are wrapped in a {@link InterfaceBeanFilterBuilder} to indicate that subclasses should
107    *    be treated as the specified class type.
108    * <p>
109    * The programmatic equivalent to this annotation is the {@link RestConfig#addBeanFilters(Class...)} method.
110    */
111   Class<?>[] beanFilters() default {};
112
113   /**
114    * Class-level POJO swaps.
115    * <p>
116    * Shortcut to add POJO swaps to the bean contexts of the objects returned by the following methods:
117    * <ul>
118    *    <li>{@link RestContext#getBeanContext()}
119    *    <li>{@link RestContext#getSerializers()}
120    *    <li>{@link RestContext#getParsers()}
121    * </ul>
122    * <p>
123    * If the specified class is an instance of {@link PojoSwap}, then that swap is added.
124    * Any other classes are wrapped in a {@link SurrogateSwap}.
125    * <p>
126    * The programmatic equivalent to this annotation is the {@link RestConfig#addPojoSwaps(Class...)} method.
127    */
128   Class<?>[] pojoSwaps() default {};
129
130   /**
131    * Class-level Java method parameter resolvers.
132    * <p>
133    * By default, the Juneau framework will automatically Java method parameters of various types (e.g.
134    * <code>RestRequest</code>, <code>Accept</code>, <code>Reader</code>).
135    * This annotation allows you to provide your own resolvers for your own class types that you want resolved.
136    * <p>
137    * For example, if you want to pass in instances of <code>MySpecialObject</code> to your Java method, define
138    * the following resolver:
139    * <p class='bcode'>
140    *    <jk>public class</jk> MyRestParam <jk>extends</jk> RestParam {
141    *
142    *       <jc>// Must have no-arg constructor!</jc>
143    *       <jk>public</jk> MyRestParam() {
144    *          <jc>// First two parameters help with Swagger doc generation.</jc>
145    *          <jk>super</jk>(<jsf>QUERY</jsf>, <js>"myparam"</js>, MySpecialObject.<jk>class</jk>);
146    *       }
147    *
148    *       <jc>// The method that creates our object.
149    *       // In this case, we're taking in a query parameter and converting it to our object.</jc>
150    *       <jk>public</jk> Object resolve(RestRequest req, RestResponse res) <jk>throws</jk> Exception {
151    *          <jk>return new</jk> MySpecialObject(req.getQuery().get(<js>"myparam"</js>));
152    *       }
153    *    }
154    * </p>
155    * <p>
156    * <b>Note:</b>{@link RestParam} classes must have no-arg constructors.
157    * <p>
158    * The programmatic equivalent to this annotation is the {@link RestConfig#addParamResolvers(Class...)} method.
159    */
160   Class<? extends RestParam>[] paramResolvers() default {};
161
162   /**
163    * Class-level properties.
164    * <p>
165    * Shortcut for specifying class-level properties on this servlet to the objects returned by the following methods:
166    * <ul>
167    *    <li>{@link RestContext#getBeanContext()}
168    *    <li>{@link RestContext#getSerializers()}
169    *    <li>{@link RestContext#getParsers()}
170    * </ul>
171    * <p>
172    * Any of the following property names can be specified:
173    * <ul>
174    *    <li>{@link RestContext}
175    *    <li>{@link BeanContext}
176    *    <li>{@link SerializerContext}
177    *    <li>{@link ParserContext}
178    *    <li>{@link JsonSerializerContext}
179    *    <li>{@link RdfSerializerContext}
180    *    <li>{@link RdfParserContext}
181    *    <li>{@link RdfCommonContext}
182    *    <li>{@link XmlSerializerContext}
183    *    <li>{@link XmlParserContext}
184    * </ul>
185    * <p>
186    * Property values will be converted to the appropriate type.
187    * <p>
188    * In some cases, properties can be overridden at runtime through the {@link RestResponse#setProperty(String, Object)} method
189    *    or through a {@link Properties @Properties} annotated method parameter.
190    * <p>
191    * The programmatic equivalent to this annotation are the {@link RestConfig#setProperty(String, Object)}/{@link RestConfig#setProperties(java.util.Map)} methods.
192    */
193   Property[] properties() default {};

195   /**
196    * Specifies a list of {@link Serializer} classes to add to the list of serializers available for this servlet.
197    * <p>
198    * This annotation can only be used on {@link Serializer} classes that have no-arg constructors.
199    * <p>
200    * The programmatic equivalent to this annotation are the {@link RestConfig#addSerializers(Class...)}/{@link RestConfig#addSerializers(Serializer...)} methods.
201    */
202   Class<? extends Serializer>[] serializers() default {};

204   /**
205    * Specifies a list of {@link Parser} classes to add to the list of parsers available for this servlet.

207    * This annotation can only be used on {@link Parser} classes that have no-arg constructors.
208    * <p>
209    * The programmatic equivalent to this annotation are the {@link RestConfig#addParsers(Class...)}/{@link RestConfig#addParsers(Parser...)} methods.
210    */
211   Class<? extends Parser>[] parsers() default {};

213   /**
214    * Specifies a list of {@link ResponseHandler} classes that know how to convert POJOs returned
215    *    by REST methods or set via {@link RestResponse#setOutput(Object)} into appropriate
216    *    HTTP responses.
217    * <p>
218    * See {@link ResponseHandler} for details.
219    * <p>
220    * The programmatic equivalent to this annotation are the {@link RestConfig#addResponseHandlers(Class...)}/{@link RestConfig#addResponseHandlers(ResponseHandler...)} methods.
221    */
222   Class<? extends ResponseHandler>[] responseHandlers() default {};

224   /**
225    * Specifies a list of {@link Encoder} to associate with this servlet.
226    * <p>
227    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
228    * <p>
229    * This annotation can only be used on {@link Encoder} classes that have no-arg constructors.
230    *
231    * <h5 class='section'>Example:</h5>
232    * <p class='bcode'>
233    *    <jc>// Servlet with automated support for GZIP compression</jc>
234    *    <ja>@RestResource</ja>(encoders={GzipEncoder.<jk>class</jk>})
235    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
236    *       ...
237    *    }
238    * </p>
239    * <p>
240    * The programmatic equivalent to this annotation are the {@link RestConfig#addEncoders(Class...)}/{@link RestConfig#addEncoders(Encoder...)} methods.
241    */
242   Class<? extends Encoder>[] encoders() default {};

244   /**
245    * Specifies default values for request headers.
246    * <p>
247    * Strings are of the format <js>"Header-Name: header-value"</js>.
248    * <p>
249    * Affects values returned by {@link RestRequest#getHeader(String)} when the header is not present on the request.
250    * <p>
251    * The most useful reason for this annotation is to provide a default <code>Accept</code> header when one is not specified
252    *    so that a particular default {@link Serializer} is picked.
253    * <p>
254    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
255    *
256    * <h5 class='section'>Example:</h5>
257    * <p class='bcode'>
258    *    <jc>// Assume "text/json" Accept value when Accept not specified</jc>
259    *    <ja>@RestResource</ja>(defaultRequestHeaders={<js>"Accept: text/json"</js>})
260    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
261    *       ...
262    *    }
263    * </p>
264    * <p>
265    * The programmatic equivalent to this annotation are the {@link RestConfig#addDefaultRequestHeader(String, Object)}/{@link RestConfig#addDefaultRequestHeaders(String...)} methods.
266    */
267   String[] defaultRequestHeaders() default {};
268
269   /**
270    * Specifies default values for response headers.
271    * <p>
272    * Strings are of the format <js>"Header-Name: header-value"</js>.
273    * <p>
274    * This is equivalent to calling {@link RestResponse#setHeader(String, String)} programmatically in each of the Java methods.
275    * <p>
276    * The header value will not be set if the header value has already been specified (hence the 'default' in the name).
277    * <p>
278    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
279    *
280    * <h5 class='section'>Example:</h5>
281    * <p class='bcode'>
282    *    <jc>// Add a version header attribute to all responses</jc>
283    *    <ja>@RestResource</ja>(defaultResponseHeaders={<js>"X-Version: 1.0"</js>})
284    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
285    *       ...
286    *    }
287    * </p>
288    * <p>
289    * The programmatic equivalent to this annotation are the {@link RestConfig#addDefaultResponseHeader(String, Object)}/{@link RestConfig#addDefaultResponseHeaders(String...)} methods.
290    */
291   String[] defaultResponseHeaders() default {};

293   /**
294    * Defines children of this resource.
295    * <p>
296    * A REST child resource is simply another servlet that is initialized as part of the parent
297    *    resource and has a servlet path directly under the parent servlet path.
298    * The main advantage to defining servlets as REST children is that you do not need
299    *    to define them in the <code>web.xml</code> file of the web application.
300    * This can cut down on the number of entries that show up in the <code>web.xml</code> file
301    *    if you are defining large numbers of servlets.
302    * <p>
303    * Child resources must specify a value for {@link #path()} that identifies the subpath of the
304    *    child resource relative to the parent path.
305    * <p>
306    * It should be noted that servlets can be nested arbitrarily deep using this technique (i.e. children can also have children).
307    *
308    * <dl>
309    *    <dt>Servlet initialization:</dt>
310    *    <dd>
311    *       <p>
312    *          A child resource will be initialized immediately after the parent servlet is initialized.  The child resource
313    *          receives the same servlet config as the parent resource.  This allows configuration information such as
314    *          servlet initialization parameters to filter to child resources.
315    *       </p>
316    *    </dd>
317    *    <dt>Runtime behavior:</dt>
318    *    <dd>
319    *       <p>
320    *          As a rule, methods defined on the <code>HttpServletRequest</code> object will behave as if
321    *          the child servlet were deployed as a top-level resource under the child's servlet path.
322    *          For example, the <code>getServletPath()</code> and <code>getPathInfo()</code> methods on the
323    *          <code>HttpServletRequest</code> object will behave as if the child resource were deployed
324    *          using the child's servlet path.
325    *          Therefore, the runtime behavior should be equivalent to deploying the child servlet in
326    *          the <code>web.xml</code> file of the web application.
327    *       </p>
328    *    </dd>
329    * </dl>
330    * <p>
331    * The programmatic equivalent to this annotation are the {@link RestConfig#addChildResource(String, Object)}/{@link RestConfig#addChildResources(Class...)}/{@link RestConfig#addChildResources(Object...)} methods.
332    */
333   Class<?>[] children() default {};
334
335   /**
336    * Identifies the URL subpath relative to the parent resource.
337    * <p>
338    * Typically, this annotation is only applicable to resources defined as children through the {@link #children()}
339    *    annotation.  However, it may be used in other ways (e.g. defining paths for top-level resources in microservices).
340    * <p>
341    * This annotation is ignored on top-level servlets (i.e. servlets defined in <code>web.xml</code> files).
342    * Therefore, implementers can optionally specify a path value for documentation purposes.
343    * <p>
344    * The programmatic equivalent to this annotation is the {@link RestConfig#setPath(String)} method.
345    */
346   String path() default "";

549    * The stylesheet to use for HTML views.
550    * <p>
551    * The name is a path to a stylesheet located in either the classpath or working directory.
552    * The resulting stylesheet becomes available through the servlet via the URL <js>"[servletpath]/style.css"</js>.
553    * <p>

561    *    chain.  Since the {@link RestServlet} class is in the <code>org.apache.juneau.rest</code> package
562    *    and the predefined styles are in the <code>org.apache.juneau.rest.styles</code> package, the paths to
563    *    the predefined styles are prefixed with <js>"styles/"</js>.
564    * <p>
565    * If the stylesheet cannot be found on the classpath, an attempt to look in the working directory
566    *    for it will be made.  This allows for stylesheets to be placed on the file system in the working
567    *    directory.

576    *       stylesheet=<js>"mystyles/mycss.css"</js>
577    *    )
578    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
579    *    }
580    * </p>
581    * <p>

583    * </p>
584    * <ol>
585    *    <li><code>com.foo.mypackage.mystyles</code> package.
586    *    <li><code>org.apache.juneau.rest.mystyles</code> package (since <code>RestServletDefault</code> is in <code>org.apache.juneau.rest</code>).
587    *    <li><code>[working-dir]/mystyles</code> directory.
588    * </ol>
