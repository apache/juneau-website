133   /**
134    * Calls {@link CloseableHttpClient#close()} on the underlying {@link CloseableHttpClient}.
135    * It's good practice to call this method after the client is no longer used.
136    *
137    * @throws IOException
138    */
139   public void close() throws IOException {
140      isClosed = true;
141      if (httpClient != null && ! keepHttpClientOpen)
142         httpClient.close();
143      if (executorService != null && executorServiceShutdownOnClose)
144         executorService.shutdown();
145      if (Boolean.getBoolean("org.apache.juneau.rest.client.RestClient.trackLifecycle"))
146         closedStack = Thread.currentThread().getStackTrace();
147   }

149   /**
150    * Same as {@link #close()}, but ignores any exceptions.
151    */
152   public void closeQuietly() {
153      isClosed = true;
154      try {
155         if (httpClient != null && ! keepHttpClientOpen)
156            httpClient.close();
157         if (executorService != null && executorServiceShutdownOnClose)
158            executorService.shutdown();
159      } catch (Throwable t) {}
160      if (Boolean.getBoolean("org.apache.juneau.rest.client.RestClient.trackLifecycle"))
161         closedStack = Thread.currentThread().getStackTrace();
162   }

282   /**
283    * Performs a REST call where the entire call is specified in a simple string.
284    * <p>
285    * This method is useful for performing callbacks when the target of a callback is passed in
286    * on an initial request, for example to signal when a long-running process has completed.
287    * <p>
288    * The call string can be any of the following formats:
289    * <ul class='spaced-list'>
290    *    <li><js>"[method] [url]"</js> - e.g. <js>"GET http://localhost/callback"</js>
291    *    <li><js>"[method] [url] [payload]"</js> - e.g. <js>"POST http://localhost/callback some text payload"</js>
292    *    <li><js>"[method] [headers] [url] [payload]"</js> - e.g. <js>"POST {'Content-Type':'text/json'} http://localhost/callback {'some':'json'}"</js>
293    * </ul>
294    * <p>
295    * The payload will always be sent using a simple {@link StringEntity}.
296    *
297    * @param callString The call string.
298    * @return A {@link RestCall} object that can be further tailored before executing the request
299    *    and getting the response as a parsed object.
300    * @throws RestCallException
301    */
302   public RestCall doCallback(String callString) throws RestCallException {
303      String s = callString;
304      try {
305         RestCall rc = null;
306         String method = null, uri = null, content = null;
307         ObjectMap h = null;
308         int i = s.indexOf(' ');
309         if (i != -1) {
310            method = s.substring(0, i).trim();
311            s = s.substring(i).trim();
312            if (s.length() > 0) {
313               if (s.charAt(0) == '{') {
314                  i = s.indexOf('}');
315                  if (i != -1) {
316                     String json = s.substring(0, i+1);
317                     h = JsonParser.DEFAULT.parse(json, ObjectMap.class);
318                     s = s.substring(i+1).trim();
319                  }
320               }
321               if (s.length() > 0) {
322                  i = s.indexOf(' ');
323                  if (i == -1)
324                     uri = s;
325                  else {
326                     uri = s.substring(0, i).trim();
327                     s = s.substring(i).trim();
328                     if (s.length() > 0)
329                        content = s;
330                  }
331               }
332            }
333         }
334         if (method != null && uri != null) {
335            rc = doCall(method, uri, content != null);
336            if (content != null)
337               rc.input(new StringEntity(content));
338            if (h != null)
339               for (Map.Entry<String,Object> e : h.entrySet())
340                  rc.header(e.getKey(), e.getValue());
341            return rc;
342         }
343      } catch (Exception e) {
344         throw new RestCallException(e);
345      }
346      throw new RestCallException("Invalid format for call string.");
347   }

353    * @param url The URL of the remote REST resource.  Can be any of the following:  {@link String}, {@link URI}, {@link URL}.

369      RestCall rc = doCall(method.name(), url, method.hasContent());
370      if (method.hasContent())
371         rc.input(content);
372      return rc;
373   }

375   /**
376    * Perform a generic REST call.
377    *
378    * @param method The method name (e.g. <js>"GET"</js>, <js>"OPTIONS"</js>).
379    * @param url The URL of the remote REST resource.  Can be any of the following:  {@link String}, {@link URI}, {@link URL}.
380    * @param hasContent Boolean flag indicating if the specified request has content associated with it.
381    * @return A {@link RestCall} object that can be further tailored before executing the request
382    *    and getting the response as a parsed object.
383    * @throws RestCallException If any authentication errors occurred.
384    */
385   public RestCall doCall(String method, Object url, boolean hasContent) throws RestCallException {
386      if (isClosed) {
387         Exception e2 = null;
388         if (closedStack != null) {
389            e2 = new Exception("Creation stack:");
390            e2.setStackTrace(closedStack);
391            throw new RestCallException("RestClient.close() has already been called.  This client cannot be reused.").initCause(e2);
392         }
393         throw new RestCallException("RestClient.close() has already been called.  This client cannot be reused.  Closed location stack trace can be displayed by setting the system property 'org.apache.juneau.rest.client.RestClient.trackCreation' to true.");
394      }
395
396      HttpRequestBase req = null;
397      RestCall restCall = null;
398      final String methodUC = method.toUpperCase(Locale.ENGLISH);
399      try {
400         if (hasContent) {
401            req = new HttpEntityEnclosingRequestBase() {
402               @Override /* HttpRequest */
403               public String getMethod() {
404                  return methodUC;
405               }
406            };
407            restCall = new RestCall(this, req, toURI(url));
408         } else {
409            req = new HttpRequestBase() {
410               @Override /* HttpRequest */
411               public String getMethod() {
412                  return methodUC;
413               }
414            };
415            restCall = new RestCall(this, req, toURI(url));
416         }
417      } catch (URISyntaxException e1) {
418         throw new RestCallException(e1);
419      }
420      for (Map.Entry<String,? extends Object> e : headers.entrySet())
421         restCall.header(e.getKey(), e.getValue());
422
423      if (parser != null && ! req.containsHeader("Accept"))
424         req.setHeader("Accept", parser.getPrimaryMediaType().toString());
425
426      return restCall;
427   }

429   /**
430    * Create a new proxy interface against a REST interface.
431    * <p>
432    * The URL to the REST interface is based on the following values:
433    * <ul>
434    *    <li>The {@link Remoteable#path() @Remoteable.path()} annotation on the interface (<code>remoteable-path</code>).
435    *    <li>The {@link RestClientBuilder#rootUrl(Object) rootUrl} on the client (<code>root-url</code>).
436    *    <li>The fully-qualified class name of the interface (<code>class-name</code>).
437    * </ul>
438    * <p>
439    * The URL calculation is as follows:
440    * <ul>
441    *    <li><code>remoteable-path</code> - If remoteable path is absolute.
442    *    <li><code>root-url/remoteable-path</code> - If remoteable path is relative and root-url has been specified.
443    *    <li><code>root-url/class-name</code> - If remoteable path is not specified.
444    * </ul>
445    * <p>
446    * If the information is not available to resolve to an absolute URL, a {@link RemoteableMetadataException} is thrown.
447    * <p>
448    * Examples:
449    * <p class='bcode'>
450    *    <jk>package</jk> org.apache.foo;
451    *
452    *    <ja>@Remoteable</ja>(path=<js>"http://hostname/resturl/myinterface1"</js>)
453    *    <jk>public interface</jk> MyInterface1 { ... }
454    *
455    *    <ja>@Remoteable</ja>(path=<js>"/myinterface2"</js>)
456    *    <jk>public interface</jk> MyInterface2 { ... }
457    *
458    *    <jk>public interface</jk> MyInterface3 { ... }
459    *
460    *    <jc>// Resolves to "http://localhost/resturl/myinterface1"</jc>
461    *    MyInterface1 i1 = <jk>new</jk> RestClientBuilder()
462    *       .build()
463    *       .getRemoteableProxy(MyInterface1.<jk>class</jk>);
464    *
465    *    <jc>// Resolves to "http://hostname/resturl/myinterface2"</jc>
466    *    MyInterface2 i2 = <jk>new</jk> RestClientBuilder()
467    *       .rootUrl(<js>"http://hostname/resturl"</js>)
468    *       .build()
469    *       .getRemoteableProxy(MyInterface2.<jk>class</jk>);
470    *
471    *    <jc>// Resolves to "http://hostname/resturl/org.apache.foo.MyInterface3"</jc>
472    *    MyInterface3 i3 = <jk>new</jk> RestClientBuilder()
473    *       .rootUrl(<js>"http://hostname/resturl"</js>)
474    *       .build()
475    *       .getRemoteableProxy(MyInterface3.<jk>class</jk>);
476    * </p>
477    * <h5 class='section'>Notes:</h5>
478    * <ul>
479    *    <li>If you plan on using your proxy in a multi-threaded environment, you'll want to use an underlying
480    *       pooling client connection manager.  The easiest way to do this is to use the {@link RestClientBuilder#pooled()}
481    *       method.  If you don't do this, you may end up seeing "Connection still allocated" exceptions.
482    * </ul>
483    *
484    * @param interfaceClass The interface to create a proxy for.
485    * @return The new proxy interface.
486    * @throws RemoteableMetadataException If the REST URI cannot be determined based on the information given.
487    */
488   public <T> T getRemoteableProxy(final Class<T> interfaceClass) {
489      return getRemoteableProxy(interfaceClass, null);
490   }

503   /**
504    * Same as {@link #getRemoteableProxy(Class, Object)} but allows you to override the serializer and parser used.
505    *
506    * @param interfaceClass The interface to create a proxy for.
507    * @param restUrl The URL of the REST interface.
508    * @param serializer The serializer used to serialize POJOs to the body of the HTTP request.
509    * @param parser The parser used to parse POJOs from the body of the HTTP response.
510    * @return The new proxy interface.
511    */
512   @SuppressWarnings({ "unchecked", "hiding" })
513   public <T> T getRemoteableProxy(final Class<T> interfaceClass, Object restUrl, final Serializer serializer, final Parser parser) {
514
515      if (restUrl == null) {
516         Remoteable r = ReflectionUtils.getAnnotation(Remoteable.class, interfaceClass);
517
518         String path = r == null ? "" : trimSlashes(r.path());
519         if (path.indexOf("://") == -1) {
520            if (path.isEmpty())
521               path = interfaceClass.getName();
522            if (rootUrl == null)
523               throw new RemoteableMetadataException(interfaceClass, "Root URI has not been specified.  Cannot construct absolute path to remoteable proxy.");
524            path = trimSlashes(rootUrl) + '/' + path;
525         }
526         restUrl = path;
527      }
528
529      final String restUrl2 = restUrl.toString();
530
531      try {
532         return (T)Proxy.newProxyInstance(
533            interfaceClass.getClassLoader(),
534            new Class[] { interfaceClass },
535            new InvocationHandler() {
536
537               final RemoteableMeta rm = new RemoteableMeta(interfaceClass, restUrl2);
538
539               @Override /* InvocationHandler */
540               public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
541                  RemoteableMethodMeta rmm = rm.getMethodMeta(method);
542
543                  if (rmm == null)
544                     throw new RuntimeException("Method is not exposed as a remoteable method.");
545
546                  try {
547                     String url = rmm.getUrl();
548                     String httpMethod = rmm.getHttpMethod();
549                     RestCall rc = (httpMethod.equals("POST") ? doPost(url) : doGet(url));
550                     rc.serializer(serializer).parser(parser);
551
552                     for (RemoteMethodArg a : rmm.getPathArgs())
553                        rc.path(a.name, args[a.index], a.serializer);
554
555                     for (RemoteMethodArg a : rmm.getQueryArgs())
556                        rc.query(a.name, args[a.index], a.skipIfNE, a.serializer);
557
558                     for (RemoteMethodArg a : rmm.getFormDataArgs())
559                        rc.formData(a.name, args[a.index], a.skipIfNE, a.serializer);
560
561                     for (RemoteMethodArg a : rmm.getHeaderArgs())
562                        rc.header(a.name, args[a.index], a.skipIfNE, a.serializer);
563
564                     if (rmm.getBodyArg() != null)
565                        rc.input(args[rmm.getBodyArg()]);
566
567                     if (rmm.getRequestBeanArgs().length > 0) {
568                        BeanSession bs = getBeanContext().createSession();
569
570                        for (RemoteMethodArg rma : rmm.getRequestBeanArgs()) {
571                           BeanMap<?> bm = bs.toBeanMap(args[rma.index]);
572
573                           for (BeanPropertyValue bpv : bm.getValues(false)) {
574                              BeanPropertyMeta pMeta = bpv.getMeta();
575                              Object val = bpv.getValue();
576
577                              Path p = pMeta.getAnnotation(Path.class);
578                              if (p != null)
579                                 rc.path(getName(p.name(), p.value(), pMeta), val, getPartSerializer(p.serializer(), rma.serializer));
580
581                              if (val != null) {
582                                 Query q1 = pMeta.getAnnotation(Query.class);
583                                 if (q1 != null)
584                                    rc.query(getName(q1.name(), q1.value(), pMeta), val, q1.skipIfEmpty(), getPartSerializer(q1.serializer(), rma.serializer));
585
586                                 QueryIfNE q2 = pMeta.getAnnotation(QueryIfNE.class);
587                                 if (q2 != null)
588                                    rc.query(getName(q2.name(), q2.value(), pMeta), val, true, getPartSerializer(q2.serializer(), rma.serializer));
589
590                                 FormData f1 = pMeta.getAnnotation(FormData.class);
591                                 if (f1 != null)
592                                    rc.formData(getName(f1.name(), f1.value(), pMeta), val, f1.skipIfEmpty(), getPartSerializer(f1.serializer(), rma.serializer));
593
594                                 FormDataIfNE f2 = pMeta.getAnnotation(FormDataIfNE.class);
595                                 if (f2 != null)
596                                    rc.formData(getName(f2.name(), f2.value(), pMeta), val, true, getPartSerializer(f2.serializer(), rma.serializer));
597
598                                 org.apache.juneau.remoteable.Header h1 = pMeta.getAnnotation(org.apache.juneau.remoteable.Header.class);
599                                 if (h1 != null)
600                                    rc.header(getName(h1.name(), h1.value(), pMeta), val, h1.skipIfEmpty(), getPartSerializer(h1.serializer(), rma.serializer));
601
602                                 HeaderIfNE h2 = pMeta.getAnnotation(HeaderIfNE.class);
603                                 if (h2 != null)
604                                    rc.header(getName(h2.name(), h2.value(), pMeta), val, true, getPartSerializer(h2.serializer(), rma.serializer));
605                              }
606                           }
607                        }
608                     }
609
610                     if (rmm.getOtherArgs().length > 0) {
611                        Object[] otherArgs = new Object[rmm.getOtherArgs().length];
612                        int i = 0;
613                        for (Integer otherArg : rmm.getOtherArgs())
614                           otherArgs[i++] = args[otherArg];
615                        rc.input(otherArgs);
616                     }
617
618                     return rc.getResponse(method.getGenericReturnType());
619
620                  } catch (RestCallException e) {
621                     // Try to throw original exception if possible.
622                     e.throwServerException(interfaceClass.getClassLoader());
623                     throw new RuntimeException(e);
624                  } catch (Exception e) {
625                     throw new RuntimeException(e);
626                  }
627               }
628         });
629      } catch (Exception e) {
630         throw new RuntimeException(e);
631      }
632   }

634   private static String getName(String name1, String name2, BeanPropertyMeta pMeta) {
635      String n = name1.isEmpty() ? name2 : name1;
636      ClassMeta<?> cm = pMeta.getClassMeta();
637      if (n.isEmpty() && (cm.isMapOrBean() || cm.isReader() || cm.isInstanceOf(NameValuePairs.class)))
638         n = "*";
639      if (n.isEmpty())
640         n = pMeta.getName();
641      return n;
642   }

644   private static PartSerializer getPartSerializer(Class c, PartSerializer c2) {
645      if (c2 != null)
646         return c2;
647      if (c == PartSerializer.class)
648         return null;
649      PartSerializer pf = partSerializerCache.get(c);
650      if (pf == null) {
651         try {
652            partSerializerCache.putIfAbsent(c, (PartSerializer)c.newInstance());
653         } catch (Exception e) {
654            throw new RuntimeException(e);
655         }
656         pf = partSerializerCache.get(c);
657      }
658      return pf;
659   }

667   URI toURI(Object url) throws URISyntaxException {
668      if (url instanceof URI)
669         return (URI)url;
670      if (url instanceof URL)
671         ((URL)url).toURI();
672      if (url instanceof URIBuilder)
673         return ((URIBuilder)url).build();
674      String s = url == null ? "" : url.toString();
675      if (rootUrl != null && ! absUrlPattern.matcher(s).matches()) {
676         if (s.isEmpty())
677            s = rootUrl;
678         else {
679            StringBuilder sb = new StringBuilder(rootUrl);
680            if (! s.startsWith("/"))
681               sb.append('/');
682            sb.append(s);
683            s = sb.toString();
684         }
685      }
686      if (s.indexOf('{') != -1)
687         s = s.replace("{", "%7B").replace("}", "%7D");
688      return new URI(s);
689   }
