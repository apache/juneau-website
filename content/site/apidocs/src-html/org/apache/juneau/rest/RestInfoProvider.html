029/**
030 * Class that provides documentation and other related information about a REST resource.
031 *
032 * <p>
033 * Subclasses can override these methods to tailor how HTTP REST resources are documented.
034 * Subclasses MUST implement a public constructor that takes in a {@link RestContext} object.
035 *
036 * <p>
037 * RestInfoProviders are associated with servlets/resources in one of the following ways:
038 * <ul>
039 *    <li>The {@link RestResource#infoProvider @RestResource.infoProvider()} annotation.
040 *    <li>The {@link RestConfig#setInfoProvider(Class)}/{@link RestConfig#setInfoProvider(RestInfoProvider)} methods.
041 * </ul>
042 */
043public class RestInfoProvider {
044
045   private final RestContext context;
046   private final String
047      siteName,
048      title,
049      description,
050      termsOfService,
051      contact,
052      license,
053      version,
054      tags,
055      externalDocs;
056   private final ConcurrentHashMap<Locale,Swagger> swaggers = new ConcurrentHashMap<>();
057
058   /**
059    * Constructor.
060    *
061    * @param context The resource context.
062    */
063   public RestInfoProvider(RestContext context) {
064      this.context = context;
065
066      Builder b = new Builder(context);
067      this.siteName = b.siteName;
068      this.title = b.title;
069      this.description = b.description;
070      this.termsOfService = b.termsOfService;
071      this.contact = b.contact;
072      this.license = b.license;
073      this.version = b.version;
074      this.tags = b.tags;
075      this.externalDocs = b.externalDocs;
076   }
077
078   private static final class Builder {
079      String
080         siteName,
081         title,
082         description,
083         termsOfService,
084         contact,
085         license,
086         version,
087         tags,
088         externalDocs;
089
090      Builder(RestContext context) {
091
092         LinkedHashMap<Class<?>,RestResource> restResourceAnnotationsParentFirst = findAnnotationsMapParentFirst(RestResource.class, context.getResource().getClass());
093
094         for (RestResource r : restResourceAnnotationsParentFirst.values()) {
095            if (! r.siteName().isEmpty())
096               siteName = r.siteName();
097            if (! r.title().isEmpty())
098               title = r.title();
099            if (! r.description().isEmpty())
100               description = r.description();
101            ResourceSwagger sr = r.swagger();
102            if (! sr.termsOfService().isEmpty())
103               termsOfService = sr.termsOfService();
104            if (! sr.contact().isEmpty())
105               contact = sr.contact();
106            if (! sr.license().isEmpty())
107               license = sr.license();
108            if (! sr.version().isEmpty())
109               version = sr.version();
110            if (! sr.tags().isEmpty())
111               tags = sr.tags();
112            if (! sr.externalDocs().isEmpty())
113               externalDocs = sr.externalDocs();
114         }
115      }
116   }

118   /**
119    * Returns the localized swagger for this REST resource.
120    *
121    * @param req The incoming HTTP request.
122    * @return A new Swagger instance.
123    * @throws RestException
124    */
125   protected Swagger getSwagger(RestRequest req) throws RestException {
126      try {
127         // If a file is defined, use that.
128         Swagger s = req.getSwaggerFromFile();
129         if (s != null)
130            return s;
131
132         s = swagger(
133            info(getTitle(req), getVersion(req))
134               .contact(getContact(req))
135               .license(getLicense(req))
136               .description(getDescription(req))
137               .termsOfService(getTermsOfService(req))
138            )
139            .consumes(context.getSupportedAcceptTypes())
140            .produces(context.getSupportedContentTypes())
141            .tags(getTags(req))
142            .externalDocs(getExternalDocs(req));
143
144         for (CallMethod sm : context.getCallMethods().values()) {
145            if (sm.isRequestAllowed(req)) {
146               Operation o = sm.getSwaggerOperation(req);
147               s.path(
148                  sm.getPathPattern(),
149                  sm.getHttpMethod().toLowerCase(),
150                  o
151               );
152            }
153         }
154         return s;
155      } catch (RestException e) {
156         throw e;
157      } catch (Exception e) {
158         throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
159      }
160   }

162   /**
163    * Returns the localized Swagger from the file system.
164    *
165    * <p>
166    * Looks for a file called <js>"{ServletClass}_{locale}.json"</js> in the same package as this servlet and returns
167    * it as a parsed {@link Swagger} object.
168    *
169    * <p>
170    * Returned objects are cached for later quick-lookup.
171    *
172    * @param locale The locale of the swagger.
173    * @return The parsed swagger object, or <jk>null</jk> if the swagger file could not be found.
174    * @throws RestException
175    */
176   protected Swagger getSwaggerFromFile(Locale locale) throws RestException {
177      Swagger s = swaggers.get(locale);
178      if (s == null) {
179         try {
180            s = context.getResource(Swagger.class, MediaType.JSON, getClass().getSimpleName() + ".json", locale);
181            swaggers.putIfAbsent(locale, s == null ? Swagger.NULL : s);
182         } catch (Exception e) {
183            throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
184         }
185      }
186      return s == Swagger.NULL ? null : s;
187   }

606         String externalDocs = context.getMessages().findFirstString(req.getLocale(), "externalDocs");
607         if (externalDocs != null)
608            return jp.parse(vr.resolve(externalDocs), ExternalDocumentation.class);
609         Swagger s = req.getSwaggerFromFile();
610         if (s != null)
611            return s.getExternalDocs();
612         return null;
613      } catch (Exception e) {
614         throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
615      }
616   }
