059    * <b>Configuration property:</b>  Enable header URL parameters.
060    *
061    * <ul>
062    *    <li><b>Name:</b> <js>"RestServlet.allowHeaderParams"</js>
063    *    <li><b>Data type:</b> <code>Boolean</code>
064    *    <li><b>Default:</b> <jk>true</jk>
065    * </ul>
066    *
067    * <p>
068    * When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query
069    * parameters.
070    * For example:  <js>"?Accept=text/json&amp;Content-Type=text/json"</js>
071    *
072    * <p>
073    * Parameter names are case-insensitive.
074    *
075    * <p>
076    * Useful for debugging REST interface using only a browser.
077    *
078    * <p>
079    * Applicable to servlet class only.
080    */
081   public static final String REST_allowHeaderParams = "RestServlet.allowHeaderParams";

083   /**
084    * <b>Configuration property:</b>  Enable <js>"method"</js> URL parameter for specific HTTP methods.
085    *
086    * <ul>
087    *    <li><b>Name:</b> <js>"RestServlet.allowMethodParam"</js>
088    *    <li><b>Data type:</b> <code>String</code>
089    *    <li><b>Default:</b> <js>""</js>
090    * </ul>
091    *
092    * <p>
093    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular
094    * GET request.
095    * For example:  <js>"?method=OPTIONS"</js>
096    *
097    * <p>
098    * Format is a comma-delimited list of HTTP method names that can be passed in as a method parameter.
099    * Parameter name is case-insensitive.
100    * Use "*" to represent all methods.
101    * For backwards compatibility, "true" also means "*".
102    *
103    * <p>
104    * Note that per the <a class="doclink"
105    * href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html">HTTP specification</a>, special care should
106    * be taken when allowing non-safe (POST, PUT, DELETE) methods to be invoked through GET requests.
107    *
108    * <p>
109    * Applicable to servlet class only.
110    *
111    * <p>
112    * Example: <js>"HEAD,OPTIONS"</js>
113    */
114   public static final String REST_allowMethodParam = "RestServlet.allowMethodParam";

413   /**
414    * Constructor.
415    *
416    * @param resource The resource class (a class annotated with {@link RestResource @RestResource}).
417    * @param config The servlet configuration object.
418    * @throws Exception If any initialization problems were encountered.
419    */
420   @SuppressWarnings("unchecked")
421   public RestContext(Object resource, RestConfig config) throws Exception {
422      super(null);
423      RestException _initException = null;
424      try {
425         this.resource = resource;
426         this.config = config;
427         this.resourceFinder = new ResourceFinder(resource.getClass());
428         this.parentContext = config.parentContext;
429
430         Builder b = new Builder(resource, config);
431         this.allowHeaderParams = b.allowHeaderParams;
432         this.allowBodyParam = b.allowBodyParam;
433         this.renderResponseStackTraces = b.renderResponseStackTraces;
434         this.useStackTraceHashes = b.useStackTraceHashes;
435         this.allowMethodParams = Collections.unmodifiableSet(b.allowMethodParams);
436         this.defaultCharset = b.defaultCharset;
437         this.paramFormat = b.paramFormat;
438         this.varResolver = b.varResolver;
439         this.configFile = b.configFile;
440         this.properties = b.properties;
441         this.beanFilters = b.beanFilters;
442         this.pojoSwaps = b.pojoSwaps;
443         this.paramResolvers = Collections.unmodifiableMap(b.paramResolvers);
444         this.serializers = b.serializers;
445         this.parsers = b.parsers;
446         this.urlEncodingSerializer = b.urlEncodingSerializer;
447         this.urlEncodingParser = b.urlEncodingParser;
448         this.encoders = b.encoders;
449         this.supportedContentTypes = toObjectArray(b.supportedContentTypes, MediaType.class);
450         this.supportedAcceptTypes = toObjectArray(b.supportedAcceptTypes, MediaType.class);
451         this.clientVersionHeader = b.clientVersionHeader;
452         this.defaultRequestHeaders = Collections.unmodifiableMap(b.defaultRequestHeaders);
453         this.defaultResponseHeaders = Collections.unmodifiableMap(b.defaultResponseHeaders);
454         this.beanContext = b.beanContext;
455         this.converters = b.converters.toArray(new RestConverter[b.converters.size()]);
456         this.guards = b.guards.toArray(new RestGuard[b.guards.size()]);
457         this.responseHandlers = toObjectArray(b.responseHandlers, ResponseHandler.class);
458         this.mimetypesFileTypeMap = b.mimetypesFileTypeMap;
459         this.favIcon = b.favIcon;
460         this.staticFilesMap = Collections.unmodifiableMap(b.staticFilesMap);
461         this.staticFilesPrefixes = b.staticFilesPrefixes;
462         this.msgs = b.messageBundle;
463         this.childResources = Collections.synchronizedMap(new LinkedHashMap<String,RestContext>());  // Not unmodifiable on purpose so that children can be replaced.
464         this.logger = b.logger;
465         this.fullPath = b.fullPath;
466         this.widgets = Collections.unmodifiableMap(b.widgets);
467
468         this.htmlTitle = b.htmlTitle;
469         this.htmlDescription = b.htmlDescription;
470         this.htmlBranding = b.htmlBranding;
471         this.htmlHeader = b.htmlHeader;
472         this.htmlLinks = b.htmlLinks;
473         this.htmlNav = b.htmlNav;
474         this.htmlAside = b.htmlAside;
475         this.htmlStyle = b.htmlStyle;
476         this.htmlStyleImport = b.htmlStyleImport;
477         this.htmlScript = b.htmlScript;
478         this.htmlFooter = b.htmlFooter;
479         this.htmlNoWrap = b.htmlNoWrap;
480         this.htmlNoResultsMessage = b.htmlNoResultsMessage;
481         this.htmlTemplate = b.htmlTemplate;
482
483         //----------------------------------------------------------------------------------------------------
484         // Initialize the child resources.
485         // Done after initializing fields above since we pass this object to the child resources.
486         //----------------------------------------------------------------------------------------------------
487         List<String> methodsFound = new LinkedList<String>();   // Temporary to help debug transient duplicate method issue.
488         Map<String,CallRouter.Builder> routers = new LinkedHashMap<String,CallRouter.Builder>();
489         Map<String,CallMethod> _javaRestMethods = new LinkedHashMap<String,CallMethod>();
490         for (java.lang.reflect.Method method : resource.getClass().getMethods()) {
491            if (method.isAnnotationPresent(RestMethod.class)) {
492               RestMethod a = method.getAnnotation(RestMethod.class);
493               methodsFound.add(method.getName() + "," + a.name() + "," + a.path());
494               try {
495                  if (! Modifier.isPublic(method.getModifiers()))
496                     throw new RestServletException("@RestMethod method {0}.{1} must be defined as public.", this.getClass().getName(), method.getName());
497
498                  CallMethod sm = new CallMethod(resource, method, this);
499                  String httpMethod = sm.getHttpMethod();
500
501                  // PROXY is a special case where a method returns an interface that we
502                  // can perform REST calls against.
503                  // We override the CallMethod.invoke() method to insert our logic.
504                  if ("PROXY".equals(httpMethod)) {
505
506                     final ClassMeta<?> interfaceClass = beanContext.getClassMeta(method.getGenericReturnType());
507                     final Map<String,Method> remoteableMethods = interfaceClass.getRemoteableMethods();
508                     if (remoteableMethods.isEmpty())
509                        throw new RestException(SC_INTERNAL_SERVER_ERROR, "Method {0} returns an interface {1} that doesn't define any remoteable methods.", getMethodSignature(method), interfaceClass.getReadableName());
510
511                     sm = new CallMethod(resource, method, this) {
512
513                        @Override
514                        int invoke(String pathInfo, RestRequest req, RestResponse res) throws RestException {
515
516                           int rc = super.invoke(pathInfo, req, res);
517                           if (rc != SC_OK)
518                              return rc;
519
520                           final Object o = res.getOutput();
521
522                           if ("GET".equals(req.getMethod())) {
523                              res.setOutput(getMethodInfo(remoteableMethods.values()));
524                              return SC_OK;
525
526                           } else if ("POST".equals(req.getMethod())) {
527                              if (pathInfo.indexOf('/') != -1)
528                                 pathInfo = pathInfo.substring(pathInfo.lastIndexOf('/')+1);
529                              pathInfo = urlDecode(pathInfo);
530                              java.lang.reflect.Method m = remoteableMethods.get(pathInfo);
531                              if (m != null) {
532                                 try {
533                                    // Parse the args and invoke the method.
534                                    Parser p = req.getBody().getParser();
535                                    Object input = p.isReaderParser() ? req.getReader() : req.getInputStream();
536                                    Object output = m.invoke(o, p.parseArgs(input, m.getGenericParameterTypes()));
537                                    res.setOutput(output);
538                                    return SC_OK;
539                                 } catch (Exception e) {
540                                    throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
541                                 }
542                              }
543                           }
544                           return SC_NOT_FOUND;
545                        }
546                     };
547
548                     _javaRestMethods.put(method.getName(), sm);
549                     addToRouter(routers, "GET", sm);
550                     addToRouter(routers, "POST", sm);
551
552                  } else {
553                     _javaRestMethods.put(method.getName(), sm);
554                     addToRouter(routers, httpMethod, sm);
555                  }
556               } catch (RestServletException e) {
557                  throw new RestServletException("Problem occurred trying to serialize methods on class {0}, methods={1}", this.getClass().getName(), JsonSerializer.DEFAULT_LAX.serialize(methodsFound)).initCause(e);
558               }
559            }
560         }
561         this.callMethods = Collections.unmodifiableMap(_javaRestMethods);
562
563         Map<String,CallRouter> _callRouters = new LinkedHashMap<String,CallRouter>();
564         for (CallRouter.Builder crb : routers.values())
565            _callRouters.put(crb.getHttpMethodName(), crb.build());
566         this.callRouters = Collections.unmodifiableMap(_callRouters);
567
568         // Initialize our child resources.
569         RestResourceResolver rrr = resolve(RestResourceResolver.class, config.resourceResolver);
570         for (Object o : config.childResources) {
571            String path = null;
572            Object r = null;
573            if (o instanceof Pair) {
574               Pair<String,Object> p = (Pair<String,Object>)o;
575               path = p.first();
576               r = p.second();
577            } else if (o instanceof Class<?>) {
578               Class<?> c = (Class<?>)o;
579               r = c;
580            } else {
581               r = o;
582            }
583
584            RestConfig childConfig = null;
585
586            if (o instanceof Class) {
587               Class<?> oc = (Class<?>)o;
588               childConfig = new RestConfig(config.inner, oc, this);
589               r = rrr.resolve(oc, childConfig);
590            } else {
591               r = o;
592               childConfig = new RestConfig(config.inner, o.getClass(), this);
593            }
594
595            if (r instanceof RestServlet) {
596               RestServlet rs = (RestServlet)r;
597               rs.init(childConfig);
598               if (rs.getContext() == null)
599                  throw new RestException(SC_INTERNAL_SERVER_ERROR, "Servlet {0} not initialized.  init(RestConfig) was not called.  This can occur if you've overridden this method but didn't call super.init(RestConfig).", rs.getClass().getName());
600               path = childConfig.path;
601               childResources.put(path, rs.getContext());
602            } else {
603
604               // Call the init(RestConfig) method.
605               java.lang.reflect.Method m2 = findPublicMethod(r.getClass(), "init", Void.class, RestConfig.class);
606               if (m2 != null)
607                  m2.invoke(r, childConfig);
608
609               RestContext rc2 = new RestContext(r, childConfig);
610
611               // Call the init(RestContext) method.
612               m2 = findPublicMethod(r.getClass(), "init", Void.class, RestContext.class);
613               if (m2 != null)
614                  m2.invoke(r, rc2);
615
616               path = childConfig.path;
617               childResources.put(path, rc2);
618            }
619         }
620
621         callHandler = config.callHandler == null ? new RestCallHandler(this) : resolve(RestCallHandler.class, config.callHandler, this);
622         infoProvider = config.infoProvider == null ? new RestInfoProvider(this) : resolve(RestInfoProvider.class, config.infoProvider, this);
623
624      } catch (RestException e) {
625         _initException = e;
626         throw e;
627      } catch (Exception e) {

698            .vars(LocalizationVar.class, RequestVar.class, SerializedRequestAttrVar.class, ServletInitParamVar.class, UrlVar.class, UrlEncodeVar.class, WidgetVar.class)

711         }

719            if (! r.messages().isEmpty()) {

722               else

741         beanContext = ps.getBeanContext();
742
743         for (Object o : sc.converters)
744            converters.add(resolve(RestConverter.class, o));
745
746         for (Object o : sc.guards)
747            guards.add(resolve(RestGuard.class, o));
748
749         for (Object o : sc.responseHandlers)
750            responseHandlers.add(resolve(ResponseHandler.class, o));

752         mimetypesFileTypeMap = sc.mimeTypes;

868   /**
869    * Resolve a static resource file.
870    *
871    * <p>
872    * The location of static resources are defined via one of the following:
873    * <ul>
874    *    <li>{@link RestResource#staticFiles() @RestResource.staticFiles()} annotation.
875    *    <li>{@link RestConfig#addStaticFiles(Class, String)} method.
876    * </ul>
877    *
878    * @param pathInfo The unencoded path info.
879    * @return The resource, or <jk>null</jk> if the resource could not be resolved.
880    * @throws IOException
881    */
882   public StreamResource resolveStaticFile(String pathInfo) throws IOException {
883      if (! staticFilesCache.containsKey(pathInfo)) {
884         String p = urlDecode(trimSlashes(pathInfo));
885         if (p.indexOf("..") != -1)
886            throw new RestException(SC_NOT_FOUND, "Invalid path");
887         for (Map.Entry<String,String> e : staticFilesMap.entrySet()) {
888            String key = trimSlashes(e.getKey());
889            if (p.startsWith(key)) {
890               String remainder = (p.equals(key) ? "" : p.substring(key.length()));
891               if (remainder.isEmpty() || remainder.startsWith("/")) {
892                  String p2 = trimSlashes(e.getValue()) + remainder;
893                  InputStream is = getResource(p2, null);
894                  if (is != null) {
895                     try {
896                        int i = p2.lastIndexOf('/');
897                        String name = (i == -1 ? p2 : p2.substring(i+1));
898                        String mediaType = mimetypesFileTypeMap.getContentType(name);
899                        ObjectMap headers = new ObjectMap().append("Cache-Control", "max-age=86400, public");
900                        staticFilesCache.put(pathInfo, new StreamResource(MediaType.forString(mediaType), headers, is));
901                        return staticFilesCache.get(pathInfo);
902                     } finally {
903                        is.close();
904                     }
905                  }
906               }
907            }
908         }
909      }
910      return staticFilesCache.get(pathInfo);
911   }

953    * matched by the specified media type.
954    *
955    * <p>
956    * Useful if you want to load predefined POJOs from JSON files in your classpath.
957    *
958    * @param c The class type of the POJO to create.
959    * @param mediaType The media type of the data in the stream (e.g. <js>"text/json"</js>)
960    * @param name The resource name (e.g. "htdocs/styles.css").
961    * @param locale Optional locale.
962    * @return The parsed resource, or <jk>null</jk> if the resource could not be found.
963    * @throws IOException
964    * @throws ServletException If the media type was unknown or the input could not be parsed into a POJO.
965    */
966   public <T> T getResource(Class<T> c, MediaType mediaType, String name, Locale locale) throws IOException, ServletException {
967      InputStream is = getResource(name, locale);
968      if (is == null)
969         return null;
970      try {
971         Parser p = parsers.getParser(mediaType);
972         if (p != null) {
973            try {
974               if (p.isReaderParser())
975                  return p.parse(new InputStreamReader(is, UTF8), c);
976               return p.parse(is, c);
977            } catch (ParseException e) {
978               throw new ServletException("Could not parse resource '' as media type '"+mediaType+"'.");
979            }
980         }
981         throw new ServletException("Unknown media type '"+mediaType+"'");
982      } catch (Exception e) {
983         throw new ServletException("Could not parse resource with name '"+name+"'", e);
984      }

1509   /**
1510    * Finds the {@link RestParam} instances to handle resolving objects on the calls to the specified Java method.
1511    *
1512    * @param method The Java method being called.
1513    * @param methodPlainParams Whether plain-params setting is specified.
1514    * @param pathPattern The parsed URL path pattern.
1515    * @return The array of resolvers.
1516    * @throws ServletException If an annotation usage error was detected.
1517    */
1518   protected RestParam[] findParams(Method method, boolean methodPlainParams, UrlPathPattern pathPattern) throws ServletException {
1519
1520      Type[] pt = method.getGenericParameterTypes();
1521      Annotation[][] pa = method.getParameterAnnotations();
1522      RestParam[] rp = new RestParam[pt.length];
1523      int attrIndex = 0;
1524
1525      for (int i = 0; i < pt.length; i++) {
1526
1527         Type t = pt[i];
1528         if (t instanceof Class) {
1529            Class<?> c = (Class<?>)t;
1530            rp[i] = paramResolvers.get(c);
1531            if (rp[i] == null)
1532               rp[i] = RestParamDefaults.STANDARD_RESOLVERS.get(c);
1533         }
1534
1535         if (rp[i] == null) {
1536            for (Annotation a : pa[i]) {
1537               if (a instanceof Header)
1538                  rp[i] = new RestParamDefaults.HeaderObject((Header)a, t);
1539               else if (a instanceof FormData)
1540                  rp[i] = new RestParamDefaults.FormDataObject(method, (FormData)a, t, methodPlainParams);
1541               else if (a instanceof Query)
1542                  rp[i] = new RestParamDefaults.QueryObject(method, (Query)a, t, methodPlainParams);
1543               else if (a instanceof HasFormData)
1544                  rp[i] = new RestParamDefaults.HasFormDataObject(method, (HasFormData)a, t);
1545               else if (a instanceof HasQuery)
1546                  rp[i] = new RestParamDefaults.HasQueryObject(method, (HasQuery)a, t);
1547               else if (a instanceof Body)
1548                  rp[i] = new RestParamDefaults.BodyObject(t);
1549               else if (a instanceof org.apache.juneau.rest.annotation.Method)
1550                  rp[i] = new RestParamDefaults.MethodObject(method, t);
1551               else if (a instanceof PathRemainder)
1552                  rp[i] = new RestParamDefaults.PathRemainderObject(method, t);
1553               else if (a instanceof Properties)
1554                  rp[i] = new RestParamDefaults.PropsObject(method, t);
1555               else if (a instanceof Messages)
1556                  rp[i] = new RestParamDefaults.MessageBundleObject();
1557            }
1558         }
1559
1560         if (rp[i] == null) {
1561            Path p = null;
1562            for (Annotation a : pa[i])
1563               if (a instanceof Path)
1564                  p = (Path)a;
1565
1566            String name = (p == null ? "" : firstNonEmpty(p.name(), p.value()));
1567
1568            if (isEmpty(name)) {
1569               int idx = attrIndex++;
1570               String[] vars = pathPattern.getVars();
1571               if (vars.length <= idx)
1572                  throw new RestServletException("Number of attribute parameters in method ''{0}'' exceeds the number of URL pattern variables.", method.getName());
1573
1574               // Check for {#} variables.
1575               String idxs = String.valueOf(idx);
1576               for (int j = 0; j < vars.length; j++)
1577                  if (isNumeric(vars[j]) && vars[j].equals(idxs))
1578                     name = vars[j];
1579
1580               if (isEmpty(name))
1581                  name = pathPattern.getVars()[idx];
1582            }
1583            rp[i] = new RestParamDefaults.PathParameterObject(name, t);
1584         }
1585      }
1586
1587      return rp;
1588   }

1853   /**
1854    * Takes in an object of type T or a Class<T> and either casts or constructs a T.
1855    */
1856   @SuppressWarnings("unchecked")
1857   private static <T> T resolve(Class<T> c, Object o, Object...cArgs) throws RestServletException {
1858      if (c.isInstance(o))
1859         return (T)o;
1860      if (! (o instanceof Class))
1861         throw new RestServletException("Invalid object type passed to resolve:  ''{0}''.  Must be an object of type T or a Class<? extend T>.", o.getClass());
1862      Constructor<T> n = findPublicConstructor((Class<T>)o, cArgs);
1863      if (n == null)
1864         throw new RestServletException("Could not find public constructor for class ''{0}'' that takes in args {1}", c, JsonSerializer.DEFAULT_LAX.toString(getClasses(cArgs)));
1865      try {
1866         return n.newInstance(cArgs);
1867      } catch (Exception e) {
1868         throw new RestServletException("Exception occurred while constructing class ''{0}''", c).initCause(e);
1869      }
1870   }
