030/**
031 * Session object that lives for the duration of a single use of {@link RdfSerializer}.
032 *
033 * <p>
034 * This class is NOT thread safe.  
035 * It is typically discarded after one-time use although it can be reused within the same thread.
036 */
037@SuppressWarnings({ "rawtypes", "unchecked" })
038public final class RdfSerializerSession extends WriterSerializerSession {
039
040   private final String rdfLanguage;
041   private final Namespace juneauNs, juneauBpNs;
042   private final boolean
043      addLiteralTypes,
044      addRootProperty,
045      useXmlNamespaces,
046      looseCollections,
047      autoDetectNamespaces,
048      addBeanTypeProperties;
049   private final Property pRoot, pValue;
050   private final Model model;
051   private final RDFWriter writer;
052   private final RdfCollectionFormat collectionFormat;
053   private final Namespace[] namespaces;

055   /**
056    * Create a new session using properties specified in the context.
057    *
058    * @param ctx
059    *    The context creating this session object.
060    *    The context contains all the configuration settings for this object.
061    * @param args
062    *    Runtime arguments.
063    *    These specify session-level information such as locale and URI context.
064    *    It also include session-level properties that override the properties defined on the bean and
065    *    serializer contexts.
066    */
067   protected RdfSerializerSession(RdfSerializerContext ctx, SerializerSessionArgs args) {
068      super(ctx, args);
069      ObjectMap jenaSettings = new ObjectMap();
070      jenaSettings.put("rdfXml.tab", isUseWhitespace() ? 2 : 0);
071      jenaSettings.put("rdfXml.attributeQuoteChar", Character.toString(getQuoteChar()));
072      jenaSettings.putAll(ctx.jenaSettings);
073      ObjectMap p = getProperties();
074      if (p.isEmpty()) {
075         this.rdfLanguage = ctx.rdfLanguage;
076         this.juneauNs = ctx.juneauNs;
077         this.juneauBpNs = ctx.juneauBpNs;
078         this.addLiteralTypes = ctx.addLiteralTypes;
079         this.addRootProperty = ctx.addRootProperty;
080         this.collectionFormat = ctx.collectionFormat;
081         this.looseCollections = ctx.looseCollections;
082         this.useXmlNamespaces = ctx.useXmlNamespaces;
083         this.autoDetectNamespaces = ctx.autoDetectNamespaces;
084         this.namespaces = ctx.namespaces;
085         addBeanTypeProperties = ctx.addBeanTypeProperties;
086      } else {
087         this.rdfLanguage = p.getString(RDF_language, ctx.rdfLanguage);
088         this.juneauNs = (p.containsKey(RDF_juneauNs) ? NamespaceFactory.parseNamespace(p.get(RDF_juneauNs)) : ctx.juneauNs);
089         this.juneauBpNs = (p.containsKey(RDF_juneauBpNs) ? NamespaceFactory.parseNamespace(p.get(RDF_juneauBpNs)) : ctx.juneauBpNs);
090         this.addLiteralTypes = p.getBoolean(RDF_addLiteralTypes, ctx.addLiteralTypes);
091         this.addRootProperty = p.getBoolean(RDF_addRootProperty, ctx.addRootProperty);
092         for (Map.Entry<String,Object> e : p.entrySet()) {
093            String key = e.getKey();
094            if (key.startsWith("Rdf.jena."))
095               jenaSettings.put(key.substring(9), e.getValue());
096         }
097         this.collectionFormat = RdfCollectionFormat.valueOf(p.getString(RDF_collectionFormat, "DEFAULT"));
098         this.looseCollections = p.getBoolean(RDF_looseCollections, ctx.looseCollections);
099         this.useXmlNamespaces = p.getBoolean(RDF_useXmlNamespaces, ctx.useXmlNamespaces);
100         this.autoDetectNamespaces = p.getBoolean(RDF_autoDetectNamespaces, ctx.autoDetectNamespaces);
101         this.namespaces = p.getWithDefault(RDF_namespaces, ctx.namespaces, Namespace[].class);
102         addBeanTypeProperties = p.getBoolean(RDF_addBeanTypeProperties, ctx.addBeanTypeProperties);
103      }
104      this.model = ModelFactory.createDefaultModel();
105      addModelPrefix(juneauNs);
106      addModelPrefix(juneauBpNs);
107      for (Namespace ns : this.namespaces)
108         addModelPrefix(ns);
109      this.pRoot = model.createProperty(juneauNs.getUri(), RDF_juneauNs_ROOT);
110      this.pValue = model.createProperty(juneauNs.getUri(), RDF_juneauNs_VALUE);
111      writer = model.getWriter(rdfLanguage);
112
113      // Only apply properties with this prefix!
114      String propPrefix = RdfCommonContext.LANG_PROP_MAP.get(rdfLanguage);
115      if (propPrefix == null)
116         throw new FormattedRuntimeException("Unknown RDF language encountered: ''{0}''", rdfLanguage);
117
118      for (Map.Entry<String,Object> e : jenaSettings.entrySet())
119         if (e.getKey().startsWith(propPrefix))
120            writer.setProperty(e.getKey().substring(propPrefix.length()), e.getValue());
121   }

123   /*
124    * Adds the specified namespace as a model prefix.
125    */
126   private void addModelPrefix(Namespace ns) {
127      model.setNsPrefix(ns.getName(), ns.getUri());
128   }

140   /*
141    * XML-encodes the specified string using the {@link XmlUtils#escapeText(Object)} method.
142    */
143   private String encodeTextInvalidChars(Object o) {
144      if (o == null)
145         return null;
146      String s = toString(o);
147      return XmlUtils.escapeText(s);
148   }

151    * XML-encoded the specified element name using the {@link XmlUtils#encodeElementName(Object)} method.
152    */
153   private String encodeElementName(Object o) {
154      return XmlUtils.encodeElementName(toString(o));
155   }

157   @Override /* Serializer */
158   protected void doSerialize(SerializerPipe out, Object o) throws Exception {
159
160      Resource r = null;
161
162      ClassMeta<?> cm = getClassMetaForObject(o);
163      if (looseCollections && cm != null && cm.isCollectionOrArray()) {
164         Collection c = sort(cm.isCollection() ? (Collection)o : toList(cm.getInnerClass(), o));
165         for (Object o2 : c)
166            serializeAnything(o2, false, object(), "root", null, null);
167      } else {
168         RDFNode n = serializeAnything(o, false, getExpectedRootType(o), "root", null, null);
169         if (n.isLiteral()) {
170            r = model.createResource();
171            r.addProperty(pValue, n);
172         } else {
173            r = n.asResource();
174         }
175
176         if (addRootProperty)
177            r.addProperty(pRoot, "true");
178      }
179
180      writer.write(model, out.getWriter(), "http://unknown/");
181   }

183   private RDFNode serializeAnything(Object o, boolean isURI, ClassMeta<?> eType, 
184         String attrName, BeanPropertyMeta bpm, Resource parentResource) throws Exception {
185      Model m = model;
186
187      ClassMeta<?> aType = null;       // The actual type
188      ClassMeta<?> wType = null;       // The wrapped type
189      ClassMeta<?> sType = object();   // The serialized type
190
191      aType = push(attrName, o, eType);
192
193      if (eType == null)
194         eType = object();
195
196      // Handle recursion
197      if (aType == null) {
198         o = null;
199         aType = object();
200      }
201
202      if (o != null) {
203
204         if (aType.isDelegate()) {
205            wType = aType;
206            aType = ((Delegate)o).getClassMeta();
207         }
208
209         sType = aType;
210
211         // Swap if necessary
212         PojoSwap swap = aType.getPojoSwap(this);
213         if (swap != null) {
214            o = swap.swap(this, o);
215            sType = swap.getSwapClassMeta(this);
216
217            // If the getSwapClass() method returns Object, we need to figure out
218            // the actual type now.
219            if (sType.isObject())
220               sType = getClassMetaForObject(o);
221         }
222      } else {
223         sType = eType.getSerializedClassMeta(this);
224      }
225
226      String typeName = getBeanTypeName(eType, aType, bpm);
227
228      RDFNode n = null;
229
230      if (o == null || sType.isChar() && ((Character)o).charValue() == 0) {
231         if (bpm != null) {
232            if (! isTrimNulls()) {
233               n = m.createResource(RDF_NIL);
234            }
235         } else {
236            n = m.createResource(RDF_NIL);
237         }
238
239      } else if (sType.isUri() || isURI) {
240         // Note that RDF URIs must be absolute to be valid!
241         String uri = getUri(o, null);
242         if (StringUtils.isAbsoluteUri(uri))
243            n = m.createResource(uri);
244         else
245            n = m.createLiteral(encodeTextInvalidChars(uri));
246
247      } else if (sType.isCharSequence() || sType.isChar()) {
248         n = m.createLiteral(encodeTextInvalidChars(o));
249
250      } else if (sType.isNumber() || sType.isBoolean()) {
251         if (! addLiteralTypes)
252            n = m.createLiteral(o.toString());
253         else
254            n = m.createTypedLiteral(o);
255
256      } else if (sType.isMap() || (wType != null && wType.isMap())) {
257         if (o instanceof BeanMap) {
258            BeanMap bm = (BeanMap)o;
259            Object uri = null;
260            RdfBeanMeta rbm = (RdfBeanMeta)bm.getMeta().getExtendedMeta(RdfBeanMeta.class);
261            if (rbm.hasBeanUri())
262               uri = rbm.getBeanUriProperty().get(bm, null);
263            String uri2 = getUri(uri, null);
264            n = m.createResource(uri2);
265            serializeBeanMap(bm, (Resource)n, typeName);
266         } else {
267            Map m2 = (Map)o;
268            n = m.createResource();
269            serializeMap(m2, (Resource)n, sType);
270         }
271
272      } else if (sType.isBean()) {
273         BeanMap bm = toBeanMap(o);
274         Object uri = null;
275         RdfBeanMeta rbm = (RdfBeanMeta)bm.getMeta().getExtendedMeta(RdfBeanMeta.class);
276         if (rbm.hasBeanUri())
277            uri = rbm.getBeanUriProperty().get(bm, null);
278         String uri2 = getUri(uri, null);
279         n = m.createResource(uri2);
280         serializeBeanMap(bm, (Resource)n, typeName);
281
282      } else if (sType.isCollectionOrArray() || (wType != null && wType.isCollection())) {
283         Collection c = sort(sType.isCollection() ? (Collection)o : toList(sType.getInnerClass(), o));
284         RdfCollectionFormat f = collectionFormat;
285         RdfClassMeta rcm = sType.getExtendedMeta(RdfClassMeta.class);
286         if (rcm.getCollectionFormat() != RdfCollectionFormat.DEFAULT)
287            f = rcm.getCollectionFormat();
288         if (bpm != null && bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat() != RdfCollectionFormat.DEFAULT)
289            f = bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat();
290         switch (f) {
291            case BAG: n = serializeToContainer(c, eType, m.createBag()); break;
292            case LIST: n = serializeToList(c, eType); break;
293            case MULTI_VALUED: serializeToMultiProperties(c, eType, bpm, attrName, parentResource); break;
294            default: n = serializeToContainer(c, eType, m.createSeq());
295         }
296      
297      } else if (sType.isReader() || sType.isInputStream()) {
298         n = m.createLiteral(encodeTextInvalidChars(IOUtils.read(o)));
299      
300      } else {
301         n = m.createLiteral(encodeTextInvalidChars(toString(o)));
302      }
303
304      pop();
305
306      return n;
307   }

309   private String getUri(Object uri, Object uri2) {
310      String s = null;
311      if (uri != null)
312         s = uri.toString();
313      if ((s == null || s.isEmpty()) && uri2 != null)
314         s = uri2.toString();
315      if (s == null)
316         return null;
317      return getUriResolver().resolve(s);
318   }

320   private void serializeMap(Map m, Resource r, ClassMeta<?> type) throws Exception {
321
322      m = sort(m);
323
324      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
325
326      ArrayList<Map.Entry<Object,Object>> l = new ArrayList<Map.Entry<Object,Object>>(m.entrySet());
327      Collections.reverse(l);
328      for (Map.Entry<Object,Object> me : l) {
329         Object value = me.getValue();
330
331         Object key = generalize(me.getKey(), keyType);
332
333         Namespace ns = juneauBpNs;
334         Property p = model.createProperty(ns.getUri(), encodeElementName(toString(key)));
335         RDFNode n = serializeAnything(value, false, valueType, key == null ? null : toString(key), null, r);
336         if (n != null)
337            r.addProperty(p, n);
338      }
339   }

341   private void serializeBeanMap(BeanMap<?> m, Resource r, String typeName) throws Exception {
342      List<BeanPropertyValue> l = m.getValues(isTrimNulls(), typeName != null ? createBeanTypeNameProperty(m, typeName) : null);
343      Collections.reverse(l);
344      for (BeanPropertyValue bpv : l) {
345         BeanPropertyMeta pMeta = bpv.getMeta();
346         ClassMeta<?> cMeta = pMeta.getClassMeta();
347
348         if (pMeta.getExtendedMeta(RdfBeanPropertyMeta.class).isBeanUri())
349            continue;
350
351         String key = bpv.getName();
352         Object value = bpv.getValue();
353         Throwable t = bpv.getThrown();
354         if (t != null)
355            onBeanGetterException(pMeta, t);
356
357         if (canIgnoreValue(cMeta, key, value))
358            continue;
359
360         BeanPropertyMeta bpm = bpv.getMeta();
361         Namespace ns = bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getNamespace();
362         if (ns == null && useXmlNamespaces)
363            ns = bpm.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
364         if (ns == null)
365            ns = juneauBpNs;
366         else if (autoDetectNamespaces)
367            addModelPrefix(ns);
368
369         Property p = model.createProperty(ns.getUri(), encodeElementName(key));
370         RDFNode n = serializeAnything(value, pMeta.isUri(), cMeta, key, pMeta, r);
371         if (n != null)
372            r.addProperty(p, n);
373      }
374   }

377   private Container serializeToContainer(Collection c, ClassMeta<?> type, Container list) throws Exception {
378
379      ClassMeta<?> elementType = type.getElementType();
380      for (Object e : c) {
381         RDFNode n = serializeAnything(e, false, elementType, null, null, null);
382         list = list.add(n);
383      }
384      return list;
385   }
386
387   private RDFList serializeToList(Collection c, ClassMeta<?> type) throws Exception {
388      ClassMeta<?> elementType = type.getElementType();
389      List<RDFNode> l = new ArrayList<RDFNode>(c.size());
390      for (Object e : c) {
391         l.add(serializeAnything(e, false, elementType, null, null, null));
392      }
393      return model.createList(l.iterator());
394   }
395
396   private void serializeToMultiProperties(Collection c, ClassMeta<?> sType, 
397         BeanPropertyMeta bpm, String attrName, Resource parentResource) throws Exception {
398      ClassMeta<?> elementType = sType.getElementType();
399      for (Object e : c) {
400         Namespace ns = null;
401         if (bpm != null) {
402            ns = bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getNamespace();
403            if (ns == null && useXmlNamespaces)
404               ns = bpm.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
405         }
406         if (ns == null)
407            ns = juneauBpNs;
408         else if (autoDetectNamespaces)
409            addModelPrefix(ns);
410         RDFNode n2 = serializeAnything(e, false, elementType, null, null, null);
411         Property p = model.createProperty(ns.getUri(), encodeElementName(attrName));
412         parentResource.addProperty(p, n2);
413      }
414   }
