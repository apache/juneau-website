025/**
026 * A builder for {@link PropertyStore} objects.
027 */
028public class PropertyStoreBuilder {
029   
030   // Contains a cache of all created PropertyStore objects keyed by hashcode.
031   // Used to minimize memory consumption by reusing identical PropertyStores.
032   private static final Map<Integer,PropertyStore> CACHE = new ConcurrentHashMap<>();
033
034   // Maps property suffixes (e.g. "lc") to PropertyType (e.g. LIST_CLASS)
035   static final Map<String,PropertyType> SUFFIX_MAP = new ConcurrentHashMap<>();
036   static {
037      for (PropertyType pt : PropertyType.values())
038         SUFFIX_MAP.put(pt.getSuffix(), pt);
039   }
040
041   private final Map<String,PropertyGroupBuilder> groups = new ConcurrentSkipListMap<>();
042   
043   // Previously-created property store.
044   private volatile PropertyStore propertyStore;
045   
046   // Called by PropertyStore.builder()
047   PropertyStoreBuilder(PropertyStore ps) {
048      apply(ps);
049   }
050
051   // Called by PropertyStore.create()
052   PropertyStoreBuilder() {}
053   
054   /**
055    * Creates a new {@link PropertyStore} based on the values in this builder.
056    * 
057    * @return A new {@link PropertyStore} based on the values in this builder.
058    */
059   public synchronized PropertyStore build() {
060      
061      // Reused the last one if we haven't change this builder.
062      if (propertyStore == null)
063         propertyStore = new PropertyStore(groups);
064      
065      PropertyStore ps = CACHE.get(propertyStore.hashCode());
066      if (ps == null)
067         CACHE.put(propertyStore.hashCode(), propertyStore);
068      else if (! ps.equals(propertyStore)) 
069         throw new RuntimeException("Property store mismatch!  This shouldn't happen.");
070      else
071         propertyStore = ps;
072      
073      return propertyStore;
074   }
075   
076   /**
077    * Copies all the values in the specified property store into this builder.
078    * 
079    * @param copyFrom The property store to copy the values from. 
080    * @return This object (for method chaining).
081    */
082   public synchronized PropertyStoreBuilder apply(PropertyStore copyFrom) {
083      propertyStore = null;
084      
085      if (copyFrom != null)
086         for (Map.Entry<String,PropertyGroup> e : copyFrom.groups.entrySet()) {
087            String gName = e.getKey();
088            PropertyGroupBuilder g1 = this.groups.get(gName);
089            PropertyGroup g2 = e.getValue();
090            if (g1 == null)
091               this.groups.put(gName, g2.builder());
092            else
093               g1.apply(g2);
094         }
095      return this;
096   }

098   /**
099    * Sets a configuration property value on this object.
100    *
101    * @param key
102    *    The configuration property key.
103    *    <br>(e.g <js>"BeanContext.foo.ss/add.1"</js>)
104    *    <br>If name ends with <l>/add</l>, then the specified value is added to the existing property value as an entry
105    *    in a SET or LIST property.
106    *    <br>If name ends with <l>/add.{key}</l>, then the specified value is added to the existing property value as a
107    *    key/value pair in a MAP property.
108    *    <br>If name ends with <l>/remove</l>, then the specified value is removed from the existing property property
109    *    value in a SET or LIST property.
110    * @param value
111    *    The new value.
112    *    If <jk>null</jk>, the property value is deleted.
113    *    In general, the value type can be anything.
114    * @return This object (for method chaining).
115    */
116   public synchronized PropertyStoreBuilder set(String key, Object value) {
117      propertyStore = null;
118
119      String g = group(key);
120      
121      int i = key.indexOf('/');
122      if (i != -1) {
123         String command = key.substring(i+1), arg = null;
124         String key2 = key.substring(0, i); 
125         int j = command.indexOf('.');
126         if (j != -1) {
127            arg = command.substring(j+1);
128            command = command.substring(0, j);
129         }
130
131         if ("add".equals(command)) {
132            return addTo(key2, arg, value);
133         } else if ("remove".equals(command)) {
134            if (arg != null)
135               throw new ConfigException("Invalid key specified: ''{0}''", key);
136            return removeFrom(key2, value);
137         } else {
138            throw new ConfigException("Invalid key specified: ''{0}''", key);
139         }
140      }
141
142      String n = g.isEmpty() ? key : key.substring(g.length()+1);
143
144      PropertyGroupBuilder gb = groups.get(g);
145      if (gb == null) {
146         gb = new PropertyGroupBuilder();
147         groups.put(g, gb);
148      }
149
150      gb.set(n, value);
151      
152      if (gb.isEmpty())
153         groups.remove(g);
154
155      return this;
156   }

158   /**
159    * Removes the property with the specified key.
160    * 
161    * <p>
162    * This is equivalent to calling <code>set(key, <jk>null</jk>);</code>
163    * 
164    * @param key The property key.
165    * @return This object (for method chaining).
166    */
167   public synchronized PropertyStoreBuilder remove(String key) {
168      propertyStore = null;
169      return set(key, null);
170   }

172   /**
173    * Convenience method for setting multiple properties in one call.
174    *
175    * <p>
176    * This replaces any previous configuration properties set on this store.
177    *
178    * @param newProperties The new properties to set.
179    * @return This object (for method chaining).

181   public synchronized PropertyStoreBuilder set(Map<String,Object> newProperties) {
182      propertyStore = null;
183      clear();
184      add(newProperties);
185      return this;
186   }

189    * Convenience method for setting multiple properties in one call.
190    *
191    * <p>
192    * This appends to any previous configuration properties set on this store.
193    *
194    * @param newProperties The new properties to set.
195    * @return This object (for method chaining).
196    */
197   public synchronized PropertyStoreBuilder add(Map<String,Object> newProperties) {
198      propertyStore = null;
199      
200      if (newProperties != null)
201         for (Map.Entry<String,Object> e : newProperties.entrySet()) 
202            set(e.getKey(), e.getValue());
203      
204      return this;
205   }

207   /**
208    * Adds one or more values to a SET, LIST, or MAP property.
209    *
210    * @param key The property key.
211    * @param arg 
212    *    The argument.  
213    *    <br>For SETs, this must always be <jk>null</jk>.
214    *    <br>For LISTs, this can be <jk>null</jk> or a numeric index.  
215    *       Out-of-range indexes are simply 'adjusted' to the beginning or the end of the list.
216    *       So, for example, a value of <js>"-100"</js> will always just cause the entry to be added to the beginning
217    *       of the list.
218    *    <br>For MAPs, this can be <jk>null</jk> if we're adding a map, or a string key if we're adding an entry.
219    * @param value 
220    *    The new value to add to the property.
221    *    <br>For SETs and LISTs, this can be a single value, Collection, array, or JSON array string.
222    *    <br>For MAPs, this can be a single value, Map, or JSON object string.
223    *    <br><jk>null</jk> values have the effect of removing entries.
224    * @return This object (for method chaining).
225    * @throws ConfigException If property is not a SET/LIST/MAP property, or the argument is invalid.
226    */
227   public synchronized PropertyStoreBuilder addTo(String key, String arg, Object value) {
228      propertyStore = null;
229      String g = group(key);
230      String n = g.isEmpty() ? key : key.substring(g.length()+1);
231      
232      PropertyGroupBuilder gb = groups.get(g);
233      if (gb == null) {
234         gb = new PropertyGroupBuilder();
235         groups.put(g, gb);
236      }
237      
238      gb.addTo(n, arg, value);
239      
240      if (gb.isEmpty())
241         groups.remove(g);
242
243      return this;
244   }
245
246   /**
247    * Adds a value to a SET, LIST, or MAP property.
248    * 
249    * <p>
250    * Shortcut for calling <code>addTo(key, <jk>null</jk>, value);</code>.
251    *
252    * @param key The property key.
253    * @param value 
254    *    The new value to add to the property.
255    *    <br>For SETs and LISTs, this can be a single value, Collection, array, or JSON array string.
256    *    <br>for MAPs, this can be a single value, Map, or JSON object string.
257    *    <br><jk>null</jk> values have the effect of removing entries.
258    * @return This object (for method chaining).
259    * @throws ConfigException If property is not a SET/LIST/MAP property, or the argument is invalid.
260    */
261   public synchronized PropertyStoreBuilder addTo(String key, Object value) {
262      propertyStore = null;
263      return addTo(key, null, value);
264   }
265
266   /**
267    * Removes a value from a SET or LIST property.
268    *
269    * @param key The property key.
270    * @param value The property value in the property.
271    * @return This object (for method chaining).
272    * @throws ConfigException If property is not a SET or LIST property.
273    */
274   public synchronized PropertyStoreBuilder removeFrom(String key, Object value) {
275      propertyStore = null;
276      String g = group(key);
277      String n = g.isEmpty() ? key : key.substring(g.length()+1);
278      
279      PropertyGroupBuilder gb = groups.get(g);
280
281      // Create property group anyway to generate a good error message.
282      if (gb == null) 
283         gb = new PropertyGroupBuilder();
284
285      gb.removeFrom(n, value);
286      
287      if (gb.isEmpty())
288         groups.remove(g);
289
290      return this;
291   }
292   
293   /**
294    * Peeks at a property value.
295    * 
296    * <p>
297    * Used for debugging purposes.
298    * 
299    * @param key The property key.
300    * @return The property value, or <jk>null</jk> if it doesn't exist.
301    */
302   public Object peek(String key) {
303      String g = group(key);
304      String n = g.isEmpty() ? key : key.substring(g.length()+1);
305      
306      PropertyGroupBuilder gb = groups.get(g);
307
308      // Create property group anyway to generate a good error message.
309      if (gb == null) 
310         return null;
311      
312      MutableProperty bp = gb.properties.get(n);
313      if (bp == null)
314         return null;
315      
316      return bp.peek();
317   }
318
319   /**
320    * Clears all entries in this property store.
321    */
322   public synchronized void clear() {
323      propertyStore = null;
324      this.groups.clear();
325   }

331   static class PropertyGroupBuilder {
332      final Map<String,MutableProperty> properties = new ConcurrentSkipListMap<>();
333      
334      PropertyGroupBuilder() {
335         this(Collections.EMPTY_MAP);
336      }
337      
338      synchronized void apply(PropertyGroup copyFrom) {
339         for (Map.Entry<String,Property> e : copyFrom.properties.entrySet()) {
340            String pName = e.getKey();
341            MutableProperty p1 = properties.get(pName);
342            Property p2 = e.getValue();
343            if (p1 == null)
344               properties.put(pName, p2.mutable());
345            else 
346               p1.apply(p2.value);
347         }
348      }
349
350      PropertyGroupBuilder(Map<String,Property> properties) {
351         for (Map.Entry<String,Property> p : properties.entrySet())
352            this.properties.put(p.getKey(), p.getValue().mutable());
353      }
354
355      synchronized void set(String key, Object value) {
356         MutableProperty p = properties.get(key);
357         if (p == null) {
358            p = MutableProperty.create(key, value);
359            if (! p.isEmpty())
360               properties.put(key, p);
361         } else {
362            p.set(value);
363            if (p.isEmpty()) 
364               properties.remove(key); 
365            else
366               properties.put(key, p);
367         }
368      }
369
370      synchronized void addTo(String key, String arg, Object value) {
371         MutableProperty p = properties.get(key);
372         if (p == null) {
373            p = MutableProperty.create(key, null);
374            p.add(arg, value);
375            if (! p.isEmpty())
376               properties.put(key, p);
377         } else {
378            p.add(arg, value);
379            if (p.isEmpty())
380               properties.remove(key);
381            else
382               properties.put(key, p);
383         }
384      }
385
386      synchronized void removeFrom(String key, Object value) {
387         MutableProperty p = properties.get(key);
388         if (p == null) {
389            // Create property anyway to generate a good error message.
390            p = MutableProperty.create(key, null);
391            p.remove(value);
392         } else {
393            p.remove(value);
394            if (p.isEmpty())
395               properties.remove(key);
396            else
397               properties.put(key, p);
398         }
399      }
400      
401      synchronized boolean isEmpty() {
402         return properties.isEmpty();
403      }
404      
405      synchronized PropertyGroup build() {
406         return new PropertyGroup(properties);
407      }
408   }

414   static abstract class MutableProperty {
415      final String name;
416      final PropertyType type;
417      
418      MutableProperty(String name, PropertyType type) {
419         this.name = name;
420         this.type = type;
421      }
422
423      static MutableProperty create(String name, Object value) {
424         int i = name.lastIndexOf('.');
425         String type = i == -1 ? "s" : name.substring(i+1);
426         PropertyType pt = SUFFIX_MAP.get(type);
427         
428         if (pt == null)
429            throw new ConfigException("Invalid type specified on property ''{0}''", name);
430         
431         switch (pt) {
432            case STRING:
433            case BOOLEAN:
434            case INTEGER:
435            case CLASS: 
436            case OBJECT:  return new MutableSimpleProperty(name, pt, value);
437            case SET_STRING:
438            case SET_INTEGER:
439            case SET_CLASS: return new MutableSetProperty(name, pt, value);
440            case LIST_STRING:
441            case LIST_INTEGER:
442            case LIST_CLASS:
443            case LIST_OBJECT: return new MutableListProperty(name, pt, value);
444            case MAP_STRING:
445            case MAP_INTEGER:
446            case MAP_CLASS:
447            case MAP_OBJECT: return new MutableMapProperty(name, pt, value);
448            default: return new MutableSimpleProperty(name, PropertyType.STRING, value);
449         }
450      }
451
452      abstract Property build();
453      
454      abstract boolean isEmpty();
455      
456      abstract void set(Object value);
457
458      abstract void apply(Object value);
459      
460      abstract Object peek();
461
462      void add(String arg, Object value) {
463         throw new ConfigException("Cannot add value {0} ({1}) to property ''{2}'' ({3}).",
464            JsonSerializer.DEFAULT_LAX.toString(value), value.getClass().getSimpleName(), name, type);
465      }
466
467      void remove(Object value) {
468         throw new ConfigException("Cannot remove value {0} ({1}) from property ''{2}'' ({3}).",
469            JsonSerializer.DEFAULT_LAX.toString(value), value.getClass().getSimpleName(), name, type);
470      }
471
472      Object convert(Object value) {
473         return value == null ? null : type.converter.convert(value, this);
474      }
475   }

519   @SuppressWarnings("unchecked")
520   static class MutableSetProperty extends MutableProperty {
521      private final Set<Object> value;
522
523      MutableSetProperty(String name, PropertyType type, Object value) {
524         super(name, type);
525         this.value = new ConcurrentSkipListSet<>(type.comparator());
526         set(value);
527      }
528      
529      @Override /* MutableProperty */
530      synchronized Property build() {
531         return new Property(name, Collections.unmodifiableSet(new LinkedHashSet<>(value)), type);
532      }
533
534      @Override /* MutableProperty */
535      synchronized void set(Object value) {
536         this.value.clear();
537         add(null, value);
538      }
539      
540      @Override /* MutableProperty */
541      synchronized void apply(Object values) {
542         for (Object o : ((Set<?>)values)) 
543            add(null, o);
544      }
545      
546      @Override /* MutableProperty */
547      synchronized void add(String arg, Object value) {
548         if (arg != null)
549            throw new ConfigException("Cannot use argument ''{0}'' on add command for property ''{1}'' ({2})", arg, name, type);
550         if (value != null) {
551            if (value.getClass().isArray()) {
552               for (int i = 0; i < Array.getLength(value); i++)
553                  add(null, Array.get(value, i));
554            } else if (value instanceof Collection) {
555               for (Object o : (Collection<Object>)value)
556                  add(null, o);
557            } else if (isObjectList(value)) {
558               try {
559                  add(null, new ObjectList(value.toString()));
560               } catch (Exception e) {
561                  throw new ConfigException(
562                     "Cannot add value {0} ({1}) to property ''{2}'' ({3}).",
563                     JsonSerializer.DEFAULT_LAX.toString(value), getReadableClassNameForObject(value), name, type
564                  );
565               }
566            } else {
567               value = convert(value);
568               this.value.add(value);
569            }
570         }
571      }
572      
573      @Override /* MutableProperty */
574      synchronized void remove(Object value) {
575         if (value != null) {
576            if (value.getClass().isArray()) {
577               for (int i = 0; i < Array.getLength(value); i++)
578                  remove(Array.get(value, i));
579            } else if (value instanceof Collection) {
580               for (Object o : (Collection<Object>)value)
581                  remove(o);
582            } else if (isObjectList(value)) {
583               try {
584                  remove(new ObjectList(value.toString()));
585               } catch (Exception e) {
586                  throw new ConfigException(
587                     "Cannot remove value {0} ({1}) from property ''{2}'' ({3}) because it''s not a valid JSON array.",
588                     JsonSerializer.DEFAULT_LAX.toString(value), getReadableClassNameForObject(value), name, type
589                  );
590               }              
591            } else {
592               value = convert(value);
593               this.value.remove(value);
594            }
595         }
596      }
597      
598      @Override /* MutableProperty */
599      synchronized boolean isEmpty() {
600         return this.value.isEmpty();
601      }
602
603      @Override /* MutableProperty */
604      synchronized Object peek() {
605         return value;
606      }
607   }

613   @SuppressWarnings("unchecked")
614   static class MutableListProperty extends MutableProperty {
615      private List<Object> value = new CopyOnWriteArrayList<>();
616
617      MutableListProperty(String name, PropertyType type, Object value) {
618         super(name, type);
619         set(value);
620      }
621      
622      @Override /* MutableProperty */
623      synchronized Property build() {
624         return new Property(name, Collections.unmodifiableList(CollectionUtils.reverse(new ArrayList<>(value))), type);
625      }
626
627      @Override /* MutableProperty */
628      synchronized void set(Object value) {
629         this.value.clear();
630         add(null, value);
631      }
632      
633      @Override /* MutableProperty */
634      synchronized void apply(Object values) {
635         add(null, values);
636      }
637
638      @Override /* MutableProperty */
639      synchronized void add(String arg, Object value) {
640         if (arg != null && ! StringUtils.isNumeric(arg))
641            throw new ConfigException("Invalid argument ''{0}'' on add command for property ''{1}'' ({2})", arg, name, type);
642         
643         // Note that since lists are stored in reverse-order, the index should be the compliment.
644         
645         int index = arg == null ? this.value.size() : this.value.size() - Integer.parseInt(arg);
646         if (index < 0)
647            index = 0;
648         else if (index > this.value.size())
649            index = this.value.size();
650         add(index, value);
651      }
652      
653      private synchronized int add(int index, Object value) {
654         if (value != null) {
655            
656            // Important!
657            // Arrays and collections are inserted in REVERSE order.
658            // So if you call addTo(X, "['a','b','c']").addTo(X, "['d','e','f']"), the list will end up
659            // containing ['c','b','a','f','e','d'].
660            // This ensures entries in the incoming value takes precedence in first-to-last order, but
661            // subsequent calls to addTo() are added to the end.
662            // What you get is a list ordered in least-to-most important.
663            
664            if (value.getClass().isArray()) {
665               for (int i = Array.getLength(value) - 1; i >= 0; i--)
666                  index = add(index, Array.get(value, i));
667            } else if (value instanceof Collection) {
668               for (Object o : CollectionUtils.reverseIterable((Collection<Object>)value))
669                  index = add(index, o);
670            } else if (isObjectList(value)) {
671               try {
672                  index = add(index, new ObjectList(value.toString()));
673               } catch (Exception e) {
674                  throw new ConfigException(
675                     "Cannot add value {0} ({1}) to property ''{2}'' ({3}).",
676                     JsonSerializer.DEFAULT_LAX.toString(value), getReadableClassNameForObject(value), name, type
677                  );
678               }
679            } else {
680               value = convert(value);
681               boolean replaced = this.value.remove(value);
682               if (replaced)
683                  index--;
684               this.value.add(index, value);
685               index++;
686            }
687         }
688         return index;
689      }
690
691      @Override /* MutableProperty */
692      synchronized void remove(Object value) {
693         if (value != null) {
694            if (value.getClass().isArray()) {
695               for (int i = 0; i < Array.getLength(value); i++)
696                  remove(Array.get(value, i));
697            } else if (value instanceof Collection) {
698               for (Object o : (Collection<Object>)value)
699                  remove(o);
700            } else if (isObjectList(value)) {
701               try {
702                  remove(new ObjectList(value.toString()));
703               } catch (Exception e) {
704                  throw new ConfigException(
705                     "Cannot remove value {0} ({1}) from property ''{2}'' ({3}) because it''s not a valid JSON array.",
706                     JsonSerializer.DEFAULT_LAX.toString(value), getReadableClassNameForObject(value), name, type
707                  );
708               }              
709            } else {
710               value = convert(value);
711               this.value.remove(value);
712            }
713         }
714      }
715
716      @Override /* MutableProperty */
717      synchronized boolean isEmpty() {
718         return this.value.isEmpty();
719      }
720
721      @Override /* MutableProperty */
722      synchronized Object peek() {
723         return value;
724      }
725   }

731   @SuppressWarnings("rawtypes")
732   static class MutableMapProperty extends MutableProperty {
733      private Map<String,Object> value = new ConcurrentHashMap<>();
734      
735      MutableMapProperty(String name, PropertyType type, Object value) {
736         super(name, type);
737         set(value);
738      }
739      
740      @Override /* MutableProperty */
741      synchronized Property build() {
742         return new Property(name, Collections.unmodifiableMap(new TreeMap<>(value)), type);
743      }
744
745      @Override /* MutableProperty */
746      synchronized void set(Object value) {
747         this.value.clear();
748         add(null, value);
749      }
750
751      @SuppressWarnings("unchecked")
752      @Override /* MutableProperty */
753      synchronized void apply(Object values) {
754         for (Map.Entry<String,Object> e : ((Map<String,Object>)values).entrySet()) 
755            add(e.getKey(), e.getValue());
756      }
757
758      @Override /* MutableProperty */
759      synchronized void add(String arg, Object value) {
760         if (arg != null) {
761            value = convert(value);
762            if (value == null)
763               this.value.remove(arg);
764            else
765               this.value.put(arg, value);
766            
767         } else if (value != null) {
768            if (value instanceof Map) {
769               Map m = (Map)value;
770               for (Map.Entry e : (Set<Map.Entry>)m.entrySet())
771                  if (e.getKey() != null)
772                     add(e.getKey().toString(), e.getValue());
773            } else if (isObjectMap(value)) {
774               try {
775                  add(null, new ObjectMap(value.toString()));
776               } catch (Exception e) {
777                  throw new ConfigException(
778                     "Cannot add {0} ({1}) to property ''{2}'' ({3}) because it''s not a valid JSON object.",
779                     JsonSerializer.DEFAULT_LAX.toString(value), getReadableClassNameForObject(value), name, type
780                  );
781               }
782            } else {
783               throw new ConfigException(
784                  "Cannot add {0} ({1}) to property ''{2}'' ({3}).",
785                  JsonSerializer.DEFAULT_LAX.toString(value), getReadableClassNameForObject(value), name, type
786               );
787            }
788         }
789      }
790
791      @Override /* MutableProperty */
792      synchronized boolean isEmpty() {
793         return this.value.isEmpty();
794      }
795
796      @Override /* MutableProperty */
797      synchronized Object peek() {
798         return value;
799      }
800   }

807   static boolean isObjectList(Object o) {
808      if (o instanceof CharSequence) {
809         String s = o.toString();
810         return (s.startsWith("[") && s.endsWith("]") && BeanContext.DEFAULT != null);
811      }
812      return false;
813   }
814
815   static boolean isObjectMap(Object o) {
816      if (o instanceof CharSequence) {
817         String s = o.toString();
818         return (s.startsWith("{") && s.endsWith("}") && BeanContext.DEFAULT != null);
819      }
820      return false;
821   }

826      return key.substring(0, key.indexOf('.'));
