013package org.apache.juneau.parser;
014
015import org.apache.juneau.*;
016import org.apache.juneau.json.*;
017
018/**
019 * Configurable properties common to all parsers.
020 */
021public class ParserContext extends BeanContext {
022
023   /**
024    * <b>Configuration property:</b>  Trim parsed strings.
025    * <p>
026    * <ul>
027    *    <li><b>Name:</b> <js>"Parser.trimStrings"</js>
028    *    <li><b>Data type:</b> <code>Boolean</code>
029    *    <li><b>Default:</b> <jk>false</jk>
030    *    <li><b>Session-overridable:</b> <jk>true</jk>
031    * </ul>
032    * <p>
033    * If <jk>true</jk>, string values will be trimmed of whitespace using {@link String#trim()} before being added to
034    * the POJO.
035    */
036   public static final String PARSER_trimStrings = "Parser.trimStrings";
037
038   /**
039    * <b>Configuration property:</b>  Strict mode.
040    * <p>
041    * <ul>
042    *    <li><b>Name:</b> <js>"Parser.strict"</js>
043    *    <li><b>Data type:</b> <code>Boolean</code>
044    *    <li><b>Default:</b> <jk>false</jk>
045    *    <li><b>Session-overridable:</b> <jk>true</jk>
046    * </ul>
047    * <p>
048    * If <jk>true</jk>, strict mode for the parser is enabled.
049    * <p>
050    * Strict mode can mean different things for different parsers.
051    * However, all reader-based parsers will
052    * <p>
053    * <table class='styled'>
054    *    <tr><th>Parser class</th><th>Strict behavior</th></tr>
055    *    <tr>
056    *       <td>All reader-based parsers</td>
057    *       <td>
058    *          When enabled, throws {@link ParseException ParseExceptions} on malformed charset input.
059    *          Otherwise, malformed input is ignored.
060    *       </td>
061    *    </tr>
062    *    <tr>
063    *       <td>{@link JsonParser}</td>
064    *       <td>
065    *          When enabled, throws exceptions on the following invalid JSON syntax:
066    *          <ul>
067    *             <li>Unquoted attributes.
068    *             <li>Missing attribute values.
069    *             <li>Concatenated strings.
070    *             <li>Javascript comments.
071    *             <li>Numbers and booleans when Strings are expected.
072    *             <li>Numbers valid in Java but not JSON (e.g. octal notation, etc...)
073    *          </ul>
074    *       </td>
075    *    </tr>
076    * </table>
077    */
078   public static final String PARSER_strict = "Parser.strict";
079
080   /**
081    * <b>Configuration property:</b>  Input stream charset.
082    * <p>
083    * <ul>
084    *    <li><b>Name:</b> <js>"Parser.inputStreamCharset"</js>
085    *    <li><b>Data type:</b> <code>String</code>
086    *    <li><b>Default:</b> <js>"UTF-8"</js>
087    *    <li><b>Session-overridable:</b> <jk>true</jk>
088    * </ul>
089    * <p>
090    * The character set to use for converting <code>InputStreams</code> and byte arrays to readers.
091    * <p>
092    * Used when passing in input streams and byte arrays to {@link Parser#parse(Object, Class)}.
093    */
094   public static final String PARSER_inputStreamCharset = "Parser.inputStreamCharset";
095
096   /**
097    * <b>Configuration property:</b>  File charset.
098    * <p>
099    * <ul>
100    *    <li><b>Name:</b> <js>"Parser.fileCharset"</js>
101    *    <li><b>Data type:</b> <code>String</code>
102    *    <li><b>Default:</b> <js>"default"</js>
103    *    <li><b>Session-overridable:</b> <jk>true</jk>
104    * </ul>
105    * <p>
106    * The character set to use for reading <code>Files</code> from the file system.
107    * <p>
108    * Used when passing in files to {@link Parser#parse(Object, Class)}.
109    * <p>
110    * <js>"default"</js> can be used to indicate the JVM default file system charset.
111    */
112   public static final String PARSER_fileCharset = "Parser.fileCharset";
113
114   /**
115    * <b>Configuration property:</b>  Parser listener.
116    * <p>
117    * <ul>
118    *    <li><b>Name:</b> <js>"Parser.listener"</js>
119    *    <li><b>Data type:</b> <code>Class&lt;? extends ParserListener&gt;</code>
120    *    <li><b>Default:</b> <jk>null</jk>
121    *    <li><b>Session-overridable:</b> <jk>true</jk>
122    * </ul>
123    * <p>
124    * Class used to listen for errors and warnings that occur during parsing.
125    */
126   public static final String PARSER_listener = "PARSER.listener";
127
128   final boolean trimStrings, strict;
129   final String inputStreamCharset, fileCharset;
130   final Class<? extends ParserListener> listener;
131
132   /**
133    * Constructor.
134    *
135    * @param ps The property store that created this context.
136    */
137   @SuppressWarnings("unchecked")
138   public ParserContext(PropertyStore ps) {
139      super(ps);
140      this.trimStrings = ps.getProperty(PARSER_trimStrings, boolean.class, false);
141      this.strict = ps.getProperty(PARSER_strict, boolean.class, false);
142      this.inputStreamCharset = ps.getProperty(PARSER_inputStreamCharset, String.class, "UTF-8");
143      this.fileCharset = ps.getProperty(PARSER_fileCharset, String.class, "default");
144      this.listener = ps.getProperty(PARSER_listener, Class.class, null);
145   }
146
147   @Override /* Context */
148   public ObjectMap asMap() {
149      return super.asMap()
150         .append("ParserContext", new ObjectMap()
151            .append("trimStrings", trimStrings)
152            .append("strict", strict)
153            .append("inputStreamCharset", inputStreamCharset)
154            .append("fileCharset", fileCharset)
155            .append("listener", listener)
156         );
157   }
158}
