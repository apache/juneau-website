013package org.apache.juneau.xml.annotation;
014
015/**
016 * XML format to use when serializing a POJO.
017 */
018public enum XmlFormat {
019
020   /**
021    * Normal formatting (default).
022    * <p>
023    * On a bean class, implies {@link #ELEMENTS} meaning bean properties will be serialized as child elements by default.
024    * <p>
025    * On a bean property, implies {@link #ELEMENT} meaning the bean property will be serialized as a child element.
026    */
027   DEFAULT,
028
029   /**
030    * Render a bean property as an attribute instead of an element.
031    * <p>
032    * Only applicable for bean properties, not bean classes.
033    * <p>
034    * Can only be applied to properties (methods/fields) of class types that can be convertible to <code>Strings</code>.
035    */
036   ATTR,
037
038   /**
039    * Render property as attributes instead of an element.
040    * <p>
041    * On a bean class, implies bean properties will be serialized as attributes instead of child elements by default.
042    * <p>
043    * On bean properties, implies that the bean property value itself should be serialized as attributes on the bean
044    * element.
045    * The bean property data type must be of class type <code>Map&lt;Object,Object&gt;</code> where both
046    * objects are convertible to <code>Strings</code>.
047    */
048   ATTRS,
049
050   /**
051    * Render property as an element instead of an attribute.
052    * <p>
053    * Only applicable for bean properties, not bean classes.
054    * <p>
055    * Used to override the behavior of the {@link #ATTRS} format applied to the bean class.
056    */
057   ELEMENT,
058
059   /**
060    * Render property value directly as the contents of the element.
061    * <p>
062    * On a bean class, implies that bean properties will be serialized as child elements.
063    * Note that this is equivalent to {@link #DEFAULT}.
064    * <p>
065    * Only applicable for objects of type array/Collection.
066    * <p>
067    * On a bean property, implies that the bean property value itself should be serialized as child elements of the
068    * bean element.
069    */
070   ELEMENTS,
071
072   /**
073    * Same as {@link #ELEMENTS} except primitive types (e.g. string/boolean/number/null) are not wrapped in elements.
074    * <p>
075    * Only applicable for bean properties, not bean classes.
076    * <p>
077    * Only applicable for objects of type array/Collection.
078    * <p>
079    * Use of this format may cause data type loss during parsing if the types cannot be inferred through reflection.
080    */
081   MIXED,
082
083   /**
084    * Same as {@link XmlFormat#MIXED}, but whitespace in text nodes are not trimmed during parsing.
085    * <p>
086    * An example use is HTML5 <xt>&lt;pre&gt;</xt> where whitespace should not be discarded.
087    */
088   MIXED_PWS,
089
090   /**
091    * Render property value as the text content of the element.
092    * <p>
093    * Similar to {@link #MIXED} but value must be a single value, not a collection.
094    * <p>
095    * Only applicable for bean properties, not bean classes.
096    * <p>
097    * Use of this format may cause data type loss during parsing if the type cannot be inferred through reflection.
098    */
099   TEXT,
100
101   /**
102    * Same as {@link XmlFormat#TEXT}, but whitespace in text node is not trimmed during parsing.
103    */
104   TEXT_PWS,
105
106   /**
107    * Same as {@link #TEXT} except the content is expected to be fully-formed XML that will
108    * get serialized as-is.
109    * <p>
110    * During parsing, this XML text will be re-serialized and set on the property.
111    * <p>
112    * Only applicable for bean properties, not bean classes.
113    * <p>
114    * Use of this format may cause data type loss during parsing if the type cannot be inferred through reflection.
115    */
116   XMLTEXT,
117
118   /**
119    * Prevents collections and arrays from being enclosed in <xt>&lt;array&gt;</xt> elements.
120    * <p>
121    * Can only be applied to properties (methods/fields) of type collection or array, or collection classes.
122    */
123   COLLAPSED,
124
125   /**
126    * Identifies a void element.
127    * <p>
128    * Only applicable for bean classes.
129    * <p>
130    * Identifies an element that never contains content.
131    * <p>
132    * The main difference in behavior is how non-void empty elements are handled
133    * in the HTML serializer.
134    * Void elements are serialized as collapsed nodes (e.g. <js>"&lt;br/&gt;"</js>)
135    * whereas non-void empty elements are serialized with an end tag (e.g. "&lt;p&gt;&lt;/p&gt;").
136    */
137   VOID;
138
139   /**
140    * Returns <jk>true</jk> if this format is one of those specified.
141    *
142    * @param formats The formats to match against.
143    * @return <jk>true</jk> if this format is one of those specified.
144    */
145   public boolean isOneOf(XmlFormat...formats) {
146      for (XmlFormat format : formats)
147         if (format == this)
148            return true;
149      return false;
150   }
151}
