031/**
032 * Core class of the Juneau architecture.
033 * <p>
034 * This class servers multiple purposes:
035 * <ul class='spaced-list'>
036 *    <li>Provides the ability to wrap beans inside {@link Map} interfaces.
037 *    <li>Serves as a repository for metadata on POJOs, such as associated {@link BeanFilter beanFilters}, {@link PropertyNamer property namers}, etc...
038 *       which are used to tailor how POJOs are serialized and parsed.
039 *    <li>Serves as a common utility class for all {@link Serializer Serializers} and {@link Parser Parsers}
040 *          for serializing and parsing Java beans.
041 * </ul>
042 * <p>
043 * All serializer and parser contexts extend from this context.
044 *
045 * <h5 class='topic'>Bean Contexts</h5>
046 * Bean contexts are created through the {@link PropertyStore#getContext(Class)} method.
047 * These context objects are read-only, reusable, and thread-safe.
048 * The {@link PropertyStore} class will typically cache copies of <code>Context</code> objects based on
049 *    the current settings on the factory.
050 * <p>
051 * Each bean context maintains a cache of {@link ClassMeta} objects that describe information about classes encountered.
052 * These <code>ClassMeta</code> objects are time-consuming to construct.
053 * Therefore, instances of {@link BeanContext} that share the same <js>"BeanContext.*"</js> property values share
054 *    the same cache.  This allows for efficient reuse of <code>ClassMeta</code> objects so that the information about
055 *    classes only needs to be calculated once.
056 * Because of this, many of the properties defined on the {@link BeanContext} class cannot be overridden on the session.
057 *
058 * <h5 class='topic'>Bean Sessions</h5>
059 * <p>
060 * Whereas <code>BeanContext</code> objects are permanent, unchangeable, cached, and thread-safe,
061 *    {@link BeanSession} objects are ephemeral and not thread-safe.
062 * They are meant to be used as quickly-constructed scratchpads for creating bean maps.
063 * {@link BeanMap} objects can only be created through the session.
064 *
065 * <h5 class='topic'>BeanContext configuration properties</h5>
066 * <code>BeanContexts</code> have several configuration properties that can be used to tweak behavior on how beans are handled.
067 * These are denoted as the static <jsf>BEAN_*</jsf> fields on this class.
068 * <p>
069 * Some settings (e.g. {@link BeanContext#BEAN_beansRequireDefaultConstructor}) are used to differentiate between bean and non-bean classes.
070 * Attempting to create a bean map around one of these objects will throw a {@link BeanRuntimeException}.
071 * The purpose for this behavior is so that the serializers can identify these non-bean classes and convert them to plain strings using the {@link Object#toString()} method.
072 * <p>
073 * Some settings (e.g. {@link BeanContext#BEAN_beanFieldVisibility}) are used to determine what kinds of properties are detected on beans.
074 * <p>
075 * Some settings (e.g. {@link BeanContext#BEAN_beanMapPutReturnsOldValue}) change the runtime behavior of bean maps.
076 * <p>
077 * Settings are specified using the {@link PropertyStore#setProperty(String, Object)} method and related convenience methods.
078 *
079 * <h5 class='section'>Example:</h5>
080 * <p class='bcode'>
081 *    <jc>// Construct a context from scratch.</jc>
082 *    BeanContext beanContext = PropertyStore.<jsm>create</jsm>()
083 *       .property(BeanContext.<jsf>BEAN_beansRequireDefaultConstructor</jsf>, <jk>true</jk>)
084 *       .notBeanClasses(Foo.<jk>class</jk>)
085 *       .getBeanContext();
086 *
087 *    <jc>// Clone an existing property store.</jc>
088 *    BeanContext beanContext = PropertyStore.<jsm>create</jsm>(otherConfig)
089 *       .property(BeanContext.<jsf>BEAN_beansRequireDefaultConstructor</jsf>, <jk>true</jk>)
090 *       .notBeanClasses(Foo.<jk>class</jk>)
091 *       .getBeanContext();
092 * </p>
093 *
094 * <h5 class='topic'>Bean Maps</h5>
095 * <p>
096 * {@link BeanMap BeanMaps} are wrappers around Java beans that allow properties to be retrieved and
097 *    set using the common {@link Map#put(Object,Object)} and {@link Map#get(Object)} methods.<br>
098 * <br>
099 * Bean maps are created in two ways...
100 * <ol>
101 *    <li> {@link BeanSession#toBeanMap(Object) BeanSession.toBeanMap()} - Wraps an existing bean inside a {@code Map} wrapper.
102 *    <li> {@link BeanSession#newBeanMap(Class) BeanSession.newBeanMap()} - Create a new bean instance wrapped in a {@code Map} wrapper.
103 * </ol>
104 *
105 * <h5 class='section'>Example:</h5>
106 * <p class='bcode'>
107 *    <jc>// A sample bean class</jc>
108 *    <jk>public class</jk> Person {
109 *       <jk>public</jk> String getName();
110 *       <jk>public void</jk> setName(String name);
111 *       <jk>public int</jk> getAge();
112 *       <jk>public void</jk> setAge(<jk>int</jk> age);
113 *    }
114 *
115 *    <jc>// Create a new bean session</jc>
116 *    BeanSession session = BeanContext.<jsf>DEFAULT</jsf>.createSession();
117 *
118 *    <jc>// Wrap an existing bean in a new bean map</jc>
119 *    BeanMap&lt;Person&gt; m1 = session.toBeanMap(<jk>new</jk> Person());
120 *    m1.put(<js>"name"</js>, <js>"John Smith"</js>);
121 *    m1.put(<js>"age"</js>, 45);
122 *
123 *    <jc>// Create a new bean instance wrapped in a new bean map</jc>
124 *    BeanMap&lt;Person&gt; m2 = session.newBeanMap(Person.<jk>class</jk>);
125 *    m2.put(<js>"name"</js>, <js>"John Smith"</js>);
126 *    m2.put(<js>"age"</js>, 45);
127 *    Person p = m2.getBean();  <jc>// Get the bean instance that was created.</jc>
128 * </p>
129 *
130 * <h5 class='topic'>Bean Annotations</h5>
131 * <p>
132 * This package contains annotations that can be applied to
133 *    class definitions to override what properties are detected on a bean.
134 * <h5 class='section'>Example:</h5>
135 * <p class='bcode'>
136 *    <jc>// Bean class definition where only property 'name' is detected.</jc>
137 *    <ja>&#64;Bean</ja>(properties=<js>"name"</js>)
138 *    <jk>public class</jk> Person {
139 *       <jk>public</jk> String getName();
140 *       <jk>public void</jk> setName(String name);
141 *       <jk>public int</jk> getAge();
142 *       <jk>public void</jk> setAge(<jk>int</jk> age);
143 *    }
144 * <p>
145 * See {@link Bean @Bean} and {@link BeanProperty @BeanProperty} for more information.
146 *
147 * <h5 class='topic'>Beans with read-only properties</h5>
148 * <p>
149 * Bean maps can also be defined on top of beans with read-only properties by adding a
150 *    {@link BeanConstructor @BeanConstructor} annotation to one of the constructors on the
151 *    bean class.  This will allow read-only properties to be set through constructor arguments.
152 * <p>
153 * When the <code>@BeanConstructor</code> annotation is present, bean instantiation is delayed until the call to {@link BeanMap#getBean()}.
154 * Until then, bean property values are stored in a local cache until <code>getBean()</code> is called.
155 * Because of this additional caching step, parsing into read-only beans tends to be slower and use
156 *    more memory than parsing into beans with writable properties.
157 * <p>
158 * Attempting to call {@link BeanMap#put(String,Object)} on a read-only property after calling {@link BeanMap#getBean()}
159 *    will result in a {@link BeanRuntimeException} being thrown.
160 * Multiple calls to {@link BeanMap#getBean()} will return the same bean instance.
161 * <p>
162 * Beans can be defined with a combination of read-only and read-write properties.
163 * <p>
164 * See {@link BeanConstructor @BeanConstructor} for more information.
165 *
166 * <h5 class='topic'>BeanFilters and PojoSwaps</h5>
167 * <p>
168 *    {@link BeanFilter BeanFilters} and {@link PojoSwap PojoSwaps} are used to tailor how beans and POJOs are handled.<br>
169 *    <ol class='spaced-list'>
170 *       <li>{@link BeanFilter} - Allows you to tailor handling of bean classes.
171 *          This class can be considered a programmatic equivalent to the {@link Bean} annotation when
172 *          annotating classes are not possible (e.g. you don't have access to the source).
173 *          This includes specifying which properties are visible and the ability to programmatically override the execution of properties.
174 *       <li>{@link PojoSwap} - Allows you to swap out non-serializable objects with serializable replacements.
175 *    </ol>
176 * <p>
177 * See <a class='doclink' href='transform/package-summary.html#TOC'>org.apache.juneau.transform</a> for more information.
178 *
179 * <h5 class='topic'>ClassMetas</h5>
180 * <p>
181 * The {@link ClassMeta} class is a wrapper around {@link Class} object that provides cached information
182 *    about that class (e.g. whether it's a {@link Map} or {@link Collection} or bean).
183 * <p>
184 * As a general rule, it's best to reuse bean contexts (and therefore serializers and parsers too)
185 *    whenever possible since it takes some time to populate the internal {@code ClassMeta} object cache.
186 * By reusing bean contexts, the class type metadata only needs to be calculated once which significantly
187 *    improves performance.
188 * <p>
189 * See {@link ClassMeta} for more information.
190 */

977    * Determines whether the specified class is ignored as a bean class based on the various
978    *    exclusion parameters specified on this context class.
979    *
980    * @param c The class type being tested.
981    * @return <jk>true</jk> if the specified class matches any of the exclusion parameters.
982    */
983   protected final boolean isNotABean(Class<?> c) {
984      if (c.isArray() || c.isPrimitive() || c.isEnum() || c.isAnnotation())
985         return true;
986      Package p = c.getPackage();
987      if (p != null) {
988         for (String p2 : notBeanPackageNames)
989            if (p.getName().equals(p2))
990               return true;
991         for (String p2 : notBeanPackagePrefixes)
992            if (p.getName().startsWith(p2))
993               return true;
994      }
995      for (Class exclude : notBeanClasses)
996         if (isParentClass(exclude, c))
997            return true;
998      return false;
999   }

1029   /**
1030    * Construct a {@code ClassMeta} wrapper around a {@link Class} object.
1031    *
1032    * @param <T> The class type being wrapped.
1033    * @param type The class to resolve.
1034    * @return If the class is not an array, returns a cached {@link ClassMeta} object.
1035    * Otherwise, returns a new {@link ClassMeta} object every time.<br>
1036    */
1037   public final <T> ClassMeta<T> getClassMeta(Class<T> type) {
1038
1039      // If this is an array, then we want it wrapped in an uncached ClassMeta object.
1040      // Note that if it has a pojo swap, we still want to cache it so that
1041      // we can cache something like byte[] with ByteArrayBase64Swap.
1042      if (type.isArray() && findPojoSwap(type) == null)
1043         return new ClassMeta(type, this, findImplClass(type), findBeanFilter(type), findPojoSwap(type), findChildPojoSwaps(type));
1044
1045      // This can happen if we have transforms defined against String or Object.
1046      if (cmCache == null)
1047         return null;
1048
1049      ClassMeta<T> cm = cmCache.get(type);
1050      if (cm == null) {
1051
1052         synchronized (this) {

1062   /**
1063    * Used to resolve <code>ClassMetas</code> of type <code>Collection</code> and <code>Map</code> that have
1064    * <code>ClassMeta</code> values that themselves could be collections or maps.
1065    * <p>
1066    * <code>Collection</code> meta objects are assumed to be followed by zero or one meta objects indicating the element type.
1067    * <p>
1068    * <code>Map</code> meta objects are assumed to be followed by zero or two meta objects indicating the key and value types.
1069    * <p>
1070    * The array can be arbitrarily long to indicate arbitrarily complex data structures.
1071    *
1072    * <h5 class='section'>Examples:</h5>
1073    * <ul>
1074    *    <li><code>getClassMeta(String.<jk>class</jk>);</code> - A normal type.
1075    *    <li><code>getClassMeta(List.<jk>class</jk>);</code> - A list containing objects.
1076    *    <li><code>getClassMeta(List.<jk>class</jk>, String.<jk>class</jk>);</code> - A list containing strings.
1077    *    <li><code>getClassMeta(LinkedList.<jk>class</jk>, String.<jk>class</jk>);</code> - A linked-list containing strings.
1078    *    <li><code>getClassMeta(LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);</code> - A linked-list containing linked-lists of strings.
1079    *    <li><code>getClassMeta(Map.<jk>class</jk>);</code> - A map containing object keys/values.
1080    *    <li><code>getClassMeta(Map.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);</code> - A map containing string keys/values.
1081    *    <li><code>getClassMeta(Map.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);</code> - A map containing string keys and values of lists containing beans.
1082    * </ul>
1083    *
1084    * @param type The class to resolve.
1085    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
1086    * @param args The type arguments of the class if it's a collection or map.
1087    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
1088    *    <br>Ignored if the main type is not a map or collection.
1089    * @return The resolved class meta.
1090    */
1091   public final <T> ClassMeta<T> getClassMeta(Type type, Type...args) {
1092      if (type == null)
1093         return null;
1094      ClassMeta<T> cm = type instanceof Class ? getClassMeta((Class)type) : resolveClassMeta(type, null);
1095      if (args.length == 0)
1096         return cm;
1097      ClassMeta<?>[] cma = new ClassMeta[args.length+1];
1098      cma[0] = cm;
1099      for (int i = 0; i < Array.getLength(args); i++) {
1100         Type arg = (Type)Array.get(args, i);
1101         cma[i+1] = arg instanceof Class ? getClassMeta((Class)arg) : resolveClassMeta(arg, null);
1102      }
1103      return (ClassMeta<T>) getTypedClassMeta(cma, 0);
1104   }

1106   /*
1107    * Resolves the 'genericized' class meta at the specified position in the ClassMeta array.
1108    */
1109   private ClassMeta<?> getTypedClassMeta(ClassMeta<?>[] c, int pos) {
1110      ClassMeta<?> cm = c[pos++];
1111      if (cm.isCollection()) {
1112         ClassMeta<?> ce = c.length == pos ? object() : getTypedClassMeta(c, pos);
1113         return (ce.isObject() ? cm : new ClassMeta(cm, null, null, ce));
1114      } else if (cm.isMap()) {
1115          ClassMeta<?> ck = c.length == pos ? object() : c[pos++];
1116          ClassMeta<?> cv = c.length == pos ? object() : getTypedClassMeta(c, pos);
1117          return (ck.isObject() && cv.isObject() ? cm : new ClassMeta(cm, ck, cv, null));
1118      }
1119      return cm;
1120   }

1122   final ClassMeta resolveClassMeta(Type o, Map<Class<?>,Class<?>[]> typeVarImpls) {
1123      if (o == null)
1124         return null;
1125
1126      if (o instanceof ClassMeta)
1127         return (ClassMeta)o;
1128
1129      Class c = resolve(o, typeVarImpls);
1130
1131      // This can happen when trying to resolve the "E getFirst()" method on LinkedList, whose type is a TypeVariable
1132      // These should just resolve to Object.
1133      if (c == null)
1134         return object();
1135
1136      ClassMeta rawType = getClassMeta(c);
1137
1138      // If this is a Map or Collection, and the parameter types aren't part
1139      // of the class definition itself (e.g. class AddressBook extends List<Person>),
1140      // then we need to figure out the parameters.
1141      if (rawType.isMap() || rawType.isCollection()) {
1142         ClassMeta[] params = findParameters(o, c);
1143         if (params == null)
1144            return rawType;
1145         if (rawType.isMap()) {
1146            if (params.length != 2)
1147               return rawType;
1148            if (params[0].isObject() && params[1].isObject())
1149               return rawType;
1150            return new ClassMeta(rawType, params[0], params[1], null);
1151         }
1152         if (rawType.isCollection()) {
1153            if (params.length != 1)
1154               return rawType;
1155            if (params[0].isObject())
1156               return rawType;
1157            return new ClassMeta(rawType, null, null, params[0]);
1158         }
1159      }
1160
1161      return rawType;
1162   }

1164   /** 
1165    * Convert a Type to a Class if possible.
1166    * Return null if not possible.
1167    */
1168   final Class resolve(Type t, Map<Class<?>,Class<?>[]> typeVarImpls) {
1169
1170      if (t instanceof Class)
1171         return (Class)t;
1172
1173      if (t instanceof ParameterizedType)
1174         // A parameter (e.g. <String>.
1175         return (Class)((ParameterizedType)t).getRawType();
1176
1177      if (t instanceof GenericArrayType) {
1178         // An array parameter (e.g. <byte[]>).
1179         Type gatct = ((GenericArrayType)t).getGenericComponentType();
1180
1181         if (gatct instanceof Class)
1182            return Array.newInstance((Class)gatct, 0).getClass();
1183
1184         if (gatct instanceof ParameterizedType)
1185            return Array.newInstance((Class)((ParameterizedType)gatct).getRawType(), 0).getClass();
1186
1187         if (gatct instanceof GenericArrayType)
1188            return Array.newInstance(resolve(gatct, typeVarImpls), 0).getClass();
1189
1190         return null;
1191
1192      } else if (t instanceof TypeVariable) {
1193         if (typeVarImpls != null) {
1194            TypeVariable tv = (TypeVariable)t;
1195            String varName = tv.getName();
1196            int varIndex = -1;
1197            Class gc = (Class)tv.getGenericDeclaration();
1198            TypeVariable[] tvv = gc.getTypeParameters();
1199            for (int i = 0; i < tvv.length; i++) {
1200               if (tvv[i].getName().equals(varName)) {
1201                  varIndex = i;
1202               }
1203            }
1204            if (varIndex != -1) {
1205
1206               // If we couldn't find a type variable implementation, that means
1207               // the type was defined at runtime (e.g. Bean b = new Bean<Foo>();)
1208               // in which case the type is lost through erasure.
1209               // Assume java.lang.Object as the type.
1210               if (! typeVarImpls.containsKey(gc))
1211                  return null;
1212
1213               return typeVarImpls.get(gc)[varIndex];
1214            }
1215         }
1216      }
1217      return null;
1218   }

1220   final ClassMeta[] findParameters(Type o, Class c) {
1221      if (o == null)
1222         o = c;
1223
1224      // Loop until we find a ParameterizedType
1225      if (! (o instanceof ParameterizedType)) {
1226         loop: do {
1227            o = c.getGenericSuperclass();
1228            if (o instanceof ParameterizedType)
1229               break loop;
1230            for (Type t : c.getGenericInterfaces()) {
1231               o = t;
1232               if (o instanceof ParameterizedType)
1233                  break loop;
1234            }
1235            c = c.getSuperclass();
1236         } while (c != null);
1237      }
1238
1239      if (o instanceof ParameterizedType) {
1240         ParameterizedType pt = (ParameterizedType)o;
1241         if (! pt.getRawType().equals(Enum.class)) {
1242            List<ClassMeta<?>> l = new LinkedList<ClassMeta<?>>();
1243            for (Type pt2 : pt.getActualTypeArguments()) {
1244               if (pt2 instanceof WildcardType || pt2 instanceof TypeVariable)
1245                  return null;
1246               l.add(resolveClassMeta(pt2, null));
1247            }
1248            if (l.isEmpty())
1249               return null;
1250            return l.toArray(new ClassMeta[l.size()]);
1251         }
1252      }
1253
1254      return null;
1255   }

1276    * @param p The property annotation on the type if there is one.

1284      ClassMeta<T> cm = resolveClassMeta(t, typeVarImpls);
1285      ClassMeta<T> cm2 = cm;
1286      if (p != null) {
1287
1288         if (p.type() != Object.class)
1289            cm2 = resolveClassMeta(p.type(), typeVarImpls);
1290
1291         if (cm2.isMap()) {
1292            Class<?>[] pParams = (p.params().length == 0 ? new Class[]{Object.class, Object.class} : p.params());
1293            if (pParams.length != 2)
1294               throw new RuntimeException("Invalid number of parameters specified for Map (must be 2): " + pParams.length);
1295            ClassMeta<?> keyType = resolveType(pParams[0], cm2.getKeyType(), cm.getKeyType());
1296            ClassMeta<?> valueType = resolveType(pParams[1], cm2.getValueType(), cm.getValueType());
1297            if (keyType.isObject() && valueType.isObject())
1298               return cm2;
1299            return new ClassMeta<T>(cm2, keyType, valueType, null);
1300         }
1301
1302         if (cm2.isCollection()) {
1303            Class<?>[] pParams = (p.params().length == 0 ? new Class[]{Object.class} : p.params());
1304            if (pParams.length != 1)
1305               throw new RuntimeException("Invalid number of parameters specified for Collection (must be 1): " + pParams.length);
