027/**
028 * Context object that lives for the duration of a single use of {@link Serializer}.
029 * <p>
030 * Used by serializers for the following purposes:
031 * <ul class='spaced-list'>
032 *    <li>Keeping track of how deep it is in a model for indentation purposes.
033 *    <li>Ensuring infinite loops don't occur by setting a limit on how deep to traverse a model.
034 *    <li>Ensuring infinite loops don't occur from loops in the model (when detectRecursions is enabled.
035 *    <li>Allowing serializer properties to be overridden on method calls.
036 * </ul>
037 * <p>
038 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
039 */
040public class SerializerSession extends BeanSession {

072   /**
073    * Create a new session using properties specified in the context.
074    *
075    * @param ctx The context creating this session object.
076    * The context contains all the configuration settings for this object.
077    * @param output The output object.
078    * <br>Character-based serializers can handle the following output class types:
079    * <ul>
080    *    <li>{@link Writer}
081    *    <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
082    *    <li>{@link File} - Output will be written as system-default encoded stream.
083    * </ul>
084    * <br>Stream-based serializers can handle the following output class types:
085    * <ul>
086    *    <li>{@link OutputStream}
087    *    <li>{@link File}
088    * </ul>
089    * @param op The override properties.
090    * These override any context properties defined in the context.
091    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
092    * @param locale The session locale.
093    * If <jk>null</jk>, then the locale defined on the context is used.
094    * @param timeZone The session timezone.
095    * If <jk>null</jk>, then the timezone defined on the context is used.
096    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
097    */
098   public SerializerSession(SerializerContext ctx, ObjectMap op, Object output, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
099      super(ctx, op, locale, timeZone, mediaType);
100      this.javaMethod = javaMethod;
101      this.output = output;
102      if (op == null || op.isEmpty()) {
103         maxDepth = ctx.maxDepth;
104         initialDepth = ctx.initialDepth;
105         detectRecursions = ctx.detectRecursions;
106         ignoreRecursions = ctx.ignoreRecursions;
107         useWhitespace = ctx.useWhitespace;
108         addBeanTypeProperties = ctx.addBeanTypeProperties;
109         trimNulls = ctx.trimNulls;
110         trimEmptyCollections = ctx.trimEmptyCollections;
111         trimEmptyMaps = ctx.trimEmptyMaps;
112         trimStrings = ctx.trimStrings;
113         quoteChar = ctx.quoteChar;
114         relativeUriBase = ctx.relativeUriBase;
115         absolutePathUriBase = ctx.absolutePathUriBase;
116         sortCollections = ctx.sortCollections;
117         sortMaps = ctx.sortMaps;
118         abridged = ctx.abridged;
119      } else {
120         maxDepth = op.getInt(SERIALIZER_maxDepth, ctx.maxDepth);
121         initialDepth = op.getInt(SERIALIZER_initialDepth, ctx.initialDepth);
122         detectRecursions = op.getBoolean(SERIALIZER_detectRecursions, ctx.detectRecursions);
123         ignoreRecursions = op.getBoolean(SERIALIZER_ignoreRecursions, ctx.ignoreRecursions);
124         useWhitespace = op.getBoolean(SERIALIZER_useWhitespace, ctx.useWhitespace);
125         addBeanTypeProperties = op.getBoolean(SERIALIZER_addBeanTypeProperties, ctx.addBeanTypeProperties);
126         trimNulls = op.getBoolean(SERIALIZER_trimNullProperties, ctx.trimNulls);
127         trimEmptyCollections = op.getBoolean(SERIALIZER_trimEmptyCollections, ctx.trimEmptyCollections);
128         trimEmptyMaps = op.getBoolean(SERIALIZER_trimEmptyMaps, ctx.trimEmptyMaps);
129         trimStrings = op.getBoolean(SERIALIZER_trimStrings, ctx.trimStrings);
130         quoteChar = op.getString(SERIALIZER_quoteChar, ""+ctx.quoteChar).charAt(0);
131         relativeUriBase = op.getString(SERIALIZER_relativeUriBase, ctx.relativeUriBase);
132         absolutePathUriBase = op.getString(SERIALIZER_absolutePathUriBase, ctx.absolutePathUriBase);
133         sortCollections = op.getBoolean(SERIALIZER_sortCollections, ctx.sortMaps);
134         sortMaps = op.getBoolean(SERIALIZER_sortMaps, ctx.sortMaps);
135         abridged = op.getBoolean(SERIALIZER_abridged, ctx.abridged);
136      }
137
138      this.indent = initialDepth;
139      if (detectRecursions || isDebug()) {
140         set = new IdentityHashMap<Object,Object>();
141      } else {
142         set = Collections.emptyMap();
143      }
144   }

146   /**
147    * Wraps the specified output object inside an output stream.
148    * Subclasses can override this method to implement their own specialized output streams.
149    * <p>
150    * This method can be used if the output object is any of the following class types:
151    * <ul>
152    *    <li>{@link OutputStream}
153    *    <li>{@link File}
154    * </ul>
155    *
156    * @return The output object wrapped in an output stream.
157    * @throws Exception If object could not be converted to an output stream.
158    */
159   public OutputStream getOutputStream() throws Exception {
160      if (output == null)
161         throw new SerializeException("Output cannot be null.");
162      if (output instanceof OutputStream)
163         return (OutputStream)output;
164      if (output instanceof File) {
165         if (outputStream == null)
166            outputStream = new BufferedOutputStream(new FileOutputStream((File)output));
167         return outputStream;
168      }
169      throw new SerializeException("Cannot convert object of type {0} to an OutputStream.", output.getClass().getName());
170   }

173   /**
174    * Wraps the specified output object inside a writer.
175    * Subclasses can override this method to implement their own specialized writers.
176    * <p>
177    * This method can be used if the output object is any of the following class types:
178    * <ul>
179    *    <li>{@link Writer}
180    *    <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
181    *    <li>{@link File} - Output will be written as system-default encoded stream.
182    * </ul>
183    *
184    * @return The output object wrapped in a Writer.
185    * @throws Exception If object could not be converted to a writer.
186    */
187   public Writer getWriter() throws Exception {
188      if (output == null)
189         throw new SerializeException("Output cannot be null.");
190      if (output instanceof Writer)
191         return (Writer)output;
192      if (output instanceof OutputStream) {
193         if (flushOnlyWriter == null)
194            flushOnlyWriter = new OutputStreamWriter((OutputStream)output, IOUtils.UTF8);
195         return flushOnlyWriter;
196      }
197      if (output instanceof File) {
198         if (writer == null)
199            writer = new OutputStreamWriter(new BufferedOutputStream(new FileOutputStream((File)output)));
200         return writer;
201      }
202      throw new SerializeException("Cannot convert object of type {0} to a Writer.", output.getClass().getName());
203   }

377   /**
378    * Push the specified object onto the stack.
379    *
380    * @param attrName The attribute name.
381    * @param o The current object being serialized.
382    * @param eType The expected class type.
383    * @return The {@link ClassMeta} of the object so that <code>instanceof</code> operations
384    *    only need to be performed once (since they can be expensive).<br>
385    * @throws SerializeException If recursion occurred.
386    */
387   public ClassMeta<?> push(String attrName, Object o, ClassMeta<?> eType) throws SerializeException {
388      indent++;
389      isBottom = true;
390      if (o == null)
391         return null;
392      Class<?> c = o.getClass();
393      ClassMeta<?> cm = (eType != null && c == eType.getInnerClass()) ? eType : getClassMeta(c);
394      if (cm.isCharSequence() || cm.isNumber() || cm.isBoolean())
395         return cm;
396      if (detectRecursions || isDebug()) {
397         if (stack.size() > maxDepth)
398            return null;
399         if (willRecurse(attrName, o, cm))
400            return null;
401         isBottom = false;
402         stack.add(new StackElement(stack.size(), attrName, o, cm));
403         if (isDebug())
404            getLogger().info(getStack(false));
405         set.put(o, o);
406      }
407      return cm;
408   }

410   /**
411    * Returns <jk>true</jk> if {@link SerializerContext#SERIALIZER_detectRecursions} is enabled, and the specified
412    *    object is already higher up in the serialization chain.
413    *
414    * @param attrName The bean property attribute name, or some other identifier.
415    * @param o The object to check for recursion.
416    * @param cm The metadata on the object class.
417    * @return <jk>true</jk> if recursion detected.
418    * @throws SerializeException If recursion occurred.
419    */
420   public boolean willRecurse(String attrName, Object o, ClassMeta<?> cm) throws SerializeException {
421      if (! (detectRecursions || isDebug()))
422         return false;
423      if (! set.containsKey(o))
424         return false;
425      if (ignoreRecursions && ! isDebug())
426         return true;
427
428      stack.add(new StackElement(stack.size(), attrName, o, cm));
429      throw new SerializeException("Recursion occurred, stack={0}", getStack(true));
430   }

432   /**
433    * Pop an object off the stack.
434    */
435   public void pop() {
436      indent--;
437      if ((detectRecursions || isDebug()) && ! isBottom)  {
438         Object o = stack.removeLast().o;
439         Object o2 = set.remove(o);
440         if (o2 == null)
441            addWarning("Couldn't remove object of type ''{0}'' on attribute ''{1}'' from object stack.", o.getClass().getName(), stack);
442      }
443      isBottom = false;
444   }

499   /**
500    * Returns <jk>true</jk> if the specified value should not be serialized.
501    *
502    * @param cm The class type of the object being serialized.
503    * @param attrName The bean attribute name, or <jk>null</jk> if this isn't a bean attribute.
504    * @param value The object being serialized.
505    * @return <jk>true</jk> if the specified value should not be serialized.
506    * @throws SerializeException If recursion occurred.
507    */
508   public final boolean canIgnoreValue(ClassMeta<?> cm, String attrName, Object value) throws SerializeException {
509
510      if (trimNulls && value == null)
511         return true;
512
513      if (value == null)
514         return false;
515
516      if (cm == null)
517         cm = object();
518
519      if (trimEmptyCollections) {
520         if (cm.isArray() || (cm.isObject() && value.getClass().isArray())) {
521            if (((Object[])value).length == 0)
522               return true;
523         }
524         if (cm.isCollection() || (cm.isObject() && isParentClass(Collection.class, value.getClass()))) {
525            if (((Collection<?>)value).isEmpty())
526               return true;
527         }
528      }
529
530      if (trimEmptyMaps) {
531         if (cm.isMap() || (cm.isObject() && isParentClass(Map.class, value.getClass()))) {
532            if (((Map<?,?>)value).isEmpty())
533               return true;
534         }
535      }
536
537      if (trimNulls && willRecurse(attrName, value, cm))
538         return true;
539
540      return false;
541   }

567   /**
568    * Converts a String to an absolute URI based on the {@link SerializerContext#SERIALIZER_absolutePathUriBase} and
569    *    {@link SerializerContext#SERIALIZER_relativeUriBase} settings on this context.
570    *
571    * @param uri The input URI.
572    * @return The resolved URI.
573    */
574   public String resolveUri(String uri) {
575      if (uri.indexOf("://") != -1 || (absolutePathUriBase == null && relativeUriBase == null))
576         return uri;
577      StringBuilder sb = getStringBuilder();
578      if (StringUtils.startsWith(uri, '/')) {
579         if (absolutePathUriBase != null)
580            sb.append(absolutePathUriBase);
581      } else {
582         if (relativeUriBase != null) {
583            sb.append(relativeUriBase);
584            if (! uri.equals("/"))
585               sb.append("/");
586         }
587      }
588      sb.append(uri);
589      String s = sb.toString();
590      returnStringBuilder(sb);
591      return s;
592   }

611   @Override
612   public boolean close() {
613      if (super.close()) {
614         try {
615            if (outputStream != null)
616               outputStream.close();
617            if (flushOnlyWriter != null)
618               flushOnlyWriter.flush();
619            if (writer != null)
620               writer.close();
621         } catch (IOException e) {
622            throw new BeanRuntimeException(e);
623         }
624         return true;
625      }
626      return false;
627   }

629   private static class StackElement {
630      private int depth;
631      private String name;
632      private Object o;
633      private ClassMeta<?> aType;
634
635      private StackElement(int depth, String name, Object o, ClassMeta<?> aType) {
636         this.depth = depth;
637         this.name = name;
638         this.o = o;
639         this.aType = aType;
640      }
641
642      private String toString(boolean simple) {
643         StringBuilder sb = new StringBuilder().append('[').append(depth).append(']');
644         sb.append(StringUtils.isEmpty(name) ? "<noname>" : name).append(':');
645         sb.append(aType.toString(simple));
646         if (aType != aType.getSerializedClassMeta())
647            sb.append('/').append(aType.getSerializedClassMeta().toString(simple));
648         return sb.toString();
649      }
650   }

652   private String getStack(boolean full) {
653      StringBuilder sb = new StringBuilder();
654      for (StackElement e : stack) {
655         if (full) {
656            sb.append("\n\t");
657            for (int i = 1; i < e.depth; i++)
658               sb.append("  ");
659            if (e.depth > 0)
660               sb.append("->");
661            sb.append(e.toString(false));
662         } else {
663            sb.append(" > ").append(e.toString(true));
664         }
665      }
666      return sb.toString();
667   }

669   /**
670    * Returns information used to determine at what location in the parse a failure occurred.
671    *
672    * @return A map, typically containing something like <code>{line:123,column:456,currentProperty:"foobar"}</code>
673    */
674   public Map<String,Object> getLastLocation() {
675      Map<String,Object> m = new LinkedHashMap<String,Object>();
676      if (currentClass != null)
677         m.put("currentClass", currentClass);
678      if (currentProperty != null)
679         m.put("currentProperty", currentProperty);
680      if (stack != null && ! stack.isEmpty())
681         m.put("stack", stack);
682      return m;
683   }

697   /**
698    * Resolves the dictionary name for the actual type.
699    *
700    * @param eType The expected type of the bean property.
701    * @param aType The actual type of the bean property.
702    * @param pMeta The current bean property being serialized.
703    * @return The bean dictionary name, or <jk>null</jk> if a name could not be found.
704    */
705   public String getBeanTypeName(ClassMeta<?> eType, ClassMeta<?> aType, BeanPropertyMeta pMeta) {
706      if (eType == aType)
707         return null;
708
709      if (! isAddBeanTypeProperties())
710         return null;
711
712      String eTypeTn = eType.getDictionaryName();
713
714      // First see if it's defined on the actual type.
715      String tn = aType.getDictionaryName();
716      if (tn != null && ! tn.equals(eTypeTn)) {
717         return tn;
718      }
719
720      // Then see if it's defined on the expected type.
721      // The expected type might be an interface with mappings for implementation classes.
722      BeanRegistry br = eType.getBeanRegistry();
723      if (br != null) {
724         tn = br.getTypeName(aType);
725         if (tn != null && ! tn.equals(eTypeTn))
726            return tn;
727      }
728
729      // Then look on the bean property.
730      br = pMeta == null ? null : pMeta.getBeanRegistry();
731      if (br != null) {
732         tn = br.getTypeName(aType);
733         if (tn != null && ! tn.equals(eTypeTn))
734            return tn;
735      }
736
737      // Finally look in the session.
738      br = getBeanRegistry();
739      if (br != null) {
740         tn = br.getTypeName(aType);
741         if (tn != null && ! tn.equals(eTypeTn))
742            return tn;
743      }
744
745      return null;
746   }
