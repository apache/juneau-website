128   /**
129    * <b>Configuration property:</b>  Max serialization depth.
130    * <p>
131    * <ul>
132    *    <li><b>Name:</b> <js>"Serializer.maxDepth"</js>
133    *    <li><b>Data type:</b> <code>Integer</code>
134    *    <li><b>Default:</b> <code>100</code>
135    * </ul>
136    * <p>
137    * Abort serialization if specified depth is reached in the POJO tree.
138    * If this depth is exceeded, an exception is thrown.
139    * This prevents stack overflows from occurring when trying to serialize models with recursive references.
140    */
141   public static final String SERIALIZER_maxDepth = "Serializer.maxDepth";
142
143   /**
144    * <b>Configuration property:</b>  Initial depth.
145    * <p>
146    * <ul>
147    *    <li><b>Name:</b> <js>"Serializer.initialDepth"</js>
148    *    <li><b>Data type:</b> <code>Integer</code>
149    *    <li><b>Default:</b> <code>0</code>
150    * </ul>
151    * <p>
152    * The initial indentation level at the root.
153    * Useful when constructing document fragments that need to be indented at a certain level.
154    */
155   public static final String SERIALIZER_initialDepth = "Serializer.initialDepth";

157   /**
158    * <b>Configuration property:</b>  Automatically detect POJO recursions.
159    * <p>
160    * <ul>
161    *    <li><b>Name:</b> <js>"Serializer.detectRecursions"</js>
162    *    <li><b>Data type:</b> <code>Boolean</code>
163    *    <li><b>Default:</b> <jk>false</jk>
164    * </ul>
165    * <p>
166    * Specifies that recursions should be checked for during serialization.
167    * <p>
168    * Recursions can occur when serializing models that aren't true trees, but rather contain loops.
169    * <p>
170    * The behavior when recursions are detected depends on the value for {@link #SERIALIZER_ignoreRecursions}.
171    * <p>
172    * For example, if a model contains the links A-&gt;B-&gt;C-&gt;A, then the JSON generated will look like
173    *    the following when <jsf>SERIALIZER_ignoreRecursions</jsf> is <jk>true</jk>...
174    * <code>{A:{B:{C:null}}}</code><br>
175    * <p>
176    * Note:  Checking for recursion can cause a small performance penalty.
177    */
178   public static final String SERIALIZER_detectRecursions = "Serializer.detectRecursions";
179
180   /**
181    * <b>Configuration property:</b>  Ignore recursion errors.
182    * <p>
183    * <ul>
184    *    <li><b>Name:</b> <js>"Serializer.ignoreRecursions"</js>
185    *    <li><b>Data type:</b> <code>Boolean</code>
186    *    <li><b>Default:</b> <jk>false</jk>
187    * </ul>
188    * <p>
189    * Used in conjunction with {@link #SERIALIZER_detectRecursions}.
190    * Setting is ignored if <jsf>SERIALIZER_detectRecursions</jsf> is <jk>false</jk>.
191    * <p>
192    * If <jk>true</jk>, when we encounter the same object when serializing a tree,
193    *    we set the value to <jk>null</jk>.
194    * Otherwise, an exception is thrown.
195    */
196   public static final String SERIALIZER_ignoreRecursions = "Serializer.ignoreRecursions";
197

446   /**
447    * Constructor.
448    *
449    * @param cf The factory that created this context.
450    */
451   public SerializerContext(ContextFactory cf) {
452      super(cf);
453      maxDepth = cf.getProperty(SERIALIZER_maxDepth, int.class, 100);
454      initialDepth = cf.getProperty(SERIALIZER_initialDepth, int.class, 0);
455      debug = cf.getProperty(SERIALIZER_debug, boolean.class, false);
456      detectRecursions = cf.getProperty(SERIALIZER_detectRecursions, boolean.class, false);
457      ignoreRecursions = cf.getProperty(SERIALIZER_ignoreRecursions, boolean.class, false);
458      useIndentation = cf.getProperty(SERIALIZER_useIndentation, boolean.class, false);
459      addBeanTypeProperties = cf.getProperty(SERIALIZER_addBeanTypeProperties, boolean.class, false);
460      trimNulls = cf.getProperty(SERIALIZER_trimNullProperties, boolean.class, true);
461      trimEmptyCollections = cf.getProperty(SERIALIZER_trimEmptyCollections, boolean.class, false);
462      trimEmptyMaps = cf.getProperty(SERIALIZER_trimEmptyMaps, boolean.class, false);
463      trimStrings = cf.getProperty(SERIALIZER_trimStrings, boolean.class, false);
464      sortCollections = cf.getProperty(SERIALIZER_sortCollections, boolean.class, false);
465      sortMaps = cf.getProperty(SERIALIZER_sortMaps, boolean.class, false);
466      quoteChar = cf.getProperty(SERIALIZER_quoteChar, String.class, "\"").charAt(0);
467      relativeUriBase = resolveRelativeUriBase(cf.getProperty(SERIALIZER_relativeUriBase, String.class, ""));
468      absolutePathUriBase = resolveAbsolutePathUriBase(cf.getProperty(SERIALIZER_absolutePathUriBase, String.class, ""));
469   }
470
471   private String resolveRelativeUriBase(String s) {
472      if (StringUtils.isEmpty(s))
473         return null;
474      if (s.equals("/"))
475         return s;
476      else if (StringUtils.endsWith(s, '/'))
477         s = s.substring(0, s.length()-1);
478      return s;
479   }
480
481   private String resolveAbsolutePathUriBase(String s) {
482      if (StringUtils.isEmpty(s))
483         return null;
484      if (StringUtils.endsWith(s, '/'))
485         s = s.substring(0, s.length()-1);
486      return s;
487   }
