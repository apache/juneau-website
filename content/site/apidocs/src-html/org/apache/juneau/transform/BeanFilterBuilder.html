013package org.apache.juneau.transform;
014
015import java.beans.*;
016import java.util.*;
017
018import org.apache.juneau.*;
019
020/**
021 * Builder class for {@link BeanFilter} objects.
022 * <p>
023 * Bean filter builders must have a public no-arg constructor.
024 * Builder settings should be set in the constructor using the provided setters on this class.
025 *
026 * <h5 class='section'>Example:</h5>
027 * <p class='bcode'>
028 *    <jc>// Create our serializer with a bean filter.</jc>
029 *    WriterSerializer s = <jk>new</jk> JsonSerializerBuilder().beanFilters(AddressFilter.<jk>class</jk>).build();
030 *
031 *    Address a = <jk>new</jk> Address();
032 *    String json = s.serialize(a); <jc>// Serializes only street, city, state.</jc>
033 *
034 *    <jc>// Filter class defined via setters</jc>
035 *    <jk>public class</jk> AddressFilter <jk>extends</jk> BeanFilterBuilder {
036 *       <jk>public</jk> AddressFilter() {
037 *          super(Address.<jk>class</jk>);
038 *          setProperties(<js>"street"</js>,<js>"city"</js>,<js>"state"</js>);
039 *       }
040 *    }
041 * </p>
042 *
043 * <h5 class='section'>Additional information:</h5>
044 * See <a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.transform</a> for more information.
045 */
046@SuppressWarnings("hiding")
047public abstract class BeanFilterBuilder {
048
049   Class<?> beanClass;
050   String typeName;
051   String[] properties, excludeProperties;
052   Class<?> interfaceClass, stopClass;
053   boolean sortProperties;
054   PropertyNamer propertyNamer;
055   List<Class<?>> beanDictionary;
056
057   /**
058    * Constructor.
059    *
060    * @param beanClass The bean class that this filter applies to.
061    */
062   public BeanFilterBuilder(Class<?> beanClass) {
063      this.beanClass = beanClass;
064   }
065
066   /**
067    * Specifies the type name for this bean.
068    *
069    * @param typeName The dictionary name associated with this bean.
070    * @return This object (for method chaining).
071    */
072   public BeanFilterBuilder typeName(String typeName) {
073      this.typeName = typeName;
074      return this;
075   }
076
077   /**
078    * Specifies the set and order of names of properties associated with the bean class.
079    * The order specified is the same order that the entries will be returned by the {@link BeanMap#entrySet()} and related methods.
080    * Entries in the list can also contain comma-delimited lists that will be split.
081    *
082    * @param properties The properties associated with the bean class.
083    * @return This object (for method chaining).
084    */
085   public BeanFilterBuilder properties(String...properties) {
086      this.properties = properties;
087      return this;
088   }
089
090   /**
091    * Specifies the list of properties to ignore on a bean.
092    *
093    * @param excludeProperties The list of properties to ignore on a bean.
094    * @return This object (for method chaining).
095    */
096   public BeanFilterBuilder excludeProperties(String...excludeProperties) {
097      this.excludeProperties = excludeProperties;
098      return this;
099   }
100
101   /**
102    * Identifies a class to be used as the interface class for this and all subclasses.
103    * <p>
104    * When specified, only the list of properties defined on the interface class will be used during serialization.
105    * Additional properties on subclasses will be ignored.
106    * <p class='bcode'>
107    *    <jc>// Parent class</jc>
108    *    <jk>public abstract class</jk> A {
109    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
110    *    }
111    *
112    *    <jc>// Sub class</jc>
113    *    <jk>public class</jk> A1 <jk>extends</jk> A {
114    *       <jk>public</jk> String <jf>f1</jf> = <js>"f1"</js>;
115    *    }
116    *
117    *    <jc>// Filter class</jc>
118    *    <jk>public class</jk> AFilter <jk>extends</jk> BeanFilterBuilder {
119    *       <jk>public</jk> AFilter() {
120    *          super(A.<jk>class</jk>);
121    *          setInterfaceClass(A.<jk>class</jk>);
122    *       }
123    *    }
124    *
125    *    JsonSerializer s = new JsonSerializerBuilder().beanFilters(AFilter.<jk>class</jk>).build();
126    *    A1 a1 = <jk>new</jk> A1();
127    *    String r = s.serialize(a1);
128    *    <jsm>assertEquals</jsm>(<js>"{f0:'f0'}"</js>, r);  <jc>// Note f1 is not serialized</jc>
129    * </p>
130    * <p>
131    * Note that this filter can be used on the parent class so that it filters to all child classes,
132    *    or can be set individually on the child classes.
133    *
134    * @param interfaceClass The interface class to use for this bean class.
135    * @return This object (for method chaining).
136    */
137   public BeanFilterBuilder interfaceClass(Class<?> interfaceClass) {
138      this.interfaceClass = interfaceClass;
139      return this;
140   }
141
142   /**
143    * Identifies a stop class for this class and all subclasses.
144    * <p>
145    * Identical in purpose to the stop class specified by {@link Introspector#getBeanInfo(Class, Class)}.
146    * Any properties in the stop class or in its base classes will be ignored during analysis.
147    * <p>
148    * For example, in the following class hierarchy, instances of <code>C3</code> will include property <code>p3</code>, but
149    *    not <code>p1</code> or <code>p2</code>.
150    * <p class='bcode'>
151    *    <jk>public class</jk> C1 {
152    *       <jk>public int</jk> getP1();
153    *    }
154    *
155    *    <jk>public class</jk> C2 <jk>extends</jk> C1 {
156    *       <jk>public int</jk> getP2();
157    *    }
158    *
159    *    <ja>@Bean</ja>(stopClass=C2.<jk>class</jk>)
160    *    <jk>public class</jk> C3 <jk>extends</jk> C2 {
161    *       <jk>public int</jk> getP3();
162    *    }
163    * </p>
164    *
165    * @param stopClass
166    * @return This object (for method chaining).
167    */
168   public BeanFilterBuilder stopClass(Class<?> stopClass) {
169      this.stopClass = stopClass;
170      return this;
171   }
172
173   /**
174    * Sort properties in alphabetical order.
175    *
176    * @param sortProperties
177    * @return This object (for method chaining).
178    */
179   public BeanFilterBuilder sortProperties(boolean sortProperties) {
180      this.sortProperties = sortProperties;
181      return this;
182   }
183
184   /**
185    * The property namer to use to name bean properties.
186    *
187    * @param propertyNamer The property namer instance.
188    * @return This object (for method chaining).
189    */
190   public BeanFilterBuilder propertyNamer(PropertyNamer propertyNamer) {
191      this.propertyNamer = propertyNamer;
192      return this;
193   }
194
195   /**
196    * The property namer to use to name bean properties.
197    *
198    * @param c The property namer class.  Must have a public no-arg constructor.
199    * @return This object (for method chaining).
200    * @throws Exception Thrown from constructor method.
201    */
202   public BeanFilterBuilder propertyNamer(Class<? extends PropertyNamer> c) throws Exception {
203      this.propertyNamer = c.newInstance();
204      return this;
205   }
206
207   /**
208    * Sets the contents of this bean's bean dictionary.
209    *
210    * @param c The classes to set on this bean's bean dictionary.
211    * @return This object (for method chaining).
212    */
213   public BeanFilterBuilder setBeanDictionary(Class<?>...c) {
214      beanDictionary = new ArrayList<Class<?>>(Arrays.asList(c));
215      return this;
216   }
217
218   /**
219    * Adds classes to this bean's bean dictionary.
220    *
221    * @param c The classes to add to this bean's bean dictionary.
222    * @return This object (for method chaining).
223    */
224   public BeanFilterBuilder beanDictionary(Class<?>...c) {
225      if (beanDictionary == null)
226         beanDictionary = new ArrayList<Class<?>>(Arrays.asList(c));
227      else for (Class<?> cc : c)
228         beanDictionary.add(cc);
229      return this;
230   }
231
232   /**
233    * Creates a {@link BeanFilter} with settings in this builder class.
234    *
235    * @return A new {@link BeanFilter} instance.
236    */
237   public BeanFilter build() {
238      return new BeanFilter(this);
239   }
240}
