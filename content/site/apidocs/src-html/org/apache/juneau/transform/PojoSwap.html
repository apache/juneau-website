013package org.apache.juneau.transform;
014
015import java.util.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.internal.*;
019import org.apache.juneau.parser.*;
020import org.apache.juneau.serializer.*;
021
022/**
023 * Used to swap out non-serializable objects with serializable replacements during serialization, and vis-versa during parsing.
024 *
025 * <h5 class='section'>Description:</h5>
026 * <p>
027 *    <code>PojoSwaps</code> are used to extend the functionality of the serializers and parsers to be able to handle POJOs
028 *    that aren't automatically handled by the serializers or parsers.  For example, JSON does not have a standard
029 *    representation for rendering dates.  By defining a special {@code Date} swap and associating it with a serializer and
030 *    parser, you can convert a {@code Date} object to a {@code String} during serialization, and convert that {@code String} object back into
031 *    a {@code Date} object during parsing.
032 * <p>
033 *    Swaps MUST declare a public no-arg constructor so that the bean context can instantiate them.
034 * <p>
035 *    <code>PojoSwaps</code> are associated with instances of {@link BeanContext BeanContexts} by passing the swap class to
036 *    the {@link CoreApi#addPojoSwaps(Class...)} method.<br>
037 *    When associated with a bean context, fields of the specified type will automatically be converted when the
038 *    {@link BeanMap#get(Object)} or {@link BeanMap#put(String, Object)} methods are called.<br>
039 * <p>
040 *    <code>PojoSwaps</code> have two parameters:
041 *    <ol>
042 *       <li>{@code <T>} - The normal representation of an object.
043 *       <li>{@code <S>} - The swapped representation of an object.
044 *    </ol>
045 *    <br>
046 *    {@link Serializer Serializers} use swaps to convert objects of type T into objects of type S, and on calls to {@link BeanMap#get(Object)}.<br>
047 *    {@link Parser Parsers} use swaps to convert objects of type S into objects of type T, and on calls to {@link BeanMap#put(String,Object)}.
048 *
049 * <h6 class='topic'>Subtypes</h6>
050 * <p>
051 *    The following abstract subclasses are provided for common swap types:
052 *    <ol>
053 *       <li>{@link StringSwap} - Objects swapped with strings.
054 *       <li>{@link MapSwap} - Objects swapped with {@link ObjectMap ObjectMaps}.
055 *    </ol>
056 *
057 * <h6 class='topic'>Localization</h6>
058 * <p>
059 *    Swaps have access to the session locale and timezone through the {@link BeanSession#getLocale()} and {@link BeanSession#getTimeZone()}
060 *    methods.  This allows you to specify localized swap values when needed.
061 *    If using the REST server API, the locale and timezone are set based on the <code>Accept-Language</code> and <code>Time-Zone</code> headers
062 *    on the request.
063 *
064 * <h6 class='topic'>Swap Class Type {@code <S>}</h6>
065 * <p>
066 *    The swapped object representation of an object must be an object type that the serializers can
067 *    natively convert to JSON (or language-specific equivalent).  The list of valid transformed types are as follows...
068 *    <ul class='spaced-list'>
069 *       <li>{@link String}
070 *       <li>{@link Number}
071 *       <li>{@link Boolean}
072 *       <li>{@link Collection} containing anything on this list.
073 *       <li>{@link Map} containing anything on this list.
074 *       <li>A java bean with properties of anything on this list.
075 *       <li>An array of anything on this list.
076 *    </ul>
077 *
078 * <h6 class='topic'>Normal Class Type {@code <T>}</h6>
079 * <p>
080 *    The normal object representation of an object.<br>
081 *
082 * <h6 class='topic'>One-way vs. Two-way Serialization</h6>
083 * <p>
084 *    Note that while there is a unified interface for handling swaps during both serialization and parsing,
085 *    in many cases only one of the {@link #swap(BeanSession, Object)} or {@link #unswap(BeanSession, Object, ClassMeta)} methods will be defined
086 *    because the swap is one-way.  For example, a swap may be defined to convert an {@code Iterator} to a {@code ObjectList}, but
087 *    it's not possible to unswap an {@code Iterator}.  In that case, the {@code swap(Object}} method would
088 *    be implemented, but the {@code unswap(ObjectMap)} object would not, and the swap would be associated on
089 *    the serializer, but not the parser.  Also, you may choose to serialize objects like {@code Dates} to readable {@code Strings},
090 *    in which case it's not possible to reparse it back into a {@code Date}, since there is no way for the {@code Parser} to
091 *    know it's a {@code Date} from just the JSON or XML text.
092 *
093 * <h5 class='section'>Additional information:</h5>
094 *    See {@link org.apache.juneau.transform} for more information.
095 *
096 * @param <T> The normal form of the class.
097 * @param <S> The swapped form of the class.
098 */
099public abstract class PojoSwap<T,S> {
100
101   /**
102    * Represents a non-existent pojo swap.
103    */
104   @SuppressWarnings({ "unchecked", "rawtypes" })
105   public final static PojoSwap NULL = new PojoSwap(null, null) {};
106
107   private final Class<T> normalClass;
108   private final Class<?> swapClass;
109   private ClassMeta<?> swapClassMeta;
110
111   /**
112    * Constructor.
113    */
114   @SuppressWarnings("unchecked")
115   protected PojoSwap() {
116      normalClass = (Class<T>)ClassUtils.resolveParameterType(PojoSwap.class, 0, this.getClass());
117      swapClass = ClassUtils.resolveParameterType(PojoSwap.class, 1, this.getClass());
118   }
119
120   /**
121    * Constructor for when the normal and transformed classes are already known.
122    *
123    * @param normalClass The normal class (cannot be serialized).
124    * @param swapClass The transformed class (serializable).
125    */
126   protected PojoSwap(Class<T> normalClass, Class<?> swapClass) {
127      this.normalClass = normalClass;
128      this.swapClass = swapClass;
129   }
130
131   /**
132    * If this transform is to be used to serialize non-serializable POJOs, it must implement this method.
133    * <p>
134    *    The object must be converted into one of the following serializable types:
135    *    <ul class='spaced-list'>
136    *       <li>{@link String}
137    *       <li>{@link Number}
138    *       <li>{@link Boolean}
139    *       <li>{@link Collection} containing anything on this list.
140    *       <li>{@link Map} containing anything on this list.
141    *       <li>A java bean with properties of anything on this list.
142    *       <li>An array of anything on this list.
143    *    </ul>
144    * @param session The bean session to use to get the class meta.
145    *    This is always going to be the same bean context that created this swap.
146    * @param o The object to be transformed.
147    *
148    * @return The transformed object.
149    * @throws SerializeException If a problem occurred trying to convert the output.
150    */
151   public S swap(BeanSession session, T o) throws SerializeException {
152      throw new SerializeException("Swap method not implemented on PojoSwap ''{0}''", this.getClass().getName());
153   }
154
155   /**
156    * If this transform is to be used to reconstitute POJOs that aren't true Java beans, it must implement this method.
157    * @param session The bean session to use to get the class meta.
158    *    This is always going to be the same bean context that created this swap.
159    * @param f The transformed object.
160    * @param hint If possible, the parser will try to tell you the object type being created.  For example,
161    *    on a serialized date, this may tell you that the object being created must be of type {@code GregorianCalendar}.<br>
162    *    This may be <jk>null</jk> if the parser cannot make this determination.
163    *
164    * @return The narrowed object.
165    * @throws ParseException If this method is not implemented.
166    */
167   public T unswap(BeanSession session, S f, ClassMeta<?> hint) throws ParseException {
168      throw new ParseException("Unswap method not implemented on PojoSwap ''{0}''", this.getClass().getName());
169   }
170
171   /**
172    * Returns the T class, the normalized form of the class.
173    *
174    * @return The normal form of this class.
175    */
176   public Class<T> getNormalClass() {
177      return normalClass;
178   }
179
180   /**
181    * Returns the G class, the generialized form of the class.
182    * <p>
183    *    Subclasses must override this method if the generialized class is {@code Object},
184    *    meaning it can produce multiple generialized forms.
185    *
186    * @return The transformed form of this class.
187    */
188   public Class<?> getSwapClass() {
189      return swapClass;
190   }
191
192   /**
193    * Returns the {@link ClassMeta} of the transformed class type.
194    * This value is cached for quick lookup.
195    *
196    * @param beanContext The bean context to use to get the class meta.
197    *    This is always going to be the same bean context that created this swap.
198    * @return The {@link ClassMeta} of the transformed class type.
199    */
200   public ClassMeta<?> getSwapClassMeta(BeanContext beanContext) {
201      if (swapClassMeta == null)
202         swapClassMeta = beanContext.getClassMeta(swapClass);
203      return swapClassMeta;
204   }
205
206   /**
207    * Checks if the specified object is an instance of the normal class defined on this swap.
208    *
209    * @param o The object to check.
210    * @return <jk>true</jk> if the specified object is a subclass of the normal class defined on this transform.
211    *    <jk>null</jk> always return <jk>false</jk>.
212    */
213   public boolean isNormalObject(Object o) {
214      if (o == null)
215         return false;
216      return ClassUtils.isParentClass(normalClass, o.getClass());
217   }
218
219   /**
220    * Checks if the specified object is an instance of the swap class defined on this swap.
221    *
222    * @param o The object to check.
223    * @return <jk>true</jk> if the specified object is a subclass of the transformed class defined on this transform.
224    *    <jk>null</jk> always return <jk>false</jk>.
225    */
226   public boolean isSwappedObject(Object o) {
227      if (o == null)
228         return false;
229      return ClassUtils.isParentClass(swapClass, o.getClass());
230   }
231
232   //--------------------------------------------------------------------------------
233   // Overridden methods
234   //--------------------------------------------------------------------------------
235
236   @Override /* Object */
237   public String toString() {
238      return getClass().getSimpleName() + '<' + getNormalClass().getSimpleName() + "," + getSwapClass().getSimpleName() + '>';
239   }
240}
