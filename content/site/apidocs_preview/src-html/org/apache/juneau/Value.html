013package org.apache.juneau;
014
015/**
016 * Represents a simple settable value.
017 *
018 * <p>
019 * This object is not thread safe.
020 *
021 * @param <T> The value type.
022 */
023public class Value<T> {
024
025   private T t;
026   private ValueListener<T> listener;
027
028   /**
029    * Constructor.
030    */
031   public Value() {
032   }
033
034   /**
035    * Constructor.
036    *
037    * @param t Initial value.
038    */
039   public Value(T t) {
040      set(t);
041   }
042
043   /**
044    * Adds a listener for this value.
045    *
046    * @param listener The new listener for this value.
047    * @return This object (for method chaining).
048    */
049   public Value<T> listener(ValueListener<T> listener) {
050      this.listener = listener;
051      return this;
052   }
053
054   /**
055    * Sets the value.
056    *
057    * @param t The new value.
058    * @return This object (for method chaining).
059    */
060   public Value<T> set(T t) {
061      this.t = t;
062      if (listener != null)
063         listener.onSet(t);
064      return this;
065   }
066
067   /**
068    * Returns the value.
069    *
070    * @return The value, or <jk>null</jk> if it is not set.
071    */
072   public T get() {
073      return t;
074   }
075
076   /**
077    * Returns <jk>true</jk> if the value is set.
078    *
079    * @return <jk>true</jk> if the value is set.
080    */
081   public boolean isSet() {
082      return get() != null;
083   }
084}
