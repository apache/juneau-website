013package org.apache.juneau.httppart.bean;
014
015import static org.apache.juneau.internal.ClassFlags.*;
016import static org.apache.juneau.internal.ClassUtils.*;
017
018import java.lang.reflect.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.annotation.*;
023import org.apache.juneau.http.annotation.*;
024import org.apache.juneau.httppart.*;
025import org.apache.juneau.internal.*;
026
027/**
028 * Represents the metadata gathered from a parameter or class annotated with {@link Response}.
029 */
030public class ResponseBeanMeta {
031
032   /**
033    * Represents a non-existent meta object.
034    */
035   public static ResponseBeanMeta NULL = new ResponseBeanMeta(new Builder(PropertyStore.DEFAULT));
036
037   /**
038    * Create metadata from specified class.
039    *
040    * @param c The class annotated with {@link Response}.
041    * @param ps
042    *    Configuration information used to instantiate part serializers and part parsers.
043    *    <br>Can be <jk>null</jk>.
044    * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Response}.
045    */
046   public static ResponseBeanMeta create(Class<?> c, PropertyStore ps) {
047      if (! hasAnnotation(Response.class, c))
048         return null;
049      return new ResponseBeanMeta.Builder(ps).apply(c).build();
050   }
051
052   //-----------------------------------------------------------------------------------------------------------------
053   // Instance
054   //-----------------------------------------------------------------------------------------------------------------
055
056   private final ClassMeta<?> cm;
057   private final int code;
058   private final Map<String,ResponseBeanPropertyMeta> headerMethods;
059   private final Method statusMethod;
060   private final ResponseBeanPropertyMeta bodyMethod;
061   private final HttpPartSerializer partSerializer;
062   private final HttpPartSchema schema;
063   private final boolean usePartSerializer;
064
065   ResponseBeanMeta(Builder b) {
066      this.cm = b.cm;
067      this.code = b.code;
068      this.partSerializer = ClassUtils.newInstance(HttpPartSerializer.class, b.partSerializer, true, b.ps);
069      this.schema = b.schema.build();
070      this.usePartSerializer = b.usePartSerializer || partSerializer != null;
071
072      Map<String,ResponseBeanPropertyMeta> hm = new LinkedHashMap<>();
073      for (Map.Entry<String,ResponseBeanPropertyMeta.Builder> e : b.headerMethods.entrySet()) {
074         ResponseBeanPropertyMeta pm = e.getValue().build(partSerializer);
075         hm.put(e.getKey(), pm);
076
077      }
078      this.headerMethods = Collections.unmodifiableMap(hm);
079
080      this.bodyMethod = b.bodyMethod == null ? null : b.bodyMethod.build(partSerializer);
081      this.statusMethod = b.statusMethod;
082   }
083
084   static class Builder {
085      ClassMeta<?> cm;
086      int code;
087      PropertyStore ps;
088      boolean usePartSerializer;
089      Class<? extends HttpPartSerializer> partSerializer;
090      HttpPartSchemaBuilder schema = HttpPartSchema.create();
091
092      Map<String,ResponseBeanPropertyMeta.Builder> headerMethods = new LinkedHashMap<>();
093      ResponseBeanPropertyMeta.Builder bodyMethod;
094      Method statusMethod;
095
096      Builder(PropertyStore ps) {
097         this.ps = ps;
098      }
099
100      Builder apply(Class<?> c) {
101         apply(getAnnotation(Response.class, c));
102         this.cm = BeanContext.DEFAULT.getClassMeta(c);
103         for (Method m : ClassUtils.getAllMethods(c, false)) {
104            if (isAll(m, PUBLIC, HAS_NO_ARGS)) {
105               if (hasAnnotation(ResponseHeader.class, m)) {
106                  if (m.getParameterTypes().length != 0)
107                     throw new InvalidAnnotationException("@ResponseHeader annotation on method cannot have arguments.  Method=''{0}''", m);
108                  Class<?> rt = m.getReturnType();
109                  if (rt == void.class)
110                     throw new InvalidAnnotationException("Invalid return type for @ResponseHeader annotation on method.  Method=''{0}''", m);
111                  ResponseHeader a = getAnnotation(ResponseHeader.class, m);
112                  String n = a.name();
113                  if (n.isEmpty())
114                     n = m.getName();
115                  HttpPartSchemaBuilder s = HttpPartSchema.create().apply(a);
116                  headerMethods.put(n, ResponseBeanPropertyMeta.create().name(n).partType(HttpPartType.HEADER).apply(s).getter(m));
117               }
118               if (hasAnnotation(ResponseStatus.class, m)) {
119                  if (m.getParameterTypes().length != 0)
120                     throw new InvalidAnnotationException("@ResponseStatus annotation on method cannot have arguments.  Method=''{0}''", m);
121                  Class<?> rt = m.getReturnType();
122                  if (rt != int.class || rt != Integer.class)
123                     throw new InvalidAnnotationException("Invalid return type for @ResponseStatus annotation on method.  Method=''{0}''", m);
124                  statusMethod = m;
125               }
126               if (hasAnnotation(ResponseBody.class, m)) {
127                  if (m.getParameterTypes().length != 0)
128                     throw new InvalidAnnotationException("@ResponseBody annotation on method cannot have arguments.  Method=''{0}''", m);
129                  Class<?> rt = m.getReturnType();
130                  if (rt == void.class)
131                     throw new InvalidAnnotationException("Invalid return type for @ResponseBody annotation on method.  Method=''{0}''", m);
132                  ResponseBody a = getAnnotation(ResponseBody.class, m);
133                  HttpPartSchemaBuilder s = HttpPartSchema.create().apply(a);
134                  bodyMethod = ResponseBeanPropertyMeta.create().partType(HttpPartType.BODY).apply(s).getter(m);
135               }
136               if (hasAnnotation(Body.class, m))
137                  throw new InvalidAnnotationException("@Body annotation cannot be used in a @Response bean.  Use @ResponseBody instead.  Method=''{0}''", m);
138               if (hasAnnotation(Header.class, m))
139                  throw new InvalidAnnotationException("@Header annotation cannot be used in a @Response bean.  Use @ResponseHeader instead.  Method=''{0}''", m);
140            }
141         }
142         return this;
143      }
144
145      Builder apply(Response rb) {
146         if (rb != null) {
147            if (rb.partSerializer() != HttpPartSerializer.Null.class)
148               partSerializer = rb.partSerializer();
149            if (rb.usePartSerializer())
150               usePartSerializer = true;
151            if (rb.value().length > 0)
152               code = rb.value()[0];
153            if (rb.code().length > 0)
154               code = rb.code()[0];
155            schema.apply(rb.schema());
156         }
157         return this;
158      }
159
160      ResponseBeanMeta build() {
161         return new ResponseBeanMeta(this);
162      }
163   }
164
165   /**
166    * Returns the HTTP status code.
167    *
168    * @return The HTTP status code.
169    */
170   public int getCode() {
171      return code;
172   }
173
174   /**
175    * Returns the schema information about the response object.
176    *
177    * @return The schema information about the response object.
178    */
179   public HttpPartSchema getSchema() {
180      return schema;
181   }
182
183   /**
184    * Returns metadata about the <ja>@ResponseHeader</ja>-annotated methods.
185    *
186    * @return Metadata about the <ja>@ResponseHeader</ja>-annotated methods, or an empty collection if none exist.
187    */
188   public Collection<ResponseBeanPropertyMeta> getHeaderMethods() {
189      return headerMethods.values();
190   }
191
192   /**
193    * Returns metadata about the <ja>@ResponseBody</ja>-annotated method.
194    *
195    * @return Metadata about the <ja>@ResponseBody</ja>-annotated method, or <jk>null</jk> if it doesn't exist.
196    */
197   public ResponseBeanPropertyMeta getBodyMethod() {
198      return bodyMethod;
199   }
200
201   /**
202    * Returns metadata about the <ja>@ResponseBody</ja>-annotated method.
203    *
204    * @return Metadata about the <ja>@ResponseBody</ja>-annotated method, or <jk>null</jk> if it doesn't exist.
205    */
206   public Method getStatusMethod() {
207      return statusMethod;
208   }
209
210   /**
211    * Returns the flag for whether the part serializer should be used to serialize this response.
212    *
213    * @return The flag for whether the part serializer should be used to serialize this response.
214    */
215   public boolean isUsePartSerializer() {
216      return usePartSerializer;
217   }
218
219   /**
220    * Returns the part serializer to use to serialize this response.
221    *
222    * @return The part serializer to use to serialize this response.
223    */
224   public HttpPartSerializer getPartSerializer() {
225      return partSerializer;
226   }
227
228   /**
229    * Returns metadata about the class.
230    *
231    * @return Metadata about the class.
232    */
233   public ClassMeta<?> getClassMeta() {
234      return cm;
235   }
