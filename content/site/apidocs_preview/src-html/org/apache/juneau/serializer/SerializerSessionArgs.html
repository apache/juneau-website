013package org.apache.juneau.serializer;
014
015import java.lang.reflect.*;
016import java.util.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.http.*;
020
021/**
022 * Runtime arguments common to all serializer sessions.
023 *
024 * <p>
025 * This object specifies information such as session locale or URI context.
026 */
027public final class SerializerSessionArgs extends BeanSessionArgs {
028
029   /**
030    * Default serializer session args.
031    */
032   public static final SerializerSessionArgs DEFAULT = new SerializerSessionArgs();
033
034   Method javaMethod;
035   UriContext uriContext;
036   Boolean useWhitespace;
037
038   /**
039    * Constructor
040    */
041   public SerializerSessionArgs() {}
042
043   /**
044    * Constructor.
045    *
046    * @param properties
047    *    Session-level properties.
048    *    <br>These override context-level properties.
049    *    <br>Can be <jk>null</jk>.
050    * @param javaMethod
051    *    The java method that called this serializer, usually the method in a REST servlet.
052    *    <br>Can be <jk>null</jk>.
053    * @param locale
054    *    The session locale.
055    *    <br>If <jk>null</jk>, then the locale defined on the context is used.
056    * @param timeZone
057    *    The session timezone.
058    *    <br>If <jk>null</jk>, then the timezone defined on the context is used.
059    * @param mediaType
060    *    The session media type (e.g. <js>"application/json"</js>).
061    *    <br>Can be <jk>null</jk>.
062    * @param debug
063    *    Enable debug mode for this session.
064    *    <br>Can be <jk>null</jk> to use the debug setting on the bean context..
065    * @param uriContext
066    *    The URI context.
067    *    <br>Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
068    * @param useWhitespace
069    *    Override the use-whitespace flag on the serializer.
070    */
071   public SerializerSessionArgs(ObjectMap properties, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType, Boolean debug, UriContext uriContext, Boolean    useWhitespace) {
072      super(properties, locale, timeZone, mediaType, debug);
073      this.javaMethod = javaMethod;
074      this.uriContext = uriContext;
075      this.useWhitespace = useWhitespace;
076   }
077
078   /**
079    * The java method that called this serializer, usually the method in a REST servlet.
080    *
081    * @param javaMethod
082    *    The java method that called this serializer, usually the method in a REST servlet.
083    *    <br>Can be <jk>null</jk>.
084    * @return This object (for method chaining).
085    */
086   public SerializerSessionArgs javaMethod(Method javaMethod) {
087      this.javaMethod = javaMethod;
088      return this;
089   }
090
091   /**
092    * The URI context.
093    *
094    * @param uriContext
095    *    The URI context.
096    *    <br>Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
097    * @return This object (for method chaining).
098    */
099   public SerializerSessionArgs uriContext(UriContext uriContext) {
100      this.uriContext = uriContext;
101      return this;
102   }
103
104   /**
105    * Use-whitespace flag
106    *
107    * @param useWhitespace
108    *    The use-whitespace flag.
109    *    <br>Overrides the use-whitespace flag on the serializer.
110    * @return This object (for method chaining).
111    */
112   public SerializerSessionArgs useWhitespace(Boolean useWhitespace) {
113      this.useWhitespace = useWhitespace;
114      return this;
115   }
116
117   @Override /* BeanSessionArgs */
118   public SerializerSessionArgs locale(Locale locale) {
119      super.locale(locale);
120      return this;
121   }
122
123   @Override /* BeanSessionArgs */
124   public SerializerSessionArgs timeZone(TimeZone timeZone) {
125      super.timeZone(timeZone);
126      return this;
127   }
128
129   @Override /* BeanSessionArgs */
130   public SerializerSessionArgs mediaType(MediaType mediaType) {
131      super.mediaType(mediaType);
132      return this;
133   }
134
135   @Override /* BeanSessionArgs */
136   public SerializerSessionArgs debug(Boolean debug) {
137      super.debug(debug);
138      return this;
139   }
140
141   @Override /* SessionArgs */
142   public SerializerSessionArgs properties(ObjectMap properties) {
143      super.properties(properties);
144      return this;
145   }
146}
