013package org.apache.juneau.cp;
014
015import static org.apache.juneau.assertions.Assertions.*;
016import java.util.*;
017import java.util.function.*;
018
019import org.apache.juneau.*;
020import org.apache.juneau.annotation.*;
021import org.apache.juneau.reflect.*;
022
023/**
024 * Utility class for creating beans through creator methods.
025 *
026 * <p>
027 * Used for finding and invoking methods on an object that take in arbitrary parameters and returns bean instances.
028 *
029 * <p>
030 * This class is instantiated through the following methods:
031 * <ul class='javatree'>
032 *    <li class='jc'>{@link BeanStore}
033 *       <ul class='javatreec'>
034 *          <li class='jm'>{@link BeanStore#createMethodFinder(Class)}
035 *          <li class='jm'>{@link BeanStore#createMethodFinder(Class,Class)}
036 *          <li class='jm'>{@link BeanStore#createMethodFinder(Class,Object)}
037 *       </ul>
038 *    </li>
039 * </ul>
040 *
041 * <h5 class='section'>Example:</h5>
042 * <p class='bjava'>
043 *    <jc>// The bean we want to create.</jc>
044 *    <jk>public class</jk> A {}
045 *
046 *    <jc>// The bean that has a creator method for the bean above.</jc>
047 *    <jk>public class</jk> B {
048 *
049 *       <jc>// Creator method.</jc>
050 *       <jc>// Bean store must have a C bean and optionally a D bean.</jc>
051 *       <jk>public</jk> A createA(C <jv>c</jv>, Optional&lt;D&gt; <jv>d</jv>) {
052 *          <jk>return new</jk> A(<jv>c</jv>, <jv>d</jv>.orElse(<jk>null</jk>));
053 *       }
054 *    }
055 *
056 *    <jc>// Instantiate the bean with the creator method.</jc>
057 *    B <mv>b</mv> = <jk>new</jk> B();
058 *
059 *  <jc>// Create a bean store with some mapped beans.</jc>
060 *    BeanStore <mv>beanStore</mv> = BeanStore.<jsm>create</jsm>().addBean(C.<jk>class</jk>, <jk>new</jk> C());
061 *
062 *    <jc>// Instantiate the bean using the creator method.</jc>
063 *    A <mv>a</mv> = <mv>beanStore</mv>
064 *       .createMethodFinder(A.<jk>class</jk>, <mv>b</mv>)  <jc>// Looking for creator for A on b object.</jc>
065 *       .find(<js>"createA"</js>)                         <jc>// Look for method called "createA".</jc>
066 *       .thenFind(<js>"createA2"</js>)                    <jc>// Then look for method called "createA2".</jc>
067 *       .withDefault(()-&gt;<jk>new</jk> A())                        <jc>// Optionally supply a default value if method not found.</jc>
068 *       .run();                                  <jc>// Execute.</jc>
069 * </p>
070 *
071 * <ul class='seealso'>
072 *    <li class='jc'>{@link BeanStore}
073 *    <li class='extlink'>{@source}
074 * </ul>
075 *
076 * @param <T> The bean type being created.
077 */
078public class BeanCreateMethodFinder<T> {
079
080   private Class<T> beanType;
081   private final Class<?> resourceClass;
082   private final Object resource;
083   private final BeanStore beanStore;
084
085   private MethodInfo method;
086   private Object[] args;
087
088   private Supplier<T> def = ()->null;
089
090   BeanCreateMethodFinder(Class<T> beanType, Object resource, BeanStore beanStore) {
091      this.beanType = assertArgNotNull("beanType", beanType);
092      this.resource = assertArgNotNull("resource", resource);
093      this.resourceClass = resource.getClass();
094      this.beanStore = BeanStore.of(beanStore, resource);
095   }
096
097   BeanCreateMethodFinder(Class<T> beanType, Class<?> resourceClass, BeanStore beanStore) {
098      this.beanType = assertArgNotNull("beanType", beanType);
099      this.resource = null;
100      this.resourceClass = assertArgNotNull("resourceClass", resourceClass);
101      this.beanStore = BeanStore.of(beanStore);
102   }
103
104   /**
105    * Adds a bean to the lookup for parameters.
106    *
107    * @param <T2> The bean type.
108    * @param c The bean type.
109    * @param t The bean.
110    * @return This object.
111    */
112   public <T2> BeanCreateMethodFinder<T> addBean(Class<T2> c, T2 t) {
113      beanStore.addBean(c, t);
114      return this;
115   }
116
117   /**
118    * Find the method matching the specified name and optionally having the specified required parameters present.
119    *
120    * <p>
121    * In order for the method to be used, it must adhere to the following restrictions:
122    * <ul>
123    *    <li>The method must be public.
124    *    <li>The method can be static.
125    *    <li>The method name must match exactly.
126    *    <li>The method must not be deprecated or annotated with {@link BeanIgnore}.
127    *    <li>The method must have all parameter types specified in <c>requiredParams</c>.
128    *    <li>The bean store must contain beans for all parameter types.
129    *    <li>The bean store may contain beans for all {@link Optional} parameter types.
130    * </ul>
131    *
132    * <p>
133    * This method can be called multiple times with different method names or required parameters until a match is found.
134    * <br>Once a method is found, subsequent calls to this method will be no-ops.
135    *
136    * See {@link BeanStore#createMethodFinder(Class, Object)} for usage.
137    *
138    * @param methodName The method name.
139    * @param requiredParams Optional required parameters.
140    * @return This object.
141    */
142   public BeanCreateMethodFinder<T> find(String methodName, Class<?>...requiredParams) {
143      if (method == null) {
144         method = ClassInfo.of(resourceClass).getPublicMethod(
145            x -> x.isNotDeprecated()
146            && x.hasReturnType(beanType)
147            && x.hasName(methodName)
148            && x.hasNoAnnotation(BeanIgnore.class)
149            && x.hasAllArgs(requiredParams)
150            && beanStore.hasAllParams(x)
151            && (x.isStatic() || resource != null)
152         );
153         if (method != null)
154            args = beanStore.getParams(method);
155      }
156      return this;
157   }
158
159   /**
160    * Identical to {@link #find(String, Class...)} but named for fluent-style calls.
161    *
162    * @param methodName The method name.
163    * @param requiredParams Optional required parameters.
164    * @return This object.
165    */
166   public BeanCreateMethodFinder<T> thenFind(String methodName, Class<?>...requiredParams) {
167      return find(methodName, requiredParams);
168   }
169
170   /**
171    * A default value to return if no matching methods were found.
172    *
173    * @param def The default value.  Can be <jk>null</jk>.
174    * @return This object.
175    */
176   public BeanCreateMethodFinder<T> withDefault(T def) {
177      return withDefault(()->def);
178   }
179
180   /**
181    * A default value to return if no matching methods were found.
182    *
183    * @param def The default value.
184    * @return This object.
185    */
186   public BeanCreateMethodFinder<T> withDefault(Supplier<T> def) {
187      assertArgNotNull("def", def);
188      this.def = def;
189      return this;
190   }
191
192   /**
193    * Executes the matched method and returns the result.
194    *
195    * @return The object returned by the method invocation, or the default value if method was not found.
196    * @throws ExecutableException If method invocation threw an exception.
197    */
198   @SuppressWarnings("unchecked")
199   public T run() throws ExecutableException {
200      if (method != null)
201         return (T)method.invoke(resource, args);
202      return def.get();
203   }
204
205   /**
206    * Same as {@link #run()} but also executes a consumer if the returned value was not <jk>null</jk>.
207    *
208    * @param consumer The consumer of the response.
209    * @return The object returned by the method invocation, or the default value if method was not found, or {@link Optional#empty()}.
210    * @throws ExecutableException If method invocation threw an exception.
211    */
212   public T run(Consumer<? super T> consumer) throws ExecutableException {
213      T t = run();
214      if (t != null)
215         consumer.accept(t);
216      return t;
217   }
218}
