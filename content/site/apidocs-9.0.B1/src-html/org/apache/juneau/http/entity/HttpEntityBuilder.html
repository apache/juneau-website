013package org.apache.juneau.http.entity;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016
017import java.io.*;
018import java.nio.charset.*;
019import java.util.function.*;
020
021import org.apache.http.*;
022import org.apache.juneau.http.header.*;
023import org.apache.juneau.internal.*;
024
025/**
026 * Builder for {@link HttpEntity} beans.
027 *
028 * <ul class='seealso'>
029 *    <li class='link'>{@doc jm.HttpParts}
030 *    <li class='extlink'>{@source}
031 * </ul>
032 *
033 * @param <T> The bean type to create for this builder.
034 */
035@FluentSetters(returns="HttpEntityBuilder<T>")
036public class HttpEntityBuilder<T extends BasicHttpEntity> {
037
038   boolean cached, chunked;
039   Object content;
040   Supplier<?> contentSupplier;
041   ContentType contentType;
042   ContentEncoding contentEncoding;
043   Charset charset;
044   long contentLength = -1;
045
046   /** The HttpEntity implementation class. */
047   protected final Class<? extends BasicHttpEntity> implClass;
048
049   /**
050    * Constructor.
051    *
052    * @param implClass
053    *    The subclass of {@link HttpResponse} to create.
054    *    <br>This must contain a public constructor that takes in an {@link HttpEntityBuilder} object.
055    */
056   public HttpEntityBuilder(Class<T> implClass) {
057      this.implClass = implClass;
058   }
059
060   /**
061    * Copy constructor.
062    *
063    * @param impl
064    *    The implementation object of {@link HttpEntity} to copy from.
065    *    <br>This must contain a public constructor that takes in an {@link HttpEntityBuilder} object.
066    */
067   public HttpEntityBuilder(T impl) {
068      implClass = impl.getClass();
069      cached = impl.cached;
070      content = impl.content;
071      contentSupplier = impl.contentSupplier;
072      contentType = impl.contentType;
073      contentEncoding = impl.contentEncoding;
074      charset = impl.charset;
075      contentLength = impl.contentLength;
076   }
077
078   /**
079    * Instantiates the entity bean from the settings in this builder.
080    *
081    * @return A new {@link HttpEntity} bean.
082    */
083   @SuppressWarnings("unchecked")
084   public T build() {
085      try {
086         return (T) implClass.getConstructor(HttpEntityBuilder.class).newInstance(this);
087      } catch (Exception e) {
088         throw runtimeException(e);
089      }
090   }
091
092   /**
093    * Sets the content on this entity bean.
094    *
095    * @param value The entity content, can be <jk>null</jk>.
096    * @return This object.
097    */
098   @FluentSetter
099   public HttpEntityBuilder<T> content(Object value) {
100      this.content = value;
101      return this;
102   }
103
104   /**
105    * Sets the content on this entity bean from a supplier.
106    *
107    * <p>
108    * Repeatable entities such as {@link StringEntity} use this to allow the entity content to be resolved at
109    * serialization time.
110    *
111    * @param value The entity content, can be <jk>null</jk>.
112    * @return This object.
113    */
114   @FluentSetter
115   public HttpEntityBuilder<T> content(Supplier<?> value) {
116      this.contentSupplier = value == null ? ()->null : value;
117      return this;
118   }
119
120   /**
121    * Sets the content type on this entity bean.
122    *
123    * @param value The new <c>Content-Type</c> header, or <jk>null</jk> to unset.
124    * @return This object.
125    */
126   @FluentSetter
127   public HttpEntityBuilder<T> contentType(String value) {
128      return contentType(ContentType.of(value));
129   }
130
131   /**
132    * Sets the content type on this entity bean.
133    *
134    * @param value The new <c>Content-Type</c> header, or <jk>null</jk> to unset.
135    * @return This object.
136    */
137   @FluentSetter
138   public HttpEntityBuilder<T> contentType(ContentType value) {
139      contentType = value;
140      return this;
141   }
142
143   /**
144    * Sets the content length on this entity bean.
145    *
146    * @param value The new <c>Content-Length</c> header value, or <c>-1</c> to unset.
147    * @return This object.
148    */
149   @FluentSetter
150   public HttpEntityBuilder<T> contentLength(long value) {
151      contentLength = value;
152      return this;
153   }
154
155   /**
156    * Sets the content encoding header on this entity bean.
157    *
158    * @param value The new <c>Content-Encoding</c> header, or <jk>null</jk> to unset.
159    * @return This object.
160    */
161   @FluentSetter
162   public HttpEntityBuilder<T> contentEncoding(String value) {
163      return contentEncoding(ContentEncoding.of(value));
164   }
165
166   /**
167    * Sets the content encoding header on this entity bean.
168    *
169    * @param value The new <c>Content-Encoding</c> header, or <jk>null</jk> to unset.
170    * @return This object.
171    */
172   @FluentSetter
173   public HttpEntityBuilder<T> contentEncoding(ContentEncoding value) {
174      contentEncoding = value;
175      return this;
176   }
177
178   /**
179    * Sets the 'chunked' flag value to <jk>true</jk>.
180    *
181    * <ul class='notes'>
182    *    <li class='note'>If the {@link HttpEntity#getContentLength()} method returns a negative value, the HttpClient code will always
183    *       use chunked encoding.
184    * </ul>
185    *
186    * @return This object.
187    */
188   @FluentSetter
189   public HttpEntityBuilder<T> chunked() {
190      return chunked(true);
191   }
192
193   /**
194    * Sets the 'chunked' flag value.
195    *
196    * <ul class='notes'>
197    *    <li class='note'>If the {@link HttpEntity#getContentLength()} method returns a negative value, the HttpClient code will always
198    *       use chunked encoding.
199    * </ul>
200    *
201    * @param value The new value for this flag.
202    * @return This object.
203    */
204   @FluentSetter
205   public HttpEntityBuilder<T> chunked(boolean value) {
206      chunked = value;
207      return this;
208   }
209
210   /**
211    * Specifies that the contents of this resource should be cached into an internal byte array so that it can
212    * be read multiple times.
213    *
214    * @return This object.
215    * @throws IOException If entity could not be read into memory.
216    */
217   @FluentSetter
218   public HttpEntityBuilder<T> cached() throws IOException {
219      cached = true;
220      return this;
221   }
222
223   // <FluentSetters>
224
225   // </FluentSetters>
226}
