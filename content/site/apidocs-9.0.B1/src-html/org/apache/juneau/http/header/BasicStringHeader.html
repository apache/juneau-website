013package org.apache.juneau.http.header;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.util.*;
019import java.util.function.*;
020
021import org.apache.http.*;
022import org.apache.juneau.assertions.*;
023
024/**
025 * Category of headers that consist of a single string value.
026 *
027 * <p>
028 * <h5 class='figure'>Example</h5>
029 * <p class='bcode'>
030 *    Host: www.myhost.com:8080
031 * </p>
032 *
033 * <ul class='seealso'>
034 *    <li class='link'>{@doc jm.HttpParts}
035 *    <li class='extlink'>{@doc ext.RFC2616}
036 *    <li class='extlink'>{@source}
037 * </ul>
038 *
039 * @serial exclude
040 */
041public class BasicStringHeader extends BasicHeader {
042
043   //-----------------------------------------------------------------------------------------------------------------
044   // Static
045   //-----------------------------------------------------------------------------------------------------------------
046
047   private static final long serialVersionUID = 1L;
048
049   /**
050    * Static creator.
051    *
052    * @param name The header name.
053    * @param value
054    *    The header value.
055    *    <br>Can be <jk>null</jk>.
056    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
057    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
058    */
059   public static BasicStringHeader of(String name, String value) {
060      return value == null ? null : new BasicStringHeader(name, value);
061   }
062
063   /**
064    * Static creator with delayed value.
065    *
066    * <p>
067    * Header value is re-evaluated on each call to {@link #getValue()}.
068    *
069    * @param name The header name.
070    * @param value
071    *    The supplier of the header value.
072    *    <br>Can be <jk>null</jk>.
073    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
074    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
075    */
076   public static BasicStringHeader of(String name, Supplier<String> value) {
077      return value == null ? null : new BasicStringHeader(name, value);
078   }
079
080   /**
081    * Creates a {@link Header} from a name/value pair string (e.g. <js>"Foo: bar"</js>)
082    *
083    * @param pair The pair string.
084    * @return A new header bean.
085    */
086   public static BasicStringHeader ofPair(String pair) {
087      if (pair == null)
088         return null;
089      int i = pair.indexOf(':');
090      if (i == -1)
091         i = pair.indexOf('=');
092      if (i == -1)
093         return of(pair, "");
094      return of(pair.substring(0,i).trim(), pair.substring(i+1).trim());
095   }
096
097   //-----------------------------------------------------------------------------------------------------------------
098   // Instance
099   //-----------------------------------------------------------------------------------------------------------------
100
101   private final String value;
102   private final Supplier<String> supplier;
103
104   /**
105    * Constructor.
106    *
107    * @param name The header name.
108    * @param value
109    *    The header value.
110    *    <br>Can be <jk>null</jk>.
111    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
112    */
113   public BasicStringHeader(String name, String value) {
114      super(name, value);
115      this.value = value;
116      this.supplier = null;
117   }
118
119   /**
120    * Constructor with delayed value.
121    *
122    * <p>
123    * Header value is re-evaluated on each call to {@link #getValue()}.
124    *
125    * @param name The header name.
126    * @param value
127    *    The supplier of the header value.
128    *    <br>Can be <jk>null</jk>.
129    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
130    */
131   public BasicStringHeader(String name, Supplier<String> value) {
132      super(name, null);
133      this.value = null;
134      this.supplier = value;
135   }
136
137   @Override /* Header */
138   public String getValue() {
139      return stringify(value());
140   }
141
142   @Override /* BasicHeader */
143   public Optional<String> asString() {
144      return optional(value());
145   }
146
147   /**
148    * Provides the ability to perform fluent-style assertions on this header.
149    *
150    * <h5 class='section'>Examples:</h5>
151    * <p class='bjava'>
152    *    <jc>// Validates the content type header is provided.</jc>
153    *    <jv>client</jv>
154    *       .get(<jsf>URL</jsf>)
155    *       .run()
156    *       .getHeader(<js>"Content-Type"</js>).assertString().exists();
157    * </p>
158    *
159    * @return A new fluent assertion object.
160    * @throws AssertionError If assertion failed.
161    */
162   public FluentStringAssertion<BasicStringHeader> assertString() {
163      return new FluentStringAssertion<>(value(), this);
164   }
165
166
167   /**
168    * Return the value if present, otherwise return <c>other</c>.
169    *
170    * <p>
171    * This is a shortened form for calling <c>asString().orElse(<jv>other</jv>)</c>.
172    *
173    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
174    * @return The value, if present, otherwise <c>other</c>.
175    */
176   @Override
177   public String orElse(String other) {
178      String x = value();
179      return x != null ? x : other;
180   }
181
182   private String value() {
183      if (supplier != null)
184         return supplier.get();
185      return value;
186   }
187}
