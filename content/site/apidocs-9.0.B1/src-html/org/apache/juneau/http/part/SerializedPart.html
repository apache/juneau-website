013package org.apache.juneau.http.part;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.util.function.*;
019
020import org.apache.http.*;
021import org.apache.juneau.http.header.*;
022import org.apache.juneau.httppart.*;
023import org.apache.juneau.oapi.*;
024import org.apache.juneau.serializer.*;
025import org.apache.juneau.urlencoding.*;
026
027/**
028 * Subclass of {@link NameValuePair} for serializing POJOs as URL-encoded form post entries using the
029 * {@link UrlEncodingSerializer class}.
030 *
031 * <h5 class='section'>Example:</h5>
032 * <p class='bjava'>
033 *    NameValuePairs <jv>params</jv> = <jk>new</jk> NameValuePairs()
034 *       .append(<jk>new</jk> SerializedNameValuePair(<js>"myPojo"</js>, <jv>pojo</jv>, UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>))
035 *       .append(<jk>new</jk> BasicNameValuePair(<js>"someOtherParam"</js>, <js>"foobar"</js>));
036 *    <jv>request</jv>.setEntity(<jk>new</jk> UrlEncodedFormEntity(<jv>params</jv>));
037 * </p>
038 *
039 * <ul class='seealso'>
040 *    <li class='link'>{@doc jm.HttpParts}
041 *    <li class='extlink'>{@source}
042 * </ul>
043 */
044public class SerializedPart extends BasicPart implements Headerable {
045   private final Object value;
046   private HttpPartType type;
047   private HttpPartSerializerSession serializer;
048   private HttpPartSchema schema = HttpPartSchema.DEFAULT;
049   private boolean skipIfEmpty;
050
051   /**
052    * Instantiates a new instance of this object.
053    *
054    * @param name The part name.
055    * @param value
056    *    The part value.
057    *    <br>Can be any POJO.
058    * @return A new {@link SerializedPart} object, never <jk>null</jk>.
059    */
060   public static SerializedPart of(String name, Object value) {
061      return new SerializedPart(name, value, null, null, null, false);
062   }
063
064   /**
065    * Instantiates a new instance of this object.
066    *
067    * @param name The part name.
068    * @param value
069    *    The part value supplier.
070    *    <br>Can be a supplier of any POJO.
071    * @return A new {@link SerializedPart} object, never <jk>null</jk>.
072    */
073   public static SerializedPart of(String name, Supplier<?> value) {
074      return new SerializedPart(name, value, null, null, null, false);
075   }
076
077   /**
078    * Constructor.
079    *
080    * @param name The part name.
081    * @param value The POJO to serialize to The part value.
082    * @param type The HTTP part type.
083    * @param serializer
084    *    The serializer to use for serializing the value to a string value.
085    * @param schema
086    *    The schema object that defines the format of the output.
087    *    <br>If <jk>null</jk>, defaults to the schema defined on the serializer.
088    *    <br>If that's also <jk>null</jk>, defaults to {@link HttpPartSchema#DEFAULT}.
089    *    <br>Only used if serializer is schema-aware (e.g. {@link OpenApiSerializer}).
090    * @param skipIfEmpty If value is a blank string, the value should return as <jk>null</jk>.
091    */
092   public SerializedPart(String name, Object value, HttpPartType type, HttpPartSerializerSession serializer, HttpPartSchema schema, boolean skipIfEmpty) {
093      super(name, value);
094      this.value = value;
095      this.type = type;
096      this.serializer = serializer;
097      this.schema = schema;
098      this.skipIfEmpty = skipIfEmpty;
099   }
100
101   /**
102    * Copy constructor.
103    *
104    * @param copyFrom The object to copy.
105    */
106   protected SerializedPart(SerializedPart copyFrom) {
107      super(copyFrom);
108      this.value = copyFrom.value;
109      this.type = copyFrom.type;
110      this.serializer = copyFrom.serializer == null ? serializer : copyFrom.serializer;
111      this.schema = copyFrom.schema == null ? schema : copyFrom.schema;
112      this.skipIfEmpty = copyFrom.skipIfEmpty;
113   }
114
115   /**
116    * Creates a copy of this object.
117    *
118    * @return A new copy of this object.
119    */
120   public SerializedPart copy() {
121      return new SerializedPart(this);
122   }
123
124   /**
125    * Sets the HTTP part type.
126    *
127    * @param value The new value for this property.
128    * @return This object.
129    */
130   public SerializedPart type(HttpPartType value) {
131      type = value;
132      return this;
133   }
134
135   /**
136    * Sets the serializer to use for serializing the value to a string value.
137    *
138    * @param value The new value for this property.
139    * @return This object.
140    */
141   public SerializedPart serializer(HttpPartSerializer value) {
142      if (value != null)
143         return serializer(value.getPartSession());
144      return this;
145   }
146
147   /**
148    * Sets the serializer to use for serializing the value to a string value.
149    *
150    * @param value The new value for this property.
151    * @return This object.
152    */
153   public SerializedPart serializer(HttpPartSerializerSession value) {
154      serializer = value;
155      return this;
156   }
157
158   /**
159    * Sets the schema object that defines the format of the output.
160    *
161    * @param value The new value for this property.
162    * @return This object.
163    */
164   public SerializedPart schema(HttpPartSchema value) {
165      this.schema = value;
166      return this;
167   }
168
169   /**
170    * Copies this bean and sets the serializer and schema on it.
171    *
172    * @param serializer The new serializer for the bean.  Can be <jk>null</jk>.
173    * @param schema The new schema for the bean.  Can be <jk>null</jk>.
174    * @return Either a new bean with the serializer set, or this bean if
175    *    both values are <jk>null</jk> or the serializer and schema were already set.
176    */
177   public SerializedPart copyWith(HttpPartSerializerSession serializer, HttpPartSchema schema) {
178      if ((this.serializer == null && serializer != null) || (this.schema == null && schema != null)) {
179         SerializedPart p = copy();
180         if (serializer != null)
181            p.serializer(serializer);
182         if (schema != null)
183            p.schema(schema);
184         return p;
185      }
186      return this;
187   }
188
189   /**
190    * Don't serialize this pair if the value is <jk>null</jk> or an empty string.
191    *
192    * @return This object.
193    */
194   public SerializedPart skipIfEmpty() {
195      return skipIfEmpty(true);
196   }
197
198   /**
199    * Don't serialize this pair if the value is <jk>null</jk> or an empty string.
200    *
201    * @param value The new value of this setting.
202    * @return This object.
203    */
204   public SerializedPart skipIfEmpty(boolean value) {
205      this.skipIfEmpty = value;
206      return this;
207   }
208
209   @Override /* Headerable */
210   public SerializedHeader asHeader() {
211      return new SerializedHeader(getName(), value, serializer, schema, skipIfEmpty);
212   }
213
214   @Override /* NameValuePair */
215   public String getValue() {
216      try {
217         Object v = unwrap(value);
218         HttpPartSchema schema = this.schema == null ? HttpPartSchema.DEFAULT : this.schema;
219         String def = schema.getDefault();
220         if (v == null) {
221            if ((def == null && ! schema.isRequired()) || (def == null && schema.isAllowEmptyValue()))
222               return null;
223         }
224         if (isEmpty(stringify(v)) && skipIfEmpty && def == null)
225            return null;
226         return serializer == null ? stringify(v) : serializer.serialize(type, schema, v);
227      } catch (SchemaValidationException e) {
228         throw runtimeException(e, "Validation error on request {0} part ''{1}''=''{2}''", type, getName(), value);
229      } catch (SerializeException e) {
230         throw runtimeException(e, "Serialization error on request {0} part ''{1}''", type, getName());
231      }
232   }
233
234   private Object unwrap(Object o) {
235      if (o instanceof Supplier)
236         return ((Supplier<?>)o).get();
237      return o;
238   }
