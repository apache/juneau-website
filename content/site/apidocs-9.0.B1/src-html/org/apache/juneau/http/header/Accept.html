020/**
021 * Represents a parsed <l>Accept</l> HTTP request header.
022 *
023 * <p>
024 * Content-Types that are acceptable for the response.
025 *
026 * <h5 class='figure'>Example</h5>
027 * <p class='bcode'>
028 *    Accept: text/plain
029 * </p>
030 *
031 * <h5 class='topic'>RFC2616 Specification</h5>
032 *
033 * The Accept request-header field can be used to specify certain media types which are acceptable for the response.
034 * Accept headers can be used to indicate that the request is specifically limited to a small set of desired types, as
035 * in the case of a request for an in-line image.
036 *
037 * <p class='bcode'>
038 *     Accept         = "Accept" ":
039 *                      #( media-range [ accept-params ] )
040 *
041 *     media-range    = ( "* /*"
042 *                      | ( type "/" "*" )
043 *                      | ( type "/" subtype )
044 *                      ) *( ";" parameter )
045 *     accept-params  = ";" "q" "=" qvalue *( accept-extension )
046 *     accept-extension = ";" token [ "=" ( token | quoted-string ) ]
047 * </p>
048 *
049 * <p>
050 * The asterisk "*" character is used to group media types into ranges, with "* /*" indicating all media types and
051 * "type/*" indicating all subtypes of that type.
052 * The media-range MAY include media type parameters that are applicable to that range.
053 *
054 * <p>
055 * Each media-range MAY be followed by one or more accept-params, beginning with the "q" parameter for indicating a
056 * relative quality factor.
057 * The first "q" parameter (if any) separates the media-range parameter(s) from the accept-params.
058 * Quality factors allow the user or user agent to indicate the relative degree of preference for that media-range,
059 * using the qvalue scale from 0 to 1 (section 3.9).
060 * The default value is q=1.
061 *
062 * <p>
063 * Note: Use of the "q" parameter name to separate media type parameters from Accept extension parameters is due to
064 * historical practice.
065 * Although this prevents any media type parameter named "q" from being used with a media range, such an event is
066 * believed to be unlikely given the lack of any "q" parameters in the IANA
067 * media type registry and the rare usage of any media type parameters in Accept.
068 * Future media types are discouraged from registering any parameter named "q".
069 *
070 * <p>
071 * The example
072 * <p class='bcode'>
073 *    Accept: audio/*; q=0.2, audio/basic
074 * </p>
075 * <p>
076 * SHOULD be interpreted as "I prefer audio/basic, but send me any audio type if it is the best available after an 80%
077 * mark-down in quality."
078 *
079 * <p>
080 * If no Accept header field is present, then it is assumed that the client accepts all media types.
081 *
082 * <p>
083 * If an Accept header field is present, and if the server cannot send a response which is acceptable according to the
084 * combined Accept field value, then the server SHOULD send a 406 (not acceptable) response.
085 *
086 * <p>
087 * A more elaborate example is
088 * <p class='bcode'>
089 *    Accept: text/plain; q=0.5, text/html,
090 *            text/x-dvi; q=0.8, text/x-c
091 * </p>
092 *
093 * <p>
094 * Verbally, this would be interpreted as "text/html and text/x-c are the preferred media types, but if they do not
095 * exist, then send the
096 * text/x-dvi entity, and if that does not exist, send the text/plain entity."
097 *
098 * <p>
099 * Media ranges can be overridden by more specific media ranges or specific media types.
100 * If more than one media range applies to a given type, the most specific reference has precedence.
101 * For example,
102 * <p class='bcode'>
103 *    Accept: text/ *, text/html, text/html;level=1, * /*
104 * </p>
105 * <p>
106 * have the following precedence:
107 * <ol>
108 *    <li>text/html;level=1
109 *    <li>text/html
110 *    <li>text/*
111 *    <li>* /*
112 * </ol>
113 *
114 * <p>
115 * The media type quality factor associated with a given type is determined by finding the media range with the highest
116 * precedence which matches that type.
117 * For example,
118 * <p class='bcode'>
119 *    Accept: text/*;q=0.3, text/html;q=0.7, text/html;level=1,
120 *            text/html;level=2;q=0.4, * /*;q=0.5
121 * </p>
122 * <p>
123 * would cause the following values to be associated:
124 * <p class='bcode'>
125 *    text/html;level=1         = 1
126 *    text/html                 = 0.7
127 *    text/plain                = 0.3
128 *    image/jpeg                = 0.5
129 *    text/html;level=2         = 0.4
130 *    text/html;level=3         = 0.7
131 * </p>
132 *
133 * <p>
134 * Note: A user agent might be provided with a default set of quality values for certain media ranges.
135 * However, unless the user agent is a closed system which cannot interact with other rendering agents, this default
136 * set ought to be configurable by the user.
137 *
138 * <ul class='seealso'>
139 *    <li class='link'>{@doc jm.HttpParts}
140 *    <li class='extlink'>{@doc ext.RFC2616}
141 *    <li class='extlink'>{@source}
142 * </ul>
143 *
144 * @serial exclude
145 */

150   // Static
151   //-----------------------------------------------------------------------------------------------------------------
152
153   private static final long serialVersionUID = 1L;
154   private static final String NAME = "Accept";
155
156   private static final Cache<String,Accept> CACHE = Cache.of(String.class, Accept.class).build();

183   /**
184    * Static creator.
185    *
186    * @param value
187    *    The header value.
188    *    <br>Must be parsable by {@link MediaRanges#of(String)}.
189    *    <br>Can be <jk>null</jk>.
190    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
191    */
192   public static Accept of(String value) {
193      return value == null ? null : CACHE.get(value, ()->new Accept(value));
194   }
195
196   /**
197    * Static creator.
198    *
199    * @param value
200    *    The header value.
201    *    <br>Can be <jk>null</jk>.
202    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
203    */
204   public static Accept of(MediaRanges value) {

206   }
207
208   /**
209    * Static creator.
210    *
211    * @param value

214    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
215    */
216   public static Accept of(MediaType value) {
217      return value == null ? null : new Accept(value.toString());
218   }
219
220   /**
221    * Static creator with delayed value.
222    *
223    * <p>
224    * Header value is re-evaluated on each call to {@link #getValue()}.
225    *
226    * @param value

230    */
231   public static Accept of(Supplier<MediaRanges> value) {
232      return value == null ? null : new Accept(value);
233   }
