038/**
039 * Represents an OpenAPI schema definition.
040 *
041 * <p>
042 * The schema definition can be applied to any HTTP parts such as bodies, headers, query/form parameters, and URL path parts.
043 * <br>The API is generic enough to apply to any path part although some attributes may only applicable for certain parts.
044 *
045 * <p>
046 * Schema objects are created via builders instantiated through the {@link #create()} method.
047 *
048 * <ul class='notes'>
049 *    <li class='note'>This class is thread safe and reusable.
050 * </ul>
051 *
052 * <ul class='seealso'>
053 *    <li class='link'>{@doc jm.OpenApiDetails}
054 *    <li class='extlink'>{@source}
055 * </ul>
056 */
057public class HttpPartSchema {

493   /**
494    * Finds the schema information for the specified method parameter.
495    *
496    * <p>
497    * This method will gather all the schema information from the annotations at the following locations:
498    * <ul>
499    *    <li>The method parameter.
500    *    <li>The method parameter class.
501    *    <li>The method parameter parent classes and interfaces.
502    * </ul>
503    *
504    * @param c
505    *    The annotation to look for.
506    *    <br>Valid values:
507    *    <ul>
508    *       <li>{@link Content}
509    *       <li>{@link Header}
510    *       <li>{@link Query}
511    *       <li>{@link FormData}
512    *       <li>{@link Path}
513    *       <li>{@link Response}
514    *       <li>{@link HasQuery}
515    *       <li>{@link HasFormData}
516    *    </ul>
517    * @param mpi The Java method parameter.
518    * @return The schema information about the parameter.
519    */
520   public static HttpPartSchema create(Class<? extends Annotation> c, ParamInfo mpi) {
521      return create().applyAll(c, mpi).build();
522   }
