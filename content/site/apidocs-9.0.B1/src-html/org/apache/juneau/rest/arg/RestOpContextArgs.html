013package org.apache.juneau.rest.arg;
014
015import org.apache.juneau.jsonschema.*;
016import org.apache.juneau.parser.*;
017import org.apache.juneau.reflect.*;
018import org.apache.juneau.rest.*;
019import org.apache.juneau.rest.annotation.*;
020import org.apache.juneau.serializer.*;
021import org.apache.juneau.utils.*;
022
023/**
024 * Resolves method parameters on {@link RestOp}-annotated Java methods of types found on the {@link RestOpContext} object.
025 *
026 * <ul class='javatree'>
027 *    <li class='jc'>{@link JsonSchemaGenerator}
028 *    <li class='jc'>{@link ParserSet}
029 *    <li class='jc'>{@link RestOpContext}
030 *    <li class='jc'>{@link SerializerSet}
031 * </ul>
032 *
033 * <ul class='seealso'>
034 *    <li class='link'>{@doc jrs.JavaMethodParameters}
035 *    <li class='extlink'>{@source}
036 * </ul>
037 */
038public class RestOpContextArgs extends SimpleRestOperationArg {
039
040   /**
041    * Static creator.
042    *
043    * @param paramInfo The Java method parameter being resolved.
044    * @return A new arg, or <jk>null</jk> if the parameter type is not one of the supported types.
045    */
046   public static RestOpContextArgs create(ParamInfo paramInfo) {
047      if (paramInfo.isType(JsonSchemaGenerator.class))
048         return new RestOpContextArgs(x->x.getJsonSchemaGenerator());
049      if (paramInfo.isType(ParserSet.class))
050         return new RestOpContextArgs(x->x.getParsers());
051      if (paramInfo.isType(RestOpContext.class))
052         return new RestOpContextArgs(x->x);
053      if (paramInfo.isType(SerializerSet.class))
054         return new RestOpContextArgs(x->x.getSerializers());
055      return null;
056   }
057
058   /**
059    * Constructor.
060    *
061    * @param <T> The function return type.
062    * @param function The function for finding the arg.
063    */
064   protected <T> RestOpContextArgs(ThrowingFunction<RestOpContext,T> function) {
065      super((session)->function.apply(session.getContext()));
066   }
067}
