110   /**
111    * Constructor.
112    *
113    * @param value
114    *    The object being tested.
115    *    <br>Can be <jk>null</jk>.
116    * @param returns
117    *    The object to return after a test method is called.
118    *    <br>If <jk>null</jk>, the test method returns this object allowing multiple test method calls to be
119    * used on the same assertion.
120    */
121   public FluentResponseBodyAssertion(ResponseContent value, R returns) {
122      this(null, value, returns);
123   }
124
125   /**
126    * Chained constructor.
127    *
128    * <p>
129    * Used when transforming one assertion into another so that the assertion config can be used by the new assertion.
130    *
131    * @param creator
132    *    The assertion that created this assertion.
133    *    <br>Should be <jk>null</jk> if this is the top-level assertion.
134    * @param value
135    *    The object being tested.
136    *    <br>Can be <jk>null</jk>.
137    * @param returns
138    *    The object to return after a test method is called.
139    *    <br>If <jk>null</jk>, the test method returns this object allowing multiple test method calls to be
140    * used on the same assertion.
141    */
142   public FluentResponseBodyAssertion(Assertion creator, ResponseContent value, R returns) {
143      super(creator, value, returns);
144      setThrowable(BadRequest.class);
145   }

151   /**
152    * Provides the ability to perform fluent-style assertions on this response body.
153    *
154    * <h5 class='section'>Examples:</h5>
155    * <p class='bjava'>
156    *    <jc>// Validates the response body equals the text "OK".</jc>
157    *    <jv>client</jv>
158    *       .get(<jsf>URI</jsf>)
159    *       .run()
160    *       .assertContent().is(<js>"OK"</js>);
161    *
162    *    <jc>// Validates the response body contains the text "OK".</jc>
163    *    <jv>client</jv>
164    *       .get(<jsf>URI</jsf>)
165    *       .run()
166    *       .assertContent().isContains(<js>"OK"</js>);
167    *
168    *    <jc>// Validates the response body passes a predicate test.</jc>
169    *    <jv>client</jv>
170    *       .get(<jsf>URI</jsf>)
171    *       .run()
172    *       .assertContent().is(<jv>x</jv> -&gt; <jv>x</jv>.contains(<js>"OK"</js>));
173    *
174    *    <jc>// Validates the response body matches a regular expression.</jc>
175    *    <jv>client</jv>
176    *       .get(<jsf>URI</jsf>)
177    *       .run()
178    *       .assertContent().isPattern(<js>".*OK.*"</js>);
179    *
180    *    <jc>// Validates the response body matches a regular expression using regex flags.</jc>
181    *    <jv>client</jv>
182    *       .get(<jsf>URI</jsf>)
183    *       .run()
184    *       .assertContent().isPattern(<js>".*OK.*"</js>, <jsf>MULTILINE</jsf> &amp; <jsf>CASE_INSENSITIVE</jsf>);
185    *
186    *    <jc>// Validates the response body matches a regular expression in the form of an existing Pattern.</jc>
187    *    Pattern <jv>pattern</jv> = Pattern.<jsm>compile</jsm>(<js>".*OK.*"</js>);
188    *    <jv>client</jv>
189    *       .get(<jsf>URI</jsf>)
190    *       .run()
191    *       .assertContent().isPattern(<jv>pattern</jv>);
192    * </p>
193    *
194    * <p>
195    * The assertion test returns the original response object allowing you to chain multiple requests like so:
196    * <p class='bjava'>
197    *    <jc>// Validates the response body matches a regular expression.</jc>
198    *    MyBean <jv>bean</jv> = <jv>client</jv>
199    *       .get(<jsf>URI</jsf>)
200    *       .run()
201    *       .assertContent().isPattern(<js>".*OK.*"</js>);
202    *       .assertContent().isNotPattern(<js>".*ERROR.*"</js>)
203    *       .getContent().as(MyBean.<jk>class</jk>);
204    * </p>
205    *
206    * <ul class='notes'>
207    *    <li class='note'>
208    *       If no charset was found on the <code>Content-Type</code> response header, <js>"UTF-8"</js> is assumed.
209    *  <li class='note'>
210    *    When using this method, the body is automatically cached by calling the {@link ResponseContent#cache()}.
211    *    <li class='note'>
212    *       The input stream is automatically closed after this call.
213    * </ul>
214    *
215    * @return A new fluent assertion object.
216    */
217   @Override
218   public FluentStringAssertion<R> asString() {
219      return new FluentStringAssertion<>(valueAsString(), returns());
220   }
221
222   /**
223    * Provides the ability to perform fluent-style assertions on the bytes of the response body.
224    *
225    * <h5 class='section'>Examples:</h5>
226    * <p class='bjava'>
227    *    <jc>// Validates the response body equals the text "foo".</jc>
228    *    <jv>client</jv>
229    *       .get(<jsf>URI</jsf>)
230    *       .run()
231    *       .assertContent().asBytes().asHex().is(<js>"666F6F"</js>);
232    * </p>
233    *
234    * <ul class='notes'>
235    *    <li class='note'>
236    *       If no charset was found on the <code>Content-Type</code> response header, <js>"UTF-8"</js> is assumed.
237    *  <li class='note'>
238    *    When using this method, the body is automatically cached by calling the {@link ResponseContent#cache()}.
239    *    <li class='note'>
240    *       The input stream is automatically closed after this call.
241    * </ul>
242    *
243    * @return A new fluent assertion object.
244    */
245   public FluentByteArrayAssertion<R> asBytes() {
246      return new FluentByteArrayAssertion<>(valueAsBytes(), returns());
247   }
248
249   /**
250    * Converts the body to a type using {@link ResponseContent#as(Class)} and then returns the value as an object assertion.
251    *
252    * <h5 class='section'>Examples:</h5>
253    * <p class='bjava'>
254    *    <jc>// Validates the response body as a list of strings and validates the length.</jc>
255    *    <jv>client</jv>
256    *       .get(<js>"/myBean"</js>)
257    *       .run()
258    *       .assertContent().as(List.<jk>class</jk>, String.<jk>class</jk>).is(<jv>x</jv> -&gt; <jv>x</jv>.size() &gt; 0);
259    * </p>
260    *
261    * @param <T> The object type to create.
262    * @param type The object type to create.
263    * @return A new fluent assertion object.
264    */
265   public <T> FluentAnyAssertion<T,R> as(Class<T> type) {
266      return new FluentAnyAssertion<>(valueAsType(type), returns());
267   }
268
269   /**
270    * Converts the body to a type using {@link ResponseContent#as(Type,Type...)} and then returns the value as an object assertion.
271    *
272    * <h5 class='section'>Examples:</h5>
273    * <p class='bjava'>
274    *    <jc>// Validates the response body as a list of strings and validates the length.</jc>
275    *    <jv>client</jv>
276    *       .get(<js>"/myBean"</js>)
277    *       .run()
278    *       .assertContent().as(List.<jk>class</jk>, String.<jk>class</jk>).is(<jv>x</jv> -&gt; <jv>x</jv>.size() &gt; 0);
279    * </p>
280    *
281    * <p>
282    * See {@doc jm.ComplexDataTypes Complex Data Types} for information on defining complex generic types of {@link Map Maps} and {@link Collection Collections}.
283    *
284    * @param type The object type to create.
285    * @param args Optional type arguments.
286    * @return A new fluent assertion object.
287    */
288   public FluentAnyAssertion<Object,R> as(Type type, Type...args) {
289      return new FluentAnyAssertion<>(valueAsType(type, args), returns());
290   }

296   /**
297    * Asserts that the body contains the specified value.
298    *
299    * @param value The value to check against.
300    * @return This object.

302    */
303   public R is(String value) throws AssertionError {
304      return asString().is(value);
305   }
306
307   /**
308    * Asserts that the text contains all of the specified substrings.
309    *
310    * @param values The values to check against.
311    * @return This object.
312    * @throws AssertionError If assertion failed.
313    */
314   public R isContains(String...values) throws AssertionError {
315      return asString().isContains(values);
316   }
317
318   /**
319    * Asserts that the body doesn't contain any of the specified substrings.
320    *
321    * @param values The values to check against.
322    * @return This object.
323    * @throws AssertionError If assertion failed.
324    */
325   public R isNotContains(String...values) throws AssertionError {
326      return asString().isNotContains(values);
327   }
328
329   /**
330    * Asserts that the body is empty.
331    *
332    * @return This object.
333    * @throws AssertionError If assertion failed.
334    */
335   public R isEmpty() {
336      return asString().isEmpty();
337   }

340    * Asserts that the body is not empty.
341    *
342    * @return This object.
343    * @throws AssertionError If assertion failed.
344    */
345   public R isNotEmpty() {
346      return asString().isNotEmpty();
347   }

353   @Override
354   protected String valueAsString() throws AssertionError {
355      try {
356         return value().cache().asString();
357      } catch (RestCallException e) {
358         throw error(e, "Exception occurred during call.");
359      }
360   }
361
362   private byte[] valueAsBytes() throws AssertionError {
363      try {
364         return value().cache().asBytes();
365      } catch (RestCallException e) {
366         throw error(e, "Exception occurred during call.");
367      }
368   }
369
370   private <T> T valueAsType(Type type, Type...args) throws AssertionError {
371      try {
372         return value().cache().as(type, args);
373      } catch (RestCallException e) {
374         throw error(e, "Exception occurred during call.");
375      }
376   }
