460   /**
461    * Applies {@link RestDelete} annotations to a {@link org.apache.juneau.rest.RestOpContext.Builder}.
462    */
463   public static class RestOpContextApply extends AnnotationApplier<RestDelete,RestOpContext.Builder> {
464
465      /**
466       * Constructor.
467       *
468       * @param vr The resolver for resolving values in annotations.
469       */
470      public RestOpContextApply(VarResolverSession vr) {
471         super(RestDelete.class, RestOpContext.Builder.class, vr);
472      }
473
474      @Override
475      public void apply(AnnotationInfo<RestDelete> ai, RestOpContext.Builder b) {
476         RestDelete a = ai.inner();
477
478         b.httpMethod("delete");
479
480         classes(a.encoders()).ifPresent(x -> b.encoders().set(x));
481         type(a.contextClass()).ifPresent(x -> b.type(x));
482         stream(a.defaultRequestHeaders()).map(x -> stringHeader(x)).forEach(x -> b.defaultRequestHeaders().setDefault(x));
483         stream(a.defaultResponseHeaders()).map(x -> stringHeader(x)).forEach(x -> b.defaultResponseHeaders().setDefault(x));
484         stream(a.defaultRequestAttributes()).map(x -> BasicNamedAttribute.ofPair(x)).forEach(x -> b.defaultRequestAttributes().add(x));
485         stream(a.defaultRequestQueryData()).map(x -> basicPart(x)).forEach(x -> b.defaultRequestQueryData().setDefault(x));
486         string(a.defaultAccept()).map(x -> accept(x)).ifPresent(x -> b.defaultRequestHeaders().setDefault(x));
487         b.guards().append(a.guards());
488         b.matchers().append(a.matchers());
489         string(a.clientVersion()).ifPresent(x -> b.clientVersion(x));
490         string(a.defaultCharset()).map(Charset::forName).ifPresent(x -> b.defaultCharset(x));
491         stream(a.path()).forEach(x -> b.path(x));
492         string(a.value()).ifPresent(x -> b.path(x));
493         cdl(a.rolesDeclared()).forEach(x -> b.rolesDeclared(x));
494         string(a.roleGuard()).ifPresent(x -> b.roleGuard(x));
495         string(a.debug()).map(Enablement::fromString).ifPresent(x -> b.debug(x));
496      }
497   }
