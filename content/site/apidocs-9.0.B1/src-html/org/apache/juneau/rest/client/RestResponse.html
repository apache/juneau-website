036/**
037 * Represents a response from a remote REST resource.
038 *
039 * <p>
040 * Instances of this class are created by calling the {@link RestRequest#run()} method.
041 *
042 * <ul class='seealso'>
043 *    <li class='link'>{@doc juneau-rest-client}
044 *    <li class='extlink'>{@source}
045 * </ul>
046 */
047public class RestResponse implements HttpResponse {
048
049   private final RestClient client;
050   private final RestRequest request;
051   private final HttpResponse response;
052   private final Parser parser;
053   private ResponseContent responseContent;
054   private boolean isClosed;
055   private HeaderList headers;
056
057   private Map<HttpPartParser,HttpPartParserSession> partParserSessions = new IdentityHashMap<>();
058   private HttpPartParserSession partParserSession;
059
060   /**
061    * Constructor.
062    *
063    * @param client The RestClient that created this response.
064    * @param request The REST request.
065    * @param response The HTTP response.  Can be <jk>null</jk>.
066    * @param parser The overridden parser passed into {@link RestRequest#parser(Parser)}.
067    */
068   protected RestResponse(RestClient client, RestRequest request, HttpResponse response, Parser parser) {
069      this.client = client;
070      this.request = request;
071      this.parser = parser;
072      this.response = response == null ? new BasicHttpResponse(null, 0, null) : response;
073      this.responseContent = new ResponseContent(client, request, this, parser);
074      this.headers = HeaderList.of(this.response.getAllHeaders());
075   }

218    * Shortcut for retrieving the response charset from the <l>Content-Type</l> header.
219    *
220    * @return The response charset.
221    * @throws RestCallException If REST call failed.
222    */
223   public String getCharacterEncoding() throws RestCallException {
224      Optional<ContentType> ct = getContentType();
225      String s = null;
226      if (ct.isPresent())
227         s = getContentType().get().getParameter("charset");
228      return StringUtils.isEmpty(s) ? "utf-8" : s;

263   /**
264    * Provides the ability to perform fluent-style assertions on a response header.
265    *
266    * <h5 class='section'>Examples:</h5>
267    * <p class='bjava'>
268    *    <jc>// Validates the content type header is provided.</jc>
269    *    <jv>client</jv>
270    *       .get(<jsf>URI</jsf>)
271    *       .run()
272    *       .assertHeader(<js>"Content-Type"</js>).exists();
273    *
274    *    <jc>// Validates the content type is JSON.</jc>
275    *    <jv>client</jv>
276    *       .get(<jsf>URI</jsf>)
277    *       .run()
278    *       .assertHeader(<js>"Content-Type"</js>).is(<js>"application/json"</js>);
279    *
280    *    <jc>// Validates the content type is JSON using test predicate.</jc>
281    *    <jv>client</jv>
282    *       .get(<jsf>URI</jsf>)
283    *       .run()
284    *       .assertHeader(<js>"Content-Type"</js>).is(<jv>x</jv> -&gt; <jv>x</jv>.equals(<js>"application/json"</js>));
285    *
286    *    <jc>// Validates the content type is JSON by just checking for substring.</jc>
287    *    <jv>client</jv>
288    *       .get(<jsf>URI</jsf>)
289    *       .run()
290    *       .assertHeader(<js>"Content-Type"</js>).contains(<js>"json"</js>);
291    *
292    *    <jc>// Validates the content type is JSON using regular expression.</jc>
293    *    <jv>client</jv>
294    *       .get(<jsf>URI</jsf>)
295    *       .run()
296    *       .assertHeader(<js>"Content-Type"</js>).isPattern(<js>".*json.*"</js>);
297    *
298    *    <jc>// Validates the content type is JSON using case-insensitive regular expression.</jc>
299    *    <jv>client</jv>
300    *       .get(<jsf>URI</jsf>)
301    *       .run()
302    *       .assertHeader(<js>"Content-Type"</js>).isPattern(<js>".*json.*"</js>, <jsf>CASE_INSENSITIVE</jsf>);
303    * </p>
304    *
305    * <p>
306    * The assertion test returns the original response object allowing you to chain multiple requests like so:
307    * <p class='bjava'>
308    *    <jc>// Validates the header and converts it to a bean.</jc>
309    *    MediaType <jv>mediaType</jv> = <jv>client</jv>
310    *       .get(<jsf>URI</jsf>)
311    *       .run()
312    *       .assertHeader(<js>"Content-Type"</js>).isNotEmpty()
313    *       .assertHeader(<js>"Content-Type"</js>).isPattern(<js>".*json.*"</js>)
314    *       .getHeader(<js>"Content-Type"</js>).as(MediaType.<jk>class</jk>);
315    * </p>
316    *
317    * @param name The header name.
318    * @return A new fluent assertion object.
319    */
320   public FluentResponseHeaderAssertion<RestResponse> assertHeader(String name) {
321      return new FluentResponseHeaderAssertion<>(getHeader(name), this);
322   }

339   /**
340    * Provides the ability to perform fluent-style assertions on this response body.
341    *
342    * <h5 class='section'>Examples:</h5>
343    * <p class='bjava'>
344    *    <jc>// Validates the response body equals the text "OK".</jc>
345    *    <jv>client</jv>
346    *       .get(<jsf>URI</jsf>)
347    *       .run()
348    *       .assertContent().is(<js>"OK"</js>);
349    *
350    *    <jc>// Validates the response body contains the text "OK".</jc>
351    *    <jv>client</jv>
352    *       .get(<jsf>URI</jsf>)
353    *       .run()
354    *       .assertContent().isContains(<js>"OK"</js>);
355    *
356    *    <jc>// Validates the response body passes a predicate test.</jc>
357    *    <jv>client</jv>
358    *       .get(<jsf>URI</jsf>)
359    *       .run()
360    *       .assertContent().is(<jv>x</jv> -&gt; <jv>x</jv>.contains(<js>"OK"</js>));
361    *
362    *    <jc>// Validates the response body matches a regular expression.</jc>
363    *    <jv>client</jv>
364    *       .get(<jsf>URI</jsf>)
365    *       .run()
366    *       .assertContent().isPattern(<js>".*OK.*"</js>);
367    *
368    *    <jc>// Validates the response body matches a regular expression using regex flags.</jc>
369    *    <jv>client</jv>
370    *       .get(<jsf>URI</jsf>)
371    *       .run()
372    *       .assertContent().isPattern(<js>".*OK.*"</js>, <jsf>MULTILINE</jsf> &amp; <jsf>CASE_INSENSITIVE</jsf>);
373    *
374    *    <jc>// Validates the response body matches a regular expression in the form of an existing Pattern.</jc>
375    *    Pattern <jv>pattern</jv> = Pattern.<jsm>compile</jsm>(<js>".*OK.*"</js>);
376    *    <jv>client</jv>
377    *       .get(<jsf>URI</jsf>)
378    *       .run()
379    *       .assertContent().isPattern(<jv>pattern</jv>);
380    * </p>
381    *
382    * <p>
383    * The assertion test returns the original response object allowing you to chain multiple requests like so:
384    * <p class='bjava'>
385    *    <jc>// Validates the response body matches a regular expression.</jc>
386    *    MyBean <jv>bean</jv> = <jv>client</jv>
387    *       .get(<jsf>URI</jsf>)
388    *       .run()
389    *       .assertContent().isPattern(<js>".*OK.*"</js>);
390    *       .assertContent().isNotPattern(<js>".*ERROR.*"</js>)
391    *       .getContent().as(MyBean.<jk>class</jk>);
392    * </p>
393    *
394    * <ul class='notes'>
395    *    <li class='note'>
396    *       If no charset was found on the <code>Content-Type</code> response header, <js>"UTF-8"</js> is assumed.
397    *  <li class='note'>
398    *    When using this method, the body is automatically cached by calling the {@link ResponseContent#cache()}.
399    *    <li class='note'>
400    *       The input stream is automatically closed after this call.
401    * </ul>
402    *
403    * @return A new fluent assertion object.
404    */
405   public FluentResponseBodyAssertion<RestResponse> assertContent() {
406      return new FluentResponseBodyAssertion<>(responseContent, this);
407   }

424   @SuppressWarnings("unchecked")
425   <T> T as(ResponseBeanMeta rbm) {
426      Class<T> c = (Class<T>)rbm.getClassMeta().getInnerClass();
427      final RestClient rc = this.client;
428      return (T)Proxy.newProxyInstance(
429         c.getClassLoader(),
430         new Class[] { c },
431         new InvocationHandler() {
432            @Override /* InvocationHandler */
433            public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
434               ResponseBeanPropertyMeta pm = rbm.getProperty(method.getName());
435               HttpPartParserSession pp = getPartParserSession(pm.getParser().orElse(rc.getPartParser()));
436               HttpPartSchema schema = pm.getSchema();
437               HttpPartType pt = pm.getPartType();
438               String name = pm.getPartName().orElse(null);
439               ClassMeta<?> type = rc.getBeanContext().getClassMeta(method.getGenericReturnType());
440               if (pt == RESPONSE_HEADER)
441                  return getHeader(name).parser(pp).schema(schema).as(type).orElse(null);
442               if (pt == RESPONSE_STATUS)
443                  return getStatusCode();
444               return getContent().schema(schema).as(type);
445            }
446      });
447   }

828   void close() throws RestCallException {
829      if (isClosed)
830         return;
831      isClosed = true;
832      EntityUtils.consumeQuietly(response.getEntity());
833
834      if (!request.isLoggingSuppressed() && (request.isDebug() || client.logRequestsPredicate.test(request, this))) {
835         if (client.logRequests == DetailLevel.SIMPLE) {
836            client.log(client.logRequestsLevel, "HTTP {0} {1}, {2}", request.getMethod(), request.getURI(), this.getStatusLine());
837         } else if (request.isDebug() || client.logRequests == DetailLevel.FULL) {
838            String output = getContent().asString();
839            StringBuilder sb = new StringBuilder();
840            sb.append("\n=== HTTP Call (outgoing) ======================================================");
841            sb.append("\n=== REQUEST ===\n");
842            sb.append(request.getMethod()).append(" ").append(request.getURI());
843            sb.append("\n---request headers---");
844            request.getHeaderList().forEach(x -> sb.append("\n\t").append(x));
845            if (request.hasHttpEntity()) {
846               sb.append("\n---request entity---");
847               HttpEntity e = request.getHttpEntity();
848               if (e.getContentType() != null)
849                  sb.append("\n\t").append(e.getContentType());
850               if (e.isRepeatable()) {
851                  try {
852                     sb.append("\n---request content---\n").append(EntityUtils.toString(e));
853                  } catch (Exception ex) {
854                     sb.append("\n---request content exception---\n").append(ex.getMessage());
855                  }
856               }
857            }
858            sb.append("\n=== RESPONSE ===\n").append(getStatusLine());
859            sb.append("\n---response headers---");
860            for (Header h : getAllHeaders())
861               sb.append("\n\t").append(h);
862            sb.append("\n---response content---\n").append(output);
863            sb.append("\n=== END =======================================================================");
864            client.log(client.logRequestsLevel, sb.toString());
865         }
866      }
867
868      for (RestCallInterceptor r : request.interceptors) {
869         try {
870            r.onClose(request, this);
871         } catch (RuntimeException | RestCallException e) {
872            throw e;
873         } catch (Exception e) {
874            throw new RestCallException(this, e, "Interceptor throw exception on close");
875         }
876      }
877      client.onClose(request, this);
878   }
