033/**
034 * Interface class used for logging HTTP requests.
035 *
036 * <p>
037 * The {@link Builder#type(Class)} method has been provided for easy extension of this class.
038 *
039 * <p>
040 * The following default implementations are also provided:
041 * <ul class='javatree'>
042 *    <li class='jc'>{@link BasicRestLogger} - The default logger typically used.
043 *    <li class='jc'>{@link BasicDisabledRestLogger} - A no-op logger if you want to turn off logging entirely.
044 *    <li class='jc'>{@link BasicTestRestLogger} - A logger useful for testcases.
045 *    <li class='jc'>{@link BasicTestCaptureRestLogger} - Useful for capturing log messages for testing logging itself.
046 * </ul>
047 *
048 * <ul class='seealso'>
049 *    <li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#callLogger()}
050 *    <li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#debugEnablement()}
051 *    <li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#debugOn(String)}
052 *    <li class='ja'>{@link Rest#debug}
053 *    <li class='ja'>{@link RestOp#debug}
054 *    <li class='link'>{@doc jrs.LoggingAndDebugging}
055 *    <li class='extlink'>{@source}
056 * </ul>
057 */
058public interface RestLogger {

067   /**
068    * System property name for the default logger name to use for {@link RestLogger} objects.
069    * <p>
070    * Can also use a <c>JUNEAU_RESTLOGGER_LOGGER</c> environment variable.
071    * <p>
072    * If not specified, the default is <js>"global"</js>.
073    */
074   public static final String SP_logger = "juneau.restLogger.logger";
075
076   /**
077    * System property name for the default enablement setting for {@link RestLogger} objects.
078    * <p>
079    * Can also use a <c>JUNEAU_RESTLOGGER_ENABLED</c> environment variable.
080    * <p>
081    * The possible values are:
082    * <ul>
083    *    <li>{@link Enablement#ALWAYS "ALWAYS"} (default) - Logging is enabled.
084    *    <li>{@link Enablement#NEVER "NEVER"} - Logging is disabled.
085    *    <li>{@link Enablement#CONDITIONAL "CONDITIONALLY"} - Logging is enabled if it passes the {@link Builder#enabledTest(Predicate)} test.
086    * </ul>
087    */
088   public static final String SP_enabled = "juneau.restLogger.enabled";
089
090   /**
091    * System property name for the default request detail setting for {@link RestLogger} objects.
092    * <p>
093    * Can also use a <c>JUNEAU_RESTLOGGER_REQUESTDETAIL</c> environment variable.
094    *
095    * <ul class='values'>
096    *    <li>{@link RestLoggingDetail#STATUS_LINE "STATUS_LINE"} (default) - Log only the status line.
097    *    <li>{@link RestLoggingDetail#HEADER "HEADER"} - Log the status line and headers.
098    *    <li>{@link RestLoggingDetail#ENTITY "ENTITY"} - Log the status line and headers and content if available.
099    * </ul>
100    */
101   public static final String SP_requestDetail = "juneau.restLogger.requestDetail";
102
103   /**
104    * System property name for the default response detail setting for {@link RestLogger} objects.
105    * <p>
106    * Can also use a <c>JUNEAU_RESTLOGGER_RESPONSEDETAIL</c> environment variable.
107    *
108    * <ul class='values'>
109    *    <li>{@link RestLoggingDetail#STATUS_LINE "STATUS_LINE"} (default) - Log only the status line.
110    *    <li>{@link RestLoggingDetail#HEADER "HEADER"} - Log the status line and headers.
111    *    <li>{@link RestLoggingDetail#ENTITY "ENTITY"} - Log the status line and headers and content if available.
112    * </ul>
113    */

116   /**
117    * System property name for the logging level setting for {@link RestLogger} objects.
118    * <p>
119    * Can also use a <c>JUNEAU_RESTLOGGER_LEVEL</c> environment variable.
120    *
121    * <ul class='values'>
122    *    <li>{@link Level#OFF "OFF"} (default)
123    *    <li>{@link Level#SEVERE "SEVERE"}
124    *    <li>{@link Level#WARNING "WARNING"}
125    *    <li>{@link Level#INFO "INFO"}
126    *    <li>{@link Level#CONFIG "CONFIG"}
127    *    <li>{@link Level#FINE "FINE"}
128    *    <li>{@link Level#FINER "FINER"}
129    *    <li>{@link Level#FINEST "FINEST"}
130    * </ul>
131    */
132   public static final String SP_level = "juneau.restLogger.level";

162      /**
163       * Constructor.
164       *
165       * @param beanStore The bean store to use for creating beans.
166       */
167      protected Builder(BeanStore beanStore) {
168         super(BasicRestLogger.class, beanStore);
169         logger = Logger.getLogger(env(SP_logger, "global"));
170         enabled = env(SP_enabled, ALWAYS);
171         enabledTest = x -> false;
172         requestDetail = env(SP_requestDetail, STATUS_LINE);
173         responseDetail = env(SP_responseDetail, STATUS_LINE);
174         level = env(SP_level).map(Level::parse).orElse(OFF);
175      }
176
177      @Override /* BeanBuilder */
178      protected RestLogger buildDefault() {
179         return new BasicRestLogger(this);
180      }

185      }

192       * Specifies the logger to use for logging the request.
193       *
194       * <p>
195       * If not specified, the logger name is determined in the following order:
196       * <ol>
197       *    <li><js>{@link RestLogger#SP_logger "juneau.restLogger.logger"} system property.
198       *    <li><js>{@link RestLogger#SP_logger "JUNEAU_RESTLOGGER_LOGGER"} environment variable.
199       *    <li><js>"global"</js>.
200       * </ol>
201       *
202       * <p>
203       * The {@link BasicRestLogger#getLogger()} method can also be overridden to provide different logic.
204       *
205       * @param value
206       *    The logger to use for logging the request.
207       * @return This object.
208       */
209      public Builder logger(Logger value) {
210         logger = value;
211         return this;
212      }
213
214      /**
215       * Specifies the logger to use for logging the request.
216       *
217       * <p>
218       * Shortcut for calling <c>logger(Logger.<jsm>getLogger</jsm>(value))</c>.
219       *
220       * <p>
221       * If not specified, the logger name is determined in the following order:
222       * <ol>
223       *    <li><js>{@link RestLogger#SP_logger "juneau.restLogger.logger"} system property.
224       *    <li><js>{@link RestLogger#SP_logger "JUNEAU_RESTLOGGER_LOGGER"} environment variable.
225       *    <li><js>"global"</js>.
226       * </ol>
227       *
228       * <p>
229       * The {@link BasicRestLogger#getLogger()} method can also be overridden to provide different logic.
230       *
231       * @param value
232       *    The logger to use for logging the request.
233       * @return This object.
234       */
235      public Builder logger(String value) {
236         logger = value == null ? null :Logger.getLogger(value);
237         return this;
238      }

240      /**
241       * Same as {@link #logger(Logger)} but only sets the value if it's currently <jk>null</jk>.
242       *
243       * @param value The logger to use for logging the request.
244       * @return This object.
245       */
246      public Builder loggerOnce(Logger value) {
247         if (logger == null)
248            logger = value;
249         return this;
250      }
251
252      /**
253       * Specifies the thrown exception store to use for getting stack trace information (hash IDs and occurrence counts).

257       *    <br>If <jk>null</jk>, stack trace information will not be logged.
258       * @return This object.

261         thrownStore = value;
262         return this;
263      }
264
265      /**

267       *
268       * @param value
269       *    The stack trace store.
270       *    <br>If <jk>null</jk>, stack trace information will not be logged.
271       * @return This object.
272       */
273      public Builder thrownStoreOnce(ThrownStore value) {
274         if (thrownStore == null)
275            thrownStore = value;
276         return this;
277      }
278      /**
279       * Specifies the default logging enablement setting.
280       *
281       * <p>
282       * This specifies the default logging enablement value if not set on the first matched rule or if no rules match.
283       *
284       * <p>
285       * If not specified, the setting is determined via the following:
286       * <ul>
287       *    <li><js>{@link RestLogger#SP_enabled "juneau.restLogger.enabled"} system property.
288       *    <li><js>{@link RestLogger#SP_enabled "JUNEAU_RESTLOGGER_ENABLED"} environment variable.
289       *    <li><js>"ALWAYS"</js>.
290       * </ul>
291       *
292       * <ul class='values'>
293       *    <li>{@link Enablement#ALWAYS ALWAYS} (default) - Logging is enabled.
294       *    <li>{@link Enablement#NEVER NEVER} - Logging is disabled.
295       *    <li>{@link Enablement#CONDITIONAL CONDITIONALLY} - Logging is enabled if it passes the {@link #enabledTest(Predicate)} test.
296       * </ul>
297       *
298       * <p>
299       * @param value
300       *    The default enablement flag value.  Can be <jk>null</jk> to use the default.
301       * @return This object.
302       */
303      public Builder enabled(Enablement value) {
304         enabled = value;
305         return this;
306      }
307
308      /**
309       * Specifies the default logging enablement test predicate.
310       *
311       * <p>
312       * This specifies the default logging enablement test if not set on the first matched rule or if no rules match.
313       *
314       * <p>
315       * This setting has no effect if the enablement setting is not {@link Enablement#CONDITIONAL CONDITIONALLY}.
316       *
317       * <p>
318       * The default if not specified is <c><jv>x</jv> -&gt; <jk>false</jk></c> (never log).
319       *
320       * @param value
321       *    The default enablement flag value.  Can be <jk>null</jk> to use the default.
322       * @return This object.
323       */
324      public Builder enabledTest(Predicate<HttpServletRequest> value) {
325         enabledTest = value;
326         return this;
327      }

336      }

396      /**
397       * The default logging level to use for logging the request/response.
398       *
399       * <p>
400       * This specifies the default logging level if not set on the first matched rule or if no rules match.
401       *
402       * <p>
403       * If not specified, the setting is determined via the following:
404       * <ul>
405       *    <li><js>{@link RestLogger#SP_level "juneau.restLogger.level"} system property.
406       *    <li><js>{@link RestLogger#SP_level "JUNEAU_RESTLOGGER_level"} environment variable.
407       *    <li><js>"OFF"</js>.

419      /**
420       * Adds logging rules to use when debug mode is not enabled.
421       *
422       * <p>
423       * Logging rules are matched in the order they are added.  The first to match wins.
424       *
425       * @param values The logging rules to add to the list of rules.
426       * @return This object.
427       */
428      public Builder normalRules(RestLoggerRule...values) {
429         for (RestLoggerRule rule : values)
430            normalRules.add(rule);
431         return this;
432      }
433
434      /**
435       * Adds logging rules to use when debug mode is enabled.
436       *
437       * <p>
438       * Logging rules are matched in the order they are added.  The first to match wins.
439       *
440       * @param values The logging rules to add to the list of rules.
441       * @return This object.
442       */
443      public Builder debugRules(RestLoggerRule...values) {
444         for (RestLoggerRule rule : values)
445            debugRules.add(rule);
446         return this;
447      }
448
449      /**
450       * Shortcut for adding the same rules as normal and debug rules.
451       *
452       * <p>
453       * Logging rules are matched in the order they are added.  The first to match wins.
454       *
455       * @param values The logging rules to add to the list of rules.
456       * @return This object.
457       */
458      public Builder rules(RestLoggerRule...values) {
459         return normalRules(values).debugRules(values);
460      }
