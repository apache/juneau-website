023import org.apache.juneau.http.annotation.*;
024import org.apache.juneau.httppart.*;
025import org.apache.juneau.internal.*;

027import org.apache.juneau.rest.*;
028import org.apache.juneau.rest.annotation.*;
029import org.apache.juneau.rest.httppart.*;

031/**
032 * Resolves method parameters and parameter types annotated with {@link FormData} on {@link RestOp}-annotated Java methods.
033 *
034 * <p>
035 * The parameter value is resolved using:
036 * <p class='bjava'>
037 *    <jv>opSession</jv>
038 *       .{@link RestOpSession#getRequest() getRequest}()
039 *       .{@link RestRequest#getFormParams() getFormParams}()
040 *       .{@link RequestFormParams#get(String) get}(<jv>name</jv>)
041 *       .{@link RequestFormParam#as(Class) as}(<jv>type</jv>);
042 * </p>
043 *
044 * <p>
045 * {@link HttpPartSchema schema} is derived from the {@link FormData} annotation.
046 *
047 * <p>
048 * If the {@link Schema#collectionFormat()} value is {@link HttpPartCollectionFormat#MULTI}, then the data type can be a {@link Collection} or array.
049 *
050 * <ul class='seealso'>
051 *    <li class='link'>{@doc jrs.JavaMethodParameters}
052 *    <li class='extlink'>{@source}
053 * </ul>
054 */
055public class FormDataArg implements RestOpArg {
056   private final boolean multi;
057   private final HttpPartParser partParser;
058   private final HttpPartSchema schema;
059   private final String name;
060   private final ClassInfo type;
061
062   /**
063    * Static creator.
064    *
065    * @param paramInfo The Java method parameter being resolved.
066    * @param annotations The annotations to apply to any new part parsers.
067    * @return A new {@link FormDataArg}, or <jk>null</jk> if the parameter is not annotated with {@link FormData}.
068    */
069   public static FormDataArg create(ParamInfo paramInfo, AnnotationWorkList annotations) {
070      if (paramInfo.hasAnnotation(FormData.class) || paramInfo.getParameterType().hasAnnotation(FormData.class))
071         return new FormDataArg(paramInfo, annotations);
072      return null;
073   }
074
075   /**
076    * Constructor.
077    *
078    * @param pi The Java method parameter being resolved.
079    * @param annotations The annotations to apply to any new part parsers.
080    */
081   protected FormDataArg(ParamInfo pi, AnnotationWorkList annotations) {
082      this.name = findName(pi).orElseThrow(()->new ArgException(pi, "@FormData used without name or value"));
083      this.type = pi.getParameterType();
084      this.schema = HttpPartSchema.create(FormData.class, pi);
085      Class<? extends HttpPartParser> pp = schema.getParser();
086      this.partParser = pp != null ? HttpPartParser.creator().type(pp).apply(annotations).create() : null;
087      this.multi = schema.getCollectionFormat() == HttpPartCollectionFormat.MULTI;
088
089      if (multi && ! type.isCollectionOrArray())
090         throw new ArgException(pi, "Use of multipart flag on @FormData parameter that is not an array or Collection");
091   }
092
093   @SuppressWarnings({ "rawtypes", "unchecked" })
094   @Override /* RestOpArg */
095   public Object resolve(RestOpSession opSession) throws Exception {
096      RestRequest req = opSession.getRequest();
097      HttpPartParserSession ps = partParser == null ? req.getPartParserSession() : partParser.getPartSession();
098      RequestFormParams rh = req.getFormParams();
099      BeanSession bs = req.getBeanSession();
100      ClassMeta<?> cm = bs.getClassMeta(type.innerType());
101
102      if (multi) {
103         Collection c = cm.isArray() ? list() : (Collection)(cm.canCreateNewInstance() ? cm.newInstance() : new JsonList());
104         rh.getAll(name).stream().map(x -> x.parser(ps).schema(schema).as(cm.getElementType()).orElse(null)).forEach(x -> c.add(x));
105         return cm.isArray() ? ArrayUtils.toArray(c, cm.getElementType().getInnerClass()) : c;
106      }
107
108      if (cm.isMapOrBean() && isOneOf(name, "*", "")) {
109         JsonMap m = new JsonMap();
110         for (RequestFormParam e : rh.getAll())
111            m.put(e.getName(), e.parser(ps).schema(schema == null ? null : schema.getProperty(e.getName())).as(cm.getValueType()).orElse(null));
112         return req.getBeanSession().convertToType(m, cm);
113      }
114
115      return rh.getLast(name).parser(ps).schema(schema).as(type.innerType()).orElse(null);
116   }
117}
