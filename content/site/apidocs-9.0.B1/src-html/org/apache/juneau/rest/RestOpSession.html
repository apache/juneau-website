013package org.apache.juneau.rest;
014
015import java.io.*;
016
017import org.apache.http.*;
018import org.apache.juneau.*;
019import org.apache.juneau.cp.*;
020import org.apache.juneau.http.response.*;
021import org.apache.juneau.rest.converter.*;
022import org.apache.juneau.rest.guard.*;
023import org.apache.juneau.rest.logging.*;
024
025/**
026 * A session for a single HTTP request.
027 *
028 * <p>
029 * This session object gets created by {@link RestSession} once the Java method to be invoked has been determined.
030 *
031 * <ul class='notes'>
032 *    <li class='warn'>This class is not thread safe.
033 * </ul>
034 *
035 * <ul class='seealso'>
036 *    <li class='extlink'>{@source}
037 * </ul>
038 */
039public class RestOpSession extends ContextSession {
040
041
042   //-----------------------------------------------------------------------------------------------------------------
043   // Static
044   //-----------------------------------------------------------------------------------------------------------------
045
046   /**
047    * Static creator.
048    *
049    * @param ctx The context object of the Java method being invoked.
050    * @param session The REST session object creating this object.
051    * @return A new builder.
052    */
053   public static Builder create(RestOpContext ctx, RestSession session) {
054      return new Builder(ctx, session);
055   }
056
057   //-----------------------------------------------------------------------------------------------------------------
058   // Builder
059   //-----------------------------------------------------------------------------------------------------------------
060
061   /**
062    * Builder class.
063    */
064   public static class Builder extends ContextSession.Builder {
065
066      final RestOpContext ctx;
067      final RestSession session;
068
069      /**
070       * Constructor.
071       *
072       * @param ctx The context object of the Java method being invoked.
073       * @param session The REST session object creating this object.
074       */
075      public Builder(RestOpContext ctx, RestSession session) {
076         super(ctx);
077         this.ctx = ctx;
078         this.session = session;
079      }
080
081      /**
082       * Sets the logger to use when logging this call.
083       *
084       * @param value The new value for this setting.  Can be <jk>null</jk>.
085       * @return This object.
086       */
087      public Builder logger(RestLogger value) {
088         session.logger(value);
089         return this;
090      }
091
092      /**
093       * Enables or disabled debug mode on this call.
094       *
095       * @param value The new value for this setting.
096       * @return This object.
097       * @throws IOException Occurs if request content could not be cached into memory.
098       */
099      public Builder debug(boolean value) throws IOException {
100         session.debug(value);
101         return this;
102      }
103
104      @Override /* Session.Builder */
105      public RestOpSession build() {
106         return new RestOpSession(this);
107      }
108   }
109
110   //-----------------------------------------------------------------------------------------------------------------
111   // Instance
112   //-----------------------------------------------------------------------------------------------------------------
113
114   private final RestOpContext ctx;
115   private final RestSession session;
116   private final RestRequest req;
117   private final RestResponse res;
118
119   /**
120    * Constructor.
121    *
122    * @param builder The builder for this object.
123    */
124   protected RestOpSession(Builder builder) {
125      super(builder);
126      ctx = builder.ctx;
127      session = builder.session;
128      try {
129         req = session.getBeanStore().add(RestRequest.class, ctx.createRequest(session));
130         res = session.getBeanStore().add(RestResponse.class, ctx.createResponse(session, req));
131      } catch (RuntimeException e) {
132         throw e;
133      } catch (Exception e) {
134         throw new InternalServerError(e);
135      }
136   }
137
138   @Override /* ContextSession */
139   public RestOpContext getContext() {
140      return ctx;
141   }
142
143   /**
144    * Runs this session.
145    *
146    * <p>
147    * Does the following:
148    * <ol>
149    *    <li>Runs the guards on the method.
150    *    <li>Finds the parameter values to pass to the Java method.
151    *    <li>Invokes the Java method.
152    *    <li>Sets the output and status on the response.
153    *    <li>Calls the converters on the Java method.
154    * </ol>
155    *
156    * @throws Throwable Any throwable can be thrown.
157    */
158   public void run() throws Throwable {
159
160      for (RestGuard guard : ctx.getGuards())
161         if (! guard.guard(req, res))
162            return;
163
164      ctx.getMethodInvoker().invoke(this);
165
166      if (res.hasContent())
167         for (RestConverter converter : ctx.getConverters())
168            res.setContent(converter.convert(req, res.getContent().orElse(null)));
169   }
170
171   /**
172    * Returns the REST request object for this session.
173    *
174    * @return The REST request object for this session.
175    */
176   public RestRequest getRequest() {
177      return req;
178   }
179
180   /**
181    * Returns the REST response object for this session.
182    *
183    * @return The REST response object for this session.
184    */
185   public RestResponse getResponse() {
186      return res;
187   }
188
189   /**
190    * Returns the bean store for this session.
191    *
192    * @return The bean store for this session.
193    */
194   public BeanStore getBeanStore() {
195      return session.getBeanStore();
196   }
197
198   /**
199    * Returns the context of the parent class of this Java method.
200    *
201    * @return The context of the parent class of this Java method.
202    */
203   public RestContext getRestContext() {
204      return session.getContext();
205   }
206
207   /**
208    * Returns the session of the parent class of this Java method.
209    *
210    * @return The session of the parent class of this Java method.
211    */
212   public RestSession getRestSession() {
213      return session;
214   }
215
216   /**
217    * Sets the status of the response.
218    *
219    * @param value The new status.
220    * @return This object.
221    */
222   public RestOpSession status(StatusLine value) {
223      session.status(value);
224      return this;
225   }
226
227   /**
228    * Called at the end of a call to finish any remaining tasks such as flushing buffers and logging the response.
229    *
230    * @return This object.
231    */
232   public RestOpSession finish() {
233      try {
234         res.flushBuffer();
235         req.close();
236      } catch (Exception e) {
237         session.exception(e);
238      }
239      return this;
240   }
241}
