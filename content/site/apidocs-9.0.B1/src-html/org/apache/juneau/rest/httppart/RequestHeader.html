096public class RequestHeader extends RequestHttpPart implements Header {
097
098   private final String value;
099
100   /**
101    * Constructor.
102    *
103    * @param request The request object.
104    * @param name The header name.
105    * @param value The header value.
106    */
107   public RequestHeader(RestRequest request, String name, String value) {
108      super(HEADER, request, name);
109      this.value = value;
110   }
111
112   //------------------------------------------------------------------------------------------------------------------
113   // Retrievers
114   //------------------------------------------------------------------------------------------------------------------
115
116   @Override /* RequestHttpPart */
117   public String getValue() {
118      return value;
119   }
120
121   /**
122    * Returns the value of this header as a {@link BasicCsvHeader}.
123    *
124    * @return The value of this header as a  {@link BasicCsvHeader}, never <jk>null</jk>.
125    */
126   public BasicCsvHeader asCsvHeader() {
127      return new BasicCsvHeader(getName(), getValue());
128   }
129
130   /**
131    * Returns the value of this header as a {@link BasicDateHeader}.

137   }

145      return new BasicEntityTagsHeader(getName(), getValue());
146   }
147
148   /**
149    * Returns the value of this header as a {@link BasicEntityTagHeader}.

155   }
156

158    * Returns the value of this header as a {@link BasicIntegerHeader}.

173   }

176    * Returns the value of this header as a {@link BasicLongHeader}.

178    * @return The value of this header as a {@link BasicLongHeader}, never <jk>null</jk>.

181      return new BasicLongHeader(getName(), getValue());
182   }
183

185    * Returns the value of this header as a {@link BasicStringRangesHeader}.
186    *
187    * @return The value of this header as a {@link BasicStringRangesHeader}, never <jk>null</jk>.

203    * Returns the value of this header as a {@link BasicUriHeader}.

208      return new BasicUriHeader(getName(), getValue());

215   /**
216    * Provides the ability to perform fluent-style assertions on this parameter.
217    *
218    * <h5 class='section'>Examples:</h5>
219    * <p class='bjava'>
220    *    <jv>request</jv>
221    *       .getHeader(<js>"foo"</js>)
222    *       .assertString().contains(<js>"bar"</js>);
223    * </p>
224    *
225    * <p>
226    * The assertion test returns the original object allowing you to chain multiple requests like so:
227    * <p class='bjava'>
228    *    String <jv>foo</jv> = <jv>request</jv>
229    *       .getHeader(<js>"foo"</js>)
230    *       .assertString().contains(<js>"bar"</js>)
231    *       .asString().get();
232    * </p>
233    *
234    * @return A new fluent assertion object.
235    */
236   public FluentStringAssertion<RequestHeader> assertString() {
237      return new FluentStringAssertion<>(orElse(null), this);
238   }
239
240   /**
241    * Provides the ability to perform fluent-style assertions on an integer parameter.
242    *
243    * <h5 class='section'>Examples:</h5>
244    * <p class='bjava'>
245    *    <jv>request</jv>
246    *       .getHeader(<js>"age"</js>)
247    *       .assertInteger().isGreaterThan(1);
248    * </p>
249    *
250    * @return A new fluent assertion object.
251    */
252   public FluentIntegerAssertion<RequestHeader> assertInteger() {
253      return new FluentIntegerAssertion<>(asIntegerPart().asInteger().orElse(null), this);
254   }
255

259    * <h5 class='section'>Examples:</h5>

265    *
266    * @return A new fluent assertion object.
267    */
268   public FluentLongAssertion<RequestHeader> assertLong() {
269      return new FluentLongAssertion<>(asLongPart().asLong().orElse(null), this);
270   }
271
272   /**
273    * Provides the ability to perform fluent-style assertions on a date parameter.
274    *
275    * <h5 class='section'>Examples:</h5>
276    * <p class='bjava'>
277    *    <jv>request</jv>
278    *       .getHeader(<js>"time"</js>)
279    *       .assertDate().isAfterNow();
280    * </p>
281    *
282    * @return A new fluent assertion object.
283    */
284   public FluentZonedDateTimeAssertion<RequestHeader> assertDate() {
285      return new FluentZonedDateTimeAssertion<>(asDatePart().asZonedDateTime().orElse(null), this);
286   }
287
288   /**
289    * Provides the ability to perform fluent-style assertions on comma-separated string parameters.
290    *
291    * <h5 class='section'>Examples:</h5>
292    * <p class='bjava'>
293    *    <jv>request</jv>
294    *       .getHeader(<js>"allow"</js>)
295    *       .assertCsvArray().contains(<js>"GET"</js>);
296    * </p>
297    *
298    * @return A new fluent assertion object.
299    */
300   public FluentListAssertion<String,RequestHeader> assertCsvArray() {
301      return new FluentListAssertion<>(asCsvArrayPart().asList().orElse(null), this);
302   }
303
304   /**
305    * Parses the value.
306    *
307    * @return An array of {@link HeaderElement} entries, may be empty, but is never <jk>null</jk>.
308    * @throws BasicHttpException In case of a parsing error.
309    */
310   @Override /* Header */
311   public HeaderElement[] getElements() throws BasicHttpException {
312      return new HeaderElement[0];
313   }
314
315   @Override /* Object */
316   public String toString() {
317      return getName() + ": " + getValue();
318   }
