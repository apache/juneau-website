013package org.apache.juneau;
014
015import java.lang.reflect.*;
016import java.util.*;
017import java.util.function.*;
018
019import org.apache.juneau.reflect.*;
020
021/**
022 * Represents a simple settable value.
023 *
024 * <p>
025 * Similar to an {@link Optional} but mutable.
026 *
027 * <ul class='notes'>
028 *    <li class='warn'>This class is not thread safe.
029 * </ul>
030 *
031 * <ul class='seealso'>
032 *    <li class='extlink'>{@source}
033 * </ul>
034 *
035 * @param <T> The value type.
036 */
037public class Value<T> {
038
039   //-----------------------------------------------------------------------------------------------------------------
040   // Static
041   //-----------------------------------------------------------------------------------------------------------------
042
043   /**
044    * Static creator.
045    *
046    * @param <T> The value type.
047    * @param object The object being wrapped.
048    * @return A new {@link Value} object.
049    */
050   public static <T> Value<T> of(T object) {
051      return new Value<>(object);
052   }
053
054   /**
055    * Static creator.
056    *
057    * @param <T> The value type.
058    * @return An empty {@link Value} object.
059    */
060   public static <T> Value<T> empty() {
061      return new Value<>(null);
062   }
063
064   /**
065    * Returns the generic parameter type of the Value type.
066    *
067    * @param t The type to find the parameter type of.
068    * @return The parameter type of the value, or <jk>null</jk> if the type is not a subclass of <c>Value</c>.
069    */
070   public static Type getParameterType(Type t) {
071      if (t instanceof ParameterizedType) {
072         ParameterizedType pt = (ParameterizedType)t;
073         if (pt.getRawType() == Value.class) {
074            Type[] ta = pt.getActualTypeArguments();
075            if (ta.length > 0)
076               return ta[0];
077         }
078      } else if (t instanceof Class) {
079         Class<?> c = (Class<?>)t;
080         if (Value.class.isAssignableFrom(c)) {
081            return ClassInfo.of(c).getParameterType(0, Value.class);
082         }
083      }
084
085      return null;
086   }
087
088   /**
089    * Returns the unwrapped type.
090    *
091    * @param t The type to unwrap.
092    * @return The unwrapped type, or the same type if the type isn't {@link Value}.
093    */
094   public static Type unwrap(Type t) {
095      Type x = getParameterType(t);
096      return x != null ? x : t;
097   }
098
099   /**
100    * Convenience method for checking if the specified type is this class.
101    *
102    * @param t The type to check.
103    * @return <jk>true</jk> if the specified type is this class.
104    */
105   public static boolean isType(Type t) {
106      return
107         (t instanceof ParameterizedType && ((ParameterizedType)t).getRawType() == Value.class)
108         || (t instanceof Class && Value.class.isAssignableFrom((Class<?>)t));
109   }
110
111   //-----------------------------------------------------------------------------------------------------------------
112   // Implementation
113   //-----------------------------------------------------------------------------------------------------------------
114
115
116   private T t;
117   private ValueListener<T> listener;
118
119   /**
120    * Constructor.
121    */
122   public Value() {}
123
124   /**
125    * Constructor.
126    *
127    * @param t Initial value.
128    */
129   public Value(T t) {
130      set(t);
131   }
132
133   /**
134    * Adds a listener for this value.
135    *
136    * @param listener The new listener for this value.
137    * @return This object.
138    */
139   public Value<T> listener(ValueListener<T> listener) {
140      this.listener = listener;
141      return this;
142   }
143
144   /**
145    * Sets the value.
146    *
147    * @param t The new value.
148    * @return This object.
149    */
150   public Value<T> set(T t) {
151      this.t = t;
152      if (listener != null)
153         listener.onSet(t);
154      return this;
155   }
156
157   /**
158    * Sets the value if it's not already set.
159    *
160    * @param t The new value.
161    * @return This object.
162    */
163   public Value<T> setIfEmpty(T t) {
164      if (isEmpty())
165         set(t);
166      return this;
167   }
168
169   /**
170    * Returns the value.
171    *
172    * @return The value, or <jk>null</jk> if it is not set.
173    */
174   public T get() {
175      return t;
176   }
177
178   /**
179    * Returns the value and then unsets it.
180    *
181    * @return The value before it was unset.
182    */
183   public T getAndUnset() {
184      T t2 = t;
185      t = null;
186      return t2;
187   }
188
189   /**
190    * Returns <jk>true</jk> if the value is set.
191    *
192    * @return <jk>true</jk> if the value is set.
193    */
194   public boolean isPresent() {
195      return get() != null;
196   }
197
198   /**
199    * If a value is present, invoke the specified consumer with the value, otherwise do nothing.
200    *
201    * @param consumer Block to be executed if a value is present.
202    */
203   public void ifPresent(Consumer<? super T> consumer) {
204      if (t != null)
205         consumer.accept(t);
206   }
207
208   /**
209    * Applies a mapping function against the contents of this value.
210    *
211    * @param <T2> The mapped value type.
212    * @param mapper The mapping function.
213    * @return The mapped value.
214    */
215   public <T2> Value<T2> map(Function<? super T, T2> mapper) {
216      if (t != null)
217         return Value.of(mapper.apply(t));
218      return Value.empty();
219   }
220
221   /**
222    * Returns the contents of this value or the default value if <jk>null</jk>.
223    *
224    * @param def The default value.
225    * @return The contents of this value or the default value if <jk>null</jk>.
226    */
227   public T orElse(T def) {
228      return t == null ? def : t;
229   }
230   /**
231    * Returns <jk>true</jk> if the value is empty.
232    *
233    * @return <jk>true</jk> if the value is empty.
234    */
235   public boolean isEmpty() {
236      return t == null;
237   }
238
239   /**
240    * Return the value if present, otherwise invoke {@code other} and return
241    * the result of that invocation.
242    *
243    * @param other a {@code Supplier} whose result is returned if no value
244    * is present
245    * @return the value if present otherwise the result of {@code other.get()}
246    * @throws NullPointerException if value is not present and {@code other} is
247    * null
248    */
249   public T orElseGet(Supplier<? extends T> other) {
250      return t != null ? t : other.get();
251   }
252
253   /**
254    * Return the contained value, if present, otherwise throw an exception
255    * to be created by the provided supplier.
256    *
257    * @param <X> The exception type.
258    * @param exceptionSupplier The supplier which will return the exception to
259    * be thrown
260    * @return the present value
261    * @throws X if there is no value present
262    * @throws NullPointerException if no value is present and
263    * {@code exceptionSupplier} is null
264    */
265   public <X extends Throwable> T orElseThrow(Supplier<? extends X> exceptionSupplier) throws X {
266      if (t != null)
267         return t;
268      throw exceptionSupplier.get();
269   }
270
271   @Override /* Object */
272   public String toString() {
273      return "Value("+t+")";
274   }
275}
