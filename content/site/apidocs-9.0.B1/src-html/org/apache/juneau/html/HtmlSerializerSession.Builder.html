231   /**
232    * Converts the specified output target object to an {@link HtmlWriter}.
233    *
234    * @param out The output target object.
235    * @return The output target object wrapped in an {@link HtmlWriter}.
236    * @throws IOException Thrown by underlying stream.
237    */
238   protected final HtmlWriter getHtmlWriter(SerializerPipe out) throws IOException {
239      Object output = out.getRawOutput();
240      if (output instanceof HtmlWriter)
241         return (HtmlWriter)output;
242      HtmlWriter w = new HtmlWriter(out.getWriter(), isUseWhitespace(), getMaxIndent(), isTrimStrings(), getQuoteChar(),
243         getUriResolver());
244      out.setWriter(w);
245      return w;
246   }

248   /**
249    * Returns <jk>true</jk> if the specified object is a URL.
250    *
251    * @param cm The ClassMeta of the object being serialized.
252    * @param pMeta
253    *    The property metadata of the bean property of the object.
254    *    Can be <jk>null</jk> if the object isn't from a bean property.
255    * @param o The object.
256    * @return <jk>true</jk> if the specified object is a URL.
257    */
258   public boolean isUri(ClassMeta<?> cm, BeanPropertyMeta pMeta, Object o) {
259      if (cm.isUri() || (pMeta != null && pMeta.isUri()))
260         return true;
261      if (isDetectLinksInStrings() && o instanceof CharSequence && urlPattern.matcher(o.toString()).matches())
262         return true;
263      return false;
264   }

266   /**
267    * Returns the anchor text to use for the specified URL object.
268    *
269    * @param pMeta
270    *    The property metadata of the bean property of the object.
271    *    Can be <jk>null</jk> if the object isn't from a bean property.
272    * @param o The URL object.
273    * @return The anchor text to use for the specified URL object.
274    */
275   public String getAnchorText(BeanPropertyMeta pMeta, Object o) {
276      String s = o.toString();
277      if (isDetectLabelParameters()) {
278         Matcher m = labelPattern.matcher(s);
279         if (m.find())
280            return urlDecode(m.group(1));
281      }
282      switch (getUriAnchorText()) {
283         case LAST_TOKEN:
284            s = resolveUri(s);
285            if (s.indexOf('/') != -1)
286               s = s.substring(s.lastIndexOf('/')+1);
287            if (s.indexOf('?') != -1)
288               s = s.substring(0, s.indexOf('?'));
289            if (s.indexOf('#') != -1)
290               s = s.substring(0, s.indexOf('#'));
291            if (s.isEmpty())
292               s = "/";
293            return urlDecode(s);
294         case URI_ANCHOR:
295            if (s.indexOf('#') != -1)
296               s = s.substring(s.lastIndexOf('#')+1);
297            return urlDecode(s);
298         case PROPERTY_NAME:
299            return pMeta == null ? s : pMeta.getName();
300         case URI:
301            return resolveUri(s);
302         case CONTEXT_RELATIVE:
303            return relativizeUri("context:/", s);
304         case SERVLET_RELATIVE:
305            return relativizeUri("servlet:/", s);
306         case PATH_RELATIVE:
307            return relativizeUri("request:/", s);
308         default /* TO_STRING */:
309            return s;
310      }
311   }

337   @SuppressWarnings({ "rawtypes" })
338   @Override /* XmlSerializerSession */
339   protected ContentResult serializeAnything(
340         XmlWriter out,
341         Object o,
342         ClassMeta<?> eType,
343         String keyName,
344         String elementName,
345         Namespace elementNamespace,
346         boolean addNamespaceUris,
347         XmlFormat format,
348         boolean isMixed,
349         boolean preserveWhitespace,
350         BeanPropertyMeta pMeta) throws SerializeException {
351
352      // If this is a bean, then we want to serialize it as HTML unless it's @Html(format=XML).
353      ClassMeta<?> type = push2(elementName, o, eType);
354      pop();
355
356      if (type == null)
357         type = object();
358      else if (type.isDelegate())
359         type = ((Delegate)o).getClassMeta();
360      ObjectSwap swap = type.getSwap(this);
361      if (swap != null) {
362         o = swap(swap, o);
363         type = swap.getSwapClassMeta(this);
364         if (type.isObject())
365            type = getClassMetaForObject(o);
366      }
367
368      HtmlClassMeta cHtml = getHtmlClassMeta(type);
369
370      if (type.isMapOrBean() && ! cHtml.isXml())
371         return serializeAnything(out, o, eType, elementName, pMeta, 0, false, false);
372
373      return super.serializeAnything(out, o, eType, keyName, elementName, elementNamespace, addNamespaceUris, format, isMixed, preserveWhitespace, pMeta);
374   }
375   /**
376    * Serialize the specified object to the specified writer.
377    *
378    * @param out The writer.
379    * @param o The object to serialize.
380    * @param eType The expected type of the object if this is a bean property.
381    * @param name
382    *    The attribute name of this object if this object was a field in a JSON object (i.e. key of a
383    *    {@link java.util.Map.Entry} or property name of a bean).
384    * @param pMeta The bean property being serialized, or <jk>null</jk> if we're not serializing a bean property.
385    * @param xIndent The current indentation value.
386    * @param isRoot <jk>true</jk> if this is the root element of the document.
387    * @param nlIfElement <jk>true</jk> if we should add a newline to the output before serializing only if the object is an element and not text.
388    * @return The type of content encountered.  Either simple (no whitespace) or normal (elements with whitespace).
389    * @throws SerializeException Generic serialization error occurred.
390    */
391   @SuppressWarnings({ "rawtypes", "unchecked" })
392   protected ContentResult serializeAnything(XmlWriter out, Object o,
393         ClassMeta<?> eType, String name, BeanPropertyMeta pMeta, int xIndent, boolean isRoot, boolean nlIfElement) throws SerializeException {
394
395      ClassMeta<?> aType = null;       // The actual type
396      ClassMeta<?> wType = null;     // The wrapped type (delegate)
397      ClassMeta<?> sType = object();   // The serialized type
398
399      if (eType == null)
400         eType = object();
401
402      aType = push2(name, o, eType);
403
404      // Handle recursion
405      if (aType == null) {
406         o = null;
407         aType = object();
408      }
409
410      // Handle Optional<X>
411      if (isOptional(aType)) {
412         o = getOptionalValue(o);
413         eType = getOptionalType(eType);
414         aType = getClassMetaForObject(o, object());
415      }
416
417      indent += xIndent;
418
419      ContentResult cr = CR_ELEMENTS;
420
421      // Determine the type.
422      if (o == null || (aType.isChar() && ((Character)o).charValue() == 0)) {
423         out.tag("null");
424         cr = ContentResult.CR_MIXED;
425
426      } else {
427
428         if (aType.isDelegate()) {
429            wType = aType;
430            aType = ((Delegate)o).getClassMeta();
431         }
432
433         sType = aType;
434
435         String typeName = null;
436         if (isAddBeanTypes() && ! eType.equals(aType))
437            typeName = aType.getDictionaryName();
438
439         // Swap if necessary
440         ObjectSwap swap = aType.getSwap(this);
441         if (swap != null) {
442            o = swap(swap, o);
443            sType = swap.getSwapClassMeta(this);
444
445            // If the getSwapClass() method returns Object, we need to figure out
446            // the actual type now.
447            if (sType.isObject())
448               sType = getClassMetaForObject(o);
449         }
450
451         // Handle the case where we're serializing a raw stream.
452         if (sType.isReader() || sType.isInputStream()) {
453            pop();
454            indent -= xIndent;
455            if (sType.isReader())
456               pipe((Reader)o, out, SerializerSession::handleThrown);
457            else
458               pipe((InputStream)o, out, SerializerSession::handleThrown);
459            return ContentResult.CR_MIXED;
460         }
461
462         HtmlClassMeta cHtml = getHtmlClassMeta(sType);
463         HtmlBeanPropertyMeta bpHtml = getHtmlBeanPropertyMeta(pMeta);
464
465         HtmlRender render = firstNonNull(bpHtml.getRender(), cHtml.getRender());
466
467         if (render != null) {
468            Object o2 = render.getContent(this, o);
469            if (o2 != o) {
470               indent -= xIndent;
471               pop();
472               out.nl(indent);
473               return serializeAnything(out, o2, null, typeName, null, xIndent, false, false);
474            }
475         }
476
477         if (cHtml.isXml() || bpHtml.isXml()) {
478            pop();
479            indent++;
480            if (nlIfElement)
481               out.nl(0);
482            super.serializeAnything(out, o, null, null, null, null, false, XmlFormat.MIXED, false, false, null);
483            indent -= xIndent+1;
484            return cr;
485
486         } else if (cHtml.isPlainText() || bpHtml.isPlainText()) {
487            out.w(o == null ? "null" : o.toString());
488            cr = CR_MIXED;
489
490         } else if (o == null || (sType.isChar() && ((Character)o).charValue() == 0)) {
491            out.tag("null");
492            cr = CR_MIXED;
493
494         } else if (sType.isNumber()) {
495            if (eType.isNumber() && ! isRoot)
496               out.append(o);
497            else
498               out.sTag("number").append(o).eTag("number");
499            cr = CR_MIXED;
500
501         } else if (sType.isBoolean()) {
502            if (eType.isBoolean() && ! isRoot)
503               out.append(o);
504            else
505               out.sTag("boolean").append(o).eTag("boolean");
506            cr = CR_MIXED;
507
508         } else if (sType.isMap() || (wType != null && wType.isMap())) {
509            out.nlIf(! isRoot, xIndent+1);
510            if (o instanceof BeanMap)
511               serializeBeanMap(out, (BeanMap)o, eType, pMeta);
512            else
513               serializeMap(out, (Map)o, sType, eType.getKeyType(), eType.getValueType(), typeName, pMeta);
514
515         } else if (sType.isBean()) {
516            BeanMap m = toBeanMap(o);
517            if (aType.hasAnnotation(HtmlLink.class)) {
518               Value<String> uriProperty = Value.empty(), nameProperty = Value.empty();
519               aType.forEachAnnotation(HtmlLink.class, x -> isNotEmpty(x.uriProperty()), x -> uriProperty.set(x.uriProperty()));
520               aType.forEachAnnotation(HtmlLink.class, x -> isNotEmpty(x.nameProperty()), x -> nameProperty.set(x.nameProperty()));
521               Object urlProp = m.get(uriProperty.orElse(""));
522               Object nameProp = m.get(nameProperty.orElse(""));
523
524               out.oTag("a").attrUri("href", urlProp).w('>').text(nameProp).eTag("a");
525               cr = CR_MIXED;
526            } else {
527               out.nlIf(! isRoot, xIndent+2);
528               serializeBeanMap(out, m, eType, pMeta);
529            }
530
531         } else if (sType.isCollection() || sType.isArray() || (wType != null && wType.isCollection())) {
532            out.nlIf(! isRoot, xIndent+1);
533            serializeCollection(out, o, sType, eType, name, pMeta);
534
535         } else if (isUri(sType, pMeta, o)) {
536            String label = getAnchorText(pMeta, o);
537            out.oTag("a").attrUri("href", o).w('>');
538            out.text(label);
539            out.eTag("a");
540            cr = CR_MIXED;
541
542         } else {
543            if (isRoot)
544               out.sTag("string").text(toString(o)).eTag("string");
545            else
546               out.text(toString(o));
547            cr = CR_MIXED;
548         }
549      }
550      pop();
551      indent -= xIndent;
552      return cr;
553   }

555   @SuppressWarnings({ "rawtypes", "unchecked" })
556   private void serializeMap(XmlWriter out, Map m, ClassMeta<?> sType,
557         ClassMeta<?> eKeyType, ClassMeta<?> eValueType, String typeName, BeanPropertyMeta ppMeta) throws SerializeException {
558
559      ClassMeta<?> keyType = eKeyType == null ? string() : eKeyType;
560      ClassMeta<?> valueType = eValueType == null ? object() : eValueType;
561      ClassMeta<?> aType = getClassMetaForObject(m);       // The actual type
562      HtmlClassMeta cHtml = getHtmlClassMeta(aType);
563      HtmlBeanPropertyMeta bpHtml = getHtmlBeanPropertyMeta(ppMeta);
564
565      int i = indent;
566
567      out.oTag(i, "table");
568
569      if (typeName != null && ppMeta != null && ppMeta.getClassMeta() != aType)
570         out.attr(getBeanTypePropertyName(sType), typeName);
571
572      out.append(">").nl(i+1);
573      if (isAddKeyValueTableHeaders() && ! (cHtml.isNoTableHeaders() || bpHtml.isNoTableHeaders())) {
574         out.sTag(i+1, "tr").nl(i+2);
575         out.sTag(i+2, "th").append("key").eTag("th").nl(i+3);
576         out.sTag(i+2, "th").append("value").eTag("th").nl(i+3);
577         out.ie(i+1).eTag("tr").nl(i+2);
578      }
579
580      forEachEntry(m, x -> serializeMapEntry(out, x, keyType, valueType, i, ppMeta));
581
582      out.ie(i).eTag("table").nl(i);
583   }

585   @SuppressWarnings("rawtypes")
586   private void serializeMapEntry(XmlWriter out, Map.Entry e, ClassMeta<?> keyType, ClassMeta<?> valueType, int i, BeanPropertyMeta ppMeta) throws SerializeException {
587      Object key = generalize(e.getKey(), keyType);
588      Object value = null;
589      try {
590         value = e.getValue();
591      } catch (StackOverflowError t) {
592         throw t;
593      } catch (Throwable t) {
594         onError(t, "Could not call getValue() on property ''{0}'', {1}", e.getKey(), t.getLocalizedMessage());
595      }
596
597      String link = getLink(ppMeta);
598      String style = getStyle(this, ppMeta, value);
599
600      out.sTag(i+1, "tr").nl(i+2);
601      out.oTag(i+2, "td");
602      if (style != null)
603         out.attr("style", style);
604      out.cTag();
605      if (link != null)
606         out.oTag(i+3, "a").attrUri("href", link.replace("{#}", stringify(value))).cTag();
607      ContentResult cr = serializeAnything(out, key, keyType, null, null, 2, false, false);
608      if (link != null)
609         out.eTag("a");
610      if (cr == CR_ELEMENTS)
611         out.i(i+2);
612      out.eTag("td").nl(i+2);
613      out.sTag(i+2, "td");
614      cr = serializeAnything(out, value, valueType, (key == null ? "_x0000_" : toString(key)), null, 2, false, true);
615      if (cr == CR_ELEMENTS)
616         out.ie(i+2);
617      out.eTag("td").nl(i+2);
618      out.ie(i+1).eTag("tr").nl(i+1);
619
620   }

622   private void serializeBeanMap(XmlWriter out, BeanMap<?> m, ClassMeta<?> eType, BeanPropertyMeta ppMeta) throws SerializeException {
623
624      HtmlClassMeta cHtml = getHtmlClassMeta(m.getClassMeta());
625      HtmlBeanPropertyMeta bpHtml = getHtmlBeanPropertyMeta(ppMeta);
626
627      int i = indent;
628
629      out.oTag(i, "table");
630
631      String typeName = m.getMeta().getDictionaryName();
632      if (typeName != null && eType != m.getClassMeta())
633         out.attr(getBeanTypePropertyName(m.getClassMeta()), typeName);
634
635      out.w('>').nl(i);
636      if (isAddKeyValueTableHeaders() && ! (cHtml.isNoTableHeaders() || bpHtml.isNoTableHeaders())) {
637         out.sTag(i+1, "tr").nl(i+1);
638         out.sTag(i+2, "th").append("key").eTag("th").nl(i+2);
639         out.sTag(i+2, "th").append("value").eTag("th").nl(i+2);
640         out.ie(i+1).eTag("tr").nl(i+1);
641      }
642
643      Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
644
645      m.forEachValue(checkNull, (pMeta,key,value,thrown) -> {
646         ClassMeta<?> cMeta = pMeta.getClassMeta();
647
648         if (thrown != null)
649            onBeanGetterException(pMeta, thrown);
650
651         if (canIgnoreValue(cMeta, key, value))
652            return;
653
654         String link = null, anchorText = null;
655         if (! cMeta.isCollectionOrArray()) {
656            link = m.resolveVars(getLink(pMeta));
657            anchorText = m.resolveVars(getAnchorText(pMeta));
658         }
659
660         if (anchorText != null)
661            value = anchorText;
662
663         out.sTag(i+1, "tr").nl(i+1);
664         out.sTag(i+2, "td").text(key).eTag("td").nl(i+2);
665         out.oTag(i+2, "td");
666         String style = getStyle(this, pMeta, value);
667         if (style != null)
668            out.attr("style", style);
669         out.cTag();
670
671         try {
672            if (link != null)
673               out.oTag(i+3, "a").attrUri("href", link).cTag();
674            ContentResult cr = serializeAnything(out, value, cMeta, key, pMeta, 2, false, true);
675            if (cr == CR_ELEMENTS)
676               out.i(i+2);
677            if (link != null)
678               out.eTag("a");
679         } catch (SerializeException e) {
680            throw e;
681         } catch (Error e) {
682            throw e;
683         } catch (Throwable e) {
684            onBeanGetterException(pMeta, e);
685         }
686         out.eTag("td").nl(i+2);
687         out.ie(i+1).eTag("tr").nl(i+1);
688      });
689
690      out.ie(i).eTag("table").nl(i);
691   }

693   @SuppressWarnings({ "rawtypes", "unchecked" })
694   private void serializeCollection(XmlWriter out, Object in, ClassMeta<?> sType, ClassMeta<?> eType, String name, BeanPropertyMeta ppMeta) throws SerializeException {
695
696      HtmlClassMeta cHtml = getHtmlClassMeta(sType);
697      HtmlBeanPropertyMeta bpHtml = getHtmlBeanPropertyMeta(ppMeta);
698
699      Collection c = (sType.isCollection() ? (Collection)in : toList(sType.getInnerClass(), in));
700
701      boolean isCdc = cHtml.isHtmlCdc() || bpHtml.isHtmlCdc();
702      boolean isSdc = cHtml.isHtmlSdc() || bpHtml.isHtmlSdc();
703      boolean isDc = isCdc || isSdc;
704
705      int i = indent;
706      if (c.isEmpty()) {
707         out.appendln(i, "<ul></ul>");
708         return;
709      }
710
711      String type2 = null;
712      if (sType != eType)
713         type2 = sType.getDictionaryName();
714      if (type2 == null)
715         type2 = "array";
716
717      c = sort(c);
718
719      String btpn = getBeanTypePropertyName(eType);
720
721      // Look at the objects to see how we're going to handle them.  Check the first object to see how we're going to
722      // handle this.
723      // If it's a map or bean, then we'll create a table.
724      // Otherwise, we'll create a list.
725      Object[] th = getTableHeaders(c, bpHtml);
726
727      if (th != null) {
728
729         out.oTag(i, "table").attr(btpn, type2).w('>').nl(i+1);
730         if (th.length > 0) {
731            out.sTag(i+1, "tr").nl(i+2);
732            for (Object key : th) {
733               out.sTag(i+2, "th");
734               out.text(convertToType(key, String.class));
735               out.eTag("th").nl(i+2);
736            }
737            out.ie(i+1).eTag("tr").nl(i+1);
738         } else {
739            th = null;
740         }
741
742         for (Object o : c) {
743            ClassMeta<?> cm = getClassMetaForObject(o);
744
745            if (cm != null && cm.getSwap(this) != null) {
746               ObjectSwap swap = cm.getSwap(this);
747               o = swap(swap, o);
748               cm = swap.getSwapClassMeta(this);
749            }
750
751            out.oTag(i+1, "tr");
752            String typeName = (cm == null ? null : cm.getDictionaryName());
753            String typeProperty = getBeanTypePropertyName(cm);
754
755            if (typeName != null && eType.getElementType() != cm)
756               out.attr(typeProperty, typeName);
757            out.cTag().nl(i+2);
758
759            if (cm == null) {
760               out.i(i+2);
761               serializeAnything(out, o, null, null, null, 1, false, false);
762               out.nl(0);
763
764            } else if (cm.isMap() && ! (cm.isBeanMap())) {
765               Map m2 = sort((Map)o);
766
767               if (th == null)
768                  th = m2.keySet().toArray(new Object[m2.size()]);
769
770               for (Object k : th) {
771                  out.sTag(i+2, "td");
772                  ContentResult cr = serializeAnything(out, m2.get(k), eType.getElementType(), toString(k), null, 2, false, true);
773                  if (cr == CR_ELEMENTS)
774                     out.i(i+2);
775                  out.eTag("td").nl(i+2);
776               }
777            } else {
778               BeanMap m2 = toBeanMap(o);
779
780               if (th == null)
781                  th = m2.keySet().toArray(new Object[m2.size()]);
782
783               for (Object k : th) {
784                  BeanMapEntry p = m2.getProperty(toString(k));
785                  BeanPropertyMeta pMeta = p.getMeta();
786                  if (pMeta.canRead()) {
787                     Object value = p.getValue();
788
789                     String link = null, anchorText = null;
790                     if (! pMeta.getClassMeta().isCollectionOrArray()) {
791                        link = m2.resolveVars(getLink(pMeta));
792                        anchorText = m2.resolveVars(getAnchorText(pMeta));
793                     }
794
795                     if (anchorText != null)
796                        value = anchorText;
797
798                     String style = getStyle(this, pMeta, value);
799                     out.oTag(i+2, "td");
800                     if (style != null)
801                        out.attr("style", style);
802                     out.cTag();
803                     if (link != null)
804                        out.oTag("a").attrUri("href", link).cTag();
805                     ContentResult cr = serializeAnything(out, value, pMeta.getClassMeta(), p.getKey().toString(), pMeta, 2, false, true);
806                     if (cr == CR_ELEMENTS)
807                        out.i(i+2);
808                     if (link != null)
809                        out.eTag("a");
810                     out.eTag("td").nl(i+2);
811                  }
812               }
813            }
814            out.ie(i+1).eTag("tr").nl(i+1);
815         }
816         out.ie(i).eTag("table").nl(i);
817
818      } else {
819         out.oTag(i, isDc ? "p" : "ul");
820         if (! type2.equals("array"))
821            out.attr(btpn, type2);
822         out.w('>').nl(i+1);
823         boolean isFirst = true;
824         for (Object o : c) {
825            if (isDc && ! isFirst)
826               out.append(isCdc ? ", " : " ");
827            if (! isDc)
828               out.oTag(i+1, "li");
829            String style = getStyle(this, ppMeta, o);
830            String link = getLink(ppMeta);
831            if (style != null && ! isDc)
832               out.attr("style", style);
833            if (! isDc)
834               out.cTag();
835            if (link != null)
836               out.oTag(i+2, "a").attrUri("href", link.replace("{#}", stringify(o))).cTag();
837            ContentResult cr = serializeAnything(out, o, eType.getElementType(), name, null, 1, false, true);
838            if (link != null)
839               out.eTag("a");
840            if (cr == CR_ELEMENTS)
841               out.ie(i+1);
842            if (! isDc)
843               out.eTag("li").nl(i+1);
844            isFirst = false;
845         }
846         out.ie(i).eTag(isDc ? "p" : "ul").nl(i);
847      }

875   /*
876    * Returns the table column headers for the specified collection of objects.
877    * Returns null if collection should not be serialized as a 2-dimensional table.
878    * Returns an empty array if it should be treated as a table but without headers.
879    * 2-dimensional tables are used for collections of objects that all have the same set of property names.
880    */
881   @SuppressWarnings({ "rawtypes", "unchecked" })
882   private Object[] getTableHeaders(Collection c, HtmlBeanPropertyMeta bpHtml) throws SerializeException  {
883
884      if (c.size() == 0)
885         return null;
886
887      c = sort(c);
888
889      Object o1 = null;
890      for (Object o : c)
891         if (o != null) {
892            o1 = o;
893            break;
894         }
895      if (o1 == null)
896         return null;
897
898      ClassMeta<?> cm1 = getClassMetaForObject(o1);
899
900      ObjectSwap swap = cm1.getSwap(this);
901      o1 = swap(swap, o1);
902      if (swap != null)
903         cm1 = swap.getSwapClassMeta(this);
904
905      if (cm1 == null || ! cm1.isMapOrBean() || cm1.hasAnnotation(HtmlLink.class))
906         return null;
907
908      HtmlClassMeta cHtml = getHtmlClassMeta(cm1);
909
910      if (cHtml.isNoTables() || bpHtml.isNoTables() || cHtml.isXml() || bpHtml.isXml() || canIgnoreValue(cm1, null, o1))
911         return null;
912
913      if (cHtml.isNoTableHeaders() || bpHtml.isNoTableHeaders())
914         return new Object[0];
915
916      // If it's a non-bean map, only use table if all entries are also maps.
917      if (cm1.isMap() && ! cm1.isBeanMap()) {
918
919         Set<Object> set = CollectionUtils.set();
920         for (Object o : c) {
921            o = swap(swap, o);
922            if (! canIgnoreValue(cm1, null, o)) {
923               if (! cm1.isInstance(o))
924                  return null;
925               forEachEntry((Map)o, x -> set.add(x.getKey()));
926            }
927         }
928         return set.toArray(new Object[set.size()]);
929      }
930
931      // Must be a bean or BeanMap.
932      for (Object o : c) {
933         o = swap(swap, o);
934         if (! canIgnoreValue(cm1, null, o)) {
935            if (! cm1.isInstance(o))
936               return null;
937         }
938      }
939
940      BeanMap<?> bm = toBeanMap(o1);
941      return bm.keySet().toArray(new String[bm.size()]);
942   }
