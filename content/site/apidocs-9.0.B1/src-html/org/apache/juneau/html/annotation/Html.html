013package org.apache.juneau.html.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020import org.apache.juneau.annotation.*;
021import org.apache.juneau.html.*;
022
023/**
024 * Annotation that can be applied to classes, fields, and methods to tweak how they are handled by {@link HtmlSerializer}.
025 *
026 * <p>
027 * Can be used in the following locations:
028 * <ul>
029 *    <li>Marshalled classes/methods/fields.
030 *    <li><ja>@Rest</ja>-annotated classes and <ja>@RestOp</ja>-annotated methods when an {@link #on()} value is specified.
031 * </ul>
032 *
033 * <ul class='seealso'>
034 *    <li class='link'>{@doc jm.HtmlAnnotation}
035 *    <li class='link'>{@doc jm.HtmlDetails}
036 *    <li class='extlink'>{@source}
037 * </ul>
038 */
039@Documented
040@Target({TYPE,FIELD,METHOD})
041@Retention(RUNTIME)
042@Inherited
043@Repeatable(HtmlAnnotation.Array.class)
044@ContextApply(HtmlAnnotation.Apply.class)
045public @interface Html {
046
047   /**
048    * Use the specified anchor text when serializing a URI.
049    *
050    * <p>
051    * The text can contain any bean property values resolved through variables of the form <js>"{property-name}"</js>.
052    *
053    * <h5 class='figure'>Example:</h5>
054    * <p class='bjava'>
055    *    <jc>// Produces &lt;a href&#61;'...'&gt;drive&lt;/a&gt; when serialized to HTML.</jc>
056    *    <ja>@Html</ja>(anchorText=<js>"drive"</js>)
057    *    <ja>@URI</ja> <jc>// Treat property as a URL</jc>
058    *    <jk>public</jk> String getDrive() {...}
059    * </p>
060    *
061    * <p>
062    * This overrides the behavior specified by {@link org.apache.juneau.html.HtmlSerializer.Builder#uriAnchorText(AnchorText)}.
063    *
064    * @return The annotation value.
065    */
066   String anchorText() default "";
067
068   /**
069    * Specifies what format to use for the HTML element.
070    *
071    * @return The annotation value.
072    */
073   HtmlFormat format() default HtmlFormat.HTML;
074
075   /**
076    * Adds a hyperlink to a bean property when rendered as HTML.
077    *
078    * <p>
079    * The text can contain any bean property values resolved through variables of the form <js>"{property-name}"</js>.
080    *
081    * <p>
082    * The URLs can be any of the following forms:
083    * <ul>
084    *    <li>Absolute - e.g. <js>"http://host:123/myContext/myServlet/myPath"</js>
085    *    <li>Context-root-relative - e.g. <js>"/myContext/myServlet/myPath"</js>
086    *    <li>Context-relative - e.g. <js>"context:/myServlet/myPath"</js>
087    *    <li>Servlet-relative - e.g. <js>"servlet:/myPath"</js>
088    *    <li>Path-info-relative - e.g. <js>"myPath"</js>
089    * </ul>
090    *
091    * <h5 class='figure'>Example:</h5>
092    * <p class='bjava'>
093    *    <jk>public class</jk> FileSpace {
094    *
095    *       <ja>@Html</ja>(link=<js>"servlet:/drive/{drive}"</js>)
096    *       <jk>public</jk> String getDrive() {
097    *          ...;
098    *       }
099    *    }
100    * </p>
101    *
102    * @return The annotation value.
103    */
104   String link() default "";
105
106   /**
107    * When <jk>true</jk>, don't add headers to tables.
108    *
109    * <ul class='seealso'>
110    *    <li class='jm'>{@link org.apache.juneau.html.HtmlSerializer.Builder#addKeyValueTableHeaders()}
111    * </ul>
112    *
113    * @return The annotation value.
114    */
115   boolean noTableHeaders() default false;
116
117   /**
118    * When <jk>true</jk>, collections of beans should be rendered as trees instead of tables.
119    *
120    * <p>
121    * Default is <jk>false</jk>.
122    *
123    * @return The annotation value.
124    */
125   boolean noTables() default false;
126
127   /**
128    * Dynamically apply this annotation to the specified classes/methods/fields.
129    *
130    * <p>
131    * Used in conjunction with {@link org.apache.juneau.BeanContext.Builder#applyAnnotations(Class...)} to dynamically apply an annotation to an existing class/method/field.
132    * It is ignored when the annotation is applied directly to classes/methods/fields.
133    *
134    * <h5 class='section'>Valid patterns:</h5>
135    * <ul class='spaced-list'>
136    *  <li>Classes:
137    *       <ul>
138    *          <li>Fully qualified:
139    *             <ul>
140    *                <li><js>"com.foo.MyClass"</js>
141    *             </ul>
142    *          <li>Fully qualified inner class:
143    *             <ul>
144    *                <li><js>"com.foo.MyClass$Inner1$Inner2"</js>
145    *             </ul>
146    *          <li>Simple:
147    *             <ul>
148    *                <li><js>"MyClass"</js>
149    *             </ul>
150    *          <li>Simple inner:
151    *             <ul>
152    *                <li><js>"MyClass$Inner1$Inner2"</js>
153    *                <li><js>"Inner1$Inner2"</js>
154    *                <li><js>"Inner2"</js>
155    *             </ul>
156    *       </ul>
157    *    <li>Methods:
158    *       <ul>
159    *          <li>Fully qualified with args:
160    *             <ul>
161    *                <li><js>"com.foo.MyClass.myMethod(String,int)"</js>
162    *                <li><js>"com.foo.MyClass.myMethod(java.lang.String,int)"</js>
163    *                <li><js>"com.foo.MyClass.myMethod()"</js>
164    *             </ul>
165    *          <li>Fully qualified:
166    *             <ul>
167    *                <li><js>"com.foo.MyClass.myMethod"</js>
168    *             </ul>
169    *          <li>Simple with args:
170    *             <ul>
171    *                <li><js>"MyClass.myMethod(String,int)"</js>
172    *                <li><js>"MyClass.myMethod(java.lang.String,int)"</js>
173    *                <li><js>"MyClass.myMethod()"</js>
174    *             </ul>
175    *          <li>Simple:
176    *             <ul>
177    *                <li><js>"MyClass.myMethod"</js>
178    *             </ul>
179    *          <li>Simple inner class:
180    *             <ul>
181    *                <li><js>"MyClass$Inner1$Inner2.myMethod"</js>
182    *                <li><js>"Inner1$Inner2.myMethod"</js>
183    *                <li><js>"Inner2.myMethod"</js>
184    *             </ul>
185    *       </ul>
186    *    <li>Fields:
187    *       <ul>
188    *          <li>Fully qualified:
189    *             <ul>
190    *                <li><js>"com.foo.MyClass.myField"</js>
191    *             </ul>
192    *          <li>Simple:
193    *             <ul>
194    *                <li><js>"MyClass.myField"</js>
195    *             </ul>
196    *          <li>Simple inner class:
197    *             <ul>
198    *                <li><js>"MyClass$Inner1$Inner2.myField"</js>
199    *                <li><js>"Inner1$Inner2.myField"</js>
200    *                <li><js>"Inner2.myField"</js>
201    *             </ul>
202    *       </ul>
203    *    <li>A comma-delimited list of anything on this list.
204    * </ul>
205    *
206    * <ul class='seealso'>
207    *    <li class='link'>{@doc jm.DynamicallyAppliedAnnotations}
208    * </ul>
209    *
210    * @return The annotation value.
211    */
212   String[] on() default {};
213
214   /**
215    * Dynamically apply this annotation to the specified classes.
216    *
217    * <p>
218    * Identical to {@link #on()} except allows you to specify class objects instead of a strings.
219    *
220    * <ul class='seealso'>
221    *    <li class='link'>{@doc jm.DynamicallyAppliedAnnotations}
222    * </ul>
223    *
224    * @return The annotation value.
225    */
226   Class<?>[] onClass() default {};
227
228   /**
229    * Associates an {@link HtmlRender} with a bean property for custom HTML rendering of the property.
230    *
231    * <p>
232    * This annotation applies to bean properties and classes.
233    *
234    * @return The annotation value.
235    */
236   @SuppressWarnings("rawtypes")
237   Class<? extends HtmlRender> render() default HtmlRender.class;
238}
