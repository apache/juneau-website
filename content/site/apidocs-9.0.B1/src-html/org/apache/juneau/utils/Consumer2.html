013package org.apache.juneau.utils;
014
015import java.util.function.*;
016
017/**
018 * Functional interface for consumers of 2-part arguments.
019 *
020 * @param <A> Argument 1.
021 * @param <B> Argument 2.
022 */
023@FunctionalInterface
024public interface Consumer2<A,B> {
025
026   /**
027    * Performs this operation on the given arguments.
028    *
029    * @param a Argument 1.
030    * @param b Argument 2.
031    */
032   void apply(A a, B b);
033
034   /**
035    * Returns a composed {@link Consumer} that performs, in sequence, this operation followed by the <c>after</c> operation.
036    *
037    * @param <V> The return type.
038    * @param after The operation to perform after this operation.
039    * @return A composed {@link Consumer} that performs in sequence this operation followed by the after operation.
040    */
041   default <V> Consumer2<A,B> andThen(Consumer2<? super A,? super B> after) {
042      return (A a, B b) -> {
043         apply(a, b);
044         after.apply(a, b);
045      };
046   }
047}
