013package org.apache.juneau.annotation;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016import static org.apache.juneau.collections.JsonMap.*;
017import static java.util.Arrays.*;
018
019import java.lang.annotation.*;
020import java.lang.reflect.*;
021import java.util.*;
022
023import org.apache.juneau.collections.*;
024import org.apache.juneau.internal.*;
025
026/**
027 * A concrete implementation of an annotation.
028 *
029 * <p>
030 * Follows the standard Java conventions for equality and hashcode calculation for annotations.
031 * Equivalent annotations defined programmatically and declaratively should match for equality and hashcode calculation.
032 *
033 * <p>
034 * For performance reasons, the hashcode is calculated one time and cached at the end of object creation.
035 * Constructors must call the {@link #postConstruct()} method after all fields have been set to trigger this calculation.
036 *
037 * <ul class='seealso'>
038 *    <li class='extlink'>{@source}
039 * </ul>
040 */
041public class AnnotationImpl implements Annotation {
042
043   private final Class<? extends Annotation> annotationType;
044   private int hashCode = -1;
045
046   /**
047    * Constructor.
048    *
049    * @param b The builder used to instantiate the fields of this class.
050    */
051   public AnnotationImpl(AnnotationBuilder b) {
052      this.annotationType = b.annotationType;
053   }
054
055   /**
056    * This method must be called at the end of initialization to calculate the hashCode one time.
057    */
058   protected void postConstruct() {
059      this.hashCode = AnnotationUtils.hashCode(this);
060   }
061
062   /**
063    * Implements the {@link Annotation#annotationType()} method for child classes.
064    *
065    * @return This class.
066    */
067   @Override /* Annotation */
068   public Class<? extends Annotation> annotationType() {
069      return annotationType;
070   }
071
072   @Override /* Object */
073   public int hashCode() {
074      if (hashCode == -1)
075         throw runtimeException("Programming error.  postConstruct() was never called on annotation.");
076      return hashCode;
077   }
078
079   @Override /* Object */
080   public boolean equals(Object o) {
081      if (! annotationType.isInstance(o))
082         return false;
083      return AnnotationUtils.equals(this, (Annotation)o);
084   }
085
086   /**
087    * Returns this annotation as a map of key/value pairs.
088    *
089    * <p>
090    * Useful for debugging.
091    *
092    * @return This annotation as a map of key/value pairs.
093    */
094   public JsonMap toMap() {
095      JsonMap m = create();
096      stream(annotationType().getDeclaredMethods())
097         .filter(x->x.getParameterCount() == 0 && x.getDeclaringClass().isAnnotation())
098         .sorted(Comparator.comparing(Method::getName))
099         .forEach(x -> m.append(x.getName(), safeSupplier(()->x.invoke(this))));
100      return m;
101   }
102
103   @Override /* Object */
104   public String toString() {
105      return toMap().asString();
106   }
107}
