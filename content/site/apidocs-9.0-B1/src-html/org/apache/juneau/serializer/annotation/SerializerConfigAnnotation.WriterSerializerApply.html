013package org.apache.juneau.serializer.annotation;
014
015import org.apache.juneau.*;
016import org.apache.juneau.reflect.*;
017import org.apache.juneau.serializer.*;
018import org.apache.juneau.svl.*;
019
020/**
021 * Utility classes and methods for the {@link SerializerConfig @SerializerConfig} annotation.
022 *
023 * <ul class='seealso'>
024 *    <li class='link'>{@doc jm.SerializersAndParsers}
025 *    <li class='extlink'>{@source}
026 * </ul>
027 */
028public class SerializerConfigAnnotation {
029
030   /**
031    * Applies {@link SerializerConfig} annotations to a {@link org.apache.juneau.serializer.Serializer.Builder}.
032    */
033   public static class SerializerApply extends AnnotationApplier<SerializerConfig,Serializer.Builder> {
034
035      /**
036       * Constructor.
037       *
038       * @param vr The resolver for resolving values in annotations.
039       */
040      public SerializerApply(VarResolverSession vr) {
041         super(SerializerConfig.class, Serializer.Builder.class, vr);
042      }
043
044      @Override
045      public void apply(AnnotationInfo<SerializerConfig> ai, Serializer.Builder b) {
046         SerializerConfig a = ai.inner();
047
048         bool(a.addBeanTypes()).ifPresent(x -> b.addBeanTypes(x));
049         bool(a.addRootType()).ifPresent(x -> b.addRootType(x));
050         bool(a.keepNullProperties()).ifPresent(x -> b.keepNullProperties(x));
051         type(a.listener()).ifPresent(x -> b.listener(x));
052         bool(a.sortCollections()).ifPresent(x -> b.sortCollections(x));
053         bool(a.sortMaps()).ifPresent(x -> b.sortMaps(x));
054         bool(a.trimEmptyCollections()).ifPresent(x -> b.trimEmptyCollections(x));
055         bool(a.trimEmptyMaps()).ifPresent(x -> b.trimEmptyMaps(x));
056         bool(a.trimStrings()).ifPresent(x -> b.trimStrings(x));
057         string(a.uriContext()).map(UriContext::of).ifPresent(x -> b.uriContext(x));
058         string(a.uriRelativity()).map(UriRelativity::valueOf).ifPresent(x -> b.uriRelativity(x));
059         string(a.uriResolution()).map(UriResolution::valueOf).ifPresent(x -> b.uriResolution(x));
060         bool(a.detectRecursions()).ifPresent(x -> b.detectRecursions(x));
061         bool(a.ignoreRecursions()).ifPresent(x -> b.ignoreRecursions(x));
062         integer(a.initialDepth(), "initialDepth").ifPresent(x -> b.initialDepth(x));
063         integer(a.maxDepth(), "maxDepth").ifPresent(x -> b.maxDepth(x));
064      }
065   }
066
067   /**
068    * Applies {@link SerializerConfig} annotations to a {@link org.apache.juneau.serializer.OutputStreamSerializer.Builder}.
069    */
070   public static class OutputStreamSerializerApply extends AnnotationApplier<SerializerConfig,OutputStreamSerializer.Builder> {
071
072      /**
073       * Constructor.
074       *
075       * @param vr The resolver for resolving values in annotations.
076       */
077      public OutputStreamSerializerApply(VarResolverSession vr) {
078         super(SerializerConfig.class, OutputStreamSerializer.Builder.class, vr);
079      }
080
081      @Override
082      public void apply(AnnotationInfo<SerializerConfig> ai, OutputStreamSerializer.Builder b) {
083         SerializerConfig a = ai.inner();
084
085         string(a.binaryFormat()).map(BinaryFormat::valueOf).ifPresent(x -> b.binaryFormat(x));
086      }
087   }
088
089   /**
090    * Applies {@link SerializerConfig} annotations to a {@link org.apache.juneau.serializer.WriterSerializer.Builder}.
091    */
092   public static class WriterSerializerApply extends AnnotationApplier<SerializerConfig,WriterSerializer.Builder> {
093
094      /**
095       * Constructor.
096       *
097       * @param vr The resolver for resolving values in annotations.
098       */
099      public WriterSerializerApply(VarResolverSession vr) {
100         super(SerializerConfig.class, WriterSerializer.Builder.class, vr);
101      }
102
103      @Override
104      public void apply(AnnotationInfo<SerializerConfig> ai, WriterSerializer.Builder b) {
105         SerializerConfig a = ai.inner();
106
107         charset(a.fileCharset()).ifPresent(x -> b.fileCharset(x));
108         integer(a.maxIndent(), "maxIndent").ifPresent(x -> b.maxIndent(x));
109         character(a.quoteChar(), "quoteChar").ifPresent(x -> b.quoteChar(x));
110         charset(a.streamCharset()).ifPresent(x -> b.streamCharset(x));
111         bool(a.useWhitespace()).ifPresent(x -> b.useWhitespace(x));
112      }
113   }
114}
