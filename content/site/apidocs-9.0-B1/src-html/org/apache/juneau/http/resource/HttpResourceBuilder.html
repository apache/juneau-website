027/**
028 * Builder for {@link HttpEntity} beans.
029 *
030 * <ul class='seealso'>
031 *    <li class='link'>{@doc jm.HttpParts}
032 *    <li class='extlink'>{@source}
033 * </ul>
034 *
035 * @param <T> The bean type to create for this builder.
036 */
037@FluentSetters(returns="HttpResourceBuilder<T>")
038public class HttpResourceBuilder<T extends BasicResource> {
039
040   HeaderList headers = HeaderList.EMPTY;
041   HeaderList.Builder headersBuilder;
042
043   BasicHttpEntity entity;
044   HttpEntityBuilder<?> entityBuilder;
045
046   /** The HttpEntity implementation class. */
047   protected final Class<? extends BasicResource> implClass;
048
049   /** The HttpEntity implementation class. */
050   protected final Class<? extends BasicHttpEntity> entityImplClass;
051
052   /**
053    * Constructor.
054    *
055    * @param implClass
056    *    The subclass of {@link HttpResponse} to create.
057    *    <br>This must contain a public constructor that takes in an {@link HttpResourceBuilder} object.
058    * @param entityImplClass
059    *    The subclass of {@link BasicHttpEntity} to create.
060    *    <br>This must contain a public constructor that takes in an {@link HttpEntityBuilder} object.
061    */
062   public HttpResourceBuilder(Class<T> implClass, Class<? extends BasicHttpEntity> entityImplClass) {
063      this.implClass = implClass;
064      this.entityImplClass = entityImplClass;
065   }
066
067   /**
068    * Copy constructor.
069    *
070    * @param impl
071    *    The implementation object of {@link HttpEntity} to copy from.
072    *    <br>This must contain a public constructor that takes in an {@link HttpResourceBuilder} object.
073    */
074   public HttpResourceBuilder(T impl) {
075      implClass = impl.getClass();
076      headers = impl.headers;
077      entity = impl.entity;
078      this.entityImplClass = entity.getClass();
079   }
080
081   /**
082    * Instantiates the entity bean from the settings in this builder.
083    *
084    * @return A new {@link HttpEntity} bean.
085    */
086   @SuppressWarnings("unchecked")
087   public T build() {
088      try {
089         return (T) implClass.getConstructor(HttpResourceBuilder.class).newInstance(this);
090      } catch (Exception e) {
091         throw runtimeException(e);
092      }
093   }
094
095   HeaderList headers() {
096      if (headersBuilder != null)
097         return headersBuilder.build();
098      if (headers == null)
099         return HeaderList.EMPTY;
100      return headers;
101   }
102
103   BasicHttpEntity entity() {
104      if (entityBuilder != null)
105         return entityBuilder.build();
106      if (entity == null)
107         return BasicHttpEntity.EMPTY;
108      return entity;
109   }
110
111   /**
112    * Copies the contents of the specified HTTP response to this builder.
113    *
114    * @param response The response to copy from.  Must not be null.
115    * @return This object.
116    * @throws IOException If content could not be retrieved.
117    */
118   public HttpResourceBuilder<?> copyFrom(HttpResponse response) throws IOException {
119      headers(response.getAllHeaders());
120      content(response.getEntity().getContent());
121      return this;
122   }

128   /**
129    * Returns access to the underlying builder for the HTTP entity.
130    *
131    * @return The underlying builder for the HTTP entity.
132    */
133   public HttpEntityBuilder<?> getEntity() {
134      if (entityBuilder == null) {
135         entityBuilder = entity == null ? BasicHttpEntity.create(entityImplClass) : entity.copy();
136         entity = null;
137      }
138      return entityBuilder;
139   }

151   }

241      getEntity().chunked();

243   }

246    * Sets the 'chunked' flag value.

279   /**
280    * Returns access to the underlying builder for the headers.
281    *
282    * @return The underlying builder for the headers.
283    */
284   public HeaderList.Builder getHeaders() {
285      if (headersBuilder == null) {
286         headersBuilder = headers == null ? HeaderList.create() : headers.copy();
287         headers = null;
288      }
289      return headersBuilder;
290   }
291
292   /**
293    * Sets the specified headers in this builder.
294    *
295    * @param value The new value.
296    * @return This object.
297    */
298   @FluentSetter
299   public HttpResourceBuilder<T> headers(HeaderList value) {
300      headers = value;
301      headersBuilder = null;
302      return this;
303   }
304
305   /**
306    * Adds the specified header to the end of the headers in this builder.
307    *
308    * @param value The header to add.  <jk>null</jk> values are ignored.
309    * @return This object.
310    */
311   public HttpResourceBuilder<T> header(Header value) {
312      if (value != null)
313         getHeaders().append(value);
314      return this;
315   }
316
317   /**
318    * Adds the specified header to the end of the headers in this builder.
319    *
320    * <p>
321    * This is a no-op if either the name or value is <jk>null</jk>.
322    *
323    * @param name The header name.
324    * @param value The header value.
325    * @return This object.
326    */
327   public HttpResourceBuilder<T> header(String name, String value) {
328      if (name != null && value != null)
329         getHeaders().append(name, value);
330      return this;
331   }
332
333   /**
334    * Adds the specified headers to the end of the headers in this builder.
335    *
336    * @param values The headers to add.  <jk>null</jk> headers and headers with <jk>null</jk> names or values are ignored.
337    * @return This object.
338    */
339   public HttpResourceBuilder<T> headers(Header...values) {
340      for (Header h : values) {
341         if (h != null) {
342            String n = h.getName();
343            String v = h.getValue();
344            if (isNotEmpty(n)) {
345               if (n.equalsIgnoreCase("content-type"))
346                  contentType(v);
347               else if (n.equalsIgnoreCase("content-encoding"))
348                  contentEncoding(v);
349               else if (n.equalsIgnoreCase("content-length"))
350                  contentLength(Long.parseLong(v));
351               else
352                  getHeaders().append(h);
353            }
354         }
355      }
356      return this;
357   }
358
359   /**
360    * Adds the specified headers to the end of the headers in this builder.
361    *
362    * @param values The headers to add.  <jk>null</jk> values are ignored.
363    * @return This object.
364    */
365   public HttpResourceBuilder<T> headers(List<Header> values) {
366      getHeaders().append(values);
367      return this;
368   }
