013package org.apache.juneau.http.header;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.net.*;
019import java.util.*;
020import java.util.function.*;
021
022/**
023 * Category of headers that consist of a single URL value.
024 *
025 * <p>
026 * <h5 class='figure'>Example</h5>
027 * <p class='bcode'>
028 *    Location: http://www.w3.org/pub/WWW/People.html
029 * </p>
030 *
031 * <ul class='seealso'>
032 *    <li class='link'>{@doc jm.HttpParts}
033 *    <li class='extlink'>{@doc ext.RFC2616}
034 *    <li class='extlink'>{@source}
035 * </ul>
036 *
037 * @serial exclude
038 */
039public class BasicUriHeader extends BasicHeader {
040
041   //-----------------------------------------------------------------------------------------------------------------
042   // Static
043   //-----------------------------------------------------------------------------------------------------------------
044
045   private static final long serialVersionUID = 1L;
046
047   /**
048    * Static creator.
049    *
050    * @param name The header name.
051    * @param value
052    *    The header value.
053    *    <br>Must be parsable by {@link URI#create(String)}.
054    *    <br>Can be <jk>null</jk>.
055    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
056    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
057    */
058   public static BasicUriHeader of(String name, String value) {
059      return value == null ? null : new BasicUriHeader(name, value);
060   }
061
062   /**
063    * Static creator.
064    *
065    * @param name The header name.
066    * @param value
067    *    The header value.
068    *    <br>Can be <jk>null</jk>.
069    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
070    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
071    */
072   public static BasicUriHeader of(String name, URI value) {
073      return value == null ? null : new BasicUriHeader(name, value);
074   }
075
076   /**
077    * Static creator with delayed value.
078    *
079    * <p>
080    * Header value is re-evaluated on each call to {@link #getValue()}.
081    *
082    * @param name The header name.
083    * @param value
084    *    The supplier of the header value.
085    *    <br>Can be <jk>null</jk>.
086    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
087    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
088    */
089   public static BasicUriHeader of(String name, Supplier<URI> value) {
090      return value == null ? null : new BasicUriHeader(name, value);
091   }
092
093   //-----------------------------------------------------------------------------------------------------------------
094   // Instance
095   //-----------------------------------------------------------------------------------------------------------------
096
097   private final URI value;
098   private final Supplier<URI> supplier;
099
100   /**
101    * Constructor.
102    *
103    * @param name The header name.
104    * @param value
105    *    The header value.
106    *    <br>Must be parsable by {@link URI#create(String)}.
107    *    <br>Can be <jk>null</jk>.
108    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
109    */
110   public BasicUriHeader(String name, String value) {
111      super(name, value);
112      this.value = isEmpty(value) ? null :  URI.create(value);
113      this.supplier = null;
114   }
115
116   /**
117    * Constructor.
118    *
119    * @param name The header name.
120    * @param value
121    *    The header value.
122    *    <br>Can be <jk>null</jk>.
123    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
124    */
125   public BasicUriHeader(String name, URI value) {
126      super(name, stringify(value));
127      this.value = value;
128      this.supplier = null;
129   }
130
131   /**
132    * Constructor with delayed value.
133    *
134    * <p>
135    * Header value is re-evaluated on each call to {@link #getValue()}.
136    *
137    * @param name The header name.
138    * @param value
139    *    The supplier of the header value.
140    *    <br>Can be <jk>null</jk>.
141    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
142    */
143   public BasicUriHeader(String name, Supplier<URI> value) {
144      super(name, null);
145      this.value = null;
146      this.supplier = value;
147   }
148
149   @Override /* Header */
150   public String getValue() {
151      return stringify(value());
152   }
153
154   /**
155    * Returns the header value as a {@link URI} wrapped in an {@link Optional}.
156    *
157    * @return The header value as a {@link URI} wrapped in an {@link Optional}.  Never <jk>null</jk>.
158    */
159   public Optional<URI> asUri() {
160      return optional(value());
161   }
162
163   /**
164    * Returns the header value as a {@link URI} wrapped in an {@link Optional}.
165    *
166    * @return The header value as a {@link URI} wrapped in an {@link Optional}.  Never <jk>null</jk>.
167    */
168   public URI toUri() {
169      return value();
170   }
171
172   /**
173    * Return the value if present, otherwise return <c>other</c>.
174    *
175    * <p>
176    * This is a shortened form for calling <c>asUri().orElse(<jv>other</jv>)</c>.
177    *
178    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
179    * @return The value, if present, otherwise <c>other</c>.
180    */
181   public URI orElse(URI other) {
182      URI x = value();
183      return x != null ? x : other;
184   }
185
186   private URI value() {
187      if (supplier != null)
188         return supplier.get();
189      return value;
190   }
191}
