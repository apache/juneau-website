013package org.apache.juneau.http.response;
014
015import static org.apache.juneau.http.response.Gone.*;
016
017import java.text.*;
018
019import org.apache.http.*;
020import org.apache.juneau.http.*;
021import org.apache.juneau.http.annotation.*;
022
023/**
024 * Exception representing an HTTP 410 ().
025 *
026 * <p>
027 * Indicates that the resource requested is no longer available and will not be available again.
028 * <br>This should be used when a resource has been intentionally removed and the resource should be purged.
029 * <br>Upon receiving a 410 status code, the client should not request the resource in the future.
030 * <br>Clients such as search engines should remove the resource from their indices.
031 * <br>Most use cases do not require clients and search engines to purge the resource, and a <js>"404 Not Found"</js> may be used instead.
032 *
033 * <ul class='seealso'>
034 *    <li class='link'>{@doc jm.HttpParts}
035 *    <li class='extlink'>{@source}
036 * </ul>
037 *
038 * @serial exclude
039 */
040@Response
041@StatusCode(STATUS_CODE)
042@Schema(description=REASON_PHRASE)
043public class Gone extends BasicHttpException {
044   private static final long serialVersionUID = 1L;
045
046   /** HTTP status code */
047   public static final int STATUS_CODE = 410;
048
049   /** Reason phrase */
050   public static final String REASON_PHRASE = "Gone";
051
052   /** Default status line */
053   private static final BasicStatusLine STATUS_LINE = BasicStatusLine.create().statusCode(STATUS_CODE).reasonPhrase(REASON_PHRASE).build();
054
055   /** Reusable unmodifiable instance */
056   public static final Gone INSTANCE = create().unmodifiable().build();
057
058   /**
059    * Creates a builder for this class.
060    *
061    * @return A new builder bean.
062    */
063   public static HttpExceptionBuilder<Gone> create() {
064      return new HttpExceptionBuilder<>(Gone.class).statusLine(STATUS_LINE);
065   }
066
067   /**
068    * Constructor.
069    *
070    * @param builder The builder containing the settings for this exception.
071    */
072   public Gone(HttpExceptionBuilder<?> builder) {
073      super(builder);
074   }
075
076   /**
077    * Constructor.
078    *
079    * @param cause The caused-by exception.  Can be <jk>null</jk>.
080    * @param msg The message.  Can be <jk>null</jk>.
081    * @param args The message arguments.
082    */
083   public Gone(Throwable cause, String msg, Object...args) {
084      this(create().causedBy(cause).message(msg, args));
085   }
086
087   /**
088    * Constructor.
089    */
090   public Gone() {
091      this(create());
092   }
093
094   /**
095    * Constructor.
096    *
097    * @param msg The message.  Can be <jk>null</jk>.
098    * @param args Optional {@link MessageFormat}-style arguments in the message.
099    */
100   public Gone(String msg, Object...args) {
101      this(create().message(msg, args));
102   }
103
104   /**
105    * Constructor.
106    *
107    * @param cause The cause.  Can be <jk>null</jk>.
108    */
109   public Gone(Throwable cause) {
110      this(create().causedBy(cause));
111   }
112
113   /**
114    * Constructor.
115    *
116    * <p>
117    * This is the constructor used when parsing an HTTP response.
118    *
119    * @param response The HTTP response to copy from.  Must not be <jk>null</jk>.
120    * @throws AssertionError If HTTP response status code does not match what was expected.
121    */
122   public Gone(HttpResponse response) {
123      this(create().copyFrom(response));
124      assertStatusCode(response);
125   }
126
127   /**
128    * Creates a builder for this class initialized with the contents of this bean.
129    *
130    * @return A new builder bean.
131    */
132   @Override /* BasicHttpException */
133   public HttpExceptionBuilder<Gone> copy() {
134      return new HttpExceptionBuilder<>(this);
135   }
136}
