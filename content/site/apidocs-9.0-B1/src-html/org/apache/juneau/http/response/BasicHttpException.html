031/**
032 * Basic implementation of the {@link HttpResponse} interface for error responses.
033 *
034 * <p>
035 * Although this class implements the various setters defined on the {@link HttpResponse} interface, it's in general
036 * going to be more efficient to set the status/headers/content of this bean through the builder.
037 *
038 * <p>
039 * If the <c>unmodifiable</c> flag is set on this bean, calls to the setters will throw {@link UnsupportedOperationException} exceptions.
040 *
041 * <ul class='notes'>
042 *    <li class='warn'>Beans are not thread safe unless they're marked as unmodifiable.
043 * </ul>
044 *
045 * <ul class='seealso'>
046 *    <li class='link'>{@doc jm.HttpParts}
047 *    <li class='extlink'>{@source}
048 * </ul>
049 *
050 * @serial exclude
051 */

149   /**
150    * Returns the root cause of this exception.
151    *
152    * <p>
153    * The root cause is the first exception in the init-cause parent chain that's not one of the following:
154    * <ul>
155    *    <li>{@link BasicHttpException}
156    *    <li>{@link InvocationTargetException}
157    * </ul>
158    *
159    * @return The root cause of this exception, or <jk>null</jk> if no root cause was found.
160    */
161   public Throwable getRootCause() {
162      Throwable t = this;
163      while(t != null) {
164         if (! (t instanceof BasicHttpException || t instanceof InvocationTargetException))
165            return t;
166         t = t.getCause();
167      }
168      return null;
169   }

171   /**
172    * Asserts that the specified HTTP response has the same status code as the one on the status line of this bean.
173    *
174    * @param response The HTTP response to check.  Must not be <jk>null</jk>.
175    * @throws AssertionError If status code is not what was expected.
176    */
177   protected void assertStatusCode(HttpResponse response) throws AssertionError {
178      assertArgNotNull("response", response);
179      int expected = getStatusLine().getStatusCode();
180      int actual = response.getStatusLine().getStatusCode();
181      assertInteger(actual).setMsg("Unexpected status code.  Expected:[{0}], Actual:[{1}]", expected, actual).is(expected);
182   }

184   /**
185    * Returns all error messages from all errors in this stack.
186    *
187    * <p>
188    * Typically useful if you want to render all the error messages in the stack, but don't want to render all the
189    * stack traces too.
190    *
191    * @param scrubForXssVulnerabilities
192    *    If <jk>true</jk>, replaces <js>'&lt;'</js>, <js>'&gt;'</js>, and <js>'&amp;'</js> characters with spaces.
193    * @return All error messages from all errors in this stack.
194    */
195   public String getFullStackMessage(boolean scrubForXssVulnerabilities) {
196      String msg = getMessage();
197      StringBuilder sb = new StringBuilder();
198      if (msg != null) {
199         if (scrubForXssVulnerabilities)
200            msg = msg.replace('<', ' ').replace('>', ' ').replace('&', ' ');
201         sb.append(msg);
202      }
203      Throwable e = getCause();
204      while (e != null) {
205         msg = e.getMessage();
206         if (msg != null && scrubForXssVulnerabilities)
207            msg = msg.replace('<', ' ').replace('>', ' ').replace('&', ' ');
208         String cls = e.getClass().getSimpleName();
209         if (msg == null)
210            sb.append(format("\nCaused by ({0})", cls));
211         else
212            sb.append(format("\nCaused by ({0}): {1}", cls, msg));
213         e = e.getCause();
214      }
215      return sb.toString();
216   }

218   @Override /* Throwable */
219   public String getMessage() {
220      String m = super.getMessage();
221      if (m == null && getCause() != null)
222         m = getCause().getMessage();
223      if (m == null)
224         m = statusLine().getReasonPhrase();
225      return m;
226   }

228   @Override /* Object */
229   public int hashCode() {
230      int i = 0;
231      Throwable t = this;
232      while (t != null) {
233         for (StackTraceElement e : t.getStackTrace())
234         i ^= e.hashCode();
235         t = t.getCause();
236      }
237      return i;
238   }

385   private BasicStatusLine statusLine() {
386      if (statusLine == null) {
387         statusLine = statusLineBuilder.build();
388         statusLineBuilder = null;
389      }
390      return statusLine;
391   }
392
393   private HeaderList headers() {
394      if (headers == null) {
395         headers = headersBuilder.build();
396         headersBuilder = null;
397      }
398      return headers;
399   }
400
401   private BasicStatusLine.Builder statusLineBuilder() {
402      assertModifiable();
403      if (statusLineBuilder == null) {
404         statusLineBuilder = statusLine.copy();
405         statusLine = null;
406      }
407      return statusLineBuilder;
408   }
409
410   private HeaderList.Builder headersBuilder() {
411      assertModifiable();
412      if (headersBuilder == null) {
413         headersBuilder = headers.copy();
414         headers = null;
415      }
416      return headersBuilder;
417   }
