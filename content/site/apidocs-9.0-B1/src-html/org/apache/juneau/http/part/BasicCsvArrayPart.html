013package org.apache.juneau.http.part;
014
015import static org.apache.juneau.internal.ArrayUtils.copyOf;
016import static org.apache.juneau.internal.CollectionUtils.*;
017import static org.apache.juneau.internal.StringUtils.*;
018
019import java.util.*;
020import java.util.function.*;
021
022import org.apache.http.*;
023import org.apache.juneau.assertions.*;
024
025/**
026 * A {@link NameValuePair} that consists of a comma-delimited list of string values.
027 *
028 * <ul class='seealso'>
029 *    <li class='link'>{@doc jm.HttpParts}
030 *    <li class='extlink'>{@source}
031 * </ul>
032 */
033public class BasicCsvArrayPart extends BasicPart {
034
035   //-----------------------------------------------------------------------------------------------------------------
036   // Static
037   //-----------------------------------------------------------------------------------------------------------------
038
039   private static final String[] EMPTY = new String[0];
040
041   /**
042    * Static creator.
043    *
044    * @param name The part name.
045    * @param value The part value.
046    * @return A new {@link BasicCsvArrayPart} object, or <jk>null</jk> if the name or value is <jk>null</jk>.
047    */
048   public static BasicCsvArrayPart of(String name, String...value) {
049      if (isEmpty(name) || value == null)
050         return null;
051      return new BasicCsvArrayPart(name, value);
052   }
053
054   /**
055    * Static creator with delayed value.
056    *
057    * <p>
058    * Part value is re-evaluated on each call to {@link NameValuePair#getValue()}.
059    *
060    * @param name The part name.
061    * @param value The part value supplier.
062    * @return A new {@link BasicCsvArrayPart} object, or <jk>null</jk> if the name or supplier is <jk>null</jk>.
063    */
064   public static BasicCsvArrayPart of(String name, Supplier<String[]> value) {
065      if (isEmpty(name) || value == null)
066         return null;
067      return new BasicCsvArrayPart(name, value);
068   }
069
070   //-----------------------------------------------------------------------------------------------------------------
071   // Instance
072   //-----------------------------------------------------------------------------------------------------------------
073
074   private final String[] value;
075   private final Supplier<String[]> supplier;
076   private String stringValue;
077
078   /**
079    * Constructor.
080    *
081    * @param name The part name.  Must not be <jk>null</jk>.
082    * @param value The part value.  Can be <jk>null</jk>.
083    */
084   public BasicCsvArrayPart(String name, String...value) {
085      super(name, value);
086      this.value = value;
087      this.supplier = null;
088      this.stringValue = null;
089   }
090
091   /**
092    * Constructor.
093    *
094    * @param name The part name.  Must not be <jk>null</jk>.
095    * @param value The part value supplier.  Can be <jk>null</jk> or supply <jk>null</jk>.
096    */
097   public BasicCsvArrayPart(String name, Supplier<String[]> value) {
098      super(name, value);
099      this.value = null;
100      this.supplier = value;
101      this.stringValue = null;
102   }
103
104   /**
105    * Constructor.
106    *
107    * <p>
108    * <jk>null</jk> values are treated as <jk>null</jk>.
109    * Otherwise parses as a comma-delimited list with whitespace trimmed.
110    *
111    * @param name The part name.  Must not be <jk>null</jk>.
112    * @param value The part value.  Can be <jk>null</jk>.
113    */
114   public BasicCsvArrayPart(String name, String value) {
115      super(name, value);
116      this.value = split(value);
117      this.supplier = null;
118      this.stringValue = value;
119   }
120
121   @Override /* Header */
122   public String getValue() {
123      if (supplier != null)
124         return join(supplier.get(), ',');
125      if (stringValue != null)
126         stringValue = join(value, ',');
127      return stringValue;
128   }
129
130   /**
131    * Returns <jk>true</jk> if this part contains the specified value.
132    *
133    * @param val The value to check for.
134    * @return <jk>true</jk> if this part contains the specified value.
135    */
136   public boolean contains(String val) {
137      if (val != null)
138         for (String v : value())
139            if (eq(v, val))
140               return true;
141      return false;
142   }
143
144   /**
145    * Returns <jk>true</jk> if this part contains the specified value using {@link String#equalsIgnoreCase(String)}.
146    *
147    * @param val The value to check for.
148    * @return <jk>true</jk> if this part contains the specified value.
149    */
150   public boolean containsIgnoreCase(String val) {
151      if (val != null)
152         for (String v : value())
153            if (eqic(v, val))
154               return true;
155      return false;
156   }
157
158   /**
159    * Provides the ability to perform fluent-style assertions on this part.
160    *
161    * @return A new fluent assertion object.
162    * @throws AssertionError If assertion failed.
163    */
164   public FluentListAssertion<String,BasicCsvArrayPart> assertList() {
165      return new FluentListAssertion<>(ulist(value()), this);
166   }
167
168   /**
169    * Returns The part value as a {@link List}.
170    *
171    * <p>
172    * The list is unmodifiable.
173    *
174    * @return The part value as a {@link List}, or <jk>null</jk> if the value <jk>null</jk>.
175    */
176   public List<String> toList() {
177      return ulist(value());
178   }
179
180   /**
181    * Returns The part value as a {@link List} wrapped in an {@link Optional}.
182    *
183    * <p>
184    * The list is unmodifiable.
185    *
186    * @return The part value as a {@link List} wrapped in an {@link Optional}.  Never <jk>null</jk>.
187    */
188   public Optional<List<String>> asList() {
189      return optional(toList());
190   }
191
192   /**
193    * Returns The part value as an array.
194    *
195    * <p>
196    * The array is a copy of the value of this part.
197    *
198    * @return The part value as an array, or <jk>null</jk> if the value <jk>null</jk>.
199    */
200   public String[] toArray() {
201      return copyOf(value());
202   }
203
204   /**
205    * Returns The part value as an array wrapped in an {@link Optional}.
206    *
207    * <p>
208    * Array is a copy of the value of this part.
209    *
210    * @return The part value as an array wrapped in an {@link Optional}.  Never <jk>null</jk>.
211    */
212   public Optional<String[]> asArray() {
213      return optional(copyOf(value()));
214   }
215
216   /**
217    * Return the value if present, otherwise return <c>other</c>.
218    *
219    * <p>
220    * This is a shortened form for calling <c>asArray().orElse(<jv>other</jv>)</c>.
221    *
222    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
223    * @return The value, if present, otherwise <c>other</c>.
224    */
225   public String[] orElse(String[] other) {
226      String[] x = value();
227      return x != null ? x : other;
228   }
229
230   private String[] value() {
231      if (supplier != null) {
232         String[] v = supplier.get();
233         return v != null ? v : EMPTY;
234      }
235      return value;
236   }
237}
