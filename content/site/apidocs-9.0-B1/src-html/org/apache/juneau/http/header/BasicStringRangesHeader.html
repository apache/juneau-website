013package org.apache.juneau.http.header;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.util.*;
019import java.util.function.*;
020
021/**
022 * Category of headers that consist of simple comma-delimited lists of strings with q-values.
023 *
024 * <p>
025 * <h5 class='figure'>Example</h5>
026 * <p class='bcode'>
027 *    Accept-Encoding: compress;q=0.5, gzip;q=1.0
028 * </p>
029 *
030 * <ul class='seealso'>
031 *    <li class='link'>{@doc jm.HttpParts}
032 *    <li class='extlink'>{@doc ext.RFC2616}
033 *    <li class='extlink'>{@source}
034 * </ul>
035 *
036 * @serial exclude
037 */
038public class BasicStringRangesHeader extends BasicHeader {
039
040   //-----------------------------------------------------------------------------------------------------------------
041   // Static
042   //-----------------------------------------------------------------------------------------------------------------
043
044   private static final long serialVersionUID = 1L;
045
046   /**
047    * Static creator.
048    *
049    * @param name The header name.
050    * @param value
051    *    The header value.
052    *    <br>Must be parsable by {@link StringRanges#of(String)}.
053    *    <br>Can be <jk>null</jk>.
054    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
055    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
056    */
057   public static BasicStringRangesHeader of(String name, String value) {
058      return value == null ? null : new BasicStringRangesHeader(name, value);
059   }
060
061   /**
062    * Static creator.
063    *
064    * @param name The header name.
065    * @param value
066    *    The header value.
067    *    <br>Can be <jk>null</jk>.
068    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
069    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
070    */
071   public static BasicStringRangesHeader of(String name, StringRanges value) {
072      return value == null ? null : new BasicStringRangesHeader(name, value);
073   }
074
075   /**
076    * Static creator with delayed value.
077    *
078    * <p>
079    * Header value is re-evaluated on each call to {@link #getValue()}.
080    *
081    * @param name The header name.
082    * @param value
083    *    The supplier of the header value.
084    *    <br>Can be <jk>null</jk>.
085    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
086    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
087    */
088   public static BasicStringRangesHeader of(String name, Supplier<StringRanges> value) {
089      return value == null ? null : new BasicStringRangesHeader(name, value);
090   }
091
092   //-----------------------------------------------------------------------------------------------------------------
093   // Instance
094   //-----------------------------------------------------------------------------------------------------------------
095
096   private final String stringValue;
097   private final StringRanges value;
098   private final Supplier<StringRanges> supplier;
099
100   /**
101    * Constructor.
102    *
103    * @param name The header name.
104    * @param value
105    *    The header value.
106    *    <br>Must be parsable by {@link StringRanges#of(String)}.
107    *    <br>Can be <jk>null</jk>.
108    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
109    */
110   public BasicStringRangesHeader(String name, String value) {
111      super(name, value);
112      this.stringValue = value;
113      this.value = StringRanges.of(value);
114      this.supplier = null;
115   }
116
117   /**
118    * Constructor.
119    *
120    * @param name The header name.
121    * @param value
122    *    The header value.
123    *    <br>Can be <jk>null</jk>.
124    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
125    */
126   public BasicStringRangesHeader(String name, StringRanges value) {
127      super(name, stringify(value));
128      this.stringValue = null;
129      this.value = value;
130      this.supplier = null;
131   }
132
133   /**
134    * Constructor with delayed value.
135    *
136    * <p>
137    * Header value is re-evaluated on each call to {@link #getValue()}.
138    *
139    * @param name The header name.
140    * @param value
141    *    The supplier of the header value.
142    *    <br>Can be <jk>null</jk>.
143    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
144    */
145   public BasicStringRangesHeader(String name, Supplier<StringRanges> value) {
146      super(name, null);
147      this.stringValue = null;
148      this.value = null;
149      this.supplier = value;
150   }
151
152   @Override /* Header */
153   public String getValue() {
154      return stringValue != null ? stringValue : stringify(value());
155   }
156
157   /**
158    * Returns the header value as a {@link StringRanges} wrapped in an {@link Optional}.
159    *
160    * @return The header value as a {@link StringRanges} wrapped in an {@link Optional}.  Never <jk>null</jk>.
161    */
162   public Optional<StringRanges> asStringRanges() {
163      return optional(value());
164   }
165
166   /**
167    * Returns the header value as a {@link StringRanges}.
168    *
169    * @return The header value as a {@link StringRanges}.  Can be <jk>null</jk>.
170    */
171   public StringRanges toStringRanges() {
172      return value();
173   }
174
175   /**
176    * Given a list of media types, returns the best match for this string range header.
177    *
178    * <p>
179    * Note that fuzzy matching is allowed on the media types where the string range header may
180    * contain additional subtype parts.
181    * <br>For example, given identical q-values and an string range value of <js>"text/json+activity"</js>,
182    * the media type <js>"text/json"</js> will match if <js>"text/json+activity"</js> or <js>"text/activity+json"</js>
183    * isn't found.
184    * <br>The purpose for this is to allow serializers to match when artifacts such as <c>id</c> properties are
185    * present in the header.
186    *
187    * <p>
188    * See {@doc https://www.w3.org/TR/activitypub/#retrieving-objects ActivityPub / Retrieving Objects}
189    *
190    * @param names The names to match against.
191    * @return The index into the array of the best match, or <c>-1</c> if no suitable matches could be found.
192    */
193   public int match(List<String> names) {
194      StringRanges x = value();
195      return x == null ? -1 : x.match(names);
196   }
197
198   /**
199    * Returns the {@link MediaRange} at the specified index.
200    *
201    * @param index The index position of the media range.
202    * @return The {@link MediaRange} at the specified index or <jk>null</jk> if the index is out of range.
203    */
204   public StringRange getRange(int index) {
205      StringRanges x = value();
206      return x == null ? null : x.getRange(index);
207   }
208
209   /**
210    * Return the value if present, otherwise return <c>other</c>.
211    *
212    * <p>
213    * This is a shortened form for calling <c>asArray().orElse(<jv>other</jv>)</c>.
214    *
215    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
216    * @return The value, if present, otherwise <c>other</c>.
217    */
218   public StringRanges orElse(StringRanges other) {
219      StringRanges x = value();
220      return x != null ? x : other;
221   }
222
223   private StringRanges value() {
224      if (supplier != null)
225         return supplier.get();
226      return value;
227   }
228}
