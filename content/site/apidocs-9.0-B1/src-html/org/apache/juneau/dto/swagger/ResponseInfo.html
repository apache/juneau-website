024/**
025 * Describes a single response from an API Operation.
026 *
027 * <h5 class='section'>Example:</h5>
028 * <p class='bjava'>
029 *    <jc>// Construct using SwaggerBuilder.</jc>
030 *    ResponseInfo <jv>info</jv> = <jsm>responseInfo</jsm>(<js>"A complex object array response"</js>)
031 *       .schema(
032 *          <jsm>schemaInfo</jsm>
033 *             .type(<js>"array"</js>)
034 *             .items(
035 *                <jsm>items</jsm>()
036 *                   .set(<js>"$ref"</js>, <js>"#/definitions/VeryComplexType"</js>)
037 *             )
038 *       );
039 *
040 *    <jc>// Serialize using JsonSerializer.</jc>
041 *    String <jv>json</jv> = JsonSerializer.<jsf>DEFAULT</jsf>.toString(<jv>info</jv>);
042 *
043 *    <jc>// Or just use toString() which does the same as above.</jc>
044 *    <jv>json</jv> = <jv>info</jv>.toString();
045 * </p>
046 * <p class='bjson'>
047 *    <jc>// Output</jc>
048 *    {
049 *       <js>"description"</js>: <js>"A complex object array response"</js>,
050 *       <js>"schema"</js>: {
051 *          <js>"type"</js>: <js>"array"</js>,
052 *          <js>"items"</js>: {
053 *             <js>"$ref"</js>: <js>"#/definitions/VeryComplexType"</js>
054 *          }
055 *       }
056 *    }
057 * </p>
058 *
059 * <ul class='seealso'>
060 *    <li class='link'>{@doc jd.Swagger}
061 *    <li class='extlink'>{@source}
062 * </ul>
063 */
064@Bean(properties="description,schema,headers,examples,*")
065@FluentSetters
066public class ResponseInfo extends SwaggerElement {
067
068   private String description;
069   private SchemaInfo schema;
070   private Map<String,HeaderInfo> headers;
071   private Map<String,Object> examples;
072
073   /**
074    * Default constructor.
075    */
076   public ResponseInfo() {}
077
078   /**
079    * Copy constructor.
080    *
081    * @param copyFrom The object to copy.
082    */
083   public ResponseInfo(ResponseInfo copyFrom) {
084      super(copyFrom);
085
086      this.description = copyFrom.description;
087      this.schema = copyFrom.schema == null ? null : copyFrom.schema.copy();
088
089      this.examples = copyOf(copyFrom.examples);
090
091      if (copyFrom.headers == null) {
092         this.headers = null;
093      } else {
094         this.headers = map();
095         copyFrom.headers.forEach((k,v) -> this.headers.put(k, v.copy()));
096      }
097
098   }
099
100   /**
101    * Make a deep copy of this object.
102    *
103    * @return A deep copy of this object.
104    */
105   public ResponseInfo copy() {
106      return new ResponseInfo(this);
107   }
108
109   /**
110    * Copies any non-null fields from the specified object to this object.
111    *
112    * @param r
113    *    The object to copy fields from.
114    *    <br>Can be <jk>null</jk>.
115    * @return This object.
116    */
117   public ResponseInfo copyFrom(ResponseInfo r) {
118      if (r != null) {
119         if (r.description != null)
120            description = r.description;
121         if (r.schema != null)
122            schema = r.schema;
123         if (r.headers != null)
124            headers = r.headers;
125         if (r.examples != null)
126            examples = r.examples;
127      }
128      return this;
129   }

291   @Override /* SwaggerElement */
292   public <T> T get(String property, Class<T> type) {
293      if (property == null)
294         return null;
295      switch (property) {
296         case "description": return toType(getDescription(), type);
297         case "examples": return toType(getExamples(), type);
298         case "headers": return toType(getHeaders(), type);
299         case "schema": return toType(getSchema(), type);
300         default: return super.get(property, type);
301      }
302   }
303
304   @Override /* SwaggerElement */
305   public ResponseInfo set(String property, Object value) {
306      if (property == null)
307         return this;
308      switch (property) {
309         case "description": return setDescription(stringify(value));
310         case "examples": return setExamples(mapBuilder(String.class,Object.class).sparse().addAny(value).build());
311         case "headers": return setHeaders(mapBuilder(String.class,HeaderInfo.class).sparse().addAny(value).build());
312         case "schema": return setSchema(toType(value, SchemaInfo.class));
313         default:
314            super.set(property, value);
315            return this;
316      }
317   }

319   @Override /* SwaggerElement */
320   public Set<String> keySet() {
321      Set<String> s = setBuilder(String.class)
322         .addIf(description != null, "description")
323         .addIf(examples != null, "examples")
324         .addIf(headers != null, "headers")
325         .addIf(schema != null, "schema")
326         .build();
327      return new MultiSet<>(s, super.keySet());
328   }

330   /**
331    * Resolves any <js>"$ref"</js> attributes in this element.
332    *
333    * @param swagger The swagger document containing the definitions.
334    * @param refStack Keeps track of previously-visited references so that we don't cause recursive loops.
335    * @param maxDepth
336    *    The maximum depth to resolve references.
337    *    <br>After that level is reached, <c>$ref</c> references will be left alone.
338    *    <br>Useful if you have very complex models and you don't want your swagger page to be overly-complex.
339    * @return
340    *    This object with references resolved.
341    *    <br>May or may not be the same object.
342    */
343   public ResponseInfo resolveRefs(Swagger swagger, Deque<String> refStack, int maxDepth) {
344
345      if (schema != null)
346         schema = schema.resolveRefs(swagger, refStack, maxDepth);
347
348      if (headers != null)
349         headers.entrySet().forEach(x -> x.setValue(x.getValue().resolveRefs(swagger, refStack, maxDepth)));
350
351      return this;
352   }
