013package org.apache.juneau.annotation;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016
017import java.util.*;
018
019import org.apache.juneau.*;
020import org.apache.juneau.http.header.*;
021import org.apache.juneau.reflect.*;
022import org.apache.juneau.svl.*;
023
024/**
025 * Utility classes and methods for the {@link BeanConfig @BeanConfig} annotation.
026 *
027 * <ul class='seealso'>
028 *    <li class='extlink'>{@source}
029 * </ul>
030 */
031public class BeanConfigAnnotation {
032
033   /**
034    * Applies {@link BeanConfig} annotations to a {@link org.apache.juneau.BeanContext.Builder}.
035    */
036   public static class Applier extends AnnotationApplier<BeanConfig,BeanContext.Builder> {
037
038      /**
039       * Constructor.
040       *
041       * @param vr The resolver for resolving values in annotations.
042       */
043      public Applier(VarResolverSession vr) {
044         super(BeanConfig.class, BeanContext.Builder.class, vr);
045      }
046
047      @Override
048      public void apply(AnnotationInfo<BeanConfig> ai, BeanContext.Builder b) {
049         BeanConfig a = ai.inner();
050
051         string(a.beanClassVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanClassVisibility(x));
052         string(a.beanConstructorVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanConstructorVisibility(x));
053         string(a.beanFieldVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanFieldVisibility(x));
054         string(a.beanMethodVisibility()).map(Visibility::valueOf).ifPresent(x -> b.beanMethodVisibility(x));
055         bool(a.beanMapPutReturnsOldValue()).ifPresent(x -> b.beanMapPutReturnsOldValue(x));
056         bool(a.beansRequireDefaultConstructor()).ifPresent(x -> b.beansRequireDefaultConstructor(x));
057         bool(a.beansRequireSerializable()).ifPresent(x -> b.beansRequireSerializable(x));
058         bool(a.beansRequireSettersForGetters()).ifPresent(x -> b.beansRequireSettersForGetters(x));
059         bool(a.disableBeansRequireSomeProperties()).ifPresent(x -> b.disableBeansRequireSomeProperties(x));
060         bool(a.debug()).ifPresent(x -> b.debug(x));
061         bool(a.findFluentSetters()).ifPresent(x -> b.findFluentSetters(x));
062         bool(a.ignoreInvocationExceptionsOnGetters()).ifPresent(x -> b.ignoreInvocationExceptionsOnGetters(x));
063         bool(a.ignoreInvocationExceptionsOnSetters()).ifPresent(x -> b.ignoreInvocationExceptionsOnSetters(x));
064         bool(a.disableIgnoreMissingSetters()).ifPresent(x -> b.disableIgnoreMissingSetters(x));
065         bool(a.disableIgnoreTransientFields()).ifPresent(x -> b.disableIgnoreTransientFields(x));
066         bool(a.ignoreUnknownBeanProperties()).ifPresent(x -> b.ignoreUnknownBeanProperties(x));
067         bool(a.ignoreUnknownEnumValues()).ifPresent(x -> b.ignoreUnknownEnumValues(x));
068         bool(a.disableIgnoreUnknownNullBeanProperties()).ifPresent(x -> b.disableIgnoreUnknownNullBeanProperties(x));
069         bool(a.sortProperties()).ifPresent(x -> b.sortProperties(x));
070         bool(a.useEnumNames()).ifPresent(x -> b.useEnumNames(x));
071         bool(a.disableInterfaceProxies()).ifPresent(x -> b.disableInterfaceProxies(x));
072         bool(a.useJavaBeanIntrospector()).ifPresent(x -> b.useJavaBeanIntrospector(x));
073         string(a.typePropertyName()).ifPresent(x -> b.typePropertyName(x));
074         string(a.locale()).map(Locale::forLanguageTag).ifPresent(x -> b.locale(x));
075         string(a.mediaType()).map(MediaType::of).ifPresent(x -> b.mediaType(x));
076         string(a.timeZone()).map(TimeZone::getTimeZone).ifPresent(x -> b.timeZone(x));
077         classes(a.dictionary()).ifPresent(x -> b.beanDictionary(x));
078         classes(a.dictionary_replace()).ifPresent(x -> { b.beanDictionary().clear(); b.beanDictionary(x);});
079         classes(a.swaps()).ifPresent(x -> b.swaps(x));
080         classes(a.swaps_replace()).ifPresent(x -> { b.swaps().clear(); b.swaps(x);});
081         classes(a.notBeanClasses()).ifPresent(x -> b.notBeanClasses(x));
082         classes(a.notBeanClasses_replace()).ifPresent(x -> { b.notBeanClasses().clear(); b.notBeanClasses(x);});
083         type(a.propertyNamer()).ifPresent(x -> b.propertyNamer(x));
084         alist(a.interfaces()).stream().map(x -> BeanAnnotation.create(x).interfaceClass(x).build()).forEach(x -> b.annotations(x));
085         strings(a.notBeanPackages()).ifPresent(x -> b.notBeanPackages(x));
086         strings(a.notBeanPackages_replace()).ifPresent(x -> {b.notBeanPackages().clear(); b.notBeanPackages(x);});
087      }
088   }
089}
