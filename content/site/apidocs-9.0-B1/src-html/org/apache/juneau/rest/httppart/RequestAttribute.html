013package org.apache.juneau.rest.httppart;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.util.*;
019
020import org.apache.juneau.http.response.*;
021import org.apache.juneau.rest.*;
022
023/**
024 * Represents a single request attribute on an HTTP request.
025 *
026 * <p>
027 * Typically accessed through the {@link RequestAttributes} class.
028 *
029 * <p>
030 *    Some important methods on this class are:
031 * </p>
032 * <ul class='javatree'>
033 *    <li class='jc'>{@link RequestAttribute}
034 *    <ul class='spaced-list'>
035 *       <li>Methods for retrieving simple string values:
036 *       <ul class='javatreec'>
037 *          <li class='jm'>{@link RequestAttribute#asString() asString()}
038 *          <li class='jm'>{@link RequestAttribute#get() get()}
039 *          <li class='jm'>{@link RequestAttribute#isPresent() isPresent()}
040 *          <li class='jm'>{@link RequestAttribute#orElse(Object) orElse(Object)}
041 *       </ul>
042 *       <li>Methods for retrieving as custom types:
043 *       <ul class='javatreec'>
044 *          <li class='jm'>{@link RequestAttribute#as(Class) as(Class)}
045 *       </ul>
046 *       <li>Methods for performing assertion checks:
047 *       <ul class='javatreec'>
048 *          <li class='jm'>{@link RequestAttribute#assertName() assertName()}
049 *          <li class='jm'>{@link RequestAttribute#assertValue() assertValue()}
050 *       </ul>
051 *       <li>Other methods:
052 *       <ul class='javatreec'>
053 *          <li class='jm'>{@link RequestAttribute#getName() getName()}
054 *          <li class='jm'>{@link RequestAttribute#getValue() getValue()}
055*     </ul>
056 * </ul>
057 *
058 * <ul class='seealso'>
059 *    <li class='link'>{@doc jrs.HttpParts}
060 *    <li class='extlink'>{@source}
061 * </ul>
062 */
063public class RequestAttribute extends BasicNamedAttribute {
064
065   private final RestRequest req;
066
067   /**
068    * Constructor.
069    *
070    * @param request The request object.
071    * @param name The parameter name.
072    * @param value The parameter value.
073    */
074   public RequestAttribute(RestRequest request, String name, Object value) {
075      super(name, value);
076      this.req = request;
077   }
078
079   //------------------------------------------------------------------------------------------------------------------
080   // Retrievers
081   //------------------------------------------------------------------------------------------------------------------
082
083   /**
084    * Returns the value of this part as a string.
085    *
086    * @return The value of this part as a string, or {@link Optional#empty()} if the part was not present.
087    */
088   public Optional<String> asString() {
089      return optional(stringify(getValue()));
090   }
091
092   /**
093    * Converts this part to the specified POJO.
094    *
095    * @param <T> The type to convert to.
096    * @param type The type to convert to.
097    * @return The converted type, or {@link Optional#empty()} if the part is not present.
098    * @throws BasicHttpException If value could not be parsed.
099    */
100   public <T> Optional<T> as(Class<T> type) {
101      return optional(req.getBeanSession().convertToType(getValue(), type));
102   }
103}
