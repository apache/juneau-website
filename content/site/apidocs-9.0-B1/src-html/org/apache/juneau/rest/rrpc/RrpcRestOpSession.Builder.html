013package org.apache.juneau.rest.rrpc;
014
015import static org.apache.juneau.internal.StringUtils.*;
016import static org.apache.juneau.rest.HttpRuntimeException.*;
017
018import java.io.*;
019import java.lang.reflect.*;
020
021import org.apache.juneau.http.remote.*;
022import org.apache.juneau.http.response.*;
023import org.apache.juneau.parser.*;
024import org.apache.juneau.rest.*;
025
026/**
027 * A session for a single HTTP request against an RRPC Java method.
028 *
029 * <ul class='notes'>
030 *    <li class='warn'>This class is not thread safe.
031 * </ul>
032 *
033 * <ul class='seealso'>
034 *    <li class='link'>{@doc jrs.RestRpc}
035 *    <li class='extlink'>{@source}
036 * </ul>
037 */
038public class RrpcRestOpSession extends RestOpSession {
039
040   //-----------------------------------------------------------------------------------------------------------------
041   // Static
042   //-----------------------------------------------------------------------------------------------------------------
043
044   /**
045    * Static creator.
046    *
047    * @param ctx The context of the RRPC Java Method.
048    * @param session The REST session creating this session.
049    * @return A new builder.
050    */
051   public static Builder create(RrpcRestOpContext ctx, RestSession session) {
052      return new Builder(ctx, session);
053
054   }
055
056   //-----------------------------------------------------------------------------------------------------------------
057   // Builder
058   //-----------------------------------------------------------------------------------------------------------------
059
060   /**
061    * Builder class.
062    */
063   public static class Builder extends RestOpSession.Builder {
064
065      RrpcRestOpContext ctx;
066
067      /**
068       * Constructor.
069       *
070       * @param ctx The context object of the RRPC Java method.
071       * @param session The REST session.
072       */
073      public Builder(RrpcRestOpContext ctx, RestSession session) {
074         super(ctx, session);
075         this.ctx = ctx;
076      }
077
078      @Override
079      public RrpcRestOpSession build() {
080         return new RrpcRestOpSession(this);
081      }
082
083   }
084
085   //-----------------------------------------------------------------------------------------------------------------
086   // Instance
087   //-----------------------------------------------------------------------------------------------------------------
088
089   private final RrpcRestOpContext ctx;
090
091   /**
092    * Constructor.
093    *
094    * @param builder The builder for this object.
095    */
096   protected RrpcRestOpSession(Builder builder) {
097      super(builder);
098      ctx = builder.ctx;
099   }
100
101   @Override
102   public void run() throws Throwable {
103
104      super.run();
105
106      RestRequest req = getRequest();
107      RestResponse res = getResponse();
108      RestSession session = getRestSession();
109
110      final Object o = res.hasContent() ? res.getContent(Object.class) : null;
111
112      if ("GET".equals(session.getMethod())) {
113         res.setContent(ctx.getMeta().getMethodsByPath().keySet());
114         return;
115
116      } else if ("POST".equals(session.getMethod())) {
117         String pip = session.getUrlPath().getPath();
118         if (pip.indexOf('/') != -1)
119            pip = pip.substring(pip.lastIndexOf('/')+1);
120         pip = urlDecode(pip);
121         RrpcInterfaceMethodMeta rmm = ctx.getMeta().getMethodMetaByPath(pip);
122         if (rmm != null) {
123            Method m = rmm.getJavaMethod();
124            try {
125               // Parse the args and invoke the method.
126               Parser p = req.getContent().getParserMatch().get().getParser();
127               Object[] args = null;
128               if (m.getGenericParameterTypes().length == 0)
129                  args = new Object[0];
130               else {
131                  try (Closeable in = p.isReaderParser() ? req.getReader() : req.getInputStream()) {
132                     args = p.parseArgs(in, m.getGenericParameterTypes());
133                  }
134               }
135               res.setContent(m.invoke(o, args));
136               return;
137            } catch (Exception e) {
138               throw toHttpException(e, InternalServerError.class);
139            }
140         }
141      }
142      throw new NotFound();
143   }
144}
