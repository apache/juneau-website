032/**
033 * Basic implementation of a {@link RestLogger} for logging HTTP requests.
034 *
035 * <p>
036 * Provides the following capabilities:
037 * <ul>
038 *    <li>Allows incoming HTTP requests to be logged at various {@link Enablement detail levels}.
039 *    <li>Allows rules to be defined to handle request logging differently depending on the resulting status code.
040 *    <li>Allows use of stack trace hashing to eliminate duplication of stack traces in log files.
041 *    <li>Allows customization of handling of where requests are logged to.
042 *    <li>Allows configuration via system properties or environment variables.
043 * </ul>
044 *
045 * <p>
046 * The following is an example of a logger that logs errors only when debugging is not enabled, and everything when
047 * logging is enabled.
048 *
049 * <h5 class='section'>Example:</h5>
050 * <p class='bjava'>
051 *       RestLogger <jv>logger</jv> = RestLogger
052 *          .<jsm>create</jsm>()
053 *          .logger(<js>"MyLogger"</js>)  <jc>// Use MyLogger Java logger.</jc>
054 *          .normalRules(  <jc>// Rules when debugging is not enabled.</jc>
055 *             <jsm>createRule</jsm>()  <jc>// Log 500+ errors with status-line and header information.</jc>
056 *                .statusFilter(x -&gt; x &gt;= 500)
057 *                .level(<jsf>SEVERE</jsf>)
058 *                .requestDetail(<jsf>HEADER</jsf>)
059 *                .responseDetail<jsf>(HEADER</jsf>)
060 *                .build(),
061 *             <jsm>createRule</jsm>()  <jc>// Log 400-500 errors with just status-line information.</jc>
062 *                .statusFilter(x -&gt; x &gt;= 400)
063 *                .level(<jsf>WARNING</jsf>)
064 *                .requestDetail(<jsf>STATUS_LINE</jsf>)
065 *                .responseDetail(<jsf>STATUS_LINE</jsf>)
066 *                .build()
067 *          )
068 *          .debugRules(  <jc>// Rules when debugging is enabled.</jc>
069 *             <jsm>createRule</jsm>()  <jc>// Log everything with full details.</jc>
070 *                .level(<jsf>SEVERE</jsf>)
071 *                .requestDetail(<jsf>ENTITY</jsf>)
072 *                .responseDetail(<jsf>ENTITY</jsf>)
073 *                .build()
074 *          )
075 *          .build()
076 *       ;
077 * </p>
078 *
079 * <ul class='seealso'>
080 *    <li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#callLogger()}
081 *    <li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#debugEnablement()}
082 *    <li class='jm'>{@link org.apache.juneau.rest.RestContext.Builder#debugOn(String)}
083 *    <li class='ja'>{@link Rest#debug}
084 *    <li class='ja'>{@link RestOp#debug}
085 *    <li class='link'>{@doc jrs.LoggingAndDebugging}
086 *    <li class='extlink'>{@source}
087 * </ul>
088 */
089public class BasicRestLogger implements RestLogger {
090
091   private static final RestLoggerRule DEFAULT_RULE = RestLoggerRule.create(BeanStore.INSTANCE).build();
092
093   private final Logger logger;
094   private final ThrownStore thrownStore;
095   private final RestLoggerRule[] normalRules, debugRules;
096   private final Enablement enabled;
097   private final Predicate<HttpServletRequest> enabledTest;
098   private final Level level;
099   private final RestLoggingDetail requestDetail, responseDetail;
100
101   /**
102    * Constructor.
103    *
104    * @param builder The builder object.
105    */
106   public BasicRestLogger(RestLogger.Builder builder) {
107      this.logger = builder.logger;
108      this.thrownStore = builder.thrownStore;
109      this.normalRules = builder.normalRules.toArray(new RestLoggerRule[builder.normalRules.size()]);
110      this.debugRules = builder.debugRules.toArray(new RestLoggerRule[builder.debugRules.size()]);
111      this.enabled = builder.enabled;
112      this.enabledTest = builder.enabledTest;
113      this.requestDetail = builder.requestDetail;
114      this.responseDetail = builder.responseDetail;
115      this.level = builder.level;
116   }
117
118   /**
119    * Called at the end of a servlet request to log the request.
120    *
121    * @param req The servlet request.
122    * @param res The servlet response.
123    */
124   @Override /* RestLogger */
125   public void log(HttpServletRequest req, HttpServletResponse res) {
126
127      RestLoggerRule rule = getRule(req, res);
128
129      if (! isEnabled(rule, req))
130         return;
131
132      Level level = firstNonNull(rule.getLevel(), this.level);
133
134      if (level == Level.OFF)
135         return;
136
137      Throwable e = castOrNull(req.getAttribute("Exception"), Throwable.class);
138      Long execTime = castOrNull(req.getAttribute("ExecTime"), Long.class);
139
140      RestLoggingDetail reqd = firstNonNull(rule.getRequestDetail(), requestDetail);
141      RestLoggingDetail resd = firstNonNull(rule.getResponseDetail(), responseDetail);
142
143      String method = req.getMethod();
144      int status = res.getStatus();
145      String uri = req.getRequestURI();
146      byte[] reqContent = getRequestContent(req);
147      byte[] resContent = getResponseContent(req, res);
148
149      StringBuilder sb = new StringBuilder();
150
151      if (reqd != STATUS_LINE || resd != STATUS_LINE)
152         sb.append("\n=== HTTP Call (incoming) ======================================================\n");
153
154      ThrownStats sti = getThrownStats(e);
155
156      sb.append('[').append(status);
157
158      if (sti != null) {
159         int count = sti.getCount();
160         sb.append(',').append(StringUtils.toHex8(sti.getHash())).append('.').append(count);
161         if (count > 1)
162            e = null;
163      }
164
165      sb.append("] ");
166
167      sb.append("HTTP ").append(method).append(' ').append(uri);
168
169      if (reqd != STATUS_LINE || resd != STATUS_LINE) {
170
171         if (reqd.isOneOf(HEADER, ENTITY)) {
172            String qs = req.getQueryString();
173            if (qs != null)
174               sb.append('?').append(qs);
175         }
176
177         if (reqContent != null && reqd.isOneOf(HEADER ,ENTITY))
178            sb.append("\n\tRequest length: ").append(reqContent.length).append(" bytes");
179
180         if (resd.isOneOf(HEADER, ENTITY))
181            sb.append("\n\tResponse code: ").append(status);
182
183         if (resContent != null && resd.isOneOf(HEADER, ENTITY))
184            sb.append("\n\tResponse length: ").append(resContent.length).append(" bytes");
185
186         if (execTime != null && resd.isOneOf(HEADER, ENTITY))
187            sb.append("\n\tExec time: ").append(execTime).append("ms");
188
189         if (reqd.isOneOf(HEADER, ENTITY)) {
190            Enumeration<String> hh = req.getHeaderNames();
191            if (hh.hasMoreElements()) {
192               sb.append("\n---Request Headers---");
193               while (hh.hasMoreElements()) {
194                  String h = hh.nextElement();
195                  sb.append("\n\t").append(h).append(": ").append(req.getHeader(h));
196               }
197            }
198         }
199
200         if (resd.isOneOf(HEADER, ENTITY)) {
201            Collection<String> hh = res.getHeaderNames();
202            if (hh.size() > 0) {
203               sb.append("\n---Response Headers---");
204               for (String h : hh) {
205                  sb.append("\n\t").append(h).append(": ").append(res.getHeader(h));
206               }
207            }
208         }
209
210         if (reqContent != null && reqContent.length > 0 && reqd == ENTITY) {
211            try {
212               sb.append("\n---Request Content UTF-8---");
213               sb.append("\n").append(new String(reqContent, IOUtils.UTF8));
214               sb.append("\n---Request Content Hex---");
215               sb.append("\n").append(toSpacedHex(reqContent));
216            } catch (Exception e1) {
217               sb.append("\n").append(e1.getLocalizedMessage());
218            }
219         }
220
221         if (resContent != null && resContent.length > 0 && resd == ENTITY) {
222            try {
223               sb.append("\n---Response Content UTF-8---");
224               sb.append("\n").append(new String(resContent, IOUtils.UTF8));
225               sb.append("\n---Response Content Hex---");
226               sb.append("\n").append(toSpacedHex(resContent));
227            } catch (Exception e1) {
228               sb.append(e1.getLocalizedMessage());
229            }
230         }
231         sb.append("\n=== END ======================================================================");
232      }
233
234      if (rule.isLogStackTrace() && e == null)
235         e = new Throwable("Stacktrace");
236
237      log(level, sb.toString(), e);
238
239   }
240
241   /**
242    * Given the specified servlet request/response, find the rule that applies to it.
243    *
244    * <p>
245    * This method can be overridden to provide specialized logic for finding rules.
246    *
247    * @param req The servlet request.
248    * @param res The servlet response.
249    * @return The applicable logging rule, or the default rule if not found.  Never <jk>null</jk>.
250    */
251   protected RestLoggerRule getRule(HttpServletRequest req, HttpServletResponse res) {
252      for (RestLoggerRule r : isDebug(req) ? debugRules : normalRules)
253         if (r.matches(req, res))
254            return r;
255      return DEFAULT_RULE;
256   }
257
258   /**
259    * Returns <jk>true</jk> if debug is enabled on this request.
260    *
261    * <p>
262    * Looks for the request attribute <js>"Debug"</js> to determine whether debug is enabled.
263    *
264    * <p>
265    * This method can be overridden to provide specialized logic for determining whether debug mode is enabled on a request.
266    *
267    * @param req The HTTP request being logged.
268    * @return <jk>true</jk> if debug is enabled on this request.
269    * @see org.apache.juneau.rest.RestContext.Builder#debugEnablement()
270    * @see org.apache.juneau.rest.RestContext.Builder#debugOn(String)
271    * @see Rest#debug()
272    * @see RestOp#debug()
273    */
274   protected boolean isDebug(HttpServletRequest req) {
275      return firstNonNull(castOrNull(req.getAttribute("Debug"), Boolean.class), false);
276   }
277
278   /**
279    * Returns <jk>true</jk> if logging is enabled for this request.
280    *
281    * <p>
282    * Uses the enabled and enabled-test settings on the matched rule and this logger to determine whether a REST
283    * call should be logged.
284    *
285    * <p>
286    * This method can be overridden to provide specialized logic for determining whether a REST call should be logged.
287    *
288    * @param rule The first matching rule.  Never <jk>null</jk>.
289    * @param req The HTTP request.
290    * @return <jk>true</jk> if logging is enabled for this request.
291    */
292   protected boolean isEnabled(RestLoggerRule rule, HttpServletRequest req) {
293      Enablement enabled = firstNonNull(rule.getEnabled(), this.enabled);
294      Predicate<HttpServletRequest> enabledTest = firstNonNull(rule.getEnabledTest(), this.enabledTest);
295      return enabled.isEnabled(enabledTest.test(req));
296   }

303   /**
304    * Returns the logger to use for logging REST calls.
305    *
306    * <p>
307    * Returns the logger specified in the builder, or {@link Logger#getGlobal()} if it wasn't specified.
308    *
309    * <p>
310    * This method can be overridden in subclasses to provide a different logger.
311    *
312    * @return The logger to use for logging REST calls.
313    */
314   protected Logger getLogger() {
315      return logger;
316   }
317
318   /**
319    * Logs the specified message to the logger.
320    *
321    * <p>
322    * Subclasses can override this method to capture messages being sent to the logger and handle it differently.
323    *
324    * @param level The log level.
325    * @param msg The log message.
326    * @param e The exception.
327    */
328   protected void log(Level level, String msg, Throwable e) {
329      getLogger().log(level, msg, e);
330   }
331
332   private byte[] getRequestContent(HttpServletRequest req) {
333      if (req instanceof CachingHttpServletRequest)
334         return ((CachingHttpServletRequest)req).getContent();
335      return castOrNull(req.getAttribute("RequestContent"), byte[].class);
336   }
337
338   private byte[] getResponseContent(HttpServletRequest req, HttpServletResponse res) {
339      if (res instanceof CachingHttpServletResponse)
340         return ((CachingHttpServletResponse)res).getContent();
341      return castOrNull(req.getAttribute("ResponseContent"), byte[].class);
342   }
343
344   private ThrownStats getThrownStats(Throwable e) {
345      if (e == null || thrownStore == null)
346         return null;
347      return thrownStore.getStats(e).orElse(null);
348   }
349
350   @Override /* Object */
351   public String toString() {
352      return filteredMap()
353         .append("logger", logger)
354         .append("thrownStore", thrownStore)
355         .append("enabled", enabled)
356         .append("level", level)
357         .append("requestDetail", requestDetail)
358         .append("responseDetail", responseDetail)
359         .append("normalRules", normalRules.length == 0 ? null : normalRules)
360         .append("debugRules", debugRules.length == 0 ? null : debugRules)
361         .asReadableString();
362   }
