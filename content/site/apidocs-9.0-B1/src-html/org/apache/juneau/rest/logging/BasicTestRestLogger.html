013package org.apache.juneau.rest.logging;
014
015import static org.apache.juneau.internal.StringUtils.*;
016import static org.apache.juneau.Enablement.*;
017import static java.util.logging.Level.*;
018import static org.apache.juneau.rest.logging.RestLoggingDetail.*;
019
020import javax.servlet.http.*;
021
022import org.apache.juneau.cp.*;
023import org.apache.juneau.rest.*;
024
025/**
026 * Default implementation of a {@link RestLogger} that only logs REST call errors unless no-log is enabled on the request.
027 *
028 * <p>
029 * Useful for REST tests where you know that a particular call is going to produce an error response and you don't want that
030 * response to be logged producing noisy test output.
031 *
032 * <p>
033 * Requests can be tagged as no-log (meaning don't log if there's an error) in any of the following ways:
034 * <ul>
035 *    <li>A <js>"No-Trace: true"</js> header.
036 *    <li>A <js>"noTrace=true"</js> query parameter.
037 *    <li>A <js>"NoTrace"</js> request attribute with a string value of <js>"true"</js>.
038 * </ul>
039 *
040 * <h5 class='section'>Configured Settings:</h5>
041 * <ul>
042 *    <li>Logs to the {@link RestContext#getLogger() context logger}.
043 *    <li>Only calls with status code &gt;=400 will be logged.
044 *    <li>Logs full request and response entity.
045 * </ul>
046 *
047 * <ul class='seealso'>
048 *    <li class='link'>{@doc jrs.LoggingAndDebugging}
049 *    <li class='extlink'>{@source}
050 * </ul>
051 */
052public class BasicTestRestLogger extends BasicRestLogger {
053
054   /**
055    * Constructor.
056    *
057    * @param context The context of the resource object.
058    */
059   public BasicTestRestLogger(RestContext context) {
060      super(builder().logger(context.getLogger()).thrownStore(context.getThrownStore()));
061   }
062
063   private static RestLogger.Builder builder() {
064      BeanStore bs = BeanStore.INSTANCE;
065      return RestLogger.create(bs)
066         .normalRules(  // Rules when debugging is not enabled.
067            RestLoggerRule.create(bs)  // Log 500+ errors with status-line and header information.
068               .statusFilter(x -> x >= 400)
069               .level(SEVERE)
070               .requestDetail(HEADER)
071               .responseDetail(HEADER)
072               .enabled(CONDITIONAL)
073               .enabledPredicate(x -> ! isNoTrace(x))  // Only log if it's not a no-trace request.
074               .logStackTrace()
075               .build(),
076            RestLoggerRule.create(bs)  // Log 400-500 errors with just status-line information.
077               .statusFilter(x -> x >= 400)
078               .level(WARNING)
079               .requestDetail(STATUS_LINE)
080               .responseDetail(STATUS_LINE)
081               .enabled(CONDITIONAL)
082               .enabledPredicate(x -> ! isNoTrace(x))  // Only log if it's not a no-trace request.
083               .logStackTrace()
084               .build()
085         )
086         .debugRules(  // Rules when debugging is enabled.
087            RestLoggerRule.create(bs)  // Log everything with full details.
088               .level(SEVERE)
089               .requestDetail(ENTITY)
090               .responseDetail(ENTITY)
091               .logStackTrace()
092               .build()
093         )
094      ;
095   }
096
097   private static boolean isNoTrace(HttpServletRequest req) {
098      Object o = req.getAttribute("NoTrace");
099      if (o != null)
100         return "true".equalsIgnoreCase(o.toString());
101      String s = req.getHeader("No-Trace");
102      if (s != null)
103         return "true".equalsIgnoreCase(s);
104      return emptyIfNull(req.getQueryString()).contains("noTrace=true");
105   }
106}
