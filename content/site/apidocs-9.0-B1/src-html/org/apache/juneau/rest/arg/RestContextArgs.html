030/**
031 * Resolves method parameters on {@link RestOp}-annotated Java methods of types found on the {@link RestContext} object.
032 *
033 * <ul class='javatree'>
034 *    <li class='jc'>{@link BeanContext}
035 *    <li class='jc'>{@link Config}
036 *    <li class='jc'>{@link DebugEnablement}
037 *    <li class='jc'>{@link EncoderSet}
038 *    <li class='jc'>{@link FileFinder}
039 *    <li class='jc'>{@link Logger}
040 *    <li class='jc'>{@link MethodExecStore}
041 *    <li class='jc'>{@link RestChildren}
042 *    <li class='jc'>{@link RestContext}
043 *    <li class='jc'>{@link RestContextStats}
044 *    <li class='jc'>{@link RestLogger}
045 *    <li class='jc'>{@link RestOperations}
046 *    <li class='jc'>{@link StaticFiles}
047 *    <li class='jc'>{@link ThrownStore}
048 * </ul>
049 *
050 * <ul class='seealso'>
051 *    <li class='link'>{@doc jrs.JavaMethodParameters}
052 *    <li class='extlink'>{@source}
053 * </ul>
054 */
055public class RestContextArgs extends SimpleRestOperationArg {
056
057   /**
058    * Static creator.
059    *
060    * @param paramInfo The Java method parameter being resolved.
061    * @return A new arg, or <jk>null</jk> if the parameter type is not one of the supported types.
062    */
063   public static RestContextArgs create(ParamInfo paramInfo) {
064      if (paramInfo.isType(BeanContext.class))
065         return new RestContextArgs(x->x.getBeanContext());
066      if (paramInfo.isType(Config.class))
067         return new RestContextArgs(x->x.getConfig());
068      if (paramInfo.isType(DebugEnablement.class))
069         return new RestContextArgs(x->x.getDebugEnablement());
070      if (paramInfo.isType(EncoderSet.class))
071         return new RestContextArgs(x->x.getEncoders());
072      if (paramInfo.isType(FileFinder.class))
073         return new RestContextArgs(x->x.getFileFinder());
074      if (paramInfo.isType(Logger.class))
075         return new RestContextArgs(x->x.getLogger());
076      if (paramInfo.isType(MethodExecStore.class))
077         return new RestContextArgs(x->x.getMethodExecStore());
078      if (paramInfo.isType(RestChildren.class))
079         return new RestContextArgs(x->x.getRestChildren());
080      if (paramInfo.isType(RestContext.class))
081         return new RestContextArgs(x->x);
082      if (paramInfo.isType(RestContextStats.class))
083         return new RestContextArgs(x->x.getStats());
084      if (paramInfo.isType(RestLogger.class))
085         return new RestContextArgs(x->x.getCallLogger());
086      if (paramInfo.isType(RestOperations.class))
087         return new RestContextArgs(x->x.getRestOperations());
088      if (paramInfo.isType(StaticFiles.class))
089         return new RestContextArgs(x->x.getStaticFiles());
090      if (paramInfo.isType(ThrownStore.class))
091         return new RestContextArgs(x->x.getThrownStore());
092      return null;
093   }
094
095   /**
096    * Constructor.
097    *
098    * @param <T> The function return type.
099    * @param function The function for finding the arg.
100    */
101   protected <T> RestContextArgs(ThrowingFunction<RestContext,T> function) {
102      super((session)->function.apply(session.getRestContext()));
103   }
104}
