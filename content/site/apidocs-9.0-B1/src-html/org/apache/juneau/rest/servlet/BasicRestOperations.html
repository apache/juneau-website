013package org.apache.juneau.rest.servlet;
014
015import static org.apache.juneau.http.HttpMethod.*;
016
017import java.util.*;
018
019import org.apache.juneau.annotation.*;
020import org.apache.juneau.dto.swagger.Swagger;
021import org.apache.juneau.dto.swagger.ui.*;
022import org.apache.juneau.html.annotation.*;
023import org.apache.juneau.http.annotation.*;
024import org.apache.juneau.http.resource.*;
025import org.apache.juneau.jsonschema.annotation.*;
026import org.apache.juneau.rest.*;
027import org.apache.juneau.rest.annotation.*;
028import org.apache.juneau.rest.stats.*;
029
030/**
031 * Basic REST operation methods.
032 *
033 * <p>
034 *    Defines 5 special use REST operation endpoints:
035 * </p>
036 *
037 * <p class='bjava'>
038 *    <ja>@RestGet</ja>(path=<js>"/api/*"</js></js>)
039 *    <jk>public</jk> {@link Swagger} {@link #getSwagger(RestRequest) getSwagger}({@link RestRequest} <jv>req</jv>);
040 *
041 *    <ja>@RestGet</ja>(path=<js>"/htdocs/*"</js>)
042 *    <jk>public</jk> {@link HttpResource} {@link #getHtdoc(String,Locale) getHtdoc}(<ja>@Path</ja> String <jv>path</jv>, Locale <jv>locale</jv>);
043 *
044 *    <ja>@RestGet</ja>(path=<js>"favicon.ico"</js>)
045 *    <jk>public</jk> {@link HttpResource} {@link #getFavIcon() getFavIcon}();
046 *
047 *    <ja>@RestGet</ja>(path=<js>"/stats"</js>)
048 *    <jk>public</jk> {@link RestContextStats} {@link #getStats(RestRequest) getStats}({@link RestRequest} <jv>req</jv>);
049 *
050 *    <ja>@RestOp</ja>(method=<jsf>ANY</jsf>, path=<js>"/error"</js>)
051 *    <jk>public void</jk> {@link #error() error}();
052 * </p>
053 *
054 * <p>
055 *    Implementations provided by the following classes:
056 * </p>
057 * <ul class='javatreec'>
058 *    <li class='jac'>{@link BasicRestServlet}
059 *    <li class='jac'>{@link BasicRestServletGroup}
060 *    <li class='jac'>{@link BasicRestObject}
061 *    <li class='jac'>{@link BasicRestObjectGroup}
062 * </ul>
063 *
064 * <ul class='seealso'>
065 *    <li class='link'>{@doc jrs.AnnotatedClasses}
066 *    <li class='extlink'>{@source}
067 * </ul>
068 */
069@HtmlDocConfig(
070   // Basic page navigation links.
071   navlinks={
072      "up: request:/..",
073      "api: servlet:/api",
074      "stats: servlet:/stats"
075   }
076)
077@JsonSchemaConfig(
078   // Add descriptions to the following types when not specified:
079   addDescriptionsTo="bean,collection,array,map,enum",
080   // Add example to the following types:
081   addExamplesTo="bean,collection,array,map",
082   // Don't generate schema information on the Swagger bean itself or HTML beans.
083   ignoreTypes="Swagger,org.apache.juneau.dto.html5.*",
084   // Use $ref references for bean definitions to reduce duplication in Swagger.
085   useBeanDefs="true"
086)
087public interface BasicRestOperations {
088
089   /**
090    * [GET /api] - Show resource options.
091    *
092    * @param req The HTTP request.
093    * @return A bean containing the contents for the OPTIONS page.
094    */
095   @RestGet(
096      path="/api/*",
097      summary="Swagger documentation",
098      description="Swagger documentation for this resource."
099   )
100   @HtmlDocConfig(
101      // Should override config annotations defined on class.
102      rank=10,
103      // Override the nav links for the swagger page.
104      navlinks={
105         "back: servlet:/",
106         "json: servlet:/?Accept=text/json&plainText=true"
107      },
108      // Never show aside contents of page inherited from class.
109      aside="NONE"
110   )
111   @BeanConfig(
112      // POJO swaps to apply to all serializers/parsers on this method.
113      swaps={
114         // Use the SwaggerUI swap when rendering Swagger beans.
115         // This is a per-media-type swap that only applies to text/html requests.
116         SwaggerUI.class
117      }
118   )
119   public Swagger getSwagger(RestRequest req);
120
121   /**
122    * [GET /htdocs/*] - Retrieve static file.
123    *
124    * @param path The path to retrieve.
125    * @param locale The locale of the HTTP request.
126    * @return An HTTP resource representing the static file.
127    */
128   @RestGet(
129      path="/htdocs/*",
130      summary="Static files",
131      description="Static file retrieval."
132   )
133   public HttpResource getHtdoc(@Path String path, Locale locale);
134
135   /**
136    * [GET favicon.ico] - Retrieve favorites icon image.
137    *
138    * @return A bean containing the contents for the OPTIONS page.
139    */
140   @RestGet(
141      path="favicon.ico",
142      summary="Favorites icon.",
143      description="Favorites icon."
144   )
145   public HttpResource getFavIcon();
146
147   /**
148    * [* /error] - Error occurred.
149    */
150   @RestOp(
151      method=ANY,
152      path="/error",
153      summary="Error occurred",
154      description={
155         "An error occurred during handling of the request.  ",
156         "Servlet chains will often automatically redirect to '/error' when any sort of error condition occurs ",
157         "(such as failed authentication) and will set appropriate response parameters ",
158         "(such as an WWW-Authenticate response header)."
159      }
160   )
161   public void error();
162
163   /**
164    * [GET /stats] - Timing statistics.
165    *
166    * <p>
167    * Timing statistics for method invocations on this resource.
168    *
169    * @param req The HTTP request.
170    * @return A collection of timing statistics for each annotated method on this resource.
171    */
172   @RestGet(
173      path="/stats",
174      summary="Timing statistics",
175      description="Timing statistics for method invocations on this resource."
176   )
177   @HtmlDocConfig(
178      // Should override config annotations defined on class.
179      rank=10,
180      // Override the nav links for the swagger page.
181      navlinks={
182         "back: servlet:/",
183         "json: servlet:/stats?Accept=text/json&plainText=true"
184      },
185      // Never show aside contents of page inherited from class.
186      aside="NONE"
187   )
188   public RestContextStats getStats(RestRequest req);
189}
