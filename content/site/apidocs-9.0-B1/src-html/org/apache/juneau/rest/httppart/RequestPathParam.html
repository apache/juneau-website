019

026/**
027 * Represents a single path parameter on an HTTP request.
028 *
029 * <p>
030 * Typically accessed through the {@link RequestPathParams} class.
031 *
032 * <p>
033 *    Some important methods on this class are:
034 * </p>
035 * <ul class='javatree'>
036 *    <li class='jc'>{@link RequestPathParam}
037 *    <ul class='spaced-list'>
038 *       <li>Methods for retrieving simple string values:
039 *       <ul class='javatreec'>
040 *          <li class='jm'>{@link RequestPathParam#asString() asString()}
041 *          <li class='jm'>{@link RequestPathParam#get() get()}
042 *          <li class='jm'>{@link RequestPathParam#isPresent() isPresent()}
043 *          <li class='jm'>{@link RequestPathParam#orElse(String) orElse(String)}
044 *       </ul>
045 *       <li>Methods for retrieving as other common types:
046 *       <ul class='javatreec'>
047 *          <li class='jm'>{@link RequestPathParam#asBoolean() asBoolean()}
048 *          <li class='jm'>{@link RequestPathParam#asBooleanPart() asBooleanPart()}
049 *          <li class='jm'>{@link RequestPathParam#asCsvArray() asCsvArray()}
050 *          <li class='jm'>{@link RequestPathParam#asCsvArrayPart() asCsvArrayPart)}
051 *          <li class='jm'>{@link RequestPathParam#asDate() asDate()}
052 *          <li class='jm'>{@link RequestPathParam#asDatePart() asDatePart()}
053 *          <li class='jm'>{@link RequestPathParam#asInteger() asInteger()}
054 *          <li class='jm'>{@link RequestPathParam#asIntegerPart() asIntegerPart()}
055 *          <li class='jm'>{@link RequestPathParam#asLong() asLong()}
056 *          <li class='jm'>{@link RequestPathParam#asLongPart() asLongPart()}
057 *          <li class='jm'>{@link RequestPathParam#asMatcher(Pattern) asMatcher(Pattern)}
058 *          <li class='jm'>{@link RequestPathParam#asMatcher(String) asMatcher(String)}
059 *          <li class='jm'>{@link RequestPathParam#asMatcher(String,int) asMatcher(String,int)}
060 *          <li class='jm'>{@link RequestPathParam#asStringPart() asStringPart()}
061 *          <li class='jm'>{@link RequestPathParam#asUriPart() asUriPart()}
062 *       </ul>
063 *       <li>Methods for retrieving as custom types:
064 *       <ul class='javatreec'>
065 *          <li class='jm'>{@link RequestPathParam#as(Class) as(Class)}
066 *          <li class='jm'>{@link RequestPathParam#as(ClassMeta) as(ClassMeta)}
067 *          <li class='jm'>{@link RequestPathParam#as(Type,Type...) as(Type,Type...)}
068 *          <li class='jm'>{@link RequestPathParam#parser(HttpPartParserSession) parser(HttpPartParserSession)}
069 *          <li class='jm'>{@link RequestPathParam#schema(HttpPartSchema) schema(HttpPartSchema)}
070 *       </ul>
071 *       <li>Methods for performing assertion checks:
072 *       <ul class='javatreec'>
073 *          <li class='jm'>{@link RequestPathParam#assertCsvArray() assertCsvArray()}
074 *          <li class='jm'>{@link RequestPathParam#assertDate() assertDate()}
075 *          <li class='jm'>{@link RequestPathParam#assertInteger() assertInteger()}
076 *          <li class='jm'>{@link RequestPathParam#assertLong() assertLong()}
077 *          <li class='jm'>{@link RequestPathParam#assertString() assertString()}
078 *       </ul>
079 *       <li>Other methods:
080 *       <ul class='javatreec'>
081 *          <li class='jm'>{@link RequestPathParam#getName() getName()}
082 *          <li class='jm'>{@link RequestPathParam#getValue() getValue()}
083 *       </ul>
084 * </ul>
085 *
086 * <ul class='seealso'>
087 *    <li class='link'>{@doc jrs.HttpParts}
088 *    <li class='extlink'>{@source}
089 * </ul>
090 */
091public class RequestPathParam extends RequestHttpPart implements NameValuePair {
092
093   private final String value;
094
095   /**
096    * Constructor.
097    *
098    * @param request The request object.
099    * @param name The parameter name.
100    * @param value The parameter value.
101    */
102   public RequestPathParam(RestRequest request, String name, String value) {
103      super(PATH, request, name);
104      this.value = value;
105   }
106
107   //------------------------------------------------------------------------------------------------------------------
108   // Retrievers
109   //------------------------------------------------------------------------------------------------------------------
110
111   @Override /* RequestHttpPart */
112   public String getValue() {
113      return value;
114   }
115
116   //------------------------------------------------------------------------------------------------------------------
117   // Assertions
118   //------------------------------------------------------------------------------------------------------------------
119
120   /**
121    * Provides the ability to perform fluent-style assertions on this parameter.
122    *
123    * <h5 class='section'>Examples:</h5>
124    * <p class='bjava'>
125    *    <jv>request</jv>
126    *       .getPathParam(<js>"foo"</js>)
127    *       .assertString().contains(<js>"bar"</js>);
128    * </p>
129    *
130    * <p>
131    * The assertion test returns the original object allowing you to chain multiple requests like so:
132    * <p class='bjava'>
133    *    String <jv>foo</jv> = <jv>request</jv>
134    *       .getPathParam(<js>"foo"</js>)
135    *       .assertString().contains(<js>"bar"</js>)
136    *       .asString().get();
137    * </p>
138    *
139    * @return A new fluent assertion object.
140    */
141   public FluentStringAssertion<RequestPathParam> assertString() {
142      return new FluentStringAssertion<>(orElse(null), this);
143   }
144
145   /**
146    * Provides the ability to perform fluent-style assertions on an integer parameter.
147    *
148    * <h5 class='section'>Examples:</h5>
149    * <p class='bjava'>
150    *    <jv>request</jv>
151    *       .getPathParam(<js>"age"</js>)
152    *       .assertInteger().isGreaterThan(1);
153    * </p>
154    *
155    * @return A new fluent assertion object.
156    */
157   public FluentIntegerAssertion<RequestPathParam> assertInteger() {
158      return new FluentIntegerAssertion<>(asIntegerPart().asInteger().orElse(null), this);
159   }
160
161   /**
162    * Provides the ability to perform fluent-style assertions on a long parameter.
163    *
164    * <h5 class='section'>Examples:</h5>
165    * <p class='bjava'>
166    *    <jv>request</jv>
167    *       .getPathParam(<js>"length"</js>)
168    *       .assertLong().isLessThan(100000);
169    * </p>
170    *
171    * @return A new fluent assertion object.
172    */
173   public FluentLongAssertion<RequestPathParam> assertLong() {
174      return new FluentLongAssertion<>(asLongPart().asLong().orElse(null), this);
175   }
176
177   /**
178    * Provides the ability to perform fluent-style assertions on a date parameter.
179    *
180    * <h5 class='section'>Examples:</h5>
181    * <p class='bjava'>
182    *    <jv>request</jv>
183    *       .getPathParam(<js>"time"</js>)
184    *       .assertDate().isAfterNow();
185    * </p>
186    *
187    * @return A new fluent assertion object.
188    */
189   public FluentZonedDateTimeAssertion<RequestPathParam> assertDate() {
190      return new FluentZonedDateTimeAssertion<>(asDatePart().asZonedDateTime().orElse(null), this);
191   }
192
193   /**
194    * Provides the ability to perform fluent-style assertions on comma-separated string parameters.
195    *
196    * <h5 class='section'>Examples:</h5>
197    * <p class='bjava'>
198    *    <jv>request</jv>
199    *       .getPathParam(<js>"allow"</js>)
200    *       .assertCsvArray().contains(<js>"GET"</js>);
201    * </p>
202    *
203    * @return A new fluent assertion object.
204    */
205   public FluentListAssertion<String,RequestPathParam> assertCsvArray() {
206      return new FluentListAssertion<>(asCsvArrayPart().asList().orElse(null), this);
207   }
208
209   // <FluentSetters>
210
211   @Override /* GENERATED */
212   public RequestPathParam schema(HttpPartSchema value) {
213      super.schema(value);
214      return this;
215   }
216
217   @Override /* GENERATED */
218   public RequestPathParam parser(HttpPartParserSession value) {
219      super.parser(value);
220      return this;
221   }
222   // </FluentSetters>
223}
