034/**
035 * Identical to {@link RestServlet} but doesn't extend from {@link HttpServlet}.
036 *
037 * <p>
038 * This is particularly useful in Spring Boot environments that auto-detect servlets to deploy in servlet containers,
039 * but you want this resource to be deployed as a child instead.
040 *
041 * <ul class='seealso'>
042 *    <li class='link'>{@doc jrs.AnnotatedClasses}
043 *    <li class='extlink'>{@source}
044 * </ul>
045 */
046public abstract class RestObject {
047
048   private AtomicReference<RestContext> context = new AtomicReference<>();
049
050   //-----------------------------------------------------------------------------------------------------------------
051   // Context methods.
052   //-----------------------------------------------------------------------------------------------------------------
053
054   /**
055    * Sets the context object for this servlet.
056    *
057    * @param context Sets the context object on this servlet.
058    * @throws ServletException If error occurred during post-initialiation.
059    */
060   protected void setContext(RestContext context) throws ServletException {
061      this.context.set(context);
062   }
063
064   /**
065    * Returns the read-only context object that contains all the configuration information about this resource.
066    *
067    * @return The context information on this servlet.
068    */
069   protected RestContext getContext() {
070      if (context.get() == null)
071         throw new InternalServerError("RestContext object not set on resource.");
072      return context.get();
073   }
074
075   //-----------------------------------------------------------------------------------------------------------------
076   // Convenience logger methods
077   //-----------------------------------------------------------------------------------------------------------------
078
079   /**
080    * Log a message at {@link Level#INFO} level.
081    *
082    * <p>
083    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
084    *
085    * @param msg The message to log.
086    */
087   public void log(String msg) {
088      doLog(Level.INFO, null, () -> msg);
089   }
090
091   /**
092    * Log a message.
093    *
094    * <p>
095    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
096    *
097    * @param msg The message to log.
098    * @param cause The cause.
099    */
100   public void log(String msg, Throwable cause) {
101      doLog(Level.INFO, null, () -> msg);
102   }
103
104   /**
105    * Log a message.
106    *
107    * <p>
108    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
109    *
110    * @param level The log level.
111    * @param msg The message to log.
112    * @param args Optional {@link MessageFormat}-style arguments.
113    */
114   public void log(Level level, String msg, Object...args) {
115      doLog(level, null, () -> StringUtils.format(msg, args));
116   }
117
118   /**
119    * Log a message.
120    *
121    * <p>
122    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
123    *
124    * @param level The log level.
125    * @param cause The cause.
126    * @param msg The message to log.
127    * @param args Optional {@link MessageFormat}-style arguments.
128    */
129   public void log(Level level, Throwable cause, String msg, Object...args) {
130      doLog(level, cause, () -> StringUtils.format(msg, args));
131   }
132
133   /**
134    * Main logger method.
135    *
136    * <p>
137    * The default behavior logs a message to the Java logger of the class name.
138    *
139    * <p>
140    * Subclasses can override this method to implement their own logger handling.
141    *
142    * @param level The log level.
143    * @param cause Optional throwable.
144    * @param msg The message to log.
145    */
146   protected void doLog(Level level, Throwable cause, Supplier<String> msg) {
147      RestContext c = context.get();
148      Logger logger = c == null ? null : c.getLogger();
149      if (logger == null)
150         logger = Logger.getLogger(className(this));
151      logger.log(level, cause, msg);
152   }

158   /**
159    * Method that gets called during servlet initialization.
160    *
161    * <p>
162    * This method is called from within the {@link Servlet#init(ServletConfig)} method after the {@link org.apache.juneau.rest.RestContext.Builder}
163    * object has been created and initialized with the annotations defined on the class, but before the
164    * {@link RestContext} object has been created.
165    *
166    * <p>
167    * An example of this is the <c>PetStoreResource</c> class that uses an init method to perform initialization
168    * of an internal data structure.
169    *
170    * <h5 class='figure'>Example:</h5>
171    * <p class='bjava'>
172    *    <ja>@Rest</ja>(...)
173    *    <jk>public class</jk> PetStoreResource <jk>extends</jk> BasicRestServlet <jk>implements</jk> BasicUniversalJenaConfig {
174    *
175    *       <jc>// Our database.</jc>
176    *       <jk>private</jk> Map&lt;Integer,Pet&gt; <jf>petDB</jf>;
177    *
178    *       <ja>@Override</ja>
179    *       <jk>public void</jk> onInit(RestContext.Builder <jv>builder</jv>) <jk>throws</jk> Exception {
180    *          <jc>// Load our database from a local JSON file.</jc>
181    *          <jf>petDB</jf> = JsonParser.<jsf>DEFAULT</jsf>.parse(getClass().getResourceAsStream(<js>"PetStore.json"</js>), LinkedHashMap.<jk>class</jk>, Integer.<jk>class</jk>, Pet.<jk>class</jk>);
182    *       }
183    *    }
184    * </p>
185    *
186    * <ul class='notes'>
187    *    <li class='note'>
188    *       The default implementation of this method is a no-op.
189    *    <li class='note'>
190    *       Multiple INIT methods can be defined on a class.
191    *       <br>INIT methods on parent classes are invoked before INIT methods on child classes.
192    *       <br>The order of INIT method invocations within a class is alphabetical, then by parameter count, then by parameter types.
193    *    <li class='note'>
194    *       The method can throw any exception causing initialization of the servlet to fail.
195    * </ul>
196    *
197    * @param builder Context builder which can be used to configure the servlet.
198    * @throws Exception Any exception thrown will cause servlet to fail startup.
199    */
200   @RestHook(INIT)
201   public void onInit(RestContext.Builder builder) throws Exception {}
202
203   /**
204    * Method that gets called immediately after servlet initialization.
205    *
206    * <p>
207    * This method is called from within the {@link Servlet#init(ServletConfig)} method after the {@link RestContext}
208    * object has been created.
209    *
210    * <ul class='notes'>
211    *    <li class='note'>
212    *       The default implementation of this method is a no-op.
213    *    <li class='note'>
214    *       Multiple POST_INIT methods can be defined on a class.
215    *       <br>POST_INIT methods on parent classes are invoked before POST_INIT methods on child classes.
216    *       <br>The order of POST_INIT method invocations within a class is alphabetical, then by parameter count, then by parameter types.
217    *    <li class='note'>
218    *       The method can throw any exception causing initialization of the servlet to fail.
219    * </ul>
220    *
221    * @param context The initialized context object.
222    * @throws Exception Any exception thrown will cause servlet to fail startup.
223    */
224   @RestHook(POST_INIT)
225   public void onPostInit(RestContext context) throws Exception {}
226
227   /**
228    * Identical to {@link #onPostInit(RestContext)} except the order of execution is child-resources first.
229    *
230    * <p>
231    * Use this method if you need to perform any kind of initialization on child resources before the parent resource.
232    *
233    * <p>
234    * This method is called from within the {@link Servlet#init(ServletConfig)} method after the {@link RestContext}
235    * object has been created and after the {@link HookEvent#POST_INIT} methods have been called.
236    *
237    * <p>
238    * The only valid parameter type for this method is {@link RestContext} which can be used to retrieve information
239    * about the servlet.
240    *
241    * <ul class='notes'>
242    *    <li class='note'>
243    *       The default implementation of this method is a no-op.
244    *    <li class='note'>
245    *       Multiple POST_INIT_CHILD_FIRST methods can be defined on a class.
246    *       <br>POST_INIT_CHILD_FIRST methods on parent classes are invoked before POST_INIT_CHILD_FIRST methods on child classes.
247    *       <br>The order of POST_INIT_CHILD_FIRST method invocations within a class is alphabetical, then by parameter count, then by parameter types.
248    *    <li class='note'>
249    *       The method can throw any exception causing initialization of the servlet to fail.
250    * </ul>
251    *
252    * @param context The initialized context object.
253    * @throws Exception Any exception thrown will cause servlet to fail startup.
254    */
255   @RestHook(POST_INIT_CHILD_FIRST)
256   public void onPostInitChildFirst(RestContext context) throws Exception {}
257
258   /**
259    * Method that gets called during servlet destroy.
260    *
261    * <p>
262    * This method is called from within the {@link Servlet#destroy()}.
263    *
264    * <h5 class='figure'>Example:</h5>
265    * <p class='bjava'>
266    *    <ja>@Rest</ja>(...)
267    *    <jk>public class</jk> PetStoreResource <jk>extends</jk> BasicRestServlet <jk>implements</jk> BasicUniversalJenaConfig {
268    *
269    *       <jc>// Our database.</jc>
270    *       <jk>private</jk> Map&lt;Integer,Pet&gt; <jf>petDB</jf>;
271    *
272    *       <ja>@Override</ja>
273    *       <jk>public void</jk> onDestroy(RestContext <jv>context</jv>) {
274    *          <jf>petDB</jf> = <jk>null</jk>;
275    *       }
276    *    }
277    * </p>
278    *
279    * <ul class='notes'>
280    *    <li class='note'>
281    *       The default implementation of this method is a no-op.
282    *    <li class='note'>
283    *       Multiple DESTROY methods can be defined on a class.
284    *       <br>DESTROY methods on child classes are invoked before DESTROY methods on parent classes.
285    *       <br>The order of DESTROY method invocations within a class is alphabetical, then by parameter count, then by parameter types.
286    *    <li class='note'>
287    *       In general, destroy methods should not throw any exceptions, although if any are thrown, the stack trace will be
288    *       printed to <c>System.err</c>.
289    * </ul>
290    *
291    * @param context The initialized context object.
292    * @throws Exception Any exception thrown will cause stack trace to be printed to <c>System.err</c>.
293    */
294   @RestHook(DESTROY)
295   public void onDestroy(RestContext context) throws Exception {}
296
297   /**
298    * A method that is called immediately after the <c>HttpServlet.service(HttpServletRequest, HttpServletResponse)</c>
299    * method is called.
300    *
301    * <p>
302    * Note that you only have access to the raw request and response objects at this point.
303    *
304    * <h5 class='figure'>Example:</h5>
305    * <p class='bjava'>
306    *    <ja>@Rest</ja>(...)
307    *    <jk>public class</jk> MyResource <jk>extends</jk> BasicRestServlet <jk>implements</jk> BasicUniversalConfig {
308    *
309    *       <jc>// Add a request attribute to all incoming requests.</jc>
310    *       <ja>@Override</ja>
311    *       <jk>public void</jk> onStartCall(HttpServletRequest <jv>req</jv>, HttpServletResponse <jv>res</jv>) {
312    *          <jv>req</jv>.setAttribute(<js>"foobar"</js>, <jk>new</jk> FooBar());
313    *       }
314    *    }
315    * </p>
316    *
317    * <ul class='notes'>
318    *    <li class='note'>
319    *       The default implementation of this method is a no-op.
320    *    <li class='note'>
321    *       Multiple START_CALL methods can be defined on a class.
322    *       <br>START_CALL methods on parent classes are invoked before START_CALL methods on child classes.
323    *       <br>The order of START_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
324    *    <li class='note'>
325    *       The method can throw any exception.
326    *       <br>{@link BasicHttpException HttpExceptions} can be thrown to cause a particular HTTP error status code.
327    *       <br>All other exceptions cause an HTTP 500 error status code.
328    * </ul>
329    *
330    * @param req The HTTP servlet request object.
331    * @param res The HTTP servlet response object.
332    * @throws Exception Any exception.
333    */
334   @RestHook(START_CALL)
335   public void onStartCall(HttpServletRequest req, HttpServletResponse res) throws Exception {}
336
337   /**
338    * Method that gets called immediately before the <ja>@RestOp</ja> annotated method gets called.
339    *
340    * <p>
341    * At this point, the {@link RestRequest} object has been fully initialized, and all {@link RestGuard} and
342    * {@link RestMatcher} objects have been called.
343    *
344    * <ul class='notes'>
345    *    <li class='note'>
346    *       The default implementation of this method is a no-op.
347    *    <li class='note'>
348    *       Multiple PRE_CALL methods can be defined on a class.
349    *       <br>PRE_CALL methods on parent classes are invoked before PRE_CALL methods on child classes.
350    *       <br>The order of PRE_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
351    *    <li class='note'>
352    *       The method can throw any exception.
353    *       <br>{@link BasicHttpException HttpExceptions} can be thrown to cause a particular HTTP error status code.
354    *       <br>All other exceptions cause an HTTP 500 error status code.
355    *    <li class='note'>
356    *       It's advisable not to mess around with the HTTP content itself since you may end up consuming the content
357    *       before the actual REST method has a chance to use it.
358    * </ul>
359    *
360    * @param req The request object.
361    * @param res The response object.
362    * @throws Exception Any exception.
363    */
364   @RestHook(PRE_CALL)
365   public void onPreCall(RestRequest req, RestResponse res) throws Exception {}
366
367   /**
368    * Method that gets called immediately after the <ja>@RestOp</ja> annotated method gets called.
369    *
370    * <p>
371    * At this point, the output object returned by the method call has been set on the response, but
372    * {@link RestConverter RestConverters} have not yet been executed and the response has not yet been written.
373    *
374    * <ul class='notes'>
375    *    <li class='note'>
376    *       The default implementation of this method is a no-op.
377    *    <li class='note'>
378    *       Multiple POST_CALL methods can be defined on a class.
379    *       <br>POST_CALL methods on parent classes are invoked before POST_CALL methods on child classes.
380    *       <br>The order of POST_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
381    *    <li class='note'>
382    *       The method can throw any exception, although at this point it is too late to set an HTTP error status code.
383    * </ul>
384    *
385    * @param req The request object.
386    * @param res The response object.
387    * @throws Exception Any exception.
388    */
389   @RestHook(POST_CALL)
390   public void onPostCall(RestRequest req, RestResponse res) throws Exception {}
391
392   /**
393    * Method that gets called right before we exit the servlet service method.
394    *
395    * <p>
396    * At this point, the output has been written and flushed.
397    *
398    * <p>
399    * The following attributes are set on the {@link HttpServletRequest} object that can be useful for logging purposes:
400    * <ul>
401    *    <li><js>"Exception"</js> - Any exceptions thrown during the request.
402    *    <li><js>"ExecTime"</js> - Execution time of the request.
403    * </ul>
404    *
405    * <ul class='notes'>
406    *    <li class='note'>
407    *       The default implementation of this method is a no-op.
408    *    <li class='note'>
409    *       Multiple END_CALL methods can be defined on a class.
410    *       <br>END_CALL methods on parent classes are invoked before END_CALL methods on child classes.
411    *       <br>The order of END_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
412    *    <li class='note'>
413    *       The method can throw any exception, although at this point it is too late to set an HTTP error status code.
414    *    <li class='note'>
415    *       Note that if you override a parent method, you probably need to call <code><jk>super</jk>.parentMethod(...)</code>.
416    *       <br>The method is still considered part of the parent class for ordering purposes even though it's
417    *       overridden by the child class.
418    * </ul>
419    *
420    * @param req The HTTP servlet request object.
421    * @param res The HTTP servlet response object.
422    * @throws Exception Any exception.
423    */
424   @RestHook(END_CALL)
425   public void onEndCall(HttpServletRequest req, HttpServletResponse res) throws Exception {}

431   /**
432    * Returns the current HTTP request.
433    *
434    * @return The current HTTP request, or <jk>null</jk> if it wasn't created.
435    */
436   public synchronized RestRequest getRequest() {
437      return getContext().getLocalSession().getOpSession().getRequest();
438   }

440   /**
441    * Returns the current HTTP response.
442    *
443    * @return The current HTTP response, or <jk>null</jk> if it wasn't created.

445   public synchronized RestResponse getResponse() {
446      return getContext().getLocalSession().getOpSession().getResponse();
447   }
