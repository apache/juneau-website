013package org.apache.juneau.rest.mock;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.ThrowableUtils.*;
017
018import java.io.*;
019import java.util.*;
020import java.util.logging.*;
021import java.util.logging.Formatter;
022
023import org.apache.juneau.assertions.*;
024
025/**
026 * Simplified logger for intercepting and asserting logging messages.
027 *
028 * <h5 class='figure'>Example:</h5>
029 * <p class='bjava'>
030 *    <jc>// Instantiate a mock logger.</jc>
031 *    MockLogger <jv>logger</jv> = <jk>new</jk> MockLogger();
032 *
033 *    <jc>// Associate it with a MockRestClient.</jc>
034 *    MockRestClient
035 *       .<jsm>create</jsm>(MyRestResource.<jk>class</jk>)
036 *       .simpleJson()
037 *       .logger(<jv>logger</jv>)
038 *       .logRequests(DetailLevel.<jsf>FULL</jsf>, Level.<jsf>SEVERE</jsf>)
039 *       .build()
040 *       .post(<js>"/bean"</js>, <jv>bean</jv>)
041 *       .complete();
042 *
043 *    <jc>// Assert that logging occurred.</jc>
044 *    <jv>logger</jv>.assertLastLevel(Level.<jsf>SEVERE</jsf>);
045 *    <jv>logger</jv>.assertLastMessage().is(
046 *       <js>"=== HTTP Call (outgoing) ======================================================"</js>,
047 *       <js>"=== REQUEST ==="</js>,
048 *       <js>"POST http://localhost/bean"</js>,
049 *       <js>"---request headers---"</js>,
050 *       <js>" Accept: application/json+simple"</js>,
051 *       <js>"---request entity---"</js>,
052 *       <js>" Content-Type: application/json+simple"</js>,
053 *       <js>"---request content---"</js>,
054 *       <js>"{f:1}"</js>,
055 *       <js>"=== RESPONSE ==="</js>,
056 *       <js>"HTTP/1.1 200 "</js>,
057 *       <js>"---response headers---"</js>,
058 *       <js>" Content-Type: application/json"</js>,
059 *       <js>"---response content---"</js>,
060 *       <js>"{f:1}"</js>,
061 *       <js>"=== END ======================================================================="</js>,
062 *       <js>""</js>
063 *    );
064 * </p>
065 *
066 * <ul class='seealso'>
067 *    <li class='link'>{@doc juneau-rest-mock}
068 *    <li class='extlink'>{@source}
069 * </ul>
070 */
071public class MockLogger extends Logger {
072
073   private static final String FORMAT_PROPERTY = "java.util.logging.SimpleFormatter.format";
074
075   private final List<LogRecord> logRecords = list();
076   private final ByteArrayOutputStream baos = new ByteArrayOutputStream();
077   private volatile Formatter formatter;
078   private volatile String format = "%4$s: %5$s%6$s%n";
079
080   /**
081    * Constructor.
082    */
083   public MockLogger() {
084      super("Mock", null);
085   }
086
087   /**
088    * Creator.
089    *
090    * @return A new {@link MockLogger} object.
091    */
092   public static MockLogger create() {
093      return new MockLogger();
094   }
095
096   @Override /* Logger */
097   public synchronized void log(LogRecord record) {
098      logRecords.add(record);
099      try {
100         baos.write(getFormatter().format(record).getBytes("UTF-8"));
101      } catch (Exception e) {
102         throw runtimeException(e);
103      }
104   }
105
106   private Formatter getFormatter() {
107      if (formatter == null) {
108         synchronized(this) {
109            String oldFormat = System.getProperty(FORMAT_PROPERTY);
110            System.setProperty(FORMAT_PROPERTY, format);
111            formatter = new SimpleFormatter();
112            if (oldFormat == null)
113               System.clearProperty(FORMAT_PROPERTY);
114            else
115               System.setProperty(FORMAT_PROPERTY, oldFormat);
116         }
117      }
118      return formatter;
119   }
120
121   /**
122    * Sets the level for this logger.
123    *
124    * @param level The new level for this logger.
125    * @return This object.
126    */
127   public synchronized MockLogger level(Level level) {
128      super.setLevel(level);
129      return this;
130   }
131
132   /**
133    * Specifies the format for messages sent to the log file.
134    *
135    * <p>
136    * See {@link SimpleFormatter#format(LogRecord)} for the syntax of this string.
137    *
138    * @param format The format string.
139    * @return This object.
140    */
141   public synchronized MockLogger format(String format) {
142      this.format = format;
143      return this;
144   }
145
146   /**
147    * Overrides the formatter to use for formatting messages.
148    *
149    * <p>
150    * The default uses {@link SimpleFormatter}.
151    *
152    * @param formatter The log record formatter.
153    * @return This object.
154    */
155   public synchronized MockLogger formatter(Formatter formatter) {
156      this.formatter = formatter;
157      return this;
158   }
159
160   /**
161    * Resets this logger.
162    *
163    * @return This object.
164    */
165   public synchronized MockLogger reset() {
166      logRecords.clear();
167      baos.reset();
168      return this;
169   }
170
171   /**
172    * Asserts that this logger was called.
173    *
174    * @return This object.
175    */
176   public synchronized MockLogger assertLogged() {
177      if (logRecords.isEmpty())
178         throw new AssertionError("Message not logged");
179      return this;
180   }
181
182   /**
183    * Asserts that the last message was logged at the specified level.
184    *
185    * @param level The level to match against.
186    * @return This object.
187    */
188   public synchronized MockLogger assertLastLevel(Level level) {
189      assertLogged();
190      if (last().getLevel() != level)
191         throw new AssertionError("Message logged at [" + last().getLevel() + "] instead of [" + level + "]");
192      return this;
193   }
194
195   /**
196    * Asserts that the last message matched the specified message.
197    *
198    * @return This object.
199    */
200   public synchronized FluentStringAssertion<MockLogger> assertLastMessage() {
201      assertLogged();
202      return new FluentStringAssertion<>(last().getMessage(), this);
203   }
204
205   /**
206    * Asserts that the specified number of messages have been logged.
207    *
208    * @return This object.
209    */
210   public synchronized FluentIntegerAssertion<MockLogger> assertRecordCount() {
211      return new FluentIntegerAssertion<>(logRecords.size(), this);
212   }
213
214   /**
215    * Allows you to perform fluent-style assertions on the contents of the log file.
216    *
217    * @return A new fluent-style assertion object.
218    */
219   public synchronized FluentStringAssertion<MockLogger> assertContents() {
220      return new FluentStringAssertion<>(baos.toString(), this);
221   }
222
223   private LogRecord last() {
224      if (logRecords.isEmpty())
225         throw new AssertionError("Message not logged");
226      return logRecords.get(logRecords.size()-1);
227   }
228
229   /**
230    * Returns the contents of this log file as a string.
231    */
232   @Override
233   public String toString() {
234      return baos.toString();
235   }
236}
