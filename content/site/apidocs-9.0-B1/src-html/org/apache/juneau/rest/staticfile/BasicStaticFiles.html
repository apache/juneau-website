033/**
034 * API for retrieving localized static files from either the classpath or file system.
035 *
036 * <p>
037 * Provides the same functionality as {@link BasicFileFinder} but adds support for returning files as {@link HttpResource}
038 * objects with arbitrary headers.
039 *
040 * <ul class='seealso'>
041 *    <li class='link'>{@doc jrs.StaticFiles}
042 *    <li class='extlink'>{@source}
043 * </ul>
044 */
045public class BasicStaticFiles implements StaticFiles {
046
047   private final Header[] headers;
048   private final MimetypesFileTypeMap mimeTypes;
049   private final int hashCode;
050   private final FileFinder fileFinder;
051
052   /**
053    * Creates a new builder for this object.
054    *
055    * @param beanStore The bean store to use for creating beans.
056    * @return A new builder for this object.
057    */
058   public static StaticFiles.Builder create(BeanStore beanStore) {
059      return new StaticFiles.Builder(beanStore);
060   }
061
062   /**
063    * Constructor.
064    *
065    * @param builder The builder object.
066    */
067   public BasicStaticFiles(StaticFiles.Builder builder) {
068      this.headers = builder.headers.toArray(new Header[builder.headers.size()]);
069      this.mimeTypes = builder.mimeTypes;
070      this.hashCode = HashCode.of(hashCode(), headers);
071      this.fileFinder = builder.fileFinder.build();
072   }
073
074   /**
075    * Constructor.
076    *
077    * <p>
078    * Can be used when subclassing and overriding the {@link #resolve(String, Locale)} method.
079    */
080   protected BasicStaticFiles() {
081      super();
082      this.headers = new Header[0];
083      this.mimeTypes = null;
084      this.hashCode = HashCode.of(hashCode(), headers);
085      this.fileFinder = null;
086   }
087
088   /**
089    * Resolve the specified path.
090    *
091    * <p>
092    * Subclasses can override this method to provide specialized handling.
093    *
094    * @param path The path to resolve to a static file.
095    * @param locale Optional locale.
096    * @return The resource, or <jk>null</jk> if not found.
097    */
098   @Override /* StaticFiles */
099   public Optional<HttpResource> resolve(String path, Locale locale) {
100      try {
101         Optional<InputStream> is = getStream(path);
102         if (! is.isPresent())
103            return empty();
104         return optional(
105            streamResource(is.get())
106               .header(contentType(mimeTypes == null ? null : mimeTypes.getContentType(getFileName(path))))
107               .headers(headers)
108               .build()
109         );
110      } catch (IOException e) {
111         throw new InternalServerError(e);
112      }
113   }
114
115   @Override
116   public int hashCode() {
117      return hashCode;
118   }
119
120   @Override /* Object */
121   public boolean equals(Object o) {
122      return super.equals(o) && o instanceof BasicStaticFiles && eq(this, (BasicStaticFiles)o, (x,y)->eq(x.headers, y.headers));
123   }
124
125   @Override /* FileFinder */
126   public Optional<InputStream> getStream(String name, Locale locale) throws IOException {
127      return fileFinder.getStream(name, locale);
128   }
129
130   @Override /* FileFinder */
131   public Optional<InputStream> getStream(String name) throws IOException {
132      return fileFinder.getStream(name);
133   }
134
135   @Override /* FileFinder */
136   public Optional<String> getString(String name) throws IOException {
137      return fileFinder.getString(name);
138   }
139
140   @Override /* FileFinder */
141   public Optional<String> getString(String name, Locale locale) throws IOException {
142      return fileFinder.getString(name, locale);
143   }
144
145   @Override /* Object */
146   public String toString() {
147      return filteredMap()
148         .append("headers", headers)
149         .asReadableString();
150   }
151}
