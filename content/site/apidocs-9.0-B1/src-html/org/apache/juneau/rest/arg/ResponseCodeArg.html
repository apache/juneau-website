013package org.apache.juneau.rest.arg;
014
015import java.lang.reflect.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.http.annotation.*;
019import org.apache.juneau.reflect.*;
020import org.apache.juneau.rest.*;
021import org.apache.juneau.rest.annotation.*;
022
023/**
024 * Resolves method parameters annotated with {@link StatusCode} on {@link RestOp}-annotated Java methods.
025 *
026 * <ul class='seealso'>
027 *    <li class='link'>{@doc jrs.JavaMethodParameters}
028 *    <li class='extlink'>{@source}
029 * </ul>
030 */
031public class ResponseCodeArg implements RestOpArg {
032
033   private final Type type;
034
035   /**
036    * Static creator.
037    *
038    * @param paramInfo The Java method parameter being resolved.
039    * @return A new {@link ResponseCodeArg}, or <jk>null</jk> if the parameter is not annotated with {@link StatusCode}.
040    */
041   public static ResponseCodeArg create(ParamInfo paramInfo) {
042      if (paramInfo.hasAnnotation(StatusCode.class) || paramInfo.getParameterType().hasAnnotation(StatusCode.class))
043         return new ResponseCodeArg(paramInfo);
044      return null;
045   }
046
047   /**
048    * Constructor.
049    *
050    * @param paramInfo The Java method parameter being resolved.
051    */
052   protected ResponseCodeArg(ParamInfo paramInfo) {
053      this.type = paramInfo.getParameterType().innerType();
054      Class<?> c = type instanceof Class ? (Class<?>)type : type instanceof ParameterizedType ? (Class<?>)((ParameterizedType)type).getRawType() : null;
055      if (c != Value.class || Value.getParameterType(type) != Integer.class)
056         throw new ArgException(paramInfo, "Type must be Value<Integer> on parameter annotated with @StatusCode annotation");
057   }
058
059   @SuppressWarnings({ "unchecked", "rawtypes" })
060   @Override /* RestOpArg */
061   public Object resolve(final RestOpSession opSession) throws Exception {
062      Value<Object> v = new Value();
063      v.listener(new ValueListener() {
064         @Override
065         public void onSet(Object o) {
066            opSession.getResponse().setStatus(Integer.parseInt(o.toString()));
067         }
068      });
069      return v;
070   }
071}
