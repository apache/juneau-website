013package org.apache.juneau.httppart;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016
017import org.apache.http.*;
018
019/**
020 * Represents an instance of an HTTP part.
021 *
022 * <p>
023 * Can be used to represent both request and response parts.
024 *
025 * <ul class='seealso'>
026 *    <li class='link'>{@doc jm.HttpPartSerializersParsers}
027 *    <li class='extlink'>{@source}
028 * </ul>
029 */
030public class HttpPart implements NameValuePair {
031   private final String name;
032   private final Object opart;
033   private final String spart;
034   private final HttpPartType partType;
035   private final HttpPartSchema schema;
036   private final HttpPartSerializerSession serializer;
037
038   /**
039    * Constructor.
040    *
041    * <p>
042    * Used when the part is in POJO form and needs to be converted to a String.
043    *
044    * @param name The HTTP part name (e.g. the header name).
045    * @param partType The HTTP part type.
046    * @param schema Schema information about the part.
047    * @param serializer The part serializer to use to serialize the part.
048    * @param part The part POJO being serialized.
049    */
050   public HttpPart(String name, HttpPartType partType, HttpPartSchema schema, HttpPartSerializerSession serializer, Object part) {
051      this.name = name;
052      this.partType = partType;
053      this.schema = schema;
054      this.serializer = serializer;
055      this.opart = part;
056      this.spart = null;
057   }
058
059   @Override /* NameValuePair */
060   public String getName() {
061      return name;
062   }
063
064   @Override /* NameValuePair */
065   public String getValue() {
066      if (spart != null)
067         return spart;
068      try {
069         return serializer.serialize(partType, schema, opart);
070      } catch (Exception e) {
071         throw runtimeException(e);
072      }
073   }
074}
