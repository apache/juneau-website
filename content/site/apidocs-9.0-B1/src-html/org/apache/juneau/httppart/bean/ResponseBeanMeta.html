031/**
032 * Represents the metadata gathered from a parameter or class annotated with {@link Response}.
033 *
034 * <ul class='seealso'>
035 *    <li class='link'>{@doc jm.HttpPartSerializersParsers}
036 *    <li class='extlink'>{@source}
037 * </ul>
038 */
039public class ResponseBeanMeta {
040
041   /**
042    * Represents a non-existent meta object.
043    */
044   public static ResponseBeanMeta NULL = new ResponseBeanMeta(new Builder(AnnotationWorkList.create()));
045
046   /**
047    * Create metadata from specified class.
048    *
049    * @param t The class annotated with {@link Response}.
050    * @param annotations The annotations to apply to any new part serializers or parsers.
051    * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Response}.
052    */
053   public static ResponseBeanMeta create(Type t, AnnotationWorkList annotations) {
054      ClassInfo ci = ClassInfo.of(t).unwrap(Value.class, Optional.class);
055      if (ci.hasNoAnnotation(Response.class))
056         return null;
057      Builder b = new Builder(annotations);
058      b.apply(ci.innerType());
059      ci.forEachAnnotation(Response.class, x -> true, x -> b.apply(x));
060      ci.forEachAnnotation(StatusCode.class, x -> true, x -> b.apply(x));
061      return b.build();
062   }
063
064   /**
065    * Create metadata from specified method return.
066    *
067    * @param m The method annotated with {@link Response}.
068    * @param annotations The annotations to apply to any new part serializers or parsers.
069    * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Response}.
070    */
071   public static ResponseBeanMeta create(MethodInfo m, AnnotationWorkList annotations) {
072      if (! (m.hasAnnotation(Response.class) || m.getReturnType().unwrap(Value.class,Optional.class).hasAnnotation(Response.class)))
073         return null;
074      Builder b = new Builder(annotations);
075      b.apply(m.getReturnType().unwrap(Value.class, Optional.class).innerType());
076      m.forEachAnnotation(Response.class, x -> true, x -> b.apply(x));
077      m.forEachAnnotation(StatusCode.class, x -> true, x -> b.apply(x));
078      return b.build();
079   }
080
081   /**
082    * Create metadata from specified method parameter.
083    *
084    * @param mpi The method parameter.
085    * @param annotations The annotations to apply to any new part serializers or parsers.
086    * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Response}.
087    */
088   public static ResponseBeanMeta create(ParamInfo mpi, AnnotationWorkList annotations) {
089      if (mpi.hasNoAnnotation(Response.class))
090         return null;
091      Builder b = new Builder(annotations);
092      b.apply(mpi.getParameterType().unwrap(Value.class, Optional.class).innerType());
093      mpi.forEachAnnotation(Response.class, x-> true, x -> b.apply(x));
094      mpi.forEachAnnotation(StatusCode.class, x -> true, x -> b.apply(x));
095      return b.build();
096   }
097
098   //-----------------------------------------------------------------------------------------------------------------
099   // Instance
100   //-----------------------------------------------------------------------------------------------------------------
101
102   private final ClassMeta<?> cm;
103   private final Map<String,ResponseBeanPropertyMeta> properties;
104   private final int code;
105   private final Map<String,ResponseBeanPropertyMeta> headerMethods;
106   private final ResponseBeanPropertyMeta statusMethod, contentMethod;
107   private final Optional<HttpPartSerializer> partSerializer;
108   private final Optional<HttpPartParser> partParser;
109   private final HttpPartSchema schema;
110
111   ResponseBeanMeta(Builder b) {
112      this.cm = b.cm;
113      this.code = b.code;
114      this.partSerializer = optional(b.partSerializer).map(x -> HttpPartSerializer.creator().type(x).apply(b.annotations).create());
115      this.partParser = optional(b.partParser).map(x -> HttpPartParser.creator().type(x).apply(b.annotations).create());
116      this.schema = b.schema.build();
117
118      Map<String,ResponseBeanPropertyMeta> properties = map();
119
120      Map<String,ResponseBeanPropertyMeta> hm = map();
121      b.headerMethods.forEach((k,v) -> {
122         ResponseBeanPropertyMeta pm = v.build(partSerializer, partParser);
123         hm.put(k, pm);
124         properties.put(pm.getGetter().getName(), pm);
125      });
126      this.headerMethods = unmodifiable(hm);
127
128      this.contentMethod = b.contentMethod == null ? null : b.contentMethod.schema(schema).build(partSerializer, partParser);
129      this.statusMethod = b.statusMethod == null ? null : b.statusMethod.build(empty(), empty());
130
131      if (contentMethod != null)
132         properties.put(contentMethod.getGetter().getName(), contentMethod);
133      if (statusMethod != null)
134         properties.put(statusMethod.getGetter().getName(), statusMethod);
135
136      this.properties = unmodifiable(properties);
137   }
138
139   static class Builder {
140      ClassMeta<?> cm;
141      int code;
142      AnnotationWorkList annotations;
143      Class<? extends HttpPartSerializer> partSerializer;
144      Class<? extends HttpPartParser> partParser;
145      HttpPartSchema.Builder schema = HttpPartSchema.create();
146
147      Map<String,ResponseBeanPropertyMeta.Builder> headerMethods = map();
148      ResponseBeanPropertyMeta.Builder contentMethod;
149      ResponseBeanPropertyMeta.Builder statusMethod;
150
151      Builder(AnnotationWorkList annotations) {
152         this.annotations = annotations;
153      }
154
155      Builder apply(Type t) {
156         Class<?> c = ClassUtils.toClass(t);
157         this.cm = BeanContext.DEFAULT.getClassMeta(c);
158         ClassInfo ci = cm.getInfo();
159         ci.forEachPublicMethod(x -> true, x -> {
160            assertNoInvalidAnnotations(x, Query.class, FormData.class);
161            if (x.hasAnnotation(Header.class)) {
162               assertNoArgs(x, Header.class);
163               assertReturnNotVoid(x, Header.class);
164               HttpPartSchema s = HttpPartSchema.create(x.getAnnotation(Header.class), x.getPropertyName());
165               headerMethods.put(s.getName(), ResponseBeanPropertyMeta.create(RESPONSE_HEADER, s, x));
166            } else if (x.hasAnnotation(StatusCode.class)) {
167               assertNoArgs(x, Header.class);
168               assertReturnType(x, Header.class, int.class, Integer.class);
169               statusMethod = ResponseBeanPropertyMeta.create(RESPONSE_STATUS, x);
170            } else if (x.hasAnnotation(Content.class)) {
171               if (x.hasNoParams())
172                  assertReturnNotVoid(x, Header.class);
173               else
174                  assertArgType(x, Header.class, OutputStream.class, Writer.class);
175               contentMethod = ResponseBeanPropertyMeta.create(RESPONSE_BODY, x);
176            }
177         });
178         return this;
179      }
180
181      Builder apply(Response a) {
182         if (a != null) {
183            if (isNotVoid(a.serializer()))
184               partSerializer = a.serializer();
185            if (isNotVoid(a.parser()))
186               partParser = a.parser();
187            schema.apply(a.schema());
188         }
189         return this;
190      }
191
192      Builder apply(StatusCode a) {
193         if (a != null) {
194            if (a.value().length > 0)
195               code = a.value()[0];
196         }
197         return this;
198      }
199
200      ResponseBeanMeta build() {
201         return new ResponseBeanMeta(this);
202      }
203   }

268   /**
269    * Returns metadata about the bean property with the specified method getter name.
