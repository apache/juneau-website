013package org.apache.juneau.internal;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.ThrowableUtils.*;
017
018import java.nio.charset.*;
019import java.util.*;
020import java.util.concurrent.*;
021import java.util.function.*;
022
023/**
024 * Utility methods for accessing system properties and environment variables.
025 *
026 * <ul class='seealso'>
027 *    <li class='extlink'>{@source}
028 * </ul>
029 */
030public class SystemEnv {
031
032   /**
033    * Looks up a system property or environment variable.
034    *
035    * <p>
036    * First looks in system properties.  Then converts the name to env-safe and looks in the system environment.
037    * Then returns the default if it can't be found.
038    *
039    * @param <T> The type to convert the value to.
040    * @param name The property name.
041    * @param def The default value if not found.
042    * @return The default value.
043    */
044   public static <T> T env(String name, T def) {
045      return env(name).map(x -> toType(x, def)).orElse(def);
046   }
047
048   /**
049    * Looks up a system property or environment variable.
050    *
051    * <p>
052    * First looks in system properties.  Then converts the name to env-safe and looks in the system environment.
053    * Then returns the default if it can't be found.
054    *
055    * @param name The property name.
056    * @return The value if found.
057    */
058   public static Optional<String> env(String name) {
059      String s = System.getProperty(name);
060      if (s == null)
061         s = System.getenv(envName(name));
062      return optional(s);
063   }
064
065   @SuppressWarnings({ "unchecked", "rawtypes" })
066   private static <T> T toType(String s, T def) {
067      if (s == null || def == null)
068         return null;
069      Class<T> c = (Class<T>)def.getClass();
070      if (c == String.class)
071         return (T)s;
072      if (c.isEnum())
073         return (T)Enum.valueOf((Class<? extends Enum>) c, s);
074      Function<String,T> f = (Function<String,T>)ENV_FUNCTIONS.get(c);
075      if (f == null)
076         throw runtimeException("Invalid env type: {0}", c);
077      return f.apply(s);
078   };
079
080   private static final Map<Class<?>,Function<String,?>> ENV_FUNCTIONS = new IdentityHashMap<>();
081   static {
082      ENV_FUNCTIONS.put(Boolean.class, x -> Boolean.valueOf(x));
083      ENV_FUNCTIONS.put(Charset.class, x -> Charset.forName(x));
084   }
085
086   private static final ConcurrentHashMap<String,String> PROPERTY_TO_ENV = new ConcurrentHashMap<>();
087   private static String envName(String name) {
088      String name2 = PROPERTY_TO_ENV.get(name);
089      if (name2 == null) {
090         name2 = name.toUpperCase().replace(".", "_");
091         PROPERTY_TO_ENV.put(name, name2);
092      }
093      return name2;
094   }
095}
