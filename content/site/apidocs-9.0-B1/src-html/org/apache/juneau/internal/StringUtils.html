095   public static String parseNumberString(ParserReader r) throws IOException {
096      r.mark();
097      int c = 0;
098      while (true) {
099         c = r.read();
100         if (c == -1)
101            break;
102         if (! numberChars.contains((char)c)) {
103            r.unread();
104            break;
105         }
106      }
107      return r.getMarked();
108   }

110   /**
111    * Parses a number from the specified string.
112    *
113    * @param s The string to parse the number from.
114    * @param type
115    *    The number type to created.
116    *    Can be any of the following:
117    *    <ul>
118    *       <li> Integer
119    *       <li> Double
120    *       <li> Float
121    *       <li> Long
122    *       <li> Short
123    *       <li> Byte
124    *       <li> BigInteger
125    *       <li> BigDecimal
126    *    </ul>
127    *    If <jk>null</jk> or <c>Number</c>, uses the best guess.
128    * @return The parsed number, or <jk>null</jk> if the string was null.
129    * @throws ParseException Malformed input encountered.
130    */
131   public static Number parseNumber(String s, Class<? extends Number> type) throws ParseException {
132      if (s == null)
133         return null;
134      if (s.isEmpty())
135         s = "0";
136      if (type == null)
137         type = Number.class;
138
139      try {
140         // Determine the data type if it wasn't specified.
141         boolean isAutoDetect = (type == Number.class);
142         boolean isDecimal = false;
143         if (isAutoDetect) {
144            // If we're auto-detecting, then we use either an Integer, Long, or Double depending on how
145            // long the string is.
146            // An integer range is -2,147,483,648 to 2,147,483,647
147            // An long range is -9,223,372,036,854,775,808 to +9,223,372,036,854,775,807
148            isDecimal = isDecimal(s);
149            if (isDecimal) {
150               if (s.length() > 20)
151                  type = Double.class;
152               else if (s.length() >= 10)
153                  type = Long.class;
154               else
155                  type = Integer.class;
156            }
157            else if (isFloat(s))
158               type = Double.class;
159            else
160               throw new NumberFormatException(s);
161         }
162
163         if (type == Double.class || type == Double.TYPE) {
164            Double d = Double.valueOf(s);
165            Float f = Float.valueOf(s);
166            if (isAutoDetect && (!isDecimal) && d.toString().equals(f.toString()))
167               return f;
168            return d;
169         }
170         if (type == Float.class || type == Float.TYPE)
171            return Float.valueOf(s);
172         if (type == BigDecimal.class)
173            return new BigDecimal(s);
174         if (type == Long.class || type == Long.TYPE || type == AtomicLong.class) {
175            try {
176               Long l = Long.decode(s);
177               if (type == AtomicLong.class)
178                  return new AtomicLong(l);
179               if (isAutoDetect && l >= Integer.MIN_VALUE && l <= Integer.MAX_VALUE) {
180                  // This occurs if the string is 10 characters long but is still a valid integer value.
181                  return l.intValue();
182               }
183               return l;
184            } catch (NumberFormatException e) {
185               if (isAutoDetect) {
186                  // This occurs if the string is 20 characters long but still falls outside the range of a valid long.
187                  return Double.valueOf(s);
188               }
189               throw e;
190            }
191         }
192         if (type == Integer.class || type == Integer.TYPE)
193            return Integer.decode(s);
194         if (type == Short.class || type == Short.TYPE)
195            return Short.decode(s);
196         if (type == Byte.class || type == Byte.TYPE)
197            return Byte.decode(s);
198         if (type == BigInteger.class)
199            return new BigInteger(s);
200         if (type == AtomicInteger.class)
201            return new AtomicInteger(Integer.decode(s));
202         throw new ParseException("Unsupported Number type: {0}", type.getName());
203      } catch (NumberFormatException e) {
204         throw new ParseException(e, "Invalid number: ''{0}'', class=''{1}''", s, type.getSimpleName());
205      }
206   }

253    * Returns <jk>true</jk> if the specified string is a floating point number.
254    *
255    * @param s The string to check.
256    * @return <jk>true</jk> if the specified string is a floating point number.
257    */
258   public static boolean isFloat(String s) {
259      if (s == null || s.isEmpty())
260         return false;
261      if (! firstNumberChars.contains(s.charAt(0)))
262         return (s.equals("NaN") || s.equals("Infinity"));
263      int i = 0;
264      int length = s.length();
265      char c = s.charAt(0);
266      if (c == '+' || c == '-')
267         i++;
268      if (i == length)
269         return false;
270      c = s.charAt(i++);
271      if (c == '.' || decChars.contains(c)) {
272         return fpRegex.matcher(s).matches();
273      }
274      return false;
275   }

277   /**
278    * Returns <jk>true</jk> if the specified string is numeric.
279    *
280    * @param s The string to check.
281    * @return <jk>true</jk> if the specified string is numeric.
282    */
283   public static boolean isDecimal(String s) {
284      if (s == null || s.isEmpty() || ! firstNumberChars.contains(s.charAt(0)))
285         return false;
286      int i = 0;
287      int length = s.length();
288      char c = s.charAt(0);
289      boolean isPrefixed = false;
290      if (c == '+' || c == '-') {
291         isPrefixed = true;
292         i++;
293      }
294      if (i == length)
295         return false;
296      c = s.charAt(i++);
297      if (c == '0' && length > (isPrefixed ? 2 : 1)) {
298         c = s.charAt(i++);
299         if (c == 'x' || c == 'X') {
300            for (int j = i; j < length; j++) {
301               if (! hexChars.contains(s.charAt(j)))
302                  return false;
303            }
304         } else if (octChars.contains(c)) {
305            for (int j = i; j < length; j++)
306               if (! octChars.contains(s.charAt(j)))
307                  return false;
308         } else {
309            return false;
310         }
311      } else if (c == '#') {
312         for (int j = i; j < length; j++) {
313            if (! hexChars.contains(s.charAt(j)))
314               return false;
315         }
316      } else if (decChars.contains(c)) {
317         for (int j = i; j < length; j++)
318            if (! decChars.contains(s.charAt(j)))
319               return false;
320      } else {
321         return false;
322      }
323      return true;
324   }

587   /**
588    * Same as {@link #split(String,char)} but consumes the tokens instead of creating an array.
589    *
590    * @param s The string to split.
591    * @param c The character to split on.
592    * @param consumer The consumer of the tokens.
593    */
594   public static void split(String s, char c, Consumer<String> consumer) {
595      AsciiSet escapeChars = getEscapeSet(c);
596
597      if (isEmpty(s))
598         return;
599      if (s.indexOf(c) == -1) {
600         consumer.accept(s);
601         return;
602      }
603
604      int x1 = 0, escapeCount = 0;
605      for (int i = 0; i < s.length(); i++) {
606         if (s.charAt(i) == '\\')
607            escapeCount++;
608         else if (s.charAt(i)==c && escapeCount % 2 == 0) {
609            String s2 = s.substring(x1, i);
610            String s3 = unEscapeChars(s2, escapeChars);
611            consumer.accept(s3.trim());
612            x1 = i+1;
613         }
614         if (s.charAt(i) != '\\')
615            escapeCount = 0;
616      }
617      String s2 = s.substring(x1);
618      String s3 = unEscapeChars(s2, escapeChars);
619      consumer.accept(s3.trim());
620   }

622   /**
623    * Same as {@link #split(String, char)} but limits the number of tokens returned.
624    *
625    * @param s The string to split.  Can be <jk>null</jk>.
626    * @param c The character to split on.
627    * @param limit The maximum number of tokens to return.
628    * @return The tokens, or <jk>null</jk> if the string was null.
629    */
630   public static String[] split(String s, char c, int limit) {
631
632      AsciiSet escapeChars = getEscapeSet(c);
633
634      if (s == null)
635         return null;
636      if (isEmpty(s))
637         return new String[0];
638      if (s.indexOf(c) == -1)
639         return new String[]{s};
640
641      List<String> l = new LinkedList<>();
642      char[] sArray = s.toCharArray();
643      int x1 = 0, escapeCount = 0;
644      limit--;
645      for (int i = 0; i < sArray.length && limit > 0; i++) {
646         if (sArray[i] == '\\')
647            escapeCount++;
648         else if (sArray[i]==c && escapeCount % 2 == 0) {
649            String s2 = new String(sArray, x1, i-x1);
650            String s3 = unEscapeChars(s2, escapeChars);
651            l.add(s3.trim());
652            limit--;
653            x1 = i+1;
654         }
655         if (sArray[i] != '\\')
656            escapeCount = 0;
657      }

828         S1 = 1,  // Looking for start of token.
829         S2 = 2,  // Found ', looking for end '

1414    * that would produce a 50% chance of collision is:

1432         if (c > 'z')
1433            c -= ('z'-'0'+1);
1434         sb.append((char)c);

1436      return sb.toString();

1530   /**
1531    * Simple utility for replacing variables of the form <js>"{key}"</js> with values in the specified map.
1532    *
1533    * <p>
1534    * Nested variables are supported in both the input string and map values.
1535    *
1536    * <p>
1537    * If the map does not contain the specified value, the variable is not replaced.
1538    *
1539    * <p>
1540    * <jk>null</jk> values in the map are treated as blank strings.
1541    *
1542    * @param s The string containing variables to replace.
1543    * @param m The map containing the variable values.
1544    * @return The new string with variables replaced, or the original string if it didn't have variables in it.
1545    */
1546   public static String replaceVars(String s, Map<String,Object> m) {
1547
1548      if (s == null)
1549         return null;
1550
1551      if (m == null || m.isEmpty() || s.indexOf('{') == -1)
1552         return s;
1553
1554      int S1 = 1;    // Not in variable, looking for {
1555      int S2 = 2;    // Found {, Looking for }
1556
1557      int state = S1;
1558      boolean hasInternalVar = false;
1559      int x = 0;
1560      int depth = 0;
1561      int length = s.length();
1562      StringBuilder out = new StringBuilder();
1563      for (int i = 0; i < length; i++) {
1564         char c = s.charAt(i);
1565         if (state == S1) {
1566            if (c == '{') {
1567               state = S2;
1568               x = i;
1569            } else {
1570               out.append(c);
1571            }
1572         } else /* state == S2 */ {
1573            if (c == '{') {
1574               depth++;
1575               hasInternalVar = true;
1576            } else if (c == '}') {
1577               if (depth > 0) {
1578                  depth--;
1579               } else {
1580                  String key = s.substring(x+1, i);
1581                  key = (hasInternalVar ? replaceVars(key, m) : key);
1582                  hasInternalVar = false;
1583                  if (! m.containsKey(key))
1584                     out.append('{').append(key).append('}');
1585                  else {
1586                     Object val = m.get(key);
1587                     if (val == null)
1588                        val = "";
1589                     String v = val.toString();
1590                     // If the replacement also contains variables, replace them now.
1591                     if (v.indexOf('{') != -1)
1592                        v = replaceVars(v, m);
1593                     out.append(v);
1594                  }
1595                  state = 1;
1596               }
1597            }
1598         }
1599      }
1600      return out.toString();
1601   }
