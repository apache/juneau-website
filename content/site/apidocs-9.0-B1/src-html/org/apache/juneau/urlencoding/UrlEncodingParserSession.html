217   /**
218    * Returns <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.
219    *
220    * @param pMeta The metadata on the bean property.
221    * @return <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.
222    */
223   public final boolean shouldUseExpandedParams(BeanPropertyMeta pMeta) {
224      ClassMeta<?> cm = pMeta.getClassMeta().getSerializedClassMeta(this);
225      if (cm.isCollectionOrArray()) {
226         if (isExpandedParams() || getUrlEncodingClassMeta(pMeta.getBeanMeta().getClassMeta()).isExpandedParams())
227            return true;
228      }
229      return false;
230   }

232   @Override /* ParserSession */
233   protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException, ExecutableException {
234      try (UonReader r = getUonReader(pipe, true)) {
235         return parseAnything(type, r, getOuter());
236      }
237   }

239   @Override /* ReaderParserSession */
240   protected <K,V> Map<K,V> doParseIntoMap(ParserPipe pipe, Map<K,V> m, Type keyType, Type valueType) throws Exception {
241      try (UonReader r = getUonReader(pipe, true)) {
242         if (r.peekSkipWs() == '?')
243            r.read();
244         m = parseIntoMap2(r, m, getClassMeta(Map.class, keyType, valueType), null);
245         return m;
246      }
247   }

249   private <T> T parseAnything(ClassMeta<T> eType, UonReader r, Object outer) throws IOException, ParseException, ExecutableException {
250
251      if (eType == null)
252         eType = (ClassMeta<T>)object();
253      ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)eType.getSwap(this);
254      BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)eType.getBuilderSwap(this);
255      ClassMeta<?> sType = null;
256      if (builder != null)
257         sType = builder.getBuilderClassMeta(this);
258      else if (swap != null)
259         sType = swap.getSwapClassMeta(this);
260      else
261         sType = eType;
262
263      if (sType.isOptional())
264         return (T)optional(parseAnything(eType.getElementType(), r, outer));
265
266      int c = r.peekSkipWs();
267      if (c == '?')
268         r.read();
269
270      Object o;
271
272      if (sType.isObject()) {
273         JsonMap m = new JsonMap(this);
274         parseIntoMap2(r, m, getClassMeta(Map.class, String.class, Object.class), outer);
275         if (m.containsKey("_value"))
276            o = m.get("_value");
277         else
278            o = cast(m, null, eType);
279      } else if (sType.isMap()) {
280         Map m = (sType.canCreateNewInstance() ? (Map)sType.newInstance() : newGenericMap(sType));
281         o = parseIntoMap2(r, m, sType, m);
282      } else if (builder != null) {
283         BeanMap m = toBeanMap(builder.create(this, eType));
284         m = parseIntoBeanMap(r, m);
285         o = m == null ? null : builder.build(this, m.getBean(), eType);
286      } else if (sType.canCreateNewBean(outer)) {
287         BeanMap m = newBeanMap(outer, sType.getInnerClass());
288         m = parseIntoBeanMap(r, m);
289         o = m == null ? null : m.getBean();
290      } else if (sType.isCollection() || sType.isArray() || sType.isArgs()) {
291         // ?1=foo&2=bar...
292         Collection c2 = ((sType.isArray() || sType.isArgs()) || ! sType.canCreateNewInstance(outer)) ? new JsonList(this) : (Collection)sType.newInstance();
293         Map<Integer,Object> m = new TreeMap<>();
294         parseIntoMap2(r, m, sType, c2);
295         c2.addAll(m.values());
296         if (sType.isArray())
297            o = ArrayUtils.toArray(c2, sType.getElementType().getInnerClass());
298         else if (sType.isArgs())
299            o = c2.toArray(new Object[c2.size()]);
300         else
301            o = c2;
302      } else {
303         // It could be a non-bean with _type attribute.
304         JsonMap m = new JsonMap(this);
305         parseIntoMap2(r, m, getClassMeta(Map.class, String.class, Object.class), outer);
306         if (m.containsKey(getBeanTypePropertyName(eType)))
307            o = cast(m, null, eType);
308         else if (m.containsKey("_value"))
309            o = convertToType(m.get("_value"), sType);
310         else if (sType.getProxyInvocationHandler() != null) {
311            o = newBeanMap(outer, sType.getInnerClass()).load(m).getBean();
312         } else {
313            if (sType.getNotABeanReason() != null)
314               throw new ParseException(this, "Class ''{0}'' could not be instantiated as application/x-www-form-urlencoded.  Reason: ''{1}''", sType, sType.getNotABeanReason());
315            throw new ParseException(this, "Malformed application/x-www-form-urlencoded input for class ''{0}''.", sType);
316         }
317      }
318
319      if (swap != null && o != null)
320         o = unswap(swap, o, eType);
321
322      if (outer != null)
323         setParent(eType, o, outer);
324
325      return (T)o;
326   }

328   private <K,V> Map<K,V> parseIntoMap2(UonReader r, Map<K,V> m, ClassMeta<?> type, Object outer) throws IOException, ParseException, ExecutableException {
329
330      ClassMeta<K> keyType = (ClassMeta<K>)(type.isArgs() || type.isCollectionOrArray() ? getClassMeta(Integer.class) : type.getKeyType());
331
332      int c = r.peekSkipWs();
333      if (c == -1)
334         return m;
335
336      final int S1=1; // Looking for attrName start.
337      final int S2=2; // Found attrName end, looking for =.
338      final int S3=3; // Found =, looking for valStart.
339      final int S4=4; // Looking for & or end.
340      boolean isInEscape = false;
341
342      int state = S1;
343      int argIndex = 0;
344      K currAttr = null;
345      while (c != -1) {
346         c = r.read();
347         if (! isInEscape) {
348            if (state == S1) {
349               if (c == -1)
350                  return m;
351               r.unread();
352               Object attr = parseAttr(r, true);
353               currAttr = attr == null ? null : convertAttrToType(m, trim(attr.toString()), keyType);
354               state = S2;
355               c = 0; // Avoid isInEscape if c was '\'
356            } else if (state == S2) {
357               if (c == '\u0002')
358                  state = S3;
359               else if (c == -1 || c == '\u0001') {
360                  m.put(currAttr, null);
361                  if (c == -1)
362                     return m;
363                  state = S1;
364               }
365            } else if (state == S3) {
366               if (c == -1 || c == '\u0001') {
367                  ClassMeta<V> valueType = (ClassMeta<V>)(type.isArgs() ? type.getArg(argIndex++) : type.isCollectionOrArray() ? type.getElementType() : type.getValueType());
368                  V value = convertAttrToType(m, "", valueType);
369                  m.put(currAttr, value);
370                  if (c == -1)
371                     return m;
372                  state = S1;
373               } else  {
374                  // For performance, we bypass parseAnything for string values.
375                  ClassMeta<V> valueType = (ClassMeta<V>)(type.isArgs() ? type.getArg(argIndex++) : type.isCollectionOrArray() ? type.getElementType() : type.getValueType());
376                  V value = (V)(valueType.isString() ? super.parseString(r.unread(), true) : super.parseAnything(valueType, r.unread(), outer, true, null));
377
378                  // If we already encountered this parameter, turn it into a list.
379                  if (m.containsKey(currAttr) && valueType.isObject()) {
380                     Object v2 = m.get(currAttr);
381                     if (! (v2 instanceof JsonList)) {
382                        v2 = new JsonList(v2).setBeanSession(this);
383                        m.put(currAttr, (V)v2);
384                     }
385                     ((JsonList)v2).add(value);
386                  } else {
387                     m.put(currAttr, value);
388                  }
389                  state = S4;
390                  c = 0; // Avoid isInEscape if c was '\'
391               }
392            } else if (state == S4) {
393               if (c == '\u0001')
394                  state = S1;
395               else if (c == -1) {
396                  return m;
397               }
398            }
399         }
400         isInEscape = (c == '\\' && ! isInEscape);
401      }
402      if (state == S1)
403         throw new ParseException(this, "Could not find attribute name on object.");
404      if (state == S2)
405         throw new ParseException(this, "Could not find '=' following attribute name on object.");
406      if (state == S3)
407         throw new ParseException(this, "Dangling '=' found in object entry");
408      if (state == S4)
409         throw new ParseException(this, "Could not find end of object.");
410
411      return null; // Unreachable.
412   }

414   private <T> BeanMap<T> parseIntoBeanMap(UonReader r, BeanMap<T> m) throws IOException, ParseException, ExecutableException {
415
416      int c = r.peekSkipWs();
417      if (c == -1)
418         return m;
419
420      final int S1=1; // Looking for attrName start.
421      final int S2=2; // Found attrName end, looking for =.
422      final int S3=3; // Found =, looking for valStart.
423      final int S4=4; // Looking for , or }
424      boolean isInEscape = false;
425
426      int state = S1;
427      String currAttr = "";
428      mark();
429      try {
430         while (c != -1) {
431            c = r.read();
432            if (! isInEscape) {
433               if (state == S1) {
434                  if (c == -1) {
435                     return m;
436                  }
437                  r.unread();
438                  mark();
439                  currAttr = parseAttrName(r, true);
440                  if (currAttr == null)  // Value was '%00'
441                     return null;
442                  state = S2;
443               } else if (state == S2) {
444                  if (c == '\u0002')
445                     state = S3;
446                  else if (c == -1 || c == '\u0001') {
447                     m.put(currAttr, null);
448                     if (c == -1)
449                        return m;
450                     state = S1;
451                  }
452               } else if (state == S3) {
453                  if (c == -1 || c == '\u0001') {
454                     if (! currAttr.equals(getBeanTypePropertyName(m.getClassMeta()))) {
455                        BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
456                        if (pMeta == null) {
457                           onUnknownProperty(currAttr, m, null);
458                           unmark();
459                        } else {
460                           unmark();
461                           setCurrentProperty(pMeta);
462                           // In cases of "&foo=", create an empty instance of the value if createable.
463                           // Otherwise, leave it null.
464                           ClassMeta<?> cm = pMeta.getClassMeta();
465                           if (cm.canCreateNewInstance()) {
466                              try {
467                                 pMeta.set(m, currAttr, cm.newInstance());
468                              } catch (BeanRuntimeException e) {
469                                 onBeanSetterException(pMeta, e);
470                                 throw e;
471                              }
472                           }
473                           setCurrentProperty(null);
474                        }
475                     }
476                     if (c == -1)
477                        return m;
478                     state = S1;
479                  } else {
480                     if (! currAttr.equals(getBeanTypePropertyName(m.getClassMeta()))) {
481                        BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
482                        if (pMeta == null) {
483                           onUnknownProperty(currAttr, m, parseAnything(object(), r.unread(), m.getBean(false), true, null));
484                           unmark();
485                        } else {
486                           unmark();
487                           setCurrentProperty(pMeta);
488                           if (shouldUseExpandedParams(pMeta)) {
489                              ClassMeta et = pMeta.getClassMeta().getElementType();
490                              Object value = parseAnything(et, r.unread(), m.getBean(false), true, pMeta);
491                              setName(et, value, currAttr);
492                              try {
493                                 pMeta.add(m, currAttr, value);
494                              } catch (BeanRuntimeException e) {
495                                 onBeanSetterException(pMeta, e);
496                                 throw e;
497                              }
498                           } else {
499                              ClassMeta<?> cm = pMeta.getClassMeta();
500                              Object value = parseAnything(cm, r.unread(), m.getBean(false), true, pMeta);
501                              setName(cm, value, currAttr);
502                              try {
503                                 pMeta.set(m, currAttr, value);
504                              } catch (BeanRuntimeException e) {
505                                 onBeanSetterException(pMeta, e);
506                                 throw e;
507                              }
508                           }
509                           setCurrentProperty(null);
510                        }
511                     }
512                     state = S4;
513                  }
514               } else if (state == S4) {
515                  if (c == '\u0001')
516                     state = S1;
517                  else if (c == -1) {
518                     return m;
519                  }
520               }
521            }
522            isInEscape = (c == '\\' && ! isInEscape);
523         }
524         if (state == S1)
525            throw new ParseException(this, "Could not find attribute name on object.");
526         if (state == S2)
527            throw new ParseException(this, "Could not find '=' following attribute name on object.");
528         if (state == S3)
529            throw new ParseException(this, "Could not find value following '=' on object.");
530         if (state == S4)
531            throw new ParseException(this, "Could not find end of object.");
532      } finally {
533         unmark();
534      }
535
536      return null; // Unreachable.
537   }
