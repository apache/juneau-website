013package org.apache.juneau.cp;
014
015import static org.apache.juneau.internal.ObjectUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017import static org.apache.juneau.assertions.Assertions.*;
018
019import java.net.*;
020import java.nio.file.*;
021
022import org.apache.juneau.internal.*;
023
024/**
025 * Identifies a directory located either on the classpath or file system.
026 *
027 * Used to encapsulate basic resolution and retrieval of files regardless of where they are located.
028 *
029 * <ul class='seealso'>
030 *    <li class='extlink'>{@source}
031 * </ul>
032 */
033public class LocalDir {
034
035   private final Class<?> clazz;
036   private final String clazzPath;
037   private final Path path;
038   private final int hashCode;
039
040   /**
041    * Constructor for classpath directory.
042    *
043    * @param clazz The class used to retrieve resources.
044    * @param clazzPath
045    *    The subpath.  Can be any of the following:
046    *    <ul>
047    *       <li><jk>null</jk> or an empty string - Package of the class.
048    *       <li>Absolute path (starts with <js>'/'</js>) - Relative to root package.
049    *       <li>Relative path (does not start with <js>'/'</js>) - Relative to class package.
050    *    </ul>
051    */
052   public LocalDir(Class<?> clazz, String clazzPath) {
053      this.clazz = assertArgNotNull("clazz", clazz);
054      this.clazzPath = "/".equals(clazzPath) ? "/" : nullIfEmpty(trimTrailingSlashes(clazzPath));
055      this.path = null;
056      this.hashCode = HashCode.of(clazz, clazzPath);
057   }
058
059   /**
060    * Constructor for file system directory.
061    *
062    * @param path Filesystem directory location.  Must not be <jk>null</jk>.
063    */
064   public LocalDir(Path path) {
065      this.clazz = null;
066      this.clazzPath = null;
067      this.path = assertArgNotNull("path", path);
068      this.hashCode = path.hashCode();
069   }
070
071   /**
072    * Resolves the specified path.
073    *
074    * @param path
075    *    The path to the file to resolve.
076    *    <br>Must be a non-null relative path.
077    *    <br>Does no cleanup of the path (e.g. checking for security holes or malformed values).
078    * @return The file if it exists, or <jk>null</jk> if it does not.
079    */
080   public LocalFile resolve(String path) {
081      if (clazz != null) {
082         String p = clazzPath == null ? path : ("/".equals(clazzPath) ? "" : clazzPath) + '/' + path;
083         if (isClasspathFile(clazz.getResource(p)))
084            return new LocalFile(clazz, p);
085      } else {
086         Path p = this.path.resolve(path);
087         if (Files.isReadable(p) && ! Files.isDirectory(p))
088            return new LocalFile(p);
089      }
090      return null;
091   }
092
093   /**
094    * Validates that the specified classpath resource exists and is a file.
095    * Note that the behavior of Class.getResource(path) is different when pointing to directories on the classpath.
096    * When packaged as a jar, calling Class.getResource(path) on a directory returns null.
097    * When unpackaged, calling Class.getResource(path) on a directory returns a URL starting with "file:".
098    * We perform a test to make the behavior the same regardless of whether we're packaged or not.
099    */
100   private boolean isClasspathFile(URL url) {
101      try {
102         if (url == null)
103            return false;
104         URI uri = url.toURI();
105         if (uri.toString().startsWith("file:"))
106            if (Files.isDirectory(Paths.get(uri)))
107               return false;
108      } catch (URISyntaxException e) {
109         e.printStackTrace();  // Untestable.
110         return false;
111      }
112      return true;
113   }
114
115   @Override /* Object */
116   public boolean equals(Object o) {
117      return o instanceof LocalDir && eq(this, (LocalDir)o, (x,y)->eq(x.clazz, y.clazz) && eq(x.clazzPath, y.clazzPath) && eq(x.path, y.path));
118   }
119
120   @Override /* Object */
121   public int hashCode() {
122      return hashCode;
123   }
124
125   @Override /* Object */
126   public String toString() {
127      if (clazz == null)
128         return path.toString();
129      return clazz.getName() + ":" + clazzPath;
130   }
131}
