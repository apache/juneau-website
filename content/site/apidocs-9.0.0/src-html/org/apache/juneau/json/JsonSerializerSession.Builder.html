226   /**
227    * Method that can be called from subclasses to serialize an object to JSON.
228    *
229    * <p>
230    * Used by {@link JsonSchemaSerializerSession} for serializing examples to JSON.
231    *
232    * @param o The object to serialize.
233    * @return The serialized object.
234    * @throws Exception Error occurred.
235    */
236   protected String serializeJson(Object o) throws Exception {
237      StringWriter sw = new StringWriter();
238      serializeAnything(getJsonWriter(createPipe(sw)).i(getInitialDepth()), o, getExpectedRootType(o), "root", null);
239      return sw.toString();
240   }

242   /**
243    * Workhorse method.
244    * Determines the type of object, and then calls the appropriate type-specific serialization method.
245    *
246    * @param out The output writer.
247    * @param o The object to serialize.
248    * @param eType The expected type.
249    * @param attrName The attribute name.
250    * @param pMeta The bean property currently being parsed.
251    * @return The same writer passed in.
252    * @throws SerializeException General serialization error occurred.
253    */
254   @SuppressWarnings({ "rawtypes" })
255   protected JsonWriter serializeAnything(JsonWriter out, Object o, ClassMeta<?> eType, String attrName, BeanPropertyMeta pMeta) throws SerializeException {
256
257      if (o == null) {
258         out.append("null");
259         return out;
260      }
261
262      if (eType == null)
263         eType = object();
264
265      ClassMeta<?> aType;        // The actual type
266      ClassMeta<?> sType;        // The serialized type
267
268      aType = push2(attrName, o, eType);
269      boolean isRecursion = aType == null;
270
271      // Handle recursion
272      if (aType == null) {
273         o = null;
274         aType = object();
275      }
276
277      // Handle Optional<X>
278      if (isOptional(aType)) {
279         o = getOptionalValue(o);
280         eType = getOptionalType(eType);
281         aType = getClassMetaForObject(o, object());
282      }
283
284      sType = aType;
285      String typeName = getBeanTypeName(this, eType, aType, pMeta);
286
287      // Swap if necessary
288      ObjectSwap swap = aType.getSwap(this);
289      if (swap != null) {
290         o = swap(swap, o);
291         sType = swap.getSwapClassMeta(this);
292
293         // If the getSwapClass() method returns Object, we need to figure out
294         // the actual type now.
295         if (sType.isObject())
296            sType = getClassMetaForObject(o);
297      }
298
299      String wrapperAttr = getJsonClassMeta(sType).getWrapperAttr();
300      if (wrapperAttr != null) {
301         out.w('{').cr(indent).attr(wrapperAttr).w(':').s(indent);
302         indent++;
303      }
304
305      // '\0' characters are considered null.
306      if (o == null || (sType.isChar() && ((Character)o).charValue() == 0)) {
307         out.append("null");
308      } else if (sType.isNumber() || sType.isBoolean()) {
309         out.append(o);
310      } else if (sType.isBean()) {
311         serializeBeanMap(out, toBeanMap(o), typeName);
312      } else if (sType.isUri() || (pMeta != null && pMeta.isUri())) {
313         out.uriValue(o);
314      } else if (sType.isMap()) {
315         if (o instanceof BeanMap)
316            serializeBeanMap(out, (BeanMap)o, typeName);
317         else
318            serializeMap(out, (Map)o, eType);
319      } else if (sType.isCollection()) {
320         serializeCollection(out, (Collection) o, eType);
321      } else if (sType.isArray()) {
322         serializeCollection(out, toList(sType.getInnerClass(), o), eType);
323      } else if (sType.isReader()) {
324         pipe((Reader)o, out, SerializerSession::handleThrown);
325      } else if (sType.isInputStream()) {
326         pipe((InputStream)o, out, SerializerSession::handleThrown);
327      } else {
328         out.stringValue(toString(o));
329      }
330
331      if (wrapperAttr != null) {
332         indent--;
333         out.cre(indent-1).w('}');
334      }
335
336      if (! isRecursion)
337         pop();
338      return out;
339   }

341   @SuppressWarnings({ "rawtypes", "unchecked" })
342   private SerializerWriter serializeMap(JsonWriter out, Map m, ClassMeta<?> type) throws SerializeException {
343
344      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
345
346      int i = indent;
347      out.w('{');
348
349      Flag addComma = Flag.create();
350      forEachEntry(m, x -> {
351         addComma.ifSet(()->out.w(',').smi(i)).set();
352         Object value = x.getValue();
353         Object key = generalize(x.getKey(), keyType);
354         out.cr(i).attr(toString(key)).w(':').s(i);
355         serializeAnything(out, value, valueType, (key == null ? null : toString(key)), null);
356      });
357
358      out.cre(i-1).w('}');
359
360      return out;
361   }
362
363   private SerializerWriter serializeBeanMap(JsonWriter out, BeanMap<?> m, String typeName) throws SerializeException {
364      int i = indent;
365      out.w('{');
366
367      Flag addComma = Flag.create();
368      Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
369
370      if (typeName != null) {
371         BeanPropertyMeta pm = m.getMeta().getTypeProperty();
372         out.cr(i).attr(pm.getName()).w(':').s(i).stringValue(typeName);
373         addComma.set();
374      }
375
376      m.forEachValue(checkNull, (pMeta,key,value,thrown) -> {
377         ClassMeta<?> cMeta = pMeta.getClassMeta();
378         if (thrown != null)
379            onBeanGetterException(pMeta, thrown);
380
381         if (canIgnoreValue(cMeta, key, value))
382            return;
383
384         addComma.ifSet(()->out.append(',').smi(i)).set();
385
386         out.cr(i).attr(key).w(':').s(i);
387
388         serializeAnything(out, value, cMeta, key, pMeta);
389      });
390
391      out.cre(i-1).w('}');
392      return out;
393   }
394
395   @SuppressWarnings({"rawtypes", "unchecked"})
396   private SerializerWriter serializeCollection(JsonWriter out, Collection c, ClassMeta<?> type) throws SerializeException {
397
398      ClassMeta<?> elementType = type.getElementType();
399
400      out.w('[');
401      Flag addComma = Flag.create();
402      forEachEntry(c, x -> {
403         addComma.ifSet(()->out.w(',').smi(indent)).set();
404         out.cr(indent);
405         serializeAnything(out, x, elementType, "<iterator>", null);
406      });
407
408      out.cre(indent-1).w(']');
409      return out;
410   }

412   /**
413    * Converts the specified output target object to an {@link JsonWriter}.
414    *
415    * @param out The output target object.
416    * @return The output target object wrapped in an {@link JsonWriter}.
417    * @throws IOException Thrown by underlying stream.
418    */
419   protected final JsonWriter getJsonWriter(SerializerPipe out) throws IOException {
420      Object output = out.getRawOutput();
421      if (output instanceof JsonWriter)
422         return (JsonWriter)output;
423      JsonWriter w = new JsonWriter(out.getWriter(), isUseWhitespace(), getMaxIndent(), isEscapeSolidus(), getQuoteChar(),
424         isSimpleMode(), isTrimStrings(), getUriResolver());
425      out.setWriter(w);
426      return w;
427   }
