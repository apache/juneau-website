071   /**
072    * Creates a copy of the specified annotation.
073    *
074    * @param a The annotation to copy.
075    * @param r The var resolver for resolving any variables.
076    * @return A copy of the specified annotation.
077    */
078   public static Bean copy(Bean a, VarResolverSession r) {
079      return
080         create()
081         .dictionary(a.dictionary())
082         .example(r.resolve(a.example()))
083         .excludeProperties(r.resolve(a.excludeProperties()))
084         .findFluentSetters(a.findFluentSetters())
085         .implClass(a.implClass())
086         .interceptor(a.interceptor())
087         .interfaceClass(a.interfaceClass())
088         .on(r.resolve(a.on()))
089         .onClass(a.onClass())
090         .p(r.resolve(a.p()))
091         .properties(r.resolve(a.properties()))
092         .propertyNamer(a.propertyNamer())
093         .readOnlyProperties(r.resolve(a.readOnlyProperties()))
094         .ro(r.resolve(a.ro()))
095         .sort(a.sort())
096         .stopClass(a.stopClass())
097         .typeName(r.resolve(a.typeName()))
098         .typePropertyName(r.resolve(a.typePropertyName()))
099         .wo(r.resolve(a.wo()))
100         .writeOnlyProperties(r.resolve(a.writeOnlyProperties()))
101         .xp(r.resolve(a.xp()))
102         .build();
103   }

510   /**
511    * Applies targeted {@link Bean} annotations to a {@link org.apache.juneau.BeanContext.Builder}.
512    */
513   public static class Applier extends AnnotationApplier<Bean,BeanContext.Builder> {
514
515      /**
516       * Constructor.
517       *
518       * @param vr The resolver for resolving values in annotations.
519       */
520      public Applier(VarResolverSession vr) {
521         super(Bean.class, BeanContext.Builder.class, vr);
522      }
523
524      @Override
525      public void apply(AnnotationInfo<Bean> ai, BeanContext.Builder b) {
526         Bean a = ai.inner();
527         if (isEmptyArray(a.on(), a.onClass()))
528            return;
529         b.annotations(copy(a, vr()));
530      }
531   }
