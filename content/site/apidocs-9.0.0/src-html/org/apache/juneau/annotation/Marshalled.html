013package org.apache.juneau.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020import org.apache.juneau.*;
021
022/**
023 * Annotation that can be applied to classes to control how they are marshalled.
024 *
025 * <p>
026 * Can be used in the following locations:
027 * <ul>
028 *    <li>Marshalled classes.
029 *    <li><ja>@Rest</ja>-annotated classes and <ja>@RestOp</ja>-annotated methods when an {@link #on()} value is specified.
030 * </ul>
031 *
032 * <p>
033 * This annotation is typically only applied to non-bean classes.  The {@link Bean @Bean} annotation contains equivalent
034 * functionality for bean classes.
035 *
036 * <ul class='seealso'>
037 *    <li class='extlink'>{@source}
038 * </ul>
039 */
040@Documented
041@Target({METHOD,TYPE})
042@Retention(RUNTIME)
043@Inherited
044@Repeatable(MarshalledAnnotation.Array.class)
045@ContextApply(MarshalledAnnotation.Applier.class)
046public @interface Marshalled {
047
048   /**
049    * POJO example.
050    *
051    * <p>
052    * Specifies an example of the specified class in Simplified JSON format.
053    *
054    * <p>
055    * Examples are used in cases such as POJO examples in Swagger documents.
056    *
057    * <h5 class='section'>Example:</h5>
058    * <p class='bjava'>
059    *    <ja>@Marshalled</ja>(example=<js>"{foo:'bar'}"</js>)
060    *    <jk>public class</jk> MyClass {...}
061    * </p>
062    *
063    * <ul class='notes'>
064    *    <li class='note'>
065    *       Setting applies to specified class and all subclasses.
066    *    <li class='note'>
067    *       Keys are the class of the example.
068    *       <br>Values are Simple-JSON representation of that class.
069    *    <li class='note'>
070    *       POJO examples can also be defined on classes via the following:
071    *       <ul class='spaced-list'>
072    *          <li>A static field annotated with {@link Example @Example}.
073    *          <li>A static method annotated with {@link Example @Example} with zero arguments or one {@link BeanSession} argument.
074    *          <li>A static method with name <c>example</c> with no arguments or one {@link BeanSession} argument.
075    *       </ul>
076    *    <li class='note'>
077    *       Supports {@doc jm.DefaultVarResolver} (e.g. <js>"$C{myConfigVar}"</js>).
078    * </ul>
079    *
080    * <ul class='seealso'>
081    *    <li class='ja'>{@link Example}
082    * </ul>
083    *
084    * @return The annotation value.
085    */
086   String example() default "";
087
088   /**
089    * Implementation class.
090    *
091    * <p>
092    * For interfaces and abstract classes this method can be used to specify an implementation class for the
093    * interface/abstract class so that instances of the implementation class are used when instantiated (e.g. during a
094    * parse).
095    *
096    * <h5 class='section'>Example:</h5>
097    * <p class='bjava'>
098    *    <ja>@Marshalled</ja>(implClass=MyInterfaceImpl.<jk>class</jk>)
099    *    <jk>public class</jk> MyInterface {...}
100    * <p>
101    *
102    * @return The annotation value.
103    */
104   Class<?> implClass() default void.class;
105
106   /**
107    * Dynamically apply this annotation to the specified classes.
108    *
109    * <p>
110    * Used in conjunction with {@link org.apache.juneau.BeanContext.Builder#applyAnnotations(Class...)} to dynamically apply an annotation to an existing class.
111    * It is ignored when the annotation is applied directly to classes.
112    *
113    * <p>
114    * The following example shows the equivalent methods for applying the {@link Bean @Bean} annotation to REST methods:
115    * <p class='bjava'>
116    *    <jc>// Class with explicit annotation.</jc>
117    *    <ja>@Marshalled</ja>(example=<js>"{foo:'bar'}"</js>)
118    *    <jk>public class</jk> A {...}
119    *
120    *    <jc>// Class with annotation applied via @BeanConfig</jc>
121    *    <jk>public class</jk> B {...}
122    *
123    *    <jc>// Java REST method with @BeanConfig annotation.</jc>
124    *    <ja>@RestOp</ja>(...)
125    *    <ja>@Marshalled</ja>(on=<js>"B"</js>, example=<js>"{foo:'bar'}"</js>)
126    *    <jk>public void</jk> doFoo() {...}
127    * </p>
128    *
129    * <h5 class='section'>Valid patterns:</h5>
130    * <ul class='spaced-list'>
131    *  <li>Classes:
132    *       <ul>
133    *          <li>Fully qualified:
134    *             <ul>
135    *                <li><js>"com.foo.MyClass"</js>
136    *             </ul>
137    *          <li>Fully qualified inner class:
138    *             <ul>
139    *                <li><js>"com.foo.MyClass$Inner1$Inner2"</js>
140    *             </ul>
141    *          <li>Simple:
142    *             <ul>
143    *                <li><js>"MyClass"</js>
144    *             </ul>
145    *          <li>Simple inner:
146    *             <ul>
147    *                <li><js>"MyClass$Inner1$Inner2"</js>
148    *                <li><js>"Inner1$Inner2"</js>
149    *                <li><js>"Inner2"</js>
150    *             </ul>
151    *       </ul>
152    *    <li>A comma-delimited list of anything on this list.
153    * </ul>
154    *
155    * <ul class='seealso'>
156    *    <li class='link'>{@doc jm.DynamicallyAppliedAnnotations}
157    * </ul>
158    *
159    * @return The annotation value.
160    */
161   String[] on() default {};
162
163   /**
164    * Dynamically apply this annotation to the specified classes.
165    *
166    * <p>
167    * Identical to {@link #on()} except allows you to specify class objects instead of a strings.
168    *
169    * <ul class='seealso'>
170    *    <li class='link'>{@doc jm.DynamicallyAppliedAnnotations}
171    * </ul>
172    *
173    * @return The annotation value.
174    */
175   Class<?>[] onClass() default {};
176}
