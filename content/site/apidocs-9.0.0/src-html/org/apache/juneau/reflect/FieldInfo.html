023/**
024 * Lightweight utility class for introspecting information about a field.
025 *
026 * <ul class='seealso'>
027 *    <li class='extlink'>{@source}
028 * </ul>
029 */
030public final class FieldInfo implements Comparable<FieldInfo> {
031
032   //-----------------------------------------------------------------------------------------------------------------
033   // Static
034   //-----------------------------------------------------------------------------------------------------------------
035
036   /**
037    * Convenience method for instantiating a {@link FieldInfo};
038    *
039    * @param declaringClass The class that declares this method.
040    * @param f The field being wrapped.
041    * @return A new {@link FieldInfo} object, or <jk>null</jk> if the field was null.
042    */
043   public static FieldInfo of(ClassInfo declaringClass, Field f) {
044      if (f == null)
045         return null;
046      return ClassInfo.of(declaringClass).getFieldInfo(f);
047   }
048
049   /**
050    * Convenience method for instantiating a {@link FieldInfo};
051    *
052    * @param f The field being wrapped.
053    * @return A new {@link FieldInfo} object, or <jk>null</jk> if the field was null.
054    */
055   public static FieldInfo of(Field f) {
056      if (f == null)
057         return null;
058      return ClassInfo.of(f.getDeclaringClass()).getFieldInfo(f);
059   }

102   /**
103    * Returns the specified annotation on this field.
104    *
105    * @param <A> The annotation type to look for.
106    * @param type The annotation to look for.
107    * @return The annotation, or <jk>null</jk> if not found.
108    */
109   public <A extends Annotation> A getAnnotation(Class<A> type) {
110      return getAnnotation(AnnotationProvider.DEFAULT, type);
111   }
112
113   /**
114    * Returns the specified annotation on this field.
115    *
116    * @param <A> The annotation type to look for.
117    * @param annotationProvider The annotation provider.
118    * @param type The annotation to look for.
119    * @return The annotation, or <jk>null</jk> if not found.
120    */
121   public <A extends Annotation> A getAnnotation(AnnotationProvider annotationProvider, Class<A> type) {
122      Value<A> t = Value.empty();
123      annotationProvider.forEachAnnotation(type, f, x -> true, x -> t.set(x));
124      return t.orElse(null);
125   }
126
127   /**
128    * Returns <jk>true</jk> if the specified annotation is present.
129    *
130    * @param <A> The annotation type to look for.
131    * @param type The annotation to look for.
132    * @return <jk>true</jk> if the specified annotation is present.
133    */
134   public <A extends Annotation> boolean hasAnnotation(Class<A> type) {
135      return f.isAnnotationPresent(type);
136   }
137
138   /**
139    * Returns <jk>true</jk> if the specified annotation is not present on this field.
140    *
141    * @param <A> The annotation type to look for.
142    * @param type The annotation to look for.
143    * @return <jk>true</jk> if the specified annotation is not present on this field.
144    */
145   public final <A extends Annotation> boolean hasNoAnnotation(Class<A> type) {
146      return ! hasAnnotation(type);
147   }
148
149   /**
150    * Returns <jk>true</jk> if the specified annotation is present.
151    *
152    * @param <A> The annotation type to look for.
153    * @param annotationProvider The annotation provider.
154    * @param type The annotation to look for.
155    * @return <jk>true</jk> if the specified annotation is present.
156    */
157   public <A extends Annotation> boolean hasAnnotation(AnnotationProvider annotationProvider, Class<A> type) {
158      return annotationProvider.firstAnnotation(type, f, x -> true) != null;
159   }
160
161   /**
162    * Returns <jk>true</jk> if the specified annotation is not present.
163    *
164    * @param <A> The annotation type to look for.
165    * @param annotationProvider The annotation provider.
166    * @param type The annotation to look for.
167    * @return <jk>true</jk> if the specified annotation is not present.
168    */
169   public <A extends Annotation> boolean hasNoAnnotation(AnnotationProvider annotationProvider, Class<A> type) {
170      return ! hasAnnotation(annotationProvider, type);
171   }

177   /**
178    * Returns <jk>true</jk> if all specified flags are applicable to this field.
179    *
180    * @param flags The flags to test for.
181    * @return <jk>true</jk> if all specified flags are applicable to this field.
182    */
183   public boolean isAll(ReflectFlags...flags) {
184      for (ReflectFlags f : flags) {
185         switch (f) {
186            case DEPRECATED:
187               if (isNotDeprecated())
188                  return false;
189               break;
190            case NOT_DEPRECATED:
191               if (isDeprecated())
192                  return false;
193               break;
194            case PUBLIC:
195               if (isNotPublic())
196                  return false;
197               break;
198            case NOT_PUBLIC:
199               if (isPublic())
200                  return false;
201               break;
202            case STATIC:
203               if (isNotStatic())
204                  return false;
205               break;
206            case NOT_STATIC:
207               if (isStatic())
208                  return false;
209               break;
210            case TRANSIENT:
211               if (isNotTransient())
212                  return false;
213               break;
214            case NOT_TRANSIENT:
215               if (isTransient())
216                  return false;
217               break;
218            default:
219               throw new BasicRuntimeException("Invalid flag for field: {0}", f);
220         }
221      }
222      return true;
223   }
224
225   /**
226    * Returns <jk>true</jk> if all specified flags are applicable to this field.
227    *
228    * @param flags The flags to test for.
229    * @return <jk>true</jk> if all specified flags are applicable to this field.
230    */
231   public boolean isAny(ReflectFlags...flags) {
232      for (ReflectFlags f : flags) {
233         switch (f) {
234            case DEPRECATED:
235               if (isDeprecated())
236                  return true;
237               break;
238            case NOT_DEPRECATED:
239               if (isNotDeprecated())
240                  return true;
241               break;
242            case PUBLIC:
243               if (isPublic())
244                  return true;
245               break;
246            case NOT_PUBLIC:
247               if (isNotPublic())
248                  return true;
249               break;
250            case STATIC:
251               if (isStatic())
252                  return true;
253               break;
254            case NOT_STATIC:
255               if (isNotStatic())
256                  return true;
257               break;
258            case TRANSIENT:
259               if (isTransient())
260                  return true;
261               break;
262            case NOT_TRANSIENT:
263               if (isNotTransient())
264                  return true;
265               break;
266            default:
267               throw new BasicRuntimeException("Invalid flag for field: {0}", f);
268         }
269      }
270      return false;
271   }

278    */
279   public boolean is(ReflectFlags...flags) {

369   /**
370    * Attempts to call <code>x.setAccessible(<jk>true</jk>)</code> and quietly ignores security exceptions.
371    *
372    * @return This object.
373    */
374   public FieldInfo accessible() {
375      setAccessible();
376      return this;
377   }
378
379   /**
380    * Attempts to call <code>x.setAccessible(<jk>true</jk>)</code> and quietly ignores security exceptions.
381    *
382    * @return <jk>true</jk> if call was successful.
383    */
384   public boolean setAccessible() {
385      try {
386         if (! (f.isAccessible()))
387            f.setAccessible(true);
388         return true;
389      } catch (SecurityException e) {
390         return false;
391      }
392   }

394   /**
395    * Identifies if the specified visibility matches this field.
396    *
397    * @param v The visibility to validate against.
398    * @return <jk>true</jk> if this visibility matches the modifier attribute of this field.
399    */
400   public boolean isVisible(Visibility v) {
401      return v.isVisible(f);
402   }

408   /**
409    * Returns <jk>true</jk> if this object passes the specified predicate test.
410    *
411    * @param test The test to perform.
412    * @return <jk>true</jk> if this object passes the specified predicate test.
413    */
414   public boolean matches(Predicate<FieldInfo> test) {
415      return test(test, this);
416   }
417
418   /**
419    * Performs an action on this object if the specified predicate test passes.
420    *
421    * @param test A test to apply to determine if action should be executed.  Can be <jk>null</jk>.
422    * @param action An action to perform on this object.
423    * @return This object.
424    */
425   public FieldInfo accept(Predicate<FieldInfo> test, Consumer<FieldInfo> action) {
426      if (matches(test))
427         action.accept(this);
428      return this;
429   }

431   /**
432    * Returns the type of this field.
433    *
434    * @return The type of this field.
435    */
436   public ClassInfo getType() {
437      if (type == null) {
438         synchronized(this) {
439            type = ClassInfo.of(f.getType());
440         }
441      }
442      return type;
443   }

464   /**
465    * Returns the full name of this field.
466    *
467    * <h5 class='section'>Examples:</h5>
468    * <ul>
469    *    <li><js>"com.foo.MyClass.myField"</js> - Method.
470    * </ul>
471    *
472    * @return The underlying executable name.
473    */
474   public String getFullName() {
475      StringBuilder sb = new StringBuilder(128);
476      ClassInfo dc = declaringClass;
477      Package p = dc.getPackage();
478      if (p != null)
479         sb.append(p.getName()).append('.');
480      dc.appendShortName(sb);
481      sb.append(".").append(getName());
482      return sb.toString();
483   }

485   /**
486    * Returns the field value on the specified object.
487    *
488    * @param o The object containing the field.
489    * @param <T> The object type to retrieve.
490    * @return The field value.
491    * @throws BeanRuntimeException Field was not accessible or field does not belong to object.
492    */
493   @SuppressWarnings("unchecked")
494   public <T> T get(Object o) throws BeanRuntimeException {
495      try {
496         f.setAccessible(true);
497         return (T)f.get(o);
498      } catch (Exception e) {
499         throw new BeanRuntimeException(e);
500      }
501   }
502
503   /**
504    * Same as {@link #get(Object)} but wraps the results in an {@link Optional}.
505    *
506    * @param o The object containing the field.
507    * @param <T> The object type to retrieve.
508    * @return The field value.
509    * @throws BeanRuntimeException Field was not accessible or field does not belong to object.
510    */
511   public <T> Optional<T> getOptional(Object o) throws BeanRuntimeException {
512      return Optional.ofNullable(get(o));
513   }

515   /**

517    *
518    * @param o The object containing the field.
519    * @param value The new field value.
520    * @throws BeanRuntimeException Field was not accessible or field does not belong to object.
521    */
522   public void set(Object o, Object value) throws BeanRuntimeException {
523      try {
524         f.setAccessible(true);
525         f.set(o, value);
526      } catch (Exception e) {
527         throw new BeanRuntimeException(e);
528      }
529   }
530
531   /**
532    * Sets the field value on the specified object if the value is <jk>null</jk>.
533    *
534    * @param o The object containing the field.
535    * @param value The new field value.
536    * @throws BeanRuntimeException Field was not accessible or field does not belong to object.
537    */
538   public void setIfNull(Object o, Object value) {
539      Object v = get(o);
540      if (v == null)
541         set(o, value);
542   }
