032/**
033 * Represents an annotation instance on a class and the class it was found on.
034 *
035 * <ul class='seealso'>
036 *    <li class='extlink'>{@source}
037 * </ul>
038 *
039 * @param <T> The annotation type.
040 */
041public final class AnnotationInfo<T extends Annotation> {
042
043   private final ClassInfo c;
044   private final MethodInfo m;
045   private final Package p;
046   private final T a;
047   private volatile Method[] methods;
048   final int rank;
049
050   /**
051    * Constructor.
052    *
053    * @param c The class where the annotation was found.
054    * @param m The method where the annotation was found.
055    * @param p The package where the annotation was found.
056    * @param a The annotation found.
057    */
058   AnnotationInfo(ClassInfo c, MethodInfo m, Package p, T a) {
059      this.c = c;
060      this.m = m;
061      this.p = p;
062      this.a = a;
063      this.rank = getRank(a);
064   }
065
066   private static int getRank(Object a) {
067      ClassInfo ci = ClassInfo.of(a);
068      MethodInfo mi = ci.getPublicMethod(x -> x.hasName("rank") && x.hasNoParams() && x.hasReturnType(int.class));
069      if (mi != null) {
070         try {
071            return (int)mi.invoke(a);
072         } catch (ExecutableException e) {
073            e.printStackTrace();
074         }
075      }
076      return 0;
077   }

161   /**
162    * Converts this object to a readable JSON object for debugging purposes.
163    *
164    * @return A new map showing the attributes of this object as a JSON object.
165    */
166   public JsonMap toJsonMap() {
167      JsonMap jm = new JsonMap();
168      if (c != null)
169         jm.put("class", c.getSimpleName());
170      if (m != null)
171         jm.put("method", m.getShortName());
172      if (p != null)
173         jm.put("package", p.getName());
174      JsonMap ja = new JsonMap();
175      ClassInfo ca = ClassInfo.of(a.annotationType());
176      ca.forEachDeclaredMethod(null, x -> {
177         try {
178            Object v = x.invoke(a);
179            Object d = x.inner().getDefaultValue();
180            if (ne(v, d)) {
181               if (! (ArrayUtils.isArray(v) && Array.getLength(v) == 0 && Array.getLength(d) == 0))
182                  ja.put(m.getName(), v);
183            }
184         } catch (Exception e) {
185            ja.put(m.getName(), e.getLocalizedMessage());
186         }
187      });
188      jm.put("@" + ca.getSimpleName(), ja);
189      return jm;
190   }

192   private Constructor<? extends AnnotationApplier<?,?>>[] applyConstructors;
193
194   /**
195    * If this annotation has a {@link ContextApply} annotation, consumes an instance of the specified {@link AnnotationApplier} class.
196    *
197    * @param vrs Variable resolver passed to the {@link AnnotationApplier} object.
198    * @param consumer The consumer.
199    * @return This object.
200    * @throws ExecutableException Exception occurred on invoked constructor/method/field.
201    */
202   @SuppressWarnings("unchecked")
203   public AnnotationInfo<T> getApplies(VarResolverSession vrs, Consumer<AnnotationApplier<Annotation,Object>> consumer) throws ExecutableException {
204      try {
205         if (applyConstructors == null) {
206            ContextApply cpa = a.annotationType().getAnnotation(ContextApply.class);
207            if (cpa == null)
208               applyConstructors = new Constructor[]{ AnnotationApplier.NoOp.class.getConstructor(VarResolverSession.class) };
209            else {
210               applyConstructors = new Constructor[cpa.value().length];
211               for (int i = 0; i < cpa.value().length; i++)
212                  applyConstructors[i] = (Constructor<? extends AnnotationApplier<?,?>>) cpa.value()[i].getConstructor(VarResolverSession.class);
213            }
214         }
215         for (int i = 0; i < applyConstructors.length; i++)
216            consumer.accept((AnnotationApplier<Annotation,Object>) applyConstructors[i].newInstance(vrs));
217      } catch (InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException | NoSuchMethodException | SecurityException e) {
218         throw new ExecutableException(e);
219      }
220      return this;
221   }

271   /**
272    * Returns <jk>true</jk> if this object passes the specified predicate test.
273    *
274    * @param test The test to perform.
275    * @return <jk>true</jk> if this object passes the specified predicate test.
276    */
277   public boolean matches(Predicate<AnnotationInfo<?>> test) {
278      return test(test, this);
279   }

281   /**
282    * Performs an action on this object if the specified predicate test passes.
283    *
284    * @param test A test to apply to determine if action should be executed.  Can be <jk>null</jk>.
285    * @param action An action to perform on this object.
286    * @return This object.
287    */
288   public AnnotationInfo<?> accept(Predicate<AnnotationInfo<?>> test, Consumer<AnnotationInfo<?>> action) {
289      if (matches(test))
290         action.accept(this);
291      return this;
292   }

299   /**
300    * Performs an action on all matching values on this annotation.
301    *
302    * @param <V> The annotation field type.
303    * @param type The annotation field type.
304    * @param name The annotation field name.
305    * @param test A predicate to apply to the value to determine if action should be performed.  Can be <jk>null</jk>.
306    * @param action An action to perform on the value.
307    * @return This object.
308    */
309   @SuppressWarnings("unchecked")
310   public <V> AnnotationInfo<?> forEachValue(Class<V> type, String name, Predicate<V> test, Consumer<V> action) {
311      for (Method m : _getMethods())
312         if (m.getName().equals(name) && m.getReturnType().equals(type))
313            safeRun(() -> consume(test, action, (V)m.invoke(a)));
314      return this;
315   }
316
317   /**
318    * Returns a matching value on this annotation.
319    *
320    * @param <V> The annotation field type.
321    * @param type The annotation field type.
322    * @param name The annotation field name.
323    * @param test A predicate to apply to the value to determine if value should be used.  Can be <jk>null</jk>.
324    * @return This object.
325    */
326   @SuppressWarnings("unchecked")
327   public <V> Optional<V> getValue(Class<V> type, String name, Predicate<V> test) {
328      for (Method m : _getMethods())
329         if (m.getName().equals(name) && m.getReturnType().equals(type)) {
330            try {
331               V v = (V)m.invoke(a);
332               if (test(test, v))
333                  return optional(v);
334            } catch (Exception e) {
335               e.printStackTrace(); // Shouldn't happen.
336            }
337         }
338      return empty();
339   }
340
341   Method[] _getMethods() {
342      if (methods == null)
343         synchronized(this) {
344            methods = a.annotationType().getMethods();
345         }
346      return methods;
347   }
