027/**
028 * Lightweight utility class for introspecting information about a method parameter.
029 *
030 * <ul class='seealso'>
031 *    <li class='extlink'>{@source}
032 * </ul>
033 */
034public final class ParamInfo {
035
036   private final ExecutableInfo eInfo;
037   private final Parameter p;
038   private final int index;
039   private volatile Map<Class<?>,Optional<Annotation>> annotationMap;
040
041   //-----------------------------------------------------------------------------------------------------------------
042   // Instantiation
043   //-----------------------------------------------------------------------------------------------------------------
044
045   /**
046    * Constructor.
047    *
048    * @param eInfo The constructor or method wrapper.
049    * @param p The parameter being wrapped.
050    * @param index The parameter index.
051    */
052   protected ParamInfo(ExecutableInfo eInfo, Parameter p, int index) {
053      this.eInfo = eInfo;
054      this.p = p;
055      this.index = index;
056   }
057
058   /**
059    * Returns the index position of this parameter.
060    *
061    * @return The index position of this parameter.
062    */
063   public int getIndex() {
064      return index;
065   }
066
067   /**
068    * Returns the method that this parameter belongs to.
069    *
070    * @return The method that this parameter belongs to, or <jk>null</jk> if it belongs to a constructor.
071    */
072   public MethodInfo getMethod() {
073      return eInfo.isConstructor() ? null : (MethodInfo)eInfo;
074   }
075
076   /**
077    * Returns the constructor that this parameter belongs to.
078    *
079    * @return The constructor that this parameter belongs to, or <jk>null</jk> if it belongs to a method.
080    */
081   public ConstructorInfo getConstructor() {
082      return eInfo.isConstructor() ? (ConstructorInfo)eInfo : null;
083   }
084
085   /**
086    * Returns the class type of this parameter.
087    *
088    * @return The class type of this parameter.
089    */
090   public ClassInfo getParameterType() {
091      return eInfo.getParamType(index);
092   }

098   /**
099    * Performs an action on all matching annotations declared on this parameter.
100    *
101    * @param <A> The annotation type to look for.
102    * @param type The annotation type to look for.
103    * @param filter A predicate to apply to the entries to determine if action should be performed.  Can be <jk>null</jk>.
104    * @param action An action to perform on the entry.
105    * @return This object.
106    */
107   public <A extends Annotation> ParamInfo forEachDeclaredAnnotation(Class<A> type, Predicate<A> filter, Consumer<A> action) {
108      for (Annotation a : eInfo._getParameterAnnotations(index))
109         consume(type, filter, action, a);
110      return this;
111   }
112
113   /**
114    * Returns the specified parameter annotation declared on this parameter.
115    *
116    * @param <A> The annotation type to look for.
117    * @param type The annotation type to look for.
118    * @return The specified parameter annotation declared on this parameter, or <jk>null</jk> if not found.
119    */
120   public <A extends Annotation> A getDeclaredAnnotation(Class<A> type) {
121      if (type != null)
122         for (Annotation a : eInfo._getParameterAnnotations(index))
123            if (type.isInstance(a))
124               return type.cast(a);
125      return null;
126   }
127
128   /**
129    * Finds the annotation of the specified type defined on this method parameter.
130    *
131    * <p>
132    * If the annotation cannot be found on the immediate method, searches methods with the same
133    * signature on the parent classes or interfaces.
134    * <br>The search is performed in child-to-parent order.
135    *
136    * <p>
137    * If still not found, searches for the annotation on the return type of the method.
138    *
139    * @param <A> The annotation type to look for.
140    * @param type The annotation type to look for.
141    * @return
142    *    The annotation if found, or <jk>null</jk> if not.
143    */
144   @SuppressWarnings("unchecked")
145   public <A extends Annotation> A getAnnotation(Class<A> type) {
146      Optional<Annotation> o = annotationMap().get(type);
147      if (o == null) {
148         o = optional(findAnnotation(type));
149         annotationMap().put(type, o);
150      }
151      return o.isPresent() ? (A)o.get() : null;
152   }
153
154   /**
155    * Returns <jk>true</jk> if this parameter has the specified annotation.
156    *
157    * @param <A> The annotation type to look for.
158    * @param type The annotation type to look for.
159    * @return
160    *    The <jk>true</jk> if annotation if found.
161    */
162   public <A extends Annotation> boolean hasAnnotation(Class<A> type) {
163      return getAnnotation(type) != null;
164   }
165
166   /**
167    * Returns <jk>true</jk> if this parameter doesn't have the specified annotation.
168    *
169    * @param <A> The annotation type to look for.
170    * @param type The annotation type to look for.
171    * @return
172    *    The <jk>true</jk> if annotation if not found.
173    */
174   public <A extends Annotation> boolean hasNoAnnotation(Class<A> type) {
175      return ! hasAnnotation(type);
176   }
177
178   private <A extends Annotation> A findAnnotation(Class<A> type) {
179      if (eInfo.isConstructor()) {
180         for (Annotation a2 : eInfo._getParameterAnnotations(index))
181            if (type.isInstance(a2))
182               return type.cast(a2);
183         return eInfo.getParamType(index).unwrap(Value.class,Optional.class).getAnnotation(type);
184      }
185      MethodInfo mi = (MethodInfo)eInfo;
186      Value<A> v = Value.empty();
187      mi.forEachMatchingParentFirst(x -> true, x -> x.forEachParameterAnnotation(index, type, y -> true, y -> v.set(y)));
188      return v.orElseGet(() -> eInfo.getParamType(index).unwrap(Value.class,Optional.class).getAnnotation(type));
189   }
190
191   /**
192    * Performs an action on all matching annotations on this parameter.
193    *
194    * <p>
195    * Searches all methods with the same signature on the parent classes or interfaces
196    * and the return type on the method.
197    * <p>
198    * Results are in parent-to-child order.
199    *
200    * @param <A> The annotation type to look for.
201    * @param type The annotation type to look for.
202    * @param filter A predicate to apply to the entries to determine if action should be performed.  Can be <jk>null</jk>.
203    * @param action An action to perform on the entry.
204    * @return This object.
205    */
206   public <A extends Annotation> ParamInfo forEachAnnotation(Class<A> type, Predicate<A> filter, Consumer<A> action) {
207      return forEachAnnotation(AnnotationProvider.DEFAULT, type, filter, action);
208   }
209
210   /**
211    * Returns the first matching annotation on this method parameter.
212    *
213    * <p>
214    * Searches all methods with the same signature on the parent classes or interfaces
215    * and the return type on the method.
216    * <p>
217    * Results are in parent-to-child order.
218    *
219    * @param <A> The annotation type to look for.
220    * @param type The annotation type to look for.
221    * @param filter A predicate to apply to the entries to determine if value should be used.  Can be <jk>null</jk>.
222    * @return A list of all matching annotations found or an empty list if none found.
223    */
224   @SuppressWarnings("unchecked")
225   public <A extends Annotation> A getAnnotation(Class<A> type, Predicate<A> filter) {
226      if (eInfo.isConstructor) {
227         ClassInfo ci = eInfo.getParamType(index).unwrap(Value.class,Optional.class);
228         A o = ci.getAnnotation(type, filter);
229         if (o != null)
230            return o;
231         for (Annotation a2 : eInfo._getParameterAnnotations(index))
232            if (test(type, filter, a2))
233               return (A)a2;
234      } else {
235         MethodInfo mi = (MethodInfo)eInfo;
236         ClassInfo ci = eInfo.getParamType(index).unwrap(Value.class,Optional.class);
237         A o = ci.getAnnotation(type, filter);
238         if (o != null)
239            return o;
240         Value<A> v = Value.empty();
241         mi.forEachMatchingParentFirst(x -> true, x -> x.forEachParameterAnnotation(index, type, filter, y -> v.set(y)));
242         return v.orElse(null);
243      }
244      return null;
245   }
246
247   private <A extends Annotation> ParamInfo forEachAnnotation(AnnotationProvider ap, Class<A> a, Predicate<A> filter, Consumer<A> action) {
248      if (eInfo.isConstructor) {
249         ClassInfo ci = eInfo.getParamType(index).unwrap(Value.class,Optional.class);
250         Annotation[] annotations = eInfo._getParameterAnnotations(index);
251         ci.forEachAnnotation(ap, a, filter, action);
252         for (Annotation a2 : annotations)
253            consume(a, filter, action, a2);
254      } else {
255         MethodInfo mi = (MethodInfo)eInfo;
256         ClassInfo ci = eInfo.getParamType(index).unwrap(Value.class,Optional.class);
257         ci.forEachAnnotation(ap, a, filter, action);
258         mi.forEachMatchingParentFirst(x -> true, x -> x.forEachParameterAnnotation(index, a, filter, action));
259      }
260      return this;
261   }
262
263   private Map<Class<?>,Optional<Annotation>> annotationMap() {
264      if (annotationMap == null) {
265         synchronized(this) {
266            annotationMap = new ConcurrentHashMap<>();
267         }
268      }
269      return annotationMap;
270   }

276   /**
277    * Returns <jk>true</jk> if this object passes the specified predicate test.
278    *
279    * @param test The test to perform.
280    * @return <jk>true</jk> if this object passes the specified predicate test.
281    */
282   public boolean matches(Predicate<ParamInfo> test) {
283      return test(test, this);
284   }
285
286   /**
287    * Performs an action on this object if the specified predicate test passes.
288    *
289    * @param test A test to apply to determine if action should be executed.  Can be <jk>null</jk>.
290    * @param action An action to perform on this object.
291    * @return This object.
292    */
293   public ParamInfo accept(Predicate<ParamInfo> test, Consumer<ParamInfo> action) {
294      if (matches(test))
295         action.accept(this);
296      return this;
297   }
298
299   /**
300    * Returns <jk>true</jk> if the parameter type is an exact match for the specified class.
301    *
302    * @param c The type to check.
303    * @return <jk>true</jk> if the parameter type is an exact match for the specified class.
304    */
305   public boolean isType(Class<?> c) {
306      return getParameterType().is(c);
307   }
308
309   /**
310    * Returns <jk>true</jk> if the parameter has a name provided by the class file.
311    *
312    * @return <jk>true</jk> if the parameter has a name provided by the class file.
313    */
314   public boolean hasName() {
315      return p.isNamePresent() || p.isAnnotationPresent(Name.class);
316   }
317
318   /**
319    * Returns the name of the parameter.
320    *
321    * <p>
322    * If the parameter's name is present, then this method returns the name provided by the class file.
323    * Otherwise, this method synthesizes a name of the form argN, where N is the index of the parameter in the descriptor of the method which declares the parameter.
324    *
325    * @return The name of the parameter.
326    * @see Parameter#getName()
327    */
328   public String getName() {
329      Name n = p.getAnnotation(Name.class);
330      if (n != null)
331         return n.value();
332      if (p.isNamePresent())
333         return p.getName();
334      return null;
335   }
336
337   /**
338    * Returns <jk>true</jk> if this parameter can accept the specified value.
339    *
340    * @param value The value to check.
341    * @return <jk>true</jk> if this parameter can accept the specified value.
342    */
343   public boolean canAccept(Object value) {
344      return getParameterType().isInstance(value);
345   }
346
347   @Override
348   public String toString() {
349      return (eInfo.getSimpleName()) + "[" + index + "]";
350   }
