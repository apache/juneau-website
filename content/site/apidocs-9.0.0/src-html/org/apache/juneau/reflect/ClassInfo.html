031/**
032 * Lightweight utility class for introspecting information about a class.
033 *
034 * <p>
035 * Provides various convenience methods for introspecting fields/methods/annotations
036 * that aren't provided by the standard Java reflection APIs.
037 *
038 * <p>
039 * Objects are designed to be lightweight to create and threadsafe.
040 *
041 * <h5 class='figure'>Example:</h5>
042 * <p class='bjava'>
043 *    <jc>// Wrap our class inside a ClassInfo.</jc>
044 *    ClassInfo <jv>classInfo</jv> = ClassInfo.<jsm>of</jsm>(MyClass.<jk>class</jk>);
045 *
046 *    <jc>// Get all methods in parent-to-child order, sorted alphabetically per class.</jc>
047 *    <jk>for</jk> (MethodInfo <jv>methodInfo</jv> : <jv>classInfo</jv>.getAllMethods()) {
048 *       <jc>// Do something with it.</jc>
049 *    }
050 *
051 *    <jc>// Get all class-level annotations in parent-to-child order.</jc>
052 *    <jk>for</jk> (MyAnnotation <jv>annotation</jv> : <jv>classInfo</jv>.getAnnotations(MyAnnotation.<jk>class</jk>)) {
053 *       <jc>// Do something with it.</jc>
054 *    }
055 * </p>
056 *
057 * <ul class='seealso'>
058 *    <li class='extlink'>{@source}
059 * </ul>
060 */
061public final class ClassInfo {

088    *
089    * @param c The class type.
090    * @return The constructed class info, or <jk>null</jk> if the type was <jk>null</jk>.
091    */
092   public static ClassInfo of(Class<?> c) {
093      if (c == null)
094         return null;
095      ClassInfo ci = CACHE.get(c);
096      if (ci == null) {
097         ci = new ClassInfo(c, c);
098         CACHE.put(c, ci);
099      }
100      return ci;
101   }

126   /**
127    * Same as {@link #of(Object)} but attempts to deproxify the object if it's wrapped in a CGLIB proxy.
128    *
129    * @param o The class instance.
130    * @return The constructed class info, or <jk>null</jk> if the object was <jk>null</jk>.

132   public static ClassInfo ofProxy(Object o) {
133      if (o == null)
134         return null;
135      Class<?> c = getProxyFor(o);
136      return c == null ? ClassInfo.of(o) : ClassInfo.of(c);
137   }

139   /**
140    * When this metadata is against a CGLIB proxy, this method finds the underlying "real" class.
141    *
142    * @param o The class instance.
143    * @return The non-proxy class, or <jk>null</jk> if it's not a CGLIB proxy.
144    */
145   private static Class<?> getProxyFor(Object o) {
146      Class<?> c = o.getClass();
147      String s = c.getName();
148      if (s.indexOf('$') == -1 || ! s.contains("$$EnhancerBySpringCGLIB$$"))
149         return null;
150      Value<Class<?>> v = Value.empty();
151      ClassInfo.of(c).forEachPublicMethod(
152         m -> m.hasName("getTargetClass") && m.hasNoParams() && m.hasReturnType(Class.class),
153         m -> safeRun(() -> v.set(m.invoke(o)))
154      );
155      return v.orElse(null);
156   }

211   /**
212    * Unwrap this class if it's a parameterized type of the specified type such as {@link Value} or {@link Optional}.
213    *
214    * @param wrapperTypes The parameterized types to unwrap if this class is one of those types.
215    * @return The class info on the unwrapped type, or just this type if this isn't one of the specified types.
216    */
217   public ClassInfo unwrap(Class<?>...wrapperTypes) {
218      for (Class<?> wt : wrapperTypes) {
219         if (isParameterizedTypeOf(wt)) {
220            Type t = getFirstParameterType(wt);
221            if (t != null)
222               return of(t).unwrap(wrapperTypes); // Recursively do it again.
223         }
224      }
225      return this;
226   }

228   private boolean isParameterizedTypeOf(Class<?> c) {
229      return
230         (t instanceof ParameterizedType && ((ParameterizedType)t).getRawType() == c)
231         || (t instanceof Class && c.isAssignableFrom((Class<?>)t));
232   }
233
234   private Type getFirstParameterType(Class<?> parameterizedType) {
235      if (t instanceof ParameterizedType) {
236         ParameterizedType pt = (ParameterizedType)t;
237         Type[] ta = pt.getActualTypeArguments();
238         if (ta.length > 0)
239            return ta[0];
240      } else if (t instanceof Class) /* Class that extends Optional<T> */ {
241         Class<?> c = (Class<?>)t;
242         if (c != parameterizedType && parameterizedType.isAssignableFrom(c))
243            return ClassInfo.of(c).getParameterType(0, parameterizedType);
244      }
245      return null;
246   }

365   /** Results are in child-to-parent order. */
366   ClassInfo[] _getInterfaces() {
367      if (interfaces == null) {
368         synchronized(this) {
369            Set<ClassInfo> s = set();
370            for (ClassInfo ci : _getParents())
371               for (ClassInfo ci2 : ci._getDeclaredInterfaces()) {
372                  s.add(ci2);
373                  for (ClassInfo ci3 : ci2._getInterfaces())
374                     s.add(ci3);
375               }
376            interfaces = s.toArray(new ClassInfo[s.size()]);
377         }
378      }
379      return interfaces;
380   }
381
382   /** Results are in the same order as Class.getInterfaces(). */
383   ClassInfo[] _getDeclaredInterfaces() {
384      if (declaredInterfaces == null) {
385         synchronized(this) {
386            Class<?>[] ii = c == null ? new Class[0] : c.getInterfaces();
387            ClassInfo[] l = new ClassInfo[ii.length];
388            for (int i = 0; i < ii.length; i++)
389               l[i] = of(ii[i]);
390            declaredInterfaces = l;
391         }
392      }
393      return declaredInterfaces;
394   }
395
396   /** Results are in child-to-parent order. */
397   ClassInfo[] _getParents() {
398      if (parents == null) {
399         synchronized(this) {
400            List<ClassInfo> l = list();
401            Class<?> pc = c;
402            while (pc != null && pc != Object.class) {
403               l.add(of(pc));
404               pc = pc.getSuperclass();
405            }
406            parents = l.toArray(new ClassInfo[l.size()]);
407         }
408      }
409      return parents;
410   }
411
412   /** Results are classes-before-interfaces, then child-to-parent order. */
413   ClassInfo[] _getAllParents() {
414      if (allParents == null) {
415         synchronized(this) {
416            ClassInfo[] a1 = _getParents(), a2 = _getInterfaces();
417            ClassInfo[] l = new ClassInfo[a1.length + a2.length];
418            for (int i = 0; i < a1.length; i++)
419               l[i] = a1[i];
420            for (int i = 0; i < a2.length; i++)
421               l[i+a1.length] = a2[i];
422            allParents = l;
423         }
424      }
425      return allParents;
426   }

573   MethodInfo[] _getPublicMethods() {
574      if (publicMethods == null) {
575         synchronized(this) {
576            Method[] mm = c == null ? new Method[0] : c.getMethods();
577            List<MethodInfo> l = list(mm.length);
578            for (Method m : mm)
579               if (m.getDeclaringClass() != Object.class)
580                  l.add(getMethodInfo(m));
581            l.sort(null);
582            publicMethods = l.toArray(new MethodInfo[l.size()]);
583         }
584      }
585      return publicMethods;
586   }
587
588   MethodInfo[] _getDeclaredMethods() {
589      if (declaredMethods == null) {
590         synchronized(this) {
591            Method[] mm = c == null ? new Method[0] : c.getDeclaredMethods();
592            List<MethodInfo> l = list(mm.length);
593            for (Method m : mm)
594               if (! "$jacocoInit".equals(m.getName())) // Jacoco adds its own simulated methods.
595                  l.add(getMethodInfo(m));
596            l.sort(null);
597            declaredMethods = l.toArray(new MethodInfo[l.size()]);
598         }
599      }
600      return declaredMethods;
601   }
602
603   MethodInfo[] _getAllMethods() {
604      if (allMethods == null) {
605         synchronized(this) {
606            List<MethodInfo> l = list();
607            for (ClassInfo c : _getAllParents())
608               c._appendDeclaredMethods(l);
609            allMethods = l.toArray(new MethodInfo[l.size()]);
610         }
611      }
612      return allMethods;
613   }
614
615   MethodInfo[] _getAllMethodsParentFirst() {
616      if (allMethodsParentFirst == null) {
617         synchronized(this) {
618            List<MethodInfo> l = list();
619            ClassInfo[] parents = _getAllParents();
620            for (int i = parents.length-1; i >=0; i--)
621               parents[i]._appendDeclaredMethods(l);
622            allMethodsParentFirst = l.toArray(new MethodInfo[l.size()]);
623         }
624      }
625      return allMethodsParentFirst;
626   }

628   private synchronized List<MethodInfo> _appendDeclaredMethods(List<MethodInfo> l) {
629      for (MethodInfo mi : _getDeclaredMethods())
630         l.add(mi);
631      return l;

2350    *
2351    * @param index The zero-based index of the parameter to resolve.

2353    * @return The resolved real class.

2355   @SuppressWarnings("null")
2356   public Class<?> getParameterType(int index, Class<?> pt) {
2357      assertArgNotNull("pt", pt);

2359      // We need to make up a mapping of type names.
2360      Map<Type,Type> typeMap = new HashMap<>();
2361      Class<?> cc = c;
2362      while (pt != cc.getSuperclass()) {
2363         extractTypes(typeMap, cc);
2364         cc = cc.getSuperclass();

2375      Type actualType = cpt.getActualTypeArguments()[index];
2376
2377      if (typeMap.containsKey(actualType))
2378         actualType = typeMap.get(actualType);

2380      if (actualType instanceof Class) {

2385         if (gct instanceof ParameterizedType)
2386            return Array.newInstance((Class<?>)((ParameterizedType)gct).getRawType(), 0).getClass();

2389         List<Class<?>> nestedOuterTypes = new LinkedList<>();
2390         for (Class<?> ec = cc.getEnclosingClass(); ec != null; ec = ec.getEnclosingClass()) {
2391            Class<?> outerClass = cc.getClass();
2392            nestedOuterTypes.add(outerClass);
2393            Map<Type,Type> outerTypeMap = new HashMap<>();
2394            extractTypes(outerTypeMap, outerClass);
2395            for (Map.Entry<Type,Type> entry : outerTypeMap.entrySet()) {
2396               Type key = entry.getKey(), value = entry.getValue();

2398                  TypeVariable<?> keyType = (TypeVariable<?>)key;
2399                  if (keyType.getName().equals(typeVariable.getName()) && isInnerClass(keyType.getGenericDeclaration(), typeVariable.getGenericDeclaration())) {

2401                        return (Class<?>)value;

2404               }
2405            }
