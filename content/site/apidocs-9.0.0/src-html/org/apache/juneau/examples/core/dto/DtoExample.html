013package org.apache.juneau.examples.core.dto;
014
015import org.apache.juneau.dto.atom.Feed;
016import org.apache.juneau.dto.swagger.Swagger;
017import org.apache.juneau.html.HtmlSerializer;
018import org.apache.juneau.http.header.*;
019import org.apache.juneau.json.*;
020
021import static org.apache.juneau.dto.atom.AtomBuilder.*;
022import static org.apache.juneau.dto.atom.AtomBuilder.link;
023import static org.apache.juneau.dto.html5.HtmlBuilder.*;
024import static org.apache.juneau.dto.swagger.SwaggerBuilder.*;
025
026import java.net.*;
027
028/**
029 * Sample class which shows the usage of DTO module which is a
030 * Sub module of the core.
031 *
032 * <ul class='seealso'>
033 *    <li class='extlink'>{@source}
034 * </ul>
035 */
036public class DtoExample {
037
038   /**
039    * DTO Samples
040    *
041    * @param args Unused.
042    * @throws Exception Unused.
043    */
044   @SuppressWarnings("unused")
045   public static void main(String[] args) throws Exception {
046
047      //Produces
048      /**
049       * <table>
050       * <tr>
051       * <th>c1</th>
052       * <th>c2</th>
053       * </tr>
054       * <tr>
055       * <td>v1</td>
056       * <td>v2</td>
057       * </tr>
058       * </table>
059       */
060      Object mytable =
061         table(
062            tr(
063               th("c1"),
064               th("c2")
065            ),
066            tr(
067               td("v1"),
068               td("v2")
069            )
070         );
071
072      String html = HtmlSerializer.DEFAULT.serialize(mytable);
073
074      Object mainJsp =
075         form().action("main.jsp").method("GET")
076         .children(
077            input("text").name("first_name").value("apache"), br(),
078            input("text").name("last_name").value("juneau"), br(),
079            button("submit", "Submit"),
080            button("reset", "Reset")
081         );
082
083      /**
084       * <form action='main.jsp' method='POST'>
085       * Position (1-10000): <input name='pos' type='number'
086       * value='1'/><br/>
087       * Limit (1-10000): <input name='pos' type='number'
088       * value='100'/><br/>
089       * <button type='submit'>Submit</button>
090       * <button type='reset'>Reset</button>
091       * </form>
092       */
093      html = HtmlSerializer.DEFAULT.serialize(mainJsp);
094
095      /**
096       * Produces
097       * {
098       *    a:{action:'main.jsp',method:'GET'},
099       *    c:[
100       *    {a:{type:'text',name:'first_name',value:'apache'}},{},
101       *    {a:{type:'text',name:'last_name',value:'juneau'}},{},
102       *    {a:{type:'submit'},c:['Submit']},
103       *    {a:{type:'reset'},c:['Reset']}
104       *    ]
105       * }
106       */
107      html =  SimpleJsonSerializer.DEFAULT.serialize(mainJsp);
108
109      Feed feed =
110         feed("tag:foo.org", "Title", "2016-12-31T05:02:03Z")
111         .setSubtitle(text("html").setText("Subtitle"))
112         .setLinks(
113            link("alternate", "text/html", "http://foo.org/").setHreflang("en"),
114            link("self", "application/atom+xml", "http://foo.org/feed.atom")
115         )
116         .setGenerator(
117            generator("Example Toolkit").setUri("http://www.foo.org/").setVersion("1.0")
118         )
119         .setEntries(
120            entry("tag:foo.org", "Title", "2016-12-31T05:02:03Z")
121            .setLinks(
122               link("alternate", "text/html", "http://foo.org/2005/04/02/atom"),
123               link("enclosure", "audio/mpeg", "http://foo.org/audio/foobar.mp3").setLength(1337)
124            )
125            .setPublished("2016-12-31T05:02:03Z")
126            .setAuthors(
127               person("John Smith").setUri(new URI("http://foo.org/")).setEmail("foo@foo.org")
128            )
129            .setContributors(
130               person("John Smith"),
131               person("Jane Smith")
132            )
133            .setContent(
134               content("xhtml")
135               .setLang("en")
136               .setBase("http://foo.org/")
137               .setText("<div><p><i>[Sample content]</i></p></div>")
138            )
139         );
140
141      Swagger swagger = swagger()
142         .setSwagger("2.0")
143         .setInfo(
144            info("Swagger Petstore", "1.0.0")
145            .setDescription("This is a sample server Petstore server.")
146            .setTermsOfService("http://swagger.io/terms/")
147            .setContact(
148               contact().setEmail("apiteam@swagger.io")
149            )
150            .setLicense(
151               license("Apache 2.0").setUrl(URI.create("http://www.apache.org/licenses/LICENSE-2.0.html"))
152            )
153         )
154         .addPath("/pet", "post",
155            operation()
156            .setTags("pet")
157            .setSummary("Add a new pet to the store")
158            .setDescription("")
159            .setOperationId("addPet")
160            .setConsumes(MediaType.JSON, MediaType.XML)
161            .setProduces(MediaType.JSON, MediaType.XML)
162            .setParameters(
163               parameterInfo("body", "body")
164               .setDescription("Pet object that needs to be added to the store")
165               .setRequired(true)
166            )
167            .addResponse("405", responseInfo("Invalid input"))
168         );
169
170      // Serialize to Swagger/JSON
171      /**
172       * Produces
173       * {
174       *  "swagger": "2.0",
175       *  "info": {
176       *      "title": "Swagger Petstore",
177       *      "description": "This is a sample server Petstore server.",
178       *      "version": "1.0.0",
179       *      "termsOfService": "http://swagger.io/terms/",
180       *      "contact": {
181       *          "email": "apiteam@swagger.io"
182       *      },
183       *      "license": {
184       *          "name": "Apache 2.0",
185       *          "url": "http://www.apache.org/licenses/LICENSE-2.0.html"
186       *      }
187       *  },
188       * "paths": {
189       *      "/pet": {
190       *          "post": {
191       *              "tags": [
192       *                  "pet"
193       *               ],
194       *              "summary": "Add a new pet to the store",
195       *              "description": "",
196       *              "operationId": "addPet",
197       *              "consumes": [
198       *                  "application/json",
199       *                  "text/xml"
200       *              ],
201       *              "produces": [
202       *                  "application/json",
203       *                  "text/xml"
204       *              ],
205       *              "parameters": [
206       *                  {
207       *                      "in": "body",
208       *                      "name": "body",
209       *                      "description": "Pet object that needs to be added to the store",
210       *                      "required": true
211       *                  }
212       *              ],
213       *              "responses": {
214       *                  "405": {
215       *                      "description": "Invalid input"
216       *                  }
217       *              }
218       *         }
219       *      }
220       *  },
221       *  }
222       */
223      String swaggerJson = JsonSerializer.DEFAULT_READABLE.serialize(swagger);
224
225   }
