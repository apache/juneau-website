047/**
048 * Entry point for Juneau microservice that implements a REST interface using Jetty on a single port.
049 *
050 * <h5 class='topic'>Jetty Server Details</h5>
051 *
052 * The Jetty server is created by the {@link #createServer()} method and started with the {@link #startServer()} method.
053 * These methods can be overridden to provided customized behavior.
054 *
055 * <h5 class='topic'>Defining REST Resources</h5>
056 *
057 * Top-level REST resources are defined in the <c>jetty.xml</c> file as normal servlets.
058 *
059 * <ul class='seealso'>
060 *    <li class='link'>{@doc juneau-microservice-jetty}
061 *    <li class='extlink'>{@source}
062 * </ul>
063 */
064public class JettyMicroservice extends Microservice {

159      /**
160       * Specifies the contents or location of the <c>jetty.xml</c> file used by the Jetty server.
161       *
162       * <p>
163       * If you do not specify this value, it is pulled from the following in the specified order:
164       * <ul class='spaced-list'>
165       *    <li>
166       *       <c>Jetty/config</c> setting in the config file.
167       *       <c>Jetty-Config</c> setting in the manifest file.
168       * </ul>
169       *
170       * <p>
171       * By default, we look for the <c>jetty.xml</c> file in the following locations:
172       * <ul class='spaced-list'>
173       *    <li><c>jetty.xml</c> in home directory.
174       *    <li><c>files/jetty.xml</c> in home directory.
175       *    <li><c>/jetty.xml</c> in classpath.
176       *    <li><c>/files/jetty.xml</c> in classpath.
177       * </ul>
178       *
179       * @param jettyXml
180       *    The contents or location of the file.
181       *    <br>Can be any of the following:
182       *    <ul>
183       *       <li>{@link String} - Relative path to file on file system or classpath.
184       *       <li>{@link File} - File on file system.
185       *       <li>{@link InputStream} - Raw contents as <c>UTF-8</c> encoded stream.
186       *       <li>{@link Reader} - Raw contents.
187       *    </ul>
188       *
189       * @param resolveVars
190       *    If <jk>true</jk>, SVL variables in the file will automatically be resolved.
191       * @return This object.
192       * @throws IOException Thrown by underlying stream.
193       */
194      public Builder jettyXml(Object jettyXml, boolean resolveVars) throws IOException {
195         if (jettyXml instanceof String)
196            this.jettyXml = read(resolveFile(jettyXml.toString()));
197         else if (jettyXml instanceof File)
198            this.jettyXml = read((File)jettyXml);
199         else if (jettyXml instanceof InputStream)
200            this.jettyXml = read((InputStream)jettyXml);
201         else if (jettyXml instanceof Reader)
202            this.jettyXml = read((Reader)jettyXml);
203         else
204            throw new BasicRuntimeException("Invalid object type passed to jettyXml(Object): {0}", className(jettyXml));
205         this.jettyXmlResolveVars = resolveVars;
206         return this;
207      }

511   @Override /* Microservice */
512   public synchronized JettyMicroservice stop() throws Exception {
513      final Logger logger = getLogger();
514      final Messages mb2 = messages;
515      Thread t = new Thread("JettyMicroserviceStop") {
516         @Override /* Thread */
517         public void run() {
518            try {
519               if (server == null || server.isStopping() || server.isStopped())
520                  return;
521               listener.onStopServer(JettyMicroservice.this);
522               out(mb2, "StoppingServer");
523               server.stop();
524               out(mb2, "ServerStopped");
525               listener.onPostStopServer(JettyMicroservice.this);
526            } catch (Exception e) {
527               logger.log(Level.WARNING, e.getLocalizedMessage(), e);
528            }
529         }
530      };
531      t.start();
532      try {
533         t.join();
534      } catch (InterruptedException e) {
535         e.printStackTrace();
536      }
537      super.stop();
538
539      return this;
540   }

547   /**
548    * Returns the port that this microservice started up on.
549    * <p>
550    * The value is determined by looking at the <c>Server/Connectors[ServerConnector]/port</c> value in the
551    * Jetty configuration.
552    *
553    * @return The port that this microservice started up on.
554    */
555   public int getPort() {
556      for (Connector c : getServer().getConnectors())
557         if (c instanceof ServerConnector)
558            return ((ServerConnector)c).getPort();
559      throw new RuntimeException("Could not locate ServerConnector in Jetty server.");
560   }
561
562   /**
563    * Returns the context path that this microservice is using.
564    * <p>
565    * The value is determined by looking at the <c>Server/Handlers[ServletContextHandler]/contextPath</c> value
566    * in the Jetty configuration.
567    *
568    * @return The context path that this microservice is using.
569    */
570   public String getContextPath() {
571      for (Handler h : getServer().getHandlers()) {
572         if (h instanceof HandlerCollection)
573            for (Handler h2 : ((HandlerCollection)h).getChildHandlers())
574               if (h2 instanceof ServletContextHandler)
575                  return ((ServletContextHandler)h2).getContextPath();
576         if (h instanceof ServletContextHandler)
577            return ((ServletContextHandler)h).getContextPath();
578      }
579      throw new RuntimeException("Could not locate ServletContextHandler in Jetty server.");
580   }

582   /**
583    * Returns whether this microservice is using <js>"http"</js> or <js>"https"</js>.
584    * <p>
585    * The value is determined by looking for the existence of an SSL Connection Factorie by looking for the
586    * <c>Server/Connectors[ServerConnector]/ConnectionFactories[SslConnectionFactory]</c> value in the Jetty
587    * configuration.
588    *
589    * @return Whether this microservice is using <js>"http"</js> or <js>"https"</js>.
590    */
591   public String getProtocol() {
592      for (Connector c : getServer().getConnectors())
593         if (c instanceof ServerConnector)
594            for (ConnectionFactory cf : ((ServerConnector)c).getConnectionFactories())
595               if (cf instanceof SslConnectionFactory)
596                  return "https";
597      return "http";
598   }

629   /**
630    * Method used to create (but not start) an instance of a Jetty server.
631    *
632    * <p>
633    * Subclasses can override this method to customize the Jetty server before it is started.
634    *
635    * <p>
636    * The default implementation is configured by the following values in the config file
637    * if a jetty.xml is not specified via a <c>REST/jettyXml</c> setting:
638    * <p class='bini'>
639    *    <cc>#================================================================================
640    *    # Jetty settings
641    *    #================================================================================</cc>
642    *    <cs>[Jetty]</cs>
643    *
644    *    <cc># Path of the jetty.xml file used to configure the Jetty server.</cc>
645    *    <ck>config</ck> = jetty.xml
646    *
647    *    <cc># Resolve Juneau variables in the jetty.xml file.</cc>
648    *    <ck>resolveVars</ck> = true
649    *
650    *    <cc># Port to use for the jetty server.
651    *    # You can specify multiple ports.  The first available will be used.  '0' indicates to try a random port.
652    *    # The resulting available port gets set as the system property "availablePort" which can be referenced in the
653    *    # jetty.xml file as "$S{availablePort}" (assuming resolveVars is enabled).</cc>
654    *    <ck>port</ck> = 10000,0,0,0
655    * </p>
656    *
657    * @return The newly-created server.
658    * @throws ParseException Configuration file contains malformed input.
659    * @throws IOException File could not be read.
660    * @throws ExecutableException Exception occurred on invoked constructor/method/field.
661    */
662   public Server createServer() throws ParseException, IOException, ExecutableException {
663      listener.onCreateServer(this);
664
665      Config cf = getConfig();
666      JsonMap mf = getManifest();
667      VarResolver vr = getVarResolver();
668
669      int[] ports = ObjectUtils.firstNonNull(builder.ports, cf.get("Jetty/port").as(int[].class).orElseGet(()->mf.getWithDefault("Jetty-Port", new int[]{8000}, int[].class)));
670      int availablePort = findOpenPort(ports);
671
672      if (System.getProperty("availablePort") == null)
673         System.setProperty("availablePort", String.valueOf(availablePort));
674
675      String jettyXml = builder.jettyXml;
676      String jettyConfig = cf.get("Jetty/config").orElse(mf.getString("Jetty-Config", "jetty.xml"));
677      boolean resolveVars = ObjectUtils.firstNonNull(builder.jettyXmlResolveVars, cf.get("Jetty/resolveVars").asBoolean().orElse(false));
678
679      if (jettyXml == null)
680         jettyXml = IOUtils.loadSystemResourceAsString("jetty.xml", ".", "files");
681      if (jettyXml == null)
682         throw new BasicRuntimeException("jetty.xml file ''{0}'' was not found on the file system or classpath.", jettyConfig);
683
684      if (resolveVars)
685         jettyXml = vr.resolve(jettyXml);
686
687      getLogger().info(jettyXml);
688
689      try {
690         server = factory.create(jettyXml);
691      } catch (Exception e2) {
692         throw new ExecutableException(e2);
693      }
694
695      for (String s : cf.get("Jetty/servlets").asStringArray().orElse(new String[0])) {
696         try {
697            ClassInfo c = ClassInfo.of(Class.forName(s));
698            if (c.isChildOf(RestServlet.class)) {
699               RestServlet rs = (RestServlet)c.newInstance();
700               addServlet(rs, rs.getPath());
701            } else {
702               throw new BasicRuntimeException("Invalid servlet specified in Jetty/servlets.  Must be a subclass of RestServlet: {0}", s);
703            }
704         } catch (ClassNotFoundException e1) {
705            throw new ExecutableException(e1);
706         }
707      }
708
709      cf.get("Jetty/servletMap").asMap().orElse(EMPTY_MAP).forEach((k,v) -> {
710         try {
711            ClassInfo c = ClassInfo.of(Class.forName(v.toString()));
712            if (c.isChildOf(Servlet.class)) {
713               Servlet rs = (Servlet)c.newInstance();
714               addServlet(rs, k);
715            } else {
716               throw new BasicRuntimeException("Invalid servlet specified in Jetty/servletMap.  Must be a subclass of Servlet: {0}", v);
717            }
718         } catch (ClassNotFoundException e1) {
719            throw new ExecutableException(e1);
720         }
721      });
722
723      cf.get("Jetty/servletAttributes").asMap().orElse(EMPTY_MAP).forEach((k,v) -> addServletAttribute(k, v));
724
725      builder.servlets.forEach((k,v) -> addServlet(v, k));
726
727      builder.servletAttributes.forEach((k,v) -> addServletAttribute(k, v));
728
729      if (System.getProperty("juneau.serverPort") == null)
730         System.setProperty("juneau.serverPort", String.valueOf(availablePort));
731
732      return server;
733   }

746   /**
747    * Adds an arbitrary servlet to this microservice.
748    *
749    * @param servlet The servlet instance.
750    * @param pathSpec The context path of the servlet.
751    * @return This object.
752    * @throws RuntimeException if {@link #createServer()} has not previously been called.
753    */
754   public JettyMicroservice addServlet(Servlet servlet, String pathSpec) {
755      ServletHolder sh = new ServletHolder(servlet);
756      if (pathSpec != null && ! pathSpec.endsWith("/*"))
757         pathSpec = trimTrailingSlashes(pathSpec) + "/*";
758      getServletContextHandler().addServlet(sh, pathSpec);
759      return this;
760   }

762   /**
763    * Finds and returns the servlet context handler define in the Jetty container.
764    *
765    * @return The servlet context handler.
766    * @throws RuntimeException if context handler is not defined.
767    */
768   protected ServletContextHandler getServletContextHandler() {
769      for (Handler h : getServer().getHandlers()) {
770         ServletContextHandler sch = getServletContextHandler(h);
771         if (sch != null)
772            return sch;
773      }
774      throw new RuntimeException("Servlet context handler not found in jetty server.");
775   }

801   /**
802    * Method used to start the Jetty server created by {@link #createServer()}.
803    *
804    * <p>
805    * Subclasses can override this method to customize server startup.
806    *
807    * @return The port that this server started on.
808    * @throws Exception Error occurred.

810   protected int startServer() throws Exception {

812      server.start();
813      out(messages, "ServerStarted", getPort());
814      listener.onPostStartServer(this);
815      return getPort();

822   private static ServletContextHandler getServletContextHandler(Handler h) {
823      if (h instanceof ServletContextHandler)
824         return (ServletContextHandler)h;
825      if (h instanceof HandlerCollection) {
826         for (Handler h2 : ((HandlerCollection)h).getHandlers()) {
827            ServletContextHandler sch = getServletContextHandler(h2);
828            if (sch != null)
829               return sch;
830         }
831      }
832      return null;
833   }

835   private static int findOpenPort(int[] ports) {
836      for (int port : ports) {
837         // If port is 0, try a random port between ports[0] and 32767.
838         if (port == 0)
839            port = new Random().nextInt(32767 - ports[0] + 1) + ports[0];
840         try (ServerSocket ss = new ServerSocket(port)) {
841            return port;
842         } catch (IOException e) {}
843      }
844      return 0;
845   }
