013package org.apache.juneau.microservice.jetty.resources;
014
015import java.io.*;
016
017import org.apache.juneau.html.annotation.HtmlDocConfig;
018import org.apache.juneau.http.response.*;
019import org.apache.juneau.microservice.jetty.*;
020import org.apache.juneau.rest.*;
021import org.apache.juneau.rest.annotation.*;
022import org.apache.juneau.rest.beans.*;
023import org.apache.juneau.rest.servlet.BasicRestServlet;
024
025/**
026 * Microservice debug utilities.
027 *
028 * <ul class='seealso'>
029 *    <li class='link'>{@doc juneau-microservice-jetty}
030 *    <li class='extlink'>{@source}
031 * </ul>
032 *
033 * @serial exclude
034 */
035@Rest(
036   path="/debug",
037   title="Debug",
038   description="Debug Utilities.",
039   allowedMethodParams="OPTIONS,POST"
040)
041@HtmlDocConfig(
042   navlinks={
043      "up: request:/..",
044      "jetty-thread-dump: servlet:/jetty/dump?method=POST",
045      "api: servlet:/api",
046      "stats: servlet:/stats"
047   }
048)
049public class DebugResource extends BasicRestServlet {
050   private static final long serialVersionUID = 1L;
051
052   /**
053    * [GET /] - Shows child utilities.
054    *
055    * @return Child utility links.
056    */
057   @RestGet(path="/", description="Show contents of config file.")
058   public ResourceDescriptions getChildren() {
059      return ResourceDescriptions
060         .create()
061         .append("jetty/dump", "Jetty thread dump")
062      ;
063   }
064
065   /**
066    * [GET /jetty/dump] - Generates and retrieves the jetty thread dump.
067    *
068    * @param req The request.
069    * @param res The response.
070    * @return The thread dump contents.
071    */
072   @RestGet(path="/jetty/dump", description="Generates and retrieves the jetty thread dump.")
073   public Reader getJettyDump(RestRequest req, RestResponse res) {
074      res.setContentType("text/plain");
075      return new StringReader(JettyMicroservice.getInstance().getServer().dump());
076   }
077
078   /**
079    * [POST /jetty/dump] - Generates and saves the jetty thread dump file to jetty-thread-dump.log.
080    *
081    * @param req The request.
082    * @param res The response.
083    * @throws Exception Gets converted to 500 response.
084    * @return The thread dump contents.
085    */
086   @RestPost(path="/jetty/dump", description="Generates and saves the jetty thread dump file to jetty-thread-dump.log.")
087   public Ok createJettyDump(RestRequest req, RestResponse res) throws Exception {
088      String dump = JettyMicroservice.getInstance().getServer().dump();
089      try (FileWriter fw = new FileWriter(req.getConfig().get("Logging/logDir").orElse("") + "/jetty-thread-dump.log")) {
090         fw.write(dump);
091      }
092      return Ok.OK;
093   }
094}
