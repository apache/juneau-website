013package org.apache.juneau.microservice.springboot.template;
014
015import javax.servlet.*;
016
017import org.apache.juneau.rest.annotation.*;
018import org.apache.juneau.rest.springboot.*;
019import org.springframework.boot.autoconfigure.SpringBootApplication;
020import org.springframework.boot.builder.*;
021import org.springframework.boot.web.servlet.*;
022import org.springframework.context.annotation.*;
023import org.springframework.stereotype.Controller;
024import org.springframework.web.filter.*;
025
026/**
027 * Entry point for Examples REST application when deployed as a Spring Boot application.
028 *
029 * <ul class='seealso'>
030 *    <li class='link'>{@doc my-springboot-microservice}
031 *    <li class='extlink'>{@source}
032 * </ul>
033 */
034@SpringBootApplication
035@Controller
036public class App {
037
038   /**
039    * Entry point method.
040    * @param args Command-line arguments.
041    */
042   @SuppressWarnings("resource")
043   public static void main(String[] args) {
044      new SpringApplicationBuilder(App.class).run(args);
045   }
046
047   /**
048    * Our root REST bean.
049    * <p>
050    * Note that this must extend from {@link SpringRestServlet} so that child resources can be resolved as Spring
051    * beans.
052    * <p>
053    * All REST objects are attached to this bean using the {@link Rest#children()} annotation.
054    *
055    * @return The root resources REST bean.
056    */
057   @Bean
058   public RootResources getRootResources() {
059      return new RootResources();
060   }
061
062   /**
063    * Optionally return the {@link HelloWorldResource} object as an injectable bean.
064    *
065    * @return The hello-world REST bean.
066    */
067   @Bean
068   public HelloWorldResource getHelloWorldResource() {
069      return new HelloWorldResource("Hello Spring user!");
070   }
071
072   /**
073    * Optionally return an injectable message provider for the {@link HelloWorldResource} class.
074    *
075    * @return The message provider for the hello-world REST bean.
076    */
077   @Bean
078   public HelloWorldMessageProvider getHelloWorldMessageProvider() {
079      return new HelloWorldMessageProvider("Hello Spring injection user!");
080   }
081
082   /**
083    * @param rootResources The root REST resource servlet
084    * @return The servlet registration mapped to "/*".
085    */
086   @Bean
087   public ServletRegistrationBean<Servlet> getRootServlet(RootResources rootResources) {
088      return new ServletRegistrationBean<>(rootResources, "/*");
089   }
090
091   /**
092    * We want to be able to consume url-encoded-form-post bodies, but HiddenHttpMethodFilter triggers the HTTP
093    * body to be consumed.  So disable it.
094    *
095    * @param filter The filter.
096    * @return Filter registration bean.
097    */
098   @Bean
099   public FilterRegistrationBean<HiddenHttpMethodFilter> registration(HiddenHttpMethodFilter filter) {
100      FilterRegistrationBean<HiddenHttpMethodFilter> registration = new FilterRegistrationBean<>(filter);
101      registration.setEnabled(false);
102      return registration;
103   }
104}
