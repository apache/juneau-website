034/**
035 * Basic implementation of the {@link HttpResponse} interface for error responses.
036 *
037 * <p>
038 * Although this class implements the various setters defined on the {@link HttpResponse} interface, it's in general
039 * going to be more efficient to set the status/headers/content of this bean through the builder.
040 *
041 * <p>
042 * If the <c>unmodifiable</c> flag is set on this bean, calls to the setters will throw {@link UnsupportedOperationException} exceptions.
043 *
044 * <ul class='notes'>
045 *    <li class='warn'>Beans are not thread safe unless they're marked as unmodifiable.
046 * </ul>
047 *
048 * <ul class='seealso'>
049 *    <li class='link'>{@doc juneau-rest-common}
050 *    <li class='extlink'>{@source}
051 * </ul>
052 *
053 * @serial exclude
054 */

056@FluentSetters(ignore="setUnmodifiable")

126    */
127   public BasicHttpException(HttpResponse response) {
128      super((Throwable)null);
129      Header h = response.getLastHeader("Thrown");
130      if (h != null)
131         setMessage(thrown(h.getValue()).asParts().get().get(0).getMessage());
132      setHeaders(response.getAllHeaders());
133      setContent(response.getEntity());
134      setStatusCode(response.getStatusLine().getStatusCode());
135   }

350   /**
351    * Asserts that the specified HTTP response has the same status code as the one on the status line of this bean.

357      assertArgNotNull("response", response);
358      int expected = getStatusLine().getStatusCode();

360      assertInteger(actual).setMsg("Unexpected status code.  Expected:[{0}], Actual:[{1}]", expected, actual).is(expected);

363   /**
364    * Returns the root cause of this exception.
365    *
366    * <p>
367    * The root cause is the first exception in the init-cause parent chain that's not one of the following:
368    * <ul>
369    *    <li>{@link BasicHttpException}
370    *    <li>{@link InvocationTargetException}
371    * </ul>
372    *
373    * @return The root cause of this exception, or <jk>null</jk> if no root cause was found.
374    */
375   public Throwable getRootCause() {
376      Throwable t = this;
377      while(t != null) {
378         if (! (t instanceof BasicHttpException || t instanceof InvocationTargetException))
379            return t;
380         t = t.getCause();
381      }
382      return null;
383   }

385   /**
386    * Returns all error messages from all errors in this stack.
387    *
388    * <p>
389    * Typically useful if you want to render all the error messages in the stack, but don't want to render all the
390    * stack traces too.
391    *
392    * @param scrubForXssVulnerabilities
393    *    If <jk>true</jk>, replaces <js>'&lt;'</js>, <js>'&gt;'</js>, and <js>'&amp;'</js> characters with spaces.
394    * @return All error messages from all errors in this stack.
395    */
396   public String getFullStackMessage(boolean scrubForXssVulnerabilities) {
397      String msg = getMessage();
398      StringBuilder sb = new StringBuilder();
399      if (msg != null) {
400         if (scrubForXssVulnerabilities)
401            msg = msg.replace('<', ' ').replace('>', ' ').replace('&', ' ');
402         sb.append(msg);
403      }
404      Throwable e = getCause();
405      while (e != null) {
406         msg = e.getMessage();
407         if (msg != null && scrubForXssVulnerabilities)
408            msg = msg.replace('<', ' ').replace('>', ' ').replace('&', ' ');
409         String cls = e.getClass().getSimpleName();
410         if (msg == null)
411            sb.append(format("\nCaused by ({0})", cls));
412         else
413            sb.append(format("\nCaused by ({0}): {1}", cls, msg));
414         e = e.getCause();
415      }
416      return sb.toString();
417   }

419   @Override /* Throwable */
420   public String getMessage() {
421      String m = super.getMessage();
422      if (m == null && getCause() != null)
423         m = getCause().getMessage();
424      if (m == null)
425         m = statusLine.getReasonPhrase();
426      return m;
427   }

429   @Override /* Object */
430   public int hashCode() {
431      int i = 0;
432      Throwable t = this;
433      while (t != null) {
434         for (StackTraceElement e : t.getStackTrace())
435         i ^= e.hashCode();
436         t = t.getCause();
437      }
438      return i;
439   }
