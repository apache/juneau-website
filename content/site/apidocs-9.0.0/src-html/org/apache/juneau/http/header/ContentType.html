013package org.apache.juneau.http.header;
014
015import java.util.function.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.http.annotation.Header;
019import org.apache.juneau.internal.*;
020
021/**
022 * Represents a parsed <l>Content-Type</l> HTTP request/response header.
023 *
024 * <p>
025 * The MIME type of this content.
026 *
027 * <h5 class='figure'>Example</h5>
028 * <p class='bcode'>
029 *    Content-Type: text/html; charset=utf-8
030 * </p>
031 *
032 * <h5 class='topic'>RFC2616 Specification</h5>
033 *
034 * The Content-Type entity-header field indicates the media type of the entity-body sent to the recipient or, in the
035 * case of the HEAD method, the media type that would have been sent had the request been a GET.
036 * <p class='bcode'>
037 *    Content-Type   = "Content-Type" ":" media-type
038 * </p>
039 *
040 * <p>
041 * Media types are defined in section 3.7.
042 * An example of the field is...
043 * <p class='bcode'>
044 *    Content-Type: text/html; charset=ISO-8859-4
045 * </p>
046 *
047 * <ul class='seealso'>
048 *    <li class='link'>{@doc juneau-rest-common}
049 *    <li class='extlink'>{@doc ext.RFC2616}
050 *    <li class='extlink'>{@source}
051 * </ul>
052 *
053 * @serial exclude
054 */
055@Header("Content-Type")
056public class ContentType extends BasicMediaTypeHeader {
057
058   //-----------------------------------------------------------------------------------------------------------------
059   // Static
060   //-----------------------------------------------------------------------------------------------------------------
061
062   private static final long serialVersionUID = 1L;
063   private static final String NAME = "Content-Type";
064
065   private static Cache<String,ContentType> CACHE = Cache.of(String.class, ContentType.class).build();
066
067   // Constants
068   @SuppressWarnings("javadoc")
069   public static final ContentType
070      APPLICATION_ATOM_XML = of("application/atom+xml"),
071      APPLICATION_FORM_URLENCODED = of("application/x-www-form-urlencoded"),
072      APPLICATION_JSON = of("application/json"),
073      APPLICATION_OCTET_STREAM = of("application/octet-stream"),
074      APPLICATION_SOAP_XML = of("application/soap+xml"),
075      APPLICATION_SVG_XML = of("application/svg+xml"),
076      APPLICATION_XHTML_XML = of("application/xhtml+xml"),
077      APPLICATION_XML = of("application/xml"),
078      IMAGE_BMP = of("image/bmp"),
079      IMAGE_GIF = of("image/gif"),
080      IMAGE_JPEG = of("image/jpeg"),
081      IMAGE_PNG = of("image/png"),
082      IMAGE_SVG = of("image/svg+xml"),
083      IMAGE_TIFF = of("image/tiff"),
084      IMAGE_WEBP = of("image/webp"),
085      MULTIPART_FORM_DATA = of("multipart/form-data"),
086      TEXT_HTML = of("text/html"),
087      TEXT_OPENAPI = of("text/openapi"),
088      TEXT_PLAIN = of("text/plain"),
089      TEXT_XML = of("text/xml"),
090      WILDCARD = of("*/*"),
091      NULL = new ContentType((String)null);
092
093   /**
094    * Static creator.
095    *
096    * @param value
097    *    The header value.
098    *    <br>Must be parsable by {@link MediaType#of(String)}.
099    *    <br>Can be <jk>null</jk>.
100    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
101    */
102   public static ContentType of(String value) {
103      return value == null ? null : CACHE.get(value, ()->new ContentType(value));
104   }
105
106   /**
107    * Static creator.
108    *
109    * @param value
110    *    The header value.
111    *    <br>Can be <jk>null</jk>.
112    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
113    */
114   public static ContentType of(MediaType value) {
115      return value == null ? null : new ContentType(value);
116   }
117
118   /**
119    * Static creator with delayed value.
120    *
121    * <p>
122    * Header value is re-evaluated on each call to {@link #getValue()}.
123    *
124    * @param value
125    *    The header value.
126    *    <br>Can be <jk>null</jk>.
127    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
128    */
129   public static ContentType of(Supplier<MediaType> value) {
130      return value == null ? null : new ContentType(value);
131   }
132
133   //-----------------------------------------------------------------------------------------------------------------
134   // Instance
135   //-----------------------------------------------------------------------------------------------------------------
136
137   /**
138    * Constructor.
139    *
140    * @param value
141    *    The header value.
142    *    <br>Must be parsable by {@link MediaType#of(String)}.
143    *    <br>Can be <jk>null</jk>.
144    */
145   public ContentType(String value) {
146      super(NAME, value);
147   }
148
149   /**
150    * Constructor.
151    *
152    * @param value
153    *    The header value.
154    *    <br>Can be <jk>null</jk>.
155    */
156   public ContentType(MediaType value) {
157      super(NAME, value);
158   }
159
160   /**
161    * Constructor with delayed value.
162    *
163    * <p>
164    * Header value is re-evaluated on each call to {@link #getValue()}.
165    *
166    * @param value
167    *    The supplier of the header value.
168    *    <br>Can be <jk>null</jk>.
169    */
170   public ContentType(Supplier<MediaType> value) {
171      super(NAME, value);
172   }
173}
