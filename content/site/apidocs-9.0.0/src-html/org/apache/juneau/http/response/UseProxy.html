013package org.apache.juneau.http.response;
014
015import static org.apache.juneau.http.response.UseProxy.*;
016
017import org.apache.http.*;
018import org.apache.juneau.annotation.*;
019import org.apache.juneau.http.*;
020import org.apache.juneau.http.annotation.*;
021
022/**
023 * Represents an <c>HTTP 305 Use Proxy</c> response.
024 *
025 * <p>
026 * The requested resource is available only through a proxy, the address for which is provided in the response.
027 * Many HTTP clients (such as Mozilla and Internet Explorer) do not correctly handle responses with this status code, primarily for security reasons.
028 *
029 * <ul class='seealso'>
030 *    <li class='link'>{@doc juneau-rest-common}
031 *    <li class='extlink'>{@source}
032 * </ul>
033 */
034@Response
035@StatusCode(STATUS_CODE)
036@Schema(description=REASON_PHRASE)
037public class UseProxy extends BasicHttpResponse {
038
039   /** HTTP status code */
040   public static final int STATUS_CODE = 305;
041
042   /** Reason phrase */
043   public static final String REASON_PHRASE = "Use Proxy";
044
045   private static final BasicStatusLine STATUS_LINE = BasicStatusLine.create().statusCode(STATUS_CODE).reasonPhrase(REASON_PHRASE).build();
046
047   /**
048    * Default unmodifiable instance.
049    *
050    * <br>Response body contains the reason phrase.
051    */
052   public static final UseProxy INSTANCE = create().unmodifiable().build();
053
054   /**
055    * Creates a builder for this class.
056    *
057    * @return A new builder bean.
058    */
059   public static HttpResponseBuilder<UseProxy> create() {
060      return new HttpResponseBuilder<>(UseProxy.class).statusLine(STATUS_LINE);
061   }
062
063   /**
064    * Constructor.
065    *
066    * @param builder The builder containing the settings for this exception.
067    */
068   public UseProxy(HttpResponseBuilder<?> builder) {
069      super(builder);
070   }
071
072   /**
073    * Constructor.
074    *
075    * <p>
076    * This is the constructor used when parsing an HTTP response.
077    *
078    * @param response The HTTP response to copy from.  Must not be <jk>null</jk>.
079    * @throws AssertionError If HTTP response status code does not match what was expected.
080    */
081   public UseProxy(HttpResponse response) {
082      this(create().copyFrom(response));
083      assertStatusCode(response);
084   }
085
086   /**
087    * Creates a builder for this class initialized with the contents of this bean.
088    *
089    * @return A new builder bean.
090    */
091   @Override /* BasicHttpResponse */
092   public HttpResponseBuilder<UseProxy> copy() {
093      return new HttpResponseBuilder<>(this);
094   }
095}
