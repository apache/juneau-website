027/**
028 * Represents a parsed <l>If-Range</l> HTTP request header.
029 *
030 * <p>
031 * If the entity is unchanged, send me the part(s) that I am missing; otherwise, send me the entire new entity.
032 *
033 * <h5 class='figure'>Example</h5>
034 * <p class='bcode'>
035 *    If-Range: "737060cd8c284d8af7ad3082f209582d"
036 * </p>
037 *
038 * <h5 class='topic'>RFC2616 Specification</h5>
039 *
040 * If a client has a partial copy of an entity in its cache, and wishes to have an up-to-date copy of the entire entity
041 * in its cache, it could use the Range request-header with a conditional GET (using either or both of
042 * If-Unmodified-Since and If-Match.)
043 * However, if the condition fails because the entity has been modified, the client would then have to make a second
044 * request to obtain the entire current entity-body.
045 *
046 * <p>
047 * The If-Range header allows a client to "short-circuit" the second request.
048 * Informally, its meaning is `if the entity is unchanged, send me the part(s) that I am missing; otherwise, send me
049 * the entire new entity'.
050 * <p class='bcode'>
051 *    If-Range = "If-Range" ":" ( entity-tag | HTTP-date )
052 * </p>
053 *
054 * <p>
055 * If the client has no entity tag for an entity, but does have a Last- Modified date, it MAY use that date in an
056 * If-Range header.
057 * (The server can distinguish between a valid HTTP-date and any form of entity-tag by examining no more than two
058 * characters.)
059 * The If-Range header SHOULD only be used together with a Range header, and MUST be ignored if the request does not
060 * include a Range header, or if the server does not support the sub-range operation.
061 *
062 * <p>
063 * If the entity tag given in the If-Range header matches the current entity tag for the entity, then the server SHOULD
064 * provide the specified sub-range of the entity using a 206 (Partial content) response.
065 * If the entity tag does not match, then the server SHOULD return the entire entity using a 200 (OK) response.
066 *
067 * <ul class='seealso'>
068 *    <li class='link'>{@doc jm.HttpParts}
069 *    <li class='extlink'>{@doc ext.RFC2616}
070 *    <li class='extlink'>{@source}
071 * </ul>
072 *
073 * @serial exclude
074 */
075@Header("If-Range")
076public class IfRange extends BasicDateHeader {
077
078   //-----------------------------------------------------------------------------------------------------------------
079   // Static
080   //-----------------------------------------------------------------------------------------------------------------
081
082   private static final long serialVersionUID = 1L;
083   private static final String NAME = "If-Range";
084
085   /**
086    * Static creator.
087    *
088    * @param value
089    *    The header value.
090    *    <br>Must be an RFC-1123 formated string (e.g. <js>"Sat, 29 Oct 1994 19:43:31 GMT"</js>).
091    *    <br>Can be <jk>null</jk>.
092    * @return A new header bean, or <jk>null</jk> if the name is <jk>null</jk> or empty or the value is <jk>null</jk>.
093    */
094   public static IfRange of(String value) {
095      return value == null ? null : new IfRange(value);
096   }
097
098   /**
099    * Static creator.
100    *
101    * @param value
102    *    The header value.
103    *    <br>Can be <jk>null</jk>.
104    * @return A new header bean, or <jk>null</jk> if the name is <jk>null</jk> or empty or the value is <jk>null</jk>.
105    */
106   public static IfRange of(ZonedDateTime value) {
107      return value == null ? null : new IfRange(value);
108   }
109
110   /**
111    * Static creator.
112    *
113    * @param value
114    *    The header value.
115    *    <br>Can be <jk>null</jk>.
116    * @return A new header bean, or <jk>null</jk> if the name is <jk>null</jk> or empty or the value is <jk>null</jk>.
117    */
118   public static IfRange of(EntityTag value) {
119      return value == null ? null : new IfRange(value);
120   }
121
122   /**
123    * Static creator with delayed value.
124    *
125    * <p>
126    * Header value is re-evaluated on each call to {@link #getValue()}.
127    *
128    * @param value
129    *    The supplier of the header value.
130    *    <br>Supplier must supply either {@link EntityTag} or {@link ZonedDateTime} objects.
131    *    <br>Can be <jk>null</jk>.
132    * @return A new header bean, or <jk>null</jk> if the name is <jk>null</jk> or empty or the value is <jk>null</jk>.
133    */
134   public static IfRange of(Supplier<?> value) {
135      return value == null ? null : new IfRange(value);
136   }
137
138   //-----------------------------------------------------------------------------------------------------------------
139   // Instance
140   //-----------------------------------------------------------------------------------------------------------------
141
142   private final EntityTag value;
143   private final Supplier<?> supplier;
144
145   /**
146    * Constructor.
147    *
148    * @param value
149    *    The header value.
150    *    <br>Must be an RFC-1123 formated string (e.g. <js>"Sat, 29 Oct 1994 19:43:31 GMT"</js>).
151    *    <br>Can be <jk>null</jk>.
152    */
153   public IfRange(String value) {
154      super(NAME, isEtag(value) ? null : value);
155      this.value = isEtag(value) ? EntityTag.of(value) : null;
156      this.supplier = null;
157   }
158
159   /**
160    * Constructor.
161    *
162    * @param value
163    *    The header value.
164    *    <br>Can be <jk>null</jk>.
165    */
166   public IfRange(ZonedDateTime value) {
167      super(NAME, value);
168      this.value = null;
169      this.supplier = null;
170   }
171
172   /**
173    * Constructor.
174    *
175    * @param value
176    *    The header value.
177    *    <br>Can be <jk>null</jk>.
178    */
179   public IfRange(EntityTag value) {
180      super(NAME, (String)null);
181      this.value = value;
182      this.supplier = null;
183   }
184
185   /**
186    * Constructor with delayed value.
187    *
188    * <p>
189    * Header value is re-evaluated on each call to {@link #getValue()}.
190    *
191    * @param value
192    *    The supplier of the header value.
193    *    <br>Supplier must supply either {@link EntityTag} or {@link ZonedDateTime} objects.
194    *    <br>Can be <jk>null</jk>.
195    */
196   public IfRange(Supplier<?> value) {
197      super(NAME, (String)null);
198      this.value = null;
199      this.supplier = value;
200   }
201
202   @Override /* Header */
203   public String getValue() {
204      if (supplier != null) {
205         Object o = supplier.get();
206         if (o == null)
207            return null;
208         if (o instanceof EntityTag) {
209            return o.toString();
210         } else if (o instanceof ZonedDateTime) {
211            return RFC_1123_DATE_TIME.format((ZonedDateTime)o);
212         }
213         throw runtimeException("Invalid object type returned by supplier: {0}", className(o));
214      }
215      if (value != null)
216         return stringify(value);
217      return super.getValue();
218   }
219
220   /**
221    * Returns this header as an {@link EntityTag}.
222    *
223    * @return This header as an {@link EntityTag}.
224    */
225   public Optional<EntityTag> asEntityTag() {
226      if (supplier != null) {
227         Object o = supplier.get();
228         return optional(o instanceof EntityTag ? (EntityTag)o : null);
229      }
230      return optional(value);
231   }
232
233   private static boolean isEtag(String s) {
234      return s.startsWith("\"") || s.startsWith("W/");
235   }
