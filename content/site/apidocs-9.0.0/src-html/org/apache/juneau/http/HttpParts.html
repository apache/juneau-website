422   private static final Function<ClassMeta<?>,String> HEADER_NAME_FUNCTION = x -> {
423      Value<String> n = Value.empty();
424      x.forEachAnnotation(org.apache.juneau.http.annotation.Header.class, y -> isNotEmpty(y.value()), y -> n.set(y.value()));
425      x.forEachAnnotation(org.apache.juneau.http.annotation.Header.class, y -> isNotEmpty(y.name()), y -> n.set(y.name()));
426      return n.orElse(null);
427   };

450   private static final Function<ClassMeta<?>,ConstructorInfo> CONSTRUCTOR_FUNCTION = x -> {
451      ClassInfo ci = x.getInfo();
452      ConstructorInfo cc = ci.getPublicConstructor(y -> y.hasParamTypes(String.class));
453      if (cc == null)
454         cc = ci.getPublicConstructor(y -> y.hasParamTypes(String.class, String.class));
455      return cc;
456   };

458   /**
459    * Returns the name of the specified part type.
460    *
461    * <p>
462    * Gets the name from one of the following annotations:
463    * <ul class='javatreec'>
464    *    <li class='ja'>{@link org.apache.juneau.http.annotation.Header}
465    *    <li class='ja'>{@link org.apache.juneau.http.annotation.Query}
466    *    <li class='ja'>{@link org.apache.juneau.http.annotation.FormData}
467    *    <li class='ja'>{@link org.apache.juneau.http.annotation.Path}
468    * </ul>
469    *
470    * @param partType The part type.
471    * @param type The type to check.
472    * @return The part name.  Never <jk>null</jk>.
473    */
474   public static Optional<String> getName(HttpPartType partType, ClassMeta<?> type) {
475      switch(partType) {
476         case FORMDATA: return type.getProperty("HttpPart.formData.name", FORMDATA_NAME_FUNCTION);
477         case HEADER: return type.getProperty("HttpPart.header.name", HEADER_NAME_FUNCTION);
478         case PATH: return type.getProperty("HttpPart.path.name", PATH_NAME_FUNCTION);
479         case QUERY: return type.getProperty("HttpPart.query.name", QUERY_NAME_FUNCTION);
480         default: return empty();
481      }
482   }

484   /**
485    * Returns <jk>true</jk> if the specified type is a part type.
486    *
487    * <p>
488    * A part type extends from either {@link org.apache.http.Header} or {@link org.apache.http.NameValuePair}
489    * or is annotated with {@link org.apache.juneau.http.annotation.Header}, {@link org.apache.juneau.http.annotation.Query},
490    * {@link org.apache.juneau.http.annotation.FormData}, or {@link org.apache.juneau.http.annotation.Path}.
491    *
492    * @param partType The part type.
493    * @param type The type to check.
494    * @return <jk>true</jk> if the specified type is a part type.
495    */
496   public static boolean isHttpPart(HttpPartType partType, ClassMeta<?> type) {
497      switch(partType) {
498         case PATH:
499         case QUERY:
500         case FORMDATA: return type.getProperty("HttpPart.isNameValuePair", x->x.isChildOf(NameValuePair.class)).orElse(false);
501         case HEADER: return type.getProperty("HttpPart.isHeader", x->x.isChildOf(org.apache.http.Header.class)).orElse(false);
502         default: return false;
503      }
504   }

523   /**
524    * Utility method for converting an arbitrary object to a {@link NameValuePair}.
525    *
526    * @param o
527    *    The object to cast or convert to a {@link NameValuePair}.
528    * @return Either the same object cast as a {@link NameValuePair} or converted to a {@link NameValuePair}.
529    */
530   @SuppressWarnings("rawtypes")
531   public static NameValuePair cast(Object o) {
532      if (o instanceof NameValuePair)
533         return (NameValuePair)o;
534      if (o instanceof Headerable) {
535         Header x = ((Headerable)o).asHeader();
536         return BasicPart.of(x.getName(), x.getValue());
537      }
538      if (o instanceof Map.Entry) {
539         Map.Entry e = (Map.Entry)o;
540         return BasicPart.of(stringify(e.getKey()), e.getValue());
541      }
542      throw new BasicRuntimeException("Object of type {0} could not be converted to a Part.", o == null ? null : o.getClass().getName());
543   }

545   /**
546    * Returns <jk>true</jk> if the {@link #cast(Object)} method can be used on the specified object.
547    *
548    * @param o The object to check.
549    * @return <jk>true</jk> if the {@link #cast(Object)} method can be used on the specified object.
550    */
551   public static boolean canCast(Object o) {
552      ClassInfo ci = ClassInfo.of(o);
553      return ci != null && ci.isChildOfAny(Headerable.class, NameValuePair.class, NameValuePairable.class, Map.Entry.class);
554   }
