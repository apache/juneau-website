013package org.apache.juneau.http.part;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.lang.reflect.*;
019import java.util.concurrent.*;
020
021import org.apache.juneau.http.annotation.*;
022import org.apache.juneau.httppart.*;
023import org.apache.juneau.reflect.*;
024
025/**
026 * Holds metadata about http part beans (POJOs that get serialized as HTTP parts such as form data or query parameters).
027 *
028 * <p>
029 * HTTP part beans are typically annotated with {@link Query @Query}, {@link FormData @FormData}, or {@link Path @Path}
030 * although it's not an absolute requirement.
031 *
032 * <p>
033 * HTTP part beans must have one of the following public constructors:
034 * <ul>
035 *    <li><c><jk>public</jk> X(String <jv>partValue</jv>)</c>
036 *    <li><c><jk>public</jk> X(Object <jv>partValue</jv>)</c>
037 *    <li><c><jk>public</jk> X(String <jv>partName</jv>, String <jv>partValue</jv>)</c>
038 *    <li><c><jk>public</jk> X(String <jv>partName</jv>, Object <jv>partValue</jv>)</c>
039 * </ul>
040 *
041 * <p>
042 * <h5 class='figure'>Example</h5>
043 * <p class='bjava'>
044 *    <jc>// Our part bean.</jc>
045 *    <ja>@Query</ja>(<js>"foo"</js>)
046 *    <jk>public class</jk> FooPart <jk>extends</jk> BasicStringPart {
047 *
048 *       <jk>public</jk> FooPart(String <jv>partValue</jv>) {
049 *          <jk>super</jk>(<js>"foo"</js>, <jv>partValue</jv>);
050 *       }
051 *  }
052 *
053 *  <jc>// Code to retrieve a part bean from a part list in a request.</jc>
054 *  PartList <jv>parts</jv> = <jv>httpRequest</jv>.getFormDataList();
055 *  FooPart <jv>foo</jv> = <jv>parts</jv>.get(FooPart.<jk>class</jk>);
056 * </p>
057 *
058 * <ul class='seealso'>
059 *    <li class='link'>{@doc juneau-rest-common}
060 *    <li class='extlink'>{@source}
061 * </ul>
062 *
063 * @param <T> The HTTP part bean type.
064 */
065public class PartBeanMeta<T> {
066
067   private static final ConcurrentHashMap<Class<?>,PartBeanMeta<?>> CACHE = new ConcurrentHashMap<>();
068
069   private final Class<T> type;
070   private final Constructor<T> constructor;
071   private final HttpPartSchema schema;
072
073   /**
074    * Finds the part bean meta for the specified type.
075    *
076    * @param <T> The part bean type.
077    * @param type The part bean type.
078    * @return The metadata, or <jk>null</jk> if a valid constructor could not be found.
079    */
080   @SuppressWarnings("unchecked")
081   public static <T> PartBeanMeta<T> of(Class<T> type) {
082      PartBeanMeta<?> m = CACHE.get(type);
083      if (m == null) {
084         m = new PartBeanMeta<>(type);
085         CACHE.put(type, m);
086      }
087      return (PartBeanMeta<T>)m;
088   }
089
090   private PartBeanMeta(Class<T> type) {
091      this.type = type;
092
093      ClassInfo ci = ClassInfo.of(type);
094
095      ConstructorInfo cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class));
096      if (cci == null)
097         cci = ci.getPublicConstructor(x -> x.hasParamTypes(Object.class));
098      if (cci == null)
099         cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, String.class));
100      if (cci == null)
101         cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, Object.class));
102      constructor = cci == null ? null : cci.inner();
103
104      if (ci.hasAnnotation(Query.class))
105         this.schema = HttpPartSchema.create(Query.class, type);
106      else if (ci.hasAnnotation(FormData.class))
107         this.schema = HttpPartSchema.create(FormData.class, type);
108      else if (ci.hasAnnotation(Path.class))
109         this.schema = HttpPartSchema.create(Path.class, type);
110      else
111         this.schema = HttpPartSchema.create(org.apache.juneau.http.annotation.Header.class, type);
112   }
113
114   /**
115    * Returns schema information about this part.
116    *
117    * <p>
118    * This is information pulled from {@link Query @Query}, {@link FormData @FormData}, or {@link Path @Path} annotations
119    * on the class.
120    *
121    * @return The schema information.
122    */
123   public HttpPartSchema getSchema() {
124      return schema;
125   }
126
127   /**
128    * Constructs a part bean with the specified name or value.
129    *
130    * <p>
131    * Can only be used on beans where the part name is known.
132    *
133    * @param value
134    *    The part value.
135    * @return A newly constructed bean.
136    */
137   public T construct(Object value) {
138      return construct(null, value);
139   }
140
141   /**
142    * Constructs a part bean with the specified name or value.
143    *
144    * @param name
145    *    The part name.
146    *    <br>If <jk>null</jk>, uses the value pulled from the {@link org.apache.juneau.http.annotation.Header#name() @Header(name)} or
147    *    {@link org.apache.juneau.http.annotation.Header#value() @Header(value)} annotations.
148    * @param value
149    *    The part value.
150    * @return A newly constructed bean.
151    * @throws UnsupportedOperationException If bean could not be constructed (e.g. couldn't find a constructor).
152    */
153   public T construct(String name, Object value) {
154
155      if (constructor == null)
156         throw new UnsupportedOperationException("Constructor for type "+type.getName()+" could not be found.");
157
158      if (name == null)
159         name = schema.getName();
160
161      Class<?>[] pt = constructor.getParameterTypes();
162      Object[] args = new Object[pt.length];
163      if (pt.length == 1) {
164         args[0] = pt[0] == String.class ? stringify(value) : value;
165      } else {
166         if (name == null)
167            throw new UnsupportedOperationException("Constructor for type "+type.getName()+" requires a name as the first argument.");
168         args[0] = name;
169         args[1] = pt[1] == String.class ? stringify(value) : value;
170      }
171
172      try {
173         return constructor.newInstance(args);
174      } catch (Exception e) {
175         throw asRuntimeException(e);
176      }
177   }
178}
