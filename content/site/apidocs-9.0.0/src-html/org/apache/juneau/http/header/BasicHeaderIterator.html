013package org.apache.juneau.http.header;
014
015import static org.apache.juneau.internal.ArgUtils.*;
016import java.util.NoSuchElementException;
017
018import org.apache.http.Header;
019import org.apache.http.HeaderIterator;
020import org.apache.juneau.internal.*;
021
022/**
023 * Basic implementation of a {@link HeaderIterator}.
024 *
025 * <ul class='notes'>
026 *    <li class='warn'>This class is not thread safe.
027 * </ul>
028 *
029 * <ul class='seealso'>
030 *    <li class='link'>{@doc juneau-rest-common}
031 *    <li class='extlink'>{@source}
032 * </ul>
033 */
034public class BasicHeaderIterator implements HeaderIterator {
035
036   private final Header[] entries;
037   private final String name;
038   private final boolean caseSensitive;
039
040   private int currentIndex;
041
042   /**
043    * Creates a new header iterator.
044    *
045    * @param headers An array of headers over which to iterate.
046    * @param name The name of the headers over which to iterate, or <jk>null</jk> for all.
047    * @param caseSensitive Use case-sensitive matching for part name.
048    */
049   public BasicHeaderIterator(Header[] headers, String name, boolean caseSensitive) {
050      this.entries = assertArgNotNull("headers", headers);
051      this.name = name;
052      this.caseSensitive = caseSensitive;
053      this.currentIndex = findNext(-1);
054   }
055
056   private int findNext(int pos) {
057
058      int from = pos;
059
060      int to = entries.length - 1;
061      boolean found = false;
062      while (!found && (from < to)) {
063         from++;
064         found = filter(from);
065      }
066
067      return found ? from : -1;
068   }
069
070   private boolean filter(int index) {
071      return (name == null) || eq(name, entries[index].getName());
072   }
073
074   @Override /* HeaderIterator */
075   public boolean hasNext() {
076      return (currentIndex >= 0);
077   }
078
079   @Override /* HeaderIterator */
080   public Header nextHeader() throws NoSuchElementException {
081
082      int current = currentIndex;
083
084      if (current < 0)
085         throw new NoSuchElementException("Iteration already finished.");
086
087      currentIndex = findNext(current);
088
089      return entries[current];
090   }
091
092   @Override /* HeaderIterator */
093   public final Object next() throws NoSuchElementException {
094      return nextHeader();
095   }
096
097   /**
098    * Not supported.
099    */
100   @Override /* HeaderIterator */
101   public void remove() {
102      throw new UnsupportedOperationException("Not supported.");
103   }
104
105   private boolean eq(String s1, String s2) {
106      return StringUtils.eq(!caseSensitive, s1, s2);
107   }
108}
