015import java.io.*;

017
018import org.apache.juneau.http.entity.*;

023/**
024 * Standard predefined HTTP entities.
025 *
026 * <ul class='seealso'>
027 *    <li class='link'>{@doc juneau-rest-common}
028 *    <li class='extlink'>{@source}
029 * </ul>
030 */
031public class HttpEntities {
032
033   /**
034    * Creates a new {@link ByteArrayEntity} builder.
035    *
036    * <p>
037    * Assumes no content type.
038    *
039    * @param content The entity content.  Can be <jk>null</jk>.
040    * @return A new {@link ByteArrayEntity} builder.
041    */
042   public static final HttpEntityBuilder<ByteArrayEntity> byteArrayEntity(byte[] content) {
043      return ByteArrayEntity.create().content(content);
044   }
045
046   /**
047    * Creates a new {@link ByteArrayEntity} builder.
048    *
049    * @param content The entity content.  Can be <jk>null</jk>.
050    * @param contentType The entity content type, or <jk>null</jk> if not specified.
051    * @return A new {@link ByteArrayEntity} builder.
052    */
053   public static final HttpEntityBuilder<ByteArrayEntity> byteArrayEntity(byte[] content, ContentType contentType) {
054      return ByteArrayEntity.create().content(content).contentType(contentType);
055   }
056
057   /**
058    * Creates a new {@link ByteArrayEntity} builder.
059    *

061    * Assumes no content type.
062    *
063    * @param content The entity content supplier.  Can be <jk>null</jk>.
064    * @return A new {@link ByteArrayEntity} builder.
065    */
066   public static final HttpEntityBuilder<ByteArrayEntity> byteArrayEntity(Supplier<byte[]> content) {
067      return ByteArrayEntity.create().content(content);
068   }

077   public static final HttpEntityBuilder<ByteArrayEntity> byteArrayEntity(Supplier<byte[]> content, ContentType contentType) {

126   /**
127    * Creates a new {@link SerializedEntity} object.
128    *
129    * @param content
130    *    The Java POJO representing the content.
131    *    <br>Can be <jk>null</jk>.
132    * @param serializer
133    *    The serializer to use to serialize the POJO.
134    *    <br>If <jk>null</jk>, POJO will be converted to a string using {@link Object#toString()}.
135    * @return A new {@link SerializedEntity} object.
136    */
137   public static final SerializedEntityBuilder<SerializedEntity> serializedEntity(Object content, Serializer serializer) {
138      return SerializedEntity.create().content(content).serializer(serializer);
139   }
140
141   /**
142    * Creates a new {@link SerializedEntity} object.
143    *
144    * @param content
145    *    The supplier of a Java POJO representing the content.
146    *    <br>Can be <jk>null</jk>.
147    * @param serializer
148    *    The serializer to use to serialize the POJO.
149    *    <br>If <jk>null</jk>, POJO will be converted to a string using {@link Object#toString()}.
150    * @return A new {@link SerializedEntity} object.
151    */
152   public static final SerializedEntityBuilder<SerializedEntity> serializedEntity(Supplier<?> content, Serializer serializer) {
153      return SerializedEntity.create().content(content).serializer(serializer);
154   }
155
156   /**
157    * Creates a new {@link SerializedEntity} object.
158    *
159    * @param content
160    *    The Java POJO representing the content.
161    *    <br>Can be <jk>null</jk>.
162    * @param serializer
163    *    The serializer to use to serialize the POJO.
164    *    <br>If <jk>null</jk>, POJO will be converted to a string using {@link Object#toString()}.
165    * @param schema
166    *    Optional HTTP-part schema for providing instructionst to the serializer on the format of the entity.
167    * @return A new {@link SerializedEntity} object.
168    */
169   public static final SerializedEntityBuilder<SerializedEntity> serializedEntity(Object content, Serializer serializer, HttpPartSchema   schema) {
170      return SerializedEntity.create().content(content).serializer(serializer).schema(schema);
171   }

179    * @param serializer

181    *    <br>If <jk>null</jk>, POJO will be converted to a string using {@link Object#toString()}.

184    * @return A new {@link SerializedEntity} object.
185    */
186   public static final SerializedEntityBuilder<SerializedEntity> serializedEntity(Supplier<?> content, Serializer serializer, HttpPartSchema schema) {
187      return SerializedEntity.create().content(content).serializer(serializer).schema(schema);
188   }
