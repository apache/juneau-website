013package org.apache.juneau.http.part;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.util.*;
019import java.util.function.*;
020
021import org.apache.http.*;
022import org.apache.juneau.assertions.*;
023
024/**
025 * A {@link NameValuePair} that consists of a single boolean value.
026 *
027 * <ul class='seealso'>
028 *    <li class='link'>{@doc juneau-rest-common}
029 *    <li class='extlink'>{@source}
030 * </ul>
031 */
032public class BasicBooleanPart extends BasicPart {
033
034   //-----------------------------------------------------------------------------------------------------------------
035   // Static
036   //-----------------------------------------------------------------------------------------------------------------
037
038   /**
039    * Static creator.
040    *
041    * @param name The part name.
042    * @param value The part value.
043    * @return A new {@link BasicBooleanPart} object, or <jk>null</jk> if the name or value is <jk>null</jk>.
044    */
045   public static BasicBooleanPart of(String name, Boolean value) {
046      if (isEmpty(name) || value == null)
047         return null;
048      return new BasicBooleanPart(name, value);
049   }
050
051   /**
052    * Static creator with delayed value.
053    *
054    * <p>
055    * Part value is re-evaluated on each call to {@link NameValuePair#getValue()}.
056    *
057    * @param name The part name.
058    * @param value The part value supplier.
059    * @return A new {@link BasicBooleanPart} object, or <jk>null</jk> if the name or supplier is <jk>null</jk>.
060    */
061   public static BasicBooleanPart of(String name, Supplier<Boolean> value) {
062      if (isEmpty(name) || value == null)
063         return null;
064      return new BasicBooleanPart(name, value);
065   }
066
067   //-----------------------------------------------------------------------------------------------------------------
068   // Instance
069   //-----------------------------------------------------------------------------------------------------------------
070
071   private final Boolean value;
072   private final Supplier<Boolean> supplier;
073
074   /**
075    * Constructor.
076    *
077    * @param name The part name.  Must not be <jk>null</jk>.
078    * @param value The part value.  Can be <jk>null</jk>.
079    */
080   public BasicBooleanPart(String name, Boolean value) {
081      super(name, value);
082      this.value = value;
083      this.supplier = null;
084   }
085
086   /**
087    * Constructor.
088    *
089    * @param name The part name.  Must not be <jk>null</jk>.
090    * @param value The part value supplier.  Can be <jk>null</jk> or supply <jk>null</jk>.
091    */
092   public BasicBooleanPart(String name, Supplier<Boolean> value) {
093      super(name, value);
094      this.value = null;
095      this.supplier = value;
096   }
097
098   /**
099    * Constructor.
100    *
101    * <p>
102    * <jk>null</jk> and empty values are treated as <jk>null</jk>.
103    * Otherwise parses using {@link Boolean#valueOf(String)}.
104    *
105    * @param name The part name.  Must not be <jk>null</jk>.
106    * @param value The part value.  Can be <jk>null</jk>.
107    */
108   public BasicBooleanPart(String name, String value) {
109      super(name, value);
110      this.value = isEmpty(value) ? null : Boolean.valueOf(value);
111      this.supplier = null;
112   }
113
114   @Override /* NameValuePair */
115   public String getValue() {
116      return stringify(value());
117   }
118
119   /**
120    * Returns The part value as a {@link Boolean} wrapped in an {@link Optional}.
121    *
122    * @return The part value as a {@link Boolean} wrapped in an {@link Optional}.  Never <jk>null</jk>.
123    */
124   public Optional<Boolean> asBoolean() {
125      return optional(toBoolean());
126   }
127
128   /**
129    * Returns The part value as a {@link Boolean}.
130    *
131    * @return The part value as a {@link Boolean}, or <jk>null</jk> if the value <jk>null</jk>.
132    */
133   public Boolean toBoolean() {
134      return value();
135   }
136
137   /**
138    * Provides the ability to perform fluent-style assertions on this part.
139    *
140    * @return A new fluent assertion object.
141    * @throws AssertionError If assertion failed.
142    */
143   public FluentBooleanAssertion<BasicBooleanPart> assertBoolean() {
144      return new FluentBooleanAssertion<>(value(), this);
145   }
146
147   /**
148    * Return the value if present, otherwise return <c>other</c>.
149    *
150    * <p>
151    * This is a shortened form for calling <c>asBoolean().orElse(<jv>other</jv>)</c>.
152    *
153    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
154    * @return The value, if present, otherwise <c>other</c>.
155    */
156   public Boolean orElse(Boolean other) {
157      Boolean x = value();
158      return x != null ? x : other;
159   }
160
161   private Boolean value() {
162      if (supplier != null)
163         return supplier.get();
164      return value;
165   }
166}
