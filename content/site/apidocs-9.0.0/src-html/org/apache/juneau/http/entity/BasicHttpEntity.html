013package org.apache.juneau.http.entity;
014
015import static org.apache.juneau.internal.IOUtils.*;
016
017import java.io.*;
018import java.nio.charset.*;
019import java.util.function.*;
020
021import org.apache.http.*;
022import org.apache.juneau.annotation.*;
023import org.apache.juneau.assertions.*;
024import org.apache.juneau.http.header.*;
025import org.apache.juneau.internal.*;
026
027/**
028 * A basic {@link org.apache.http.HttpEntity} implementation with additional features.
029 *
030 * Provides the following features:
031 * <ul class='spaced-list'>
032 *    <li>
033 *       Caching.
034 *    <li>
035 *       Fluent setters.
036 *    <li>
037 *       Fluent assertions.
038 *    <li>
039 *       Externally-supplied/dynamic content.
040 * </ul>
041 *
042 * <ul class='seealso'>
043 *    <li class='link'>{@doc juneau-rest-common}
044 *    <li class='extlink'>{@source}
045 * </ul>
046 */
047@BeanIgnore
048public class BasicHttpEntity implements HttpEntity {
049
050   /**
051    * An empty HttpEntity.
052    */
053   public static final BasicHttpEntity EMPTY = create(BasicHttpEntity.class).build();
054
055   final boolean cached, chunked;
056   final Object content;
057   final Supplier<?> contentSupplier;
058   final ContentType contentType;
059   final ContentEncoding contentEncoding;
060   final Charset charset;
061   final long contentLength;
062
063   /**
064    * Creates a builder for this class.
065    *
066    * @param <T> The subclass that the builder is going to create.
067    * @param implClass The subclass that the builder is going to create.
068    * @return A new builder bean.
069    */
070   public static <T extends BasicHttpEntity> HttpEntityBuilder<T> create(Class<T> implClass) {
071      return new HttpEntityBuilder<>(implClass);
072   }
073
074   /**
075    * Constructor.
076    *
077    * @param builder The builder containing the arguments for this bean.
078    */
079   public BasicHttpEntity(HttpEntityBuilder<?> builder) {
080      this.cached = builder.cached;
081      this.chunked = builder.chunked;
082      this.content = builder.content;
083      this.contentSupplier = builder.contentSupplier;
084      this.contentType = builder.contentType;
085      this.contentEncoding = builder.contentEncoding;
086      this.charset = builder.charset;
087      this.contentLength = builder.contentLength;
088   }
089
090   /**
091    * Creates a builder for this class initialized with the contents of this bean.
092    *
093    * <p>
094    * Allows you to create a modifiable copy of this bean.
095    *
096    * @return A new builder bean.
097    */
098   public HttpEntityBuilder<? extends BasicHttpEntity> copy() {
099      return new HttpEntityBuilder<>(this);
100   }
101
102   /**
103    * Converts the contents of this entity as a string.
104    *
105    * <p>
106    * Note that this may exhaust the content on non-repeatable, non-cached entities.
107    *
108    * @return The contents of this entity as a string.
109    * @throws IOException If a problem occurred while trying to read the content.
110    */
111   public String asString() throws IOException {
112      return read(getContent());
113   }
114
115   /**
116    * Converts the contents of this entity as a byte array.
117    *
118    * <p>
119    * Note that this may exhaust the content on non-repeatable, non-cached entities.
120    *
121    * @return The contents of this entity as a byte array.
122    * @throws IOException If a problem occurred while trying to read the content.
123    */
124   public byte[] asBytes() throws IOException {
125      return readBytes(getContent());
126   }
127
128   /**
129    * Returns an assertion on the contents of this entity.
130    *
131    * <p>
132    * Note that this may exhaust the content on non-repeatable, non-cached entities.
133    *
134    * @return A new fluent assertion.
135    * @throws IOException If a problem occurred while trying to read the byte array.
136    */
137   public FluentStringAssertion<BasicHttpEntity> assertString() throws IOException {
138      return new FluentStringAssertion<>(asString(), this);
139   }
140
141   /**
142    * Returns an assertion on the contents of this entity.
143    *
144    * <p>
145    * Note that this may exhaust the content on non-repeatable, non-cached entities.
146    *
147    * @return A new fluent assertion.
148    * @throws IOException If a problem occurred while trying to read the byte array.
149    */
150   public FluentByteArrayAssertion<BasicHttpEntity> assertBytes() throws IOException {
151      return new FluentByteArrayAssertion<>(asBytes(), this);
152   }
153
154   /**
155    * Returns the content of this entity.
156    *
157    * @param <T> The value type.
158    * @param def The default value if <jk>null</jk>.
159    * @return The content object.
160    */
161   @SuppressWarnings("unchecked")
162   protected <T> T contentOrElse(T def) {
163      Object o = content;
164      if (o == null && contentSupplier != null)
165         o = contentSupplier.get();
166      return (o == null ? def : (T)o);
167   }
168
169   /**
170    * Returns <jk>true</jk> if the contents of this entity are provided through an external supplier.
171    *
172    * <p>
173    * Externally supplied content generally means the content length cannot be reliably determined
174    * based on the content.
175    *
176    * @return <jk>true</jk> if the contents of this entity are provided through an external supplier.
177    */
178   protected boolean isSupplied() {
179      return contentSupplier != null;
180   }
181
182   @Override /* HttpEntity */
183   public long getContentLength() {
184      return contentLength;
185   }
186
187   @Override
188   public boolean isRepeatable() {
189      return false;
190   }
191
192   @Override
193   public boolean isChunked() {
194      return chunked;
195   }
196
197   @Override
198   public Header getContentType() {
199      return contentType;
200   }
201
202   @Override
203   public Header getContentEncoding() {
204      return contentEncoding;
205   }
206
207   @Override
208   public boolean isStreaming() {
209      return false;
210   }
211
212   @Override
213   public void consumeContent() throws IOException {}
214
215   @Override
216   public InputStream getContent() throws IOException, UnsupportedOperationException {
217      return IOUtils.EMPTY_INPUT_STREAM;
218   }
219
220   @Override
221   public void writeTo(OutputStream outStream) throws IOException {}
222
223   // <FluentSetters>
224
225   // </FluentSetters>
226}
