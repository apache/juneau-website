013package org.apache.juneau.http.header;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.lang.reflect.*;
019import java.util.concurrent.*;
020
021import org.apache.juneau.httppart.*;
022import org.apache.juneau.reflect.*;
023
024/**
025 * Holds metadata about header beans (POJOs that get serialized as HTTP headers).
026 *
027 * <p>
028 * Header beans are typically annotated with {@link org.apache.juneau.http.annotation.Header @Header} although it's not an
029 * absolute requirement.
030 *
031 * <p>
032 * Header beans must have one of the following public constructors:
033 * <ul>
034 *    <li><c><jk>public</jk> X(String <jv>headerValue</jv>)</c>
035 *    <li><c><jk>public</jk> X(Object <jv>headerValue</jv>)</c>
036 *    <li><c><jk>public</jk> X(String <jv>headerName</jv>, String <jv>headerValue</jv>)</c>
037 *    <li><c><jk>public</jk> X(String <jv>headerName</jv>, Object <jv>headerValue</jv>)</c>
038 * </ul>
039 *
040 * <p>
041 * <h5 class='figure'>Example</h5>
042 * <p class='bjava'>
043 *    <jc>// Our header bean.</jc>
044 *    <ja>@Header</ja>(<js>"Foo"</js>)
045 *    <jk>public class</jk> FooHeader <jk>extends</jk> BasicStringHeader {
046 *
047 *       <jk>public</jk> FooHeader(String <jv>headerValue</jv>) {
048 *          <jk>super</jk>(<js>"Foo"</js>, <jv>headerValue</jv>);
049 *       }
050 *  }
051 *
052 *  <jc>// Code to retrieve a header bean from a header list in a request.</jc>
053 *  HeaderList <jv>headers</jv> = <jv>httpRequest</jv>.getHeaders();
054 *  FooHeader <jv>foo</jv> = <jv>headers</jv>.get(FooHeader.<jk>class</jk>);
055 * </p>
056 *
057 * <ul class='seealso'>
058 *    <li class='link'>{@doc juneau-rest-common}
059 *    <li class='extlink'>{@source}
060 * </ul>
061 *
062 * @param <T> The header bean type.
063 */
064public class HeaderBeanMeta<T> {
065
066   private static final ConcurrentHashMap<Class<?>,HeaderBeanMeta<?>> CACHE = new ConcurrentHashMap<>();
067
068   private final Class<T> type;
069   private final Constructor<T> constructor;
070   private final HttpPartSchema schema;
071
072   /**
073    * Finds the header bean meta for the specified type.
074    *
075    * @param <T> The header bean type.
076    * @param type The header bean type.
077    * @return The metadata, or <jk>null</jk> if a valid constructor could not be found.
078    */
079   @SuppressWarnings("unchecked")
080   public static <T> HeaderBeanMeta<T> of(Class<T> type) {
081      HeaderBeanMeta<?> m = CACHE.get(type);
082      if (m == null) {
083         m = new HeaderBeanMeta<>(type);
084         CACHE.put(type, m);
085      }
086      return (HeaderBeanMeta<T>)m;
087   }
088
089   private HeaderBeanMeta(Class<T> type) {
090      this.type = type;
091
092      ClassInfo ci = ClassInfo.of(type);
093
094      ConstructorInfo cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class));
095      if (cci == null)
096         cci = ci.getPublicConstructor(x -> x.hasParamTypes(Object.class));
097      if (cci == null)
098         cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, String.class));
099      if (cci == null)
100         cci = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, Object.class));
101      constructor = cci == null ? null : cci.inner();
102
103      this.schema = HttpPartSchema.create(org.apache.juneau.http.annotation.Header.class, type);
104   }
105
106   /**
107    * Returns schema information about this header.
108    *
109    * <p>
110    * This is information pulled from {@link org.apache.juneau.http.annotation.Header @Header} annotation on the class.
111    *
112    * @return The schema information.
113    */
114   public HttpPartSchema getSchema() {
115      return schema;
116   }
117
118   /**
119    * Constructs a header bean with the specified name or value.
120    *
121    * <p>
122    * Can only be used on beans where the header name is known.
123    *
124    * @param value
125    *    The header value.
126    * @return A newly constructed bean.
127    */
128   public T construct(Object value) {
129      return construct(null, value);
130   }
131
132   /**
133    * Constructs a header bean with the specified name or value.
134    *
135    * @param name
136    *    The header name.
137    *    <br>If <jk>null</jk>, uses the value pulled from the {@link org.apache.juneau.http.annotation.Header#name() @Header(name)} or
138    *    {@link org.apache.juneau.http.annotation.Header#value() @Header(value)} annotations.
139    * @param value
140    *    The header value.
141    * @return A newly constructed bean.
142    * @throws UnsupportedOperationException If bean could not be constructed (e.g. couldn't find a constructor).
143    */
144   public T construct(String name, Object value) {
145
146      if (constructor == null)
147         throw unsupportedOperationException("Constructor for type {0} could not be found.", type.getName());
148
149      if (name == null)
150         name = schema.getName();
151
152      Class<?>[] pt = constructor.getParameterTypes();
153      Object[] args = new Object[pt.length];
154      if (pt.length == 1) {
155         args[0] = pt[0] == String.class ? stringify(value) : value;
156      } else {
157         if (name == null)
158            throw unsupportedOperationException("Constructor for type {0} requires a name as the first argument.", type.getName());
159         args[0] = name;
160         args[1] = pt[1] == String.class ? stringify(value) : value;
161      }
162
163      try {
164         return constructor.newInstance(args);
165      } catch (Exception e) {
166         throw runtimeException(e);
167      }
168   }
169}
