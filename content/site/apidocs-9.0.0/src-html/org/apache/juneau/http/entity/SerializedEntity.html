013package org.apache.juneau.http.entity;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016import static org.apache.juneau.internal.IOUtils.*;
017import static org.apache.juneau.http.HttpHeaders.*;
018
019import java.io.*;
020
021import org.apache.http.*;
022import org.apache.juneau.httppart.*;
023import org.apache.juneau.internal.*;
024import org.apache.juneau.serializer.*;
025
026/**
027 * HttpEntity for serializing POJOs as the body of HTTP requests.
028 *
029 * <ul class='seealso'>
030 *    <li class='link'>{@doc juneau-rest-common}
031 *    <li class='extlink'>{@source}
032 * </ul>
033 */
034public class SerializedEntity extends BasicHttpEntity {
035   final Serializer serializer;
036   HttpPartSchema schema;
037
038   /**
039    * Creates a new {@link SerializedEntity} builder.
040    *
041    * @return A new {@link SerializedEntity} builder.
042    */
043   public static SerializedEntityBuilder<SerializedEntity> create() {
044      return new SerializedEntityBuilder<>(SerializedEntity.class);
045   }
046
047   /**
048    * Constructor.
049    *
050    * @param builder The builder for this bean.
051    */
052   public SerializedEntity(SerializedEntityBuilder<?> builder) {
053      super(builder);
054      serializer = builder.serializer;
055      schema = builder.schema;
056   }
057
058   /**
059    * Creates a copy of this object.
060    *
061    * @return A new copy of this object.
062    */
063   @Override
064   public SerializedEntityBuilder<SerializedEntity> copy() {
065      return new SerializedEntityBuilder<>(this);
066   }
067
068   /**
069    * Copies this bean and sets the serializer and schema on it.
070    *
071    * @param serializer The new serializer for the bean.  Can be <jk>null</jk>.
072    * @param schema The new schema for the bean.  Can be <jk>null</jk>.
073    * @return Either a new bean with the serializer set, or this bean if
074    *    both values are <jk>null</jk> or the serializer and schema were already set.
075    */
076   public SerializedEntity copyWith(Serializer serializer, HttpPartSchema schema) {
077      if ((this.serializer == null && serializer != null) || (this.schema == null && schema != null)) {
078         SerializedEntityBuilder<SerializedEntity> h = copy();
079         if (serializer != null)
080            h.serializer(serializer);
081         if (schema != null)
082            h.schema(schema);
083         return h.build();
084      }
085      return this;
086   }
087
088   @Override
089   public Header getContentType() {
090      Header x = super.getContentType();
091      if (x == null && serializer != null)
092         x = contentType(serializer.getPrimaryMediaType());
093      return x;
094   }
095
096   @Override /* HttpEntity */
097   public void writeTo(OutputStream os) throws IOException {
098      try {
099         os = new NoCloseOutputStream(os);
100         Object o = contentOrElse(null);
101         if (serializer == null) {
102            try (Writer w = new OutputStreamWriter(os, UTF8)) {
103               w.write(o.toString());
104            }
105         } else {
106            SerializerSession session = serializer.createSession().schema(schema).build();
107            try (Closeable c = session.isWriterSerializer() ? new OutputStreamWriter(os, UTF8) : os) {
108               session.serialize(o, c);
109            }
110         }
111      } catch (SerializeException e) {
112         throw runtimeException(e, "Serialization error on request body.");
113      }
114   }
115
116   @Override /* BasicHttpEntity */
117   public boolean isRepeatable() {
118      return true;
119   }
120
121   @Override /* BasicHttpEntity */
122   public long getContentLength() {
123      return -1;
124   }
125
126   @Override /* BasicHttpEntity */
127   public InputStream getContent() {
128      ByteArrayOutputStream baos = new ByteArrayOutputStream();
129      try {
130         writeTo(baos);
131         return new ByteArrayInputStream(baos.toByteArray());
132      } catch (IOException e) {
133         throw runtimeException(e);
134      }
135   }
136}
