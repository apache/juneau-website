025/**
026 * Category of headers that consist of a single parameterized string value.
027 *
028 * <p>
029 * <h5 class='figure'>Example</h5>
030 * <p class='bcode'>
031 *    Content-Type: application/json;charset=utf-8
032 * </p>
033 *
034 * <ul class='seealso'>
035 *    <li class='link'>{@doc juneau-rest-common}
036 *    <li class='extlink'>{@doc ext.RFC2616}
037 *    <li class='extlink'>{@source}
038 * </ul>
039 *
040 * @serial exclude
041 */
042public class BasicMediaTypeHeader extends BasicStringHeader {

158   /**
159    * Given a list of media types, returns the best match for this <c>Content-Type</c> header.
160    *
161    * <p>
162    * Note that fuzzy matching is allowed on the media types where the <c>Content-Types</c> header may
163    * contain additional subtype parts.
164    * <br>For example, given a <c>Content-Type</c> value of <js>"text/json+activity"</js>,
165    * the media type <js>"text/json"</js> will match if <js>"text/json+activity"</js> or <js>"text/activity+json"</js>
166    * isn't found.
167    * <br>The purpose for this is to allow parsers to match when artifacts such as <c>id</c> properties are
168    * present in the header.
169    *
170    * @param mediaTypes The media types to match against.
171    * @return The index into the array of the best match, or <c>-1</c> if no suitable matches could be found.
172    */
173   public int match(List<MediaType> mediaTypes) {
174      int matchQuant = 0, matchIndex = -1;
175
176      for (int i = 0; i < mediaTypes.size(); i++) {
177         MediaType mt = mediaTypes.get(i);
178         int matchQuant2 = mt.match(orElse(MediaType.EMPTY), true);
179         if (matchQuant2 > matchQuant) {
180            matchQuant = matchQuant2;
181            matchIndex = i;
182         }
183      }
184      return matchIndex;
185   }

239   /**
240    * Returns a match metric against the specified media type where a larger number represents a better match.
241    *
242    * <p>
243    * This media type can contain <js>'*'</js> metacharacters.
244    * <br>The comparison media type must not.
245    *
246    * <ul>
247    *    <li>Exact matches (e.g. <js>"text/json"</js>/</js>"text/json"</js>) should match
248    *       better than meta-character matches (e.g. <js>"text/*"</js>/</js>"text/json"</js>)
249    *    <li>The comparison media type can have additional subtype tokens (e.g. <js>"text/json+foo"</js>)
250    *       that will not prevent a match if the <c>allowExtraSubTypes</c> flag is set.
251    *       The reverse is not true, e.g. the comparison media type must contain all subtype tokens found in the
252    *       comparing media type.
253    *       <ul>
254    *          <li>We want the {@link JsonSerializer} (<js>"text/json"</js>) class to be able to handle requests for <js>"text/json+foo"</js>.
255    *          <li>We want to make sure {@link org.apache.juneau.json.SimpleJsonSerializer} (<js>"text/json+simple"</js>) does not handle
256    *             requests for <js>"text/json"</js>.
257    *       </ul>
258    *       More token matches should result in a higher match number.
259    * </ul>
260    *
261    * The formula is as follows for <c>type/subTypes</c>:
262    * <ul>
263    *    <li>An exact match is <c>100,000</c>.
264    *    <li>Add the following for type (assuming subtype match is &lt;0):
265    *    <ul>
266    *       <li><c>10,000</c> for an exact match (e.g. <js>"text"</js>==<js>"text"</js>).
267    *       <li><c>5,000</c> for a meta match (e.g. <js>"*"</js>==<js>"text"</js>).
268    *    </ul>
269    *    <li>Add the following for subtype (assuming type match is &lt;0):
270    *    <ul>
271    *       <li><c>7,500</c> for an exact match (e.g. <js>"json+foo"</js>==<js>"json+foo"</js> or <js>"json+foo"</js>==<js>"foo+json"</js>)
272    *       <li><c>100</c> for every subtype entry match (e.g. <js>"json"</js>/<js>"json+foo"</js>)
273    *    </ul>
274    * </ul>
275    *
276    * @param o The media type to compare with.
277    * @param allowExtraSubTypes If <jk>true</jk>,
278    * @return <jk>true</jk> if the media types match.
279    */
280   public final int match(MediaType o, boolean allowExtraSubTypes) {
281      return orElse(MediaType.EMPTY).match(o, allowExtraSubTypes);
282   }

285    * Returns the additional parameters on this media type.

297   /**
298    * Returns a parameterized value of the header.
299    *
300    * <p class='bjava'>
301    *    ContentType <jv>contentType</jv> = ContentType.<jsm>of</jsm>(<js>"application/json;charset=foo"</js>);
302    *    <jsm>assertEquals</jsm>(<js>"foo"</js>, <jv>contentType</jv>.getParameter(<js>"charset"</js>);
303    * </p>
304    *
305    * @param name The header name.
306    * @return The header value, or <jk>null</jk> if the parameter is not present.
307    */
308   public String getParameter(String name) {
309      return orElse(MediaType.EMPTY).getParameter(name);
310   }
311
312   /**
313    * Return the value if present, otherwise return <c>other</c>.
314    *
315    * <p>
316    * This is a shortened form for calling <c>asMediaType().orElse(<jv>other</jv>)</c>.
317    *
318    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
319    * @return The value, if present, otherwise <c>other</c>.
320    */
321   public MediaType orElse(MediaType other) {
322      MediaType x = value();
323      return x != null ? x : other;
324   }
325
326   private MediaType parse(String value) {
327      // If this happens to be a multi-value, use the last value.
328      if (value != null) {
329         int i = value.indexOf(',');
330         if (i != -1)
331            value = value.substring(i+1);
332      }
333      return MediaType.of(value);
334   }
335
336   private MediaType value() {
337      if (supplier != null)
338         return supplier.get();
339      return value;
340   }
