013package org.apache.juneau.http.part;
014
015import static java.time.format.DateTimeFormatter.*;
016import static java.time.temporal.ChronoUnit.*;
017import static org.apache.juneau.internal.CollectionUtils.*;
018import static org.apache.juneau.internal.StringUtils.*;
019
020import java.time.*;
021import java.time.format.*;
022import java.util.*;
023import java.util.function.*;
024
025import org.apache.http.*;
026import org.apache.juneau.assertions.*;
027
028/**
029 * A {@link NameValuePair} that consist of a single HTTP-date.
030 *
031 * <ul class='seealso'>
032 *    <li class='link'>{@doc juneau-rest-common}
033 *    <li class='extlink'>{@source}
034 * </ul>
035 */
036public class BasicDatePart extends BasicPart {
037
038   //-----------------------------------------------------------------------------------------------------------------
039   // Static
040   //-----------------------------------------------------------------------------------------------------------------
041
042   /**
043    * Static creator.
044    *
045    * @param name The part name.
046    * @param value The part value.
047    * @return A new {@link BasicDatePart} object, or <jk>null</jk> if the name or value is <jk>null</jk>.
048    */
049   public static BasicDatePart of(String name, ZonedDateTime value) {
050      if (isEmpty(name) || value == null)
051         return null;
052      return new BasicDatePart(name, value);
053   }
054
055   /**
056    * Static creator with delayed value.
057    *
058    * <p>
059    * Part value is re-evaluated on each call to {@link NameValuePair#getValue()}.
060    *
061    * @param name The part name.
062    * @param value The part value supplier.
063    * @return A new {@link BasicDatePart} object, or <jk>null</jk> if the name or supplier is <jk>null</jk>.
064    */
065   public static BasicDatePart of(String name, Supplier<ZonedDateTime> value) {
066      if (isEmpty(name) || value == null)
067         return null;
068      return new BasicDatePart(name, value);
069   }
070
071   //-----------------------------------------------------------------------------------------------------------------
072   // Instance
073   //-----------------------------------------------------------------------------------------------------------------
074
075   private final ZonedDateTime value;
076   private final Supplier<ZonedDateTime> supplier;
077
078   /**
079    * Constructor.
080    *
081    * @param name The part name.  Must not be <jk>null</jk>.
082    * @param value The part value.  Can be <jk>null</jk>.
083    */
084   public BasicDatePart(String name, ZonedDateTime value) {
085      super(name, value);
086      this.value = value;
087      this.supplier = null;
088   }
089
090   /**
091    * Constructor.
092    *
093    * @param name The part name.  Must not be <jk>null</jk>.
094    * @param value The part value supplier.  Can be <jk>null</jk> or supply <jk>null</jk>.
095    */
096   public BasicDatePart(String name, Supplier<ZonedDateTime> value) {
097      super(name, value);
098      this.value = null;
099      this.supplier = value;
100   }
101
102   /**
103    * Constructor.
104    *
105    * <p>
106    * <jk>null</jk> and empty values are treated as <jk>null</jk>.
107    * Otherwise parses using {@link DateTimeFormatter#ISO_DATE_TIME}.
108    *
109    * @param name The part name.  Must not be <jk>null</jk>.
110    * @param value The part value.  Can be <jk>null</jk>.
111    */
112   public BasicDatePart(String name, String value) {
113      super(name, value);
114      this.value = isEmpty(value) ? null : ZonedDateTime.from(ISO_DATE_TIME.parse(value)).truncatedTo(SECONDS);
115      this.supplier = null;
116   }
117
118   @Override /* Header */
119   public String getValue() {
120      ZonedDateTime v = value();
121      return v == null ? null : ISO_DATE_TIME.format(v);
122   }
123
124   /**
125    * Returns The part value as a {@link ZonedDateTime} wrapped in an {@link Optional}.
126    *
127    * @return The part value as a {@link ZonedDateTime} wrapped in an {@link Optional}.  Never <jk>null</jk>.
128    */
129   public Optional<ZonedDateTime> asZonedDateTime() {
130      return optional(toZonedDateTime());
131   }
132
133   /**
134    * Returns The part value as a {@link ZonedDateTime}.
135    *
136    * @return The part value as a {@link ZonedDateTime}, or <jk>null</jk> if the value <jk>null</jk>.
137    */
138   public ZonedDateTime toZonedDateTime() {
139      return value();
140   }
141
142   /**
143    * Provides the ability to perform fluent-style assertions on this part.
144    *
145    * @return A new fluent assertion object.
146    * @throws AssertionError If assertion failed.
147    */
148   public FluentZonedDateTimeAssertion<BasicDatePart> assertZonedDateTime() {
149      return new FluentZonedDateTimeAssertion<>(value(), this);
150   }
151
152   /**
153    * Return the value if present, otherwise return <c>other</c>.
154    *
155    * <p>
156    * This is a shortened form for calling <c>asZonedDateTime().orElse(<jv>other</jv>)</c>.
157    *
158    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
159    * @return The value, if present, otherwise <c>other</c>.
160    */
161   public ZonedDateTime orElse(ZonedDateTime other) {
162      ZonedDateTime x = value();
163      return x != null ? x : other;
164   }
165
166   private ZonedDateTime value() {
167      if (supplier != null)
168         return supplier.get();
169      return value;
170   }
171}
