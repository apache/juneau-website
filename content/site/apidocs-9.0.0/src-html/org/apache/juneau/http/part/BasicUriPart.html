013package org.apache.juneau.http.part;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016import static org.apache.juneau.internal.StringUtils.*;
017
018import java.net.*;
019import java.util.*;
020import java.util.function.*;
021
022import org.apache.http.*;
023
024/**
025 * A {@link NameValuePair} that consists of a single URL value.
026 *
027 * <ul class='seealso'>
028 *    <li class='link'>{@doc juneau-rest-common}
029 *    <li class='extlink'>{@source}
030 * </ul>
031 */
032public class BasicUriPart extends BasicPart {
033
034   //-----------------------------------------------------------------------------------------------------------------
035   // Static
036   //-----------------------------------------------------------------------------------------------------------------
037
038   /**
039    * Static creator.
040    *
041    * @param name The part name.
042    * @param value The part value.
043    * @return A new {@link BasicUriPart} object, or <jk>null</jk> if the name or value is <jk>null</jk>.
044    */
045   public static BasicUriPart of(String name, URI value) {
046      if (isEmpty(name) || value == null)
047         return null;
048      return new BasicUriPart(name, value);
049   }
050
051   /**
052    * Static creator with delayed value.
053    *
054    * <p>
055    * Part value is re-evaluated on each call to {@link NameValuePair#getValue()}.
056    *
057    * @param name The part name.
058    * @param value The part value supplier.
059    * @return A new {@link BasicUriPart} object, or <jk>null</jk> if the name or supplier is <jk>null</jk>.
060    */
061   public static BasicUriPart of(String name, Supplier<URI> value) {
062      if (isEmpty(name) || value == null)
063         return null;
064      return new BasicUriPart(name, value);
065   }
066
067   //-----------------------------------------------------------------------------------------------------------------
068   // Instance
069   //-----------------------------------------------------------------------------------------------------------------
070
071   private final URI value;
072   private final Supplier<URI> supplier;
073
074   /**
075    * Constructor.
076    *
077    * @param name The part name.  Must not be <jk>null</jk>.
078    * @param value The part value.  Can be <jk>null</jk>.
079    */
080   public BasicUriPart(String name, URI value) {
081      super(name, value);
082      this.value = value;
083      this.supplier = null;
084   }
085
086   /**
087    * Constructor.
088    *
089    * @param name The part name.  Must not be <jk>null</jk>.
090    * @param value The part value supplier.  Can be <jk>null</jk> or supply <jk>null</jk>.
091    */
092   public BasicUriPart(String name, Supplier<URI> value) {
093      super(name, value);
094      this.value = null;
095      this.supplier = value;
096   }
097
098   /**
099    * Constructor.
100    *
101    * <p>
102    * <jk>null</jk> and empty values are treated as <jk>null</jk>.
103    * Otherwise parses using {@link URI#create(String)}.
104    *
105    * @param name The part name.  Must not be <jk>null</jk>.
106    * @param value The part value.  Can be <jk>null</jk>.
107    */
108   public BasicUriPart(String name, String value) {
109      super(name, value);
110      this.value = isEmpty(value) ? null : URI.create(value);
111      this.supplier = null;
112   }
113
114   /**
115    * Returns The part value as a {@link URI} wrapped in an {@link Optional}.
116    *
117    * @return The part value as a {@link URI} wrapped in an {@link Optional}.  Never <jk>null</jk>.
118    */
119   public Optional<URI> asUri() {
120      return optional(value());
121   }
122
123   /**
124    * Returns The part value as a {@link URI}.
125    *
126    * @return The part value as a {@link URI}, or <jk>null</jk> if the value <jk>null</jk>.
127    */
128   public URI toUri() {
129      return value();
130   }
131
132   /**
133    * Return the value if present, otherwise return <c>other</c>.
134    *
135    * <p>
136    * This is a shortened form for calling <c>asString().orElse(<jv>other</jv>)</c>.
137    *
138    * @param other The value to be returned if there is no value present, can be <jk>null</jk>.
139    * @return The value, if present, otherwise <c>other</c>.
140    */
141   public URI orElse(URI other) {
142      URI x = value();
143      return x != null ? x : other;
144   }
145
146   private URI value() {
147      if (supplier != null)
148         return supplier.get();
149      return value;
150   }
151}
