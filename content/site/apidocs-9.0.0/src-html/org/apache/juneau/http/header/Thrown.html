013package org.apache.juneau.http.header;
014
015import static org.apache.juneau.internal.StringUtils.*;
016import static org.apache.juneau.internal.ClassUtils.*;
017import static org.apache.juneau.internal.CollectionUtils.*;
018import java.util.*;
019import java.util.stream.*;
020
021import org.apache.juneau.http.annotation.*;
022
023/**
024 * Represents a parsed <l>Thrown</l> HTTP response header.
025 *
026 * <p>
027 * Contains exception information including name and optionally a message.
028 *
029 * <h5 class='figure'>Example</h5>
030 * <p class='bcode'>
031 *    Thrown: org.apache.juneau.http.response.NotFound;Resource was not found
032 * </p>
033 *
034 * <p>
035 * This header isn't part of the RFC2616 specification, but is provided to allow for Java exception information
036 * to be delivered to remote REST Java interfaces.
037 *
038 * <p>
039 * This header supports comma-delimited values for multiple thrown values.
040 * <p class='bcode'>
041 *    Thrown: org.apache.juneau.http.response.NotFound;Resource was not found,java.lang.RuntimeException;foo
042 * </p>
043 *
044 * <p>
045 * Note that this is equivalent to specifying multiple header values.
046 * <p class='bcode'>
047 *    Thrown: org.apache.juneau.http.response.NotFound;Resource was not found
048 *    Thrown: java.lang.RuntimeException;foo
049 * </p>
050 *
051 * <ul class='seealso'>
052 *    <li class='link'>{@doc juneau-rest-common}
053 *    <li class='extlink'>{@source}
054 * </ul>
055 *
056 * @serial exclude
057 */
058@Header("Thrown")
059public class Thrown extends BasicCsvHeader {
060
061   //-----------------------------------------------------------------------------------------------------------------
062   // Static
063   //-----------------------------------------------------------------------------------------------------------------
064
065   private static final long serialVersionUID = 1L;
066   private static final String NAME = "Thrown";
067
068   /**
069    * An empty unmodifiable Thrown header.
070    */
071   public static final Thrown EMPTY = new Thrown((String)null);
072
073   /**
074    * Static creator.
075    *
076    * @param value
077    *    The header value.
078    *    <br>Can be <jk>null</jk>.
079    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
080    */
081   public static Thrown of(String value) {
082      return value == null ? null : new Thrown(value);
083   }
084
085   /**
086    * Static creator.
087    *
088    * @param values
089    *    The header value.
090    *    <br>Can be <jk>null</jk>.
091    * @return A new header bean, or <jk>null</jk> if the value is <jk>null</jk>.
092    */
093   public static Thrown of(Throwable...values) {
094      return new Thrown(alist(values).stream().map(Part::new).collect(Collectors.toList()));
095   }
096
097   //-----------------------------------------------------------------------------------------------------------------
098   // Instance
099   //-----------------------------------------------------------------------------------------------------------------
100
101   private final List<Part> value;
102
103   /**
104    * Constructor.
105    *
106    * @param value
107    *    The header value.
108    */
109   public Thrown(String value) {
110      super(NAME, value);
111      List<Part> l = list();
112      split(value, x -> l.add(new Part(x)));
113      this.value = value == null ? null : unmodifiable(l);
114   }
115
116   /**
117    * Constructor.
118    *
119    * @param value
120    *    The header value.
121    */
122   public Thrown(List<Part> value) {
123      super(NAME, join(value, ", "));
124      this.value = unmodifiable(value);
125   }
126
127   /**
128    * Returns the class name portion of the header.
129    *
130    * @return The class name portion of the header, or <jk>null</jk> if not there.
131    */
132   public Optional<List<Part>> asParts() {
133      return optional(value);
134   }
135
136   /**
137    * Represents a single entry in this header.
138    */
139   public static class Part {
140
141      String className, message;
142      String value;
143
144      /**
145       * Constructor.
146       *
147       * @param value The header part value.
148       */
149      public Part(String value) {
150         this.value = value;
151         int i = value.indexOf(';');
152         this.className = urlDecode(i == -1 ? value.trim() : value.substring(0, i).trim());
153         this.message = urlDecode(i == -1 ? null : value.substring(i+1).trim());
154      }
155
156      /**
157       * Constructor.
158       *
159       * @param value The throwable to create the header part value from.
160       */
161      public Part(Throwable value) {
162         this.className = className(value);
163         this.message = value.getMessage();
164         this.value = urlEncode(className) + ';' + urlEncode(value.getMessage());
165      }
166
167      /**
168       * Returns the message portion of the header.
169       *
170       * @return The message portion of the header, or <jk>null</jk> if not there.
171       */
172      public String getClassName() {
173         return className;
174      }
175
176      /**
177       * Returns the message portion of the header.
178       *
179       * @return The message portion of the header, or <jk>null</jk> if not there.
180       */
181      public String getMessage() {
182         return message;
183      }
184
185      @Override /* Object */
186      public String toString() {
187         return value;
188      }
189   }
190}
