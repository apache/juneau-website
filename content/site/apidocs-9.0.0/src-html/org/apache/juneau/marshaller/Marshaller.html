015import java.io.*;
016import java.lang.reflect.*;
017import java.nio.charset.*;
018
019import org.apache.juneau.*;
020import org.apache.juneau.parser.*;
021import org.apache.juneau.parser.ParseException;
022import org.apache.juneau.serializer.*;
023
024/**
025 * Top-level class for a pairing of a {@link Serializer} and {@link Parser} into a single class with convenience read/write methods.
026 *
027 * <p>
028 *    The general idea is to combine a single serializer and parser inside a simplified API for reading and writing POJOs.
029 *
030 * <h5 class='figure'>Examples:</h5>
031 * <p class='bjava'>
032 *    <jc>// Using instance.</jc>
033 *    Marshaller <jv>json</jv> = <jk>new</jk> Json();
034 *    MyPojo <jv>myPojo</jv> = <jv>json</jv>.read(<jv>string</jv>, MyPojo.<jk>class</jk>);
035 *    String <jv>string</jv> = <jv>json</jv>.write(<jv>myPojo</jv>);
036 * </p>
037 * <p class='bjava'>
038 * <jc>// Using DEFAULT instance.</jc>
039 *    MyPojo <jv>myPojo</jv> = Json.<jsf>DEFAULT</jsf>.read(<jv>string</jv>, MyPojo.<jk>class</jk>);
040 *    String <jv>string</jv> = Json.<jsf>DEFAULT</jsf>.write(<jv>myPojo</jv>);
041 * </p>
042 *
043 * <ul class='seealso'>
044 *    <li class='link'>{@doc jm.Marshallers}
045 *    <li class='extlink'>{@source}
046 * </ul>
047 */
048public abstract class Marshaller {
049
050   private final Serializer s;
051   private final Parser p;
052
053   /**
054    * Constructor.
055    *
056    * @param s
057    *    The serializer to use for serializing output.
058    *    <br>Must not be <jk>null</jk>.
059    * @param p
060    *    The parser to use for parsing input.
061    *    <br>Must not be <jk>null</jk>.
062    */
063   protected Marshaller(Serializer s, Parser p) {
064      this.s = s;
065      this.p = p;
066   }
067
068   /**
069    * Returns the serializer associated with this marshaller.
070    *
071    * @return The serializer associated with this marshaller.
072    */
073   public Serializer getSerializer() {
074      return s;
075   }
076
077   /**
078    * Returns the parser associated with this marshaller.
079    *
080    * @return The parser associated with this marshaller.
081    */
082   public Parser getParser() {
083      return p;
084   }
085
086   /**
087    * Serializes a POJO to the specified output stream or writer.
088    *
089    * <p>
090    * Equivalent to calling <c>serializer.createSession().serialize(o, output);</c>
091    *
092    * @param object The object to serialize.
093    * @param output
094    *    The output object.
095    *    <br>Character-based serializers can handle the following output class types:
096    *    <ul>
097    *       <li>{@link Writer}
098    *       <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
099    *       <li>{@link File} - Output will be written as system-default encoded stream.
100    *       <li>{@link StringBuilder} - Output will be written to the specified string builder.
101    *    </ul>
102    *    <br>Stream-based serializers can handle the following output class types:
103    *    <ul>
104    *       <li>{@link OutputStream}
105    *       <li>{@link File}
106    *    </ul>
107    * @throws SerializeException If a problem occurred trying to convert the output.
108    * @throws IOException Thrown by underlying stream.
109    */
110   public final void write(Object object, Object output) throws SerializeException, IOException {
111      s.serialize(object, output);
112   }
113
114   /**
115    * Parses input into the specified object type.
116    *
117    * <p>
118    * The type can be a simple type (e.g. beans, strings, numbers) or parameterized type (collections/maps).
119    *
120    * <h5 class='section'>Examples:</h5>
121    * <p class='bjava'>
122    *    Marshaller <jv>marshaller</jv> = Json.<jsf>DEFAULT</jsf>;
123    *
124    *    <jc>// Parse into a linked-list of strings.</jc>
125    *    List <jv>list1</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
126    *
127    *    <jc>// Parse into a linked-list of beans.</jc>
128    *    List <jv>list2</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
129    *
130    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
131    *    List <jv>list3</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
132    *
133    *    <jc>// Parse into a map of string keys/values.</jc>
134    *    Map <jv>map1</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
135    *
136    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
137    *    Map <jv>map2</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
138    * </p>
139    *
140    * <p>
141    * <c>Collection</c> classes are assumed to be followed by zero or one objects indicating the element type.
142    *
143    * <p>
144    * <c>Map</c> classes are assumed to be followed by zero or two meta objects indicating the key and value types.
145    *
146    * <p>
147    * The array can be arbitrarily long to indicate arbitrarily complex data structures.
148    *
149    * <ul class='notes'>
150    *    <li class='note'>
151    *       Use the {@link #read(Object, Class)} method instead if you don't need a parameterized map/collection.
152    * </ul>
153    *
154    * @param <T> The class type of the object to create.
155    * @param input
156    *    The input.
157    *    <br>Character-based parsers can handle the following input class types:
158    *    <ul>
159    *       <li><jk>null</jk>
160    *       <li>{@link Reader}
161    *       <li>{@link CharSequence}
162    *       <li>{@link InputStream} containing UTF-8 encoded text (or charset defined by
163    *          {@link org.apache.juneau.parser.ReaderParser.Builder#streamCharset(Charset)} property value).
164    *       <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or charset defined by
165    *          {@link org.apache.juneau.parser.ReaderParser.Builder#streamCharset(Charset)} property value).
166    *       <li>{@link File} containing system encoded text (or charset defined by
167    *          {@link org.apache.juneau.parser.ReaderParser.Builder#fileCharset(Charset)} property value).
168    *    </ul>
169    *    <br>Stream-based parsers can handle the following input class types:
170    *    <ul>
171    *       <li><jk>null</jk>
172    *       <li>{@link InputStream}
173    *       <li><code><jk>byte</jk>[]</code>
174    *       <li>{@link File}
175    *       <li>{@link CharSequence} containing encoded bytes according to the {@link org.apache.juneau.parser.InputStreamParser.Builder#binaryFormat(BinaryFormat)} setting.
176    *    </ul>
177    * @param type
178    *    The object type to create.
179    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
180    * @param args
181    *    The type arguments of the class if it's a collection or map.
182    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
183    *    <br>Ignored if the main type is not a map or collection.
184    * @return The parsed object.
185    * @throws ParseException Malformed input encountered.
186    * @throws IOException Thrown by underlying stream.
187    * @see BeanSession#getClassMeta(Type,Type...) for argument syntax for maps and collections.
188    */
189   public final <T> T read(Object input, Type type, Type...args) throws ParseException, IOException {
190      return p.parse(input, type, args);
191   }
192
193   /**
194    * Same as {@link #read(Object, Type, Type...)} except optimized for a non-parameterized class.
195    *
196    * <p>
197    * This is the preferred parse method for simple types since you don't need to cast the results.
198    *
199    * <h5 class='section'>Examples:</h5>
200    * <p class='bjava'>
201    *    Marshaller <jv>marshaller</jv>  = Json.<jsf>DEFAULT</jsf>;
202    *
203    *    <jc>// Parse into a string.</jc>
204    *    String <jv>string</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, String.<jk>class</jk>);
205    *
206    *    <jc>// Parse into a bean.</jc>
207    *    MyBean <jv>bean</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, MyBean.<jk>class</jk>);
208    *
209    *    <jc>// Parse into a bean array.</jc>
210    *    MyBean[] <jv>beanArray</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, MyBean[].<jk>class</jk>);
211    *
212    *    <jc>// Parse into a linked-list of objects.</jc>
213    *    List <jv>list</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, LinkedList.<jk>class</jk>);
214    *
215    *    <jc>// Parse into a map of object keys/values.</jc>
216    *    Map <jv>map</jv> = <jv>marshaller</jv> .read(<jv>json</jv>, TreeMap.<jk>class</jk>);
217    * </p>
218    *
219    * @param <T> The class type of the object being created.
220    * @param input
221    *    The input.
222    *    <br>Character-based parsers can handle the following input class types:
223    *    <ul>
224    *       <li><jk>null</jk>
225    *       <li>{@link Reader}
226    *       <li>{@link CharSequence}
227    *       <li>{@link InputStream} containing UTF-8 encoded text (or charset defined by
228    *          {@link org.apache.juneau.parser.ReaderParser.Builder#streamCharset(Charset)} property value).
229    *       <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or charset defined by
230    *          {@link org.apache.juneau.parser.ReaderParser.Builder#streamCharset(Charset)} property value).
231    *       <li>{@link File} containing system encoded text (or charset defined by
232    *          {@link org.apache.juneau.parser.ReaderParser.Builder#fileCharset(Charset)} property value).
233    *    </ul>
234    *    <br>Stream-based parsers can handle the following input class types:
235    *    <ul>
236    *       <li><jk>null</jk>
237    *       <li>{@link InputStream}
238    *       <li><code><jk>byte</jk>[]</code>
239    *       <li>{@link File}
240    *       <li>{@link CharSequence} containing encoded bytes according to the {@link org.apache.juneau.parser.InputStreamParser.Builder#binaryFormat(BinaryFormat)} setting.
241    *    </ul>
242    * @param type The object type to create.
243    * @return The parsed object.
244    * @throws ParseException Malformed input encountered.
245    * @throws IOException Thrown by underlying stream.
246    */
247   public final <T> T read(Object input, Class<T> type) throws ParseException, IOException {
248      return p.parse(input, type);
249   }
