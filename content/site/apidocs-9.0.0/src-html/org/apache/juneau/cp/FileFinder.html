026/**
027 * Utility class for finding regular or localized files on the classpath and file system.
028 *
029 * <h5 class='section'>Example:</h5>
030 * <p class='bjava'>
031 *    <jc>// Constructor a file source that looks for files in the "files" working directory, then in the
032 *    // package "foo.bar", then in the package "foo.bar.files", then in the package "files".</jc>
033 *    FileFinder <jv>finder</jv> = FileFinder
034 *       .<jsm>create</jsm>()
035 *       .dir(<js>"files"</js>)
036 *       .cp(foo.bar.MyClass.<jk>class</jk>,<jk>null</jk>,<jk>true</jk>)
037 *       .cp(foo.bar.MyClass.<jk>class</jk>,<js>"files"</js>,<jk>true</jk>)
038 *       .cp(foo.bar.MyClass.<jk>class</jk>,<js>"/files"</js>,<jk>true</jk>)
039 *       .cache(1_000_000l)  <jc>// Cache files less than 1MB in size.</jc>
040 *       .ignore(Pattern.<jsm>compile</jsm>(<js>"(?i)(.*\\.(class|properties))|(package.html)"</js>)) <jc>// Ignore certain files.</jc>
041 *       .build();
042 *
043 *    <jc>// Find a normal file.</jc>
044 *    InputStream <jv>is1</jv> = <jv>finder</jv>.getStream(<js>"text.txt"</js>);
045 *
046 *    <jc>// Find a localized file called "text_ja_JP.txt".</jc>
047 *    InputStream <jv>is2</jv> = <jv>finder</jv>.getStream(<js>"text.txt"</js>, Locale.<jsf>JAPAN</jsf>);
048 * </p>
049 *
050 * <p>
051 * If the <c>locale</c> is specified, then we look for resources whose name matches that locale.
052 * For example, if looking for the resource <js>"MyResource.txt"</js> for the Japanese locale, we will look for
053 * files in the following order:
054 * <ol>
055 *    <li><js>"MyResource_ja_JP.txt"</js>
056 *    <li><js>"MyResource_ja.txt"</js>
057 *    <li><js>"MyResource.txt"</js>
058 * </ol>
059 *
060 * <p>
061 * The default implementation of this interface is {@link BasicFileFinder}.
062 * The {@link Builder#type(Class)} method is provided for instantiating other instances.
063 *
064 * <h5 class='section'>Example:</h5>
065 * <p class='bjava'>
066 *    <jk>public class</jk> MyFileFinder <jk>extends</jk> BasicFileFinder {
067 *       <jk>public</jk> MyFileFinder(FileFinder.Builder <jv>builder</jv>) {
068 *          <jk>super</jk>(<jv>builder</jv>);
069 *       }
070 *    }
071 *
072 *    <jc>// Instantiate subclass.</jc>
073 *    FileFinder <jv>myFinder</jv> = FileFinder.<jsm>create</jsm>().type(MyFileFinder.<jk>class</jk>).build();
074 * </p>
075 *
076 * <p>
077 * Subclasses must provide a public constructor that takes in any of the following arguments:
078 * <ul>
079 *    <li>{@link Builder} - The builder object.
080 *    <li>Any beans present in the bean store passed into the constructor.
081 *    <li>Any {@link Optional} beans optionally present in bean store passed into the constructor.
082 * </ul>
083 *
084 * <ul class='seealso'>
085 *    <li class='extlink'>{@source}
086 * </ul>
087 */
088public interface FileFinder {
089
090   //-----------------------------------------------------------------------------------------------------------------
091   // Static
092   //-----------------------------------------------------------------------------------------------------------------
093
094   /** Represents no file finder */
095   public abstract class Void implements FileFinder {}
096
097   /**
098    * Static creator.
099    *
100    * @param beanStore The bean store to use for creating beans.
101    * @return A new builder for this object.
102    */
103   public static Builder create(BeanStore beanStore) {
104      return new Builder(beanStore);
105   }
106
107   /**
108    * Static creator.
109    *
110    * @return A new builder for this object.
111    */
112   public static Builder create() {
113      return new Builder(BeanStore.INSTANCE);
114   }
115
116   //-----------------------------------------------------------------------------------------------------------------
117   // Builder
118   //-----------------------------------------------------------------------------------------------------------------
119
120   /**
121    * Builder class.
122    */
123   @FluentSetters
124   public static class Builder extends BeanBuilder<FileFinder> {
125
126      final Set<LocalDir> roots;
127      long cachingLimit;
128      Pattern[] include, exclude;
129
130      /**
131       * Constructor.
132       *
133       * @param beanStore The bean store to use for creating beans.
134       */
135      protected Builder(BeanStore beanStore) {
136         super(BasicFileFinder.class, beanStore);
137         roots = set();
138         cachingLimit = -1;
139         include = new Pattern[]{Pattern.compile(".*")};
140         exclude = new Pattern[0];
141      }
142
143      @Override /* BeanBuilder */
144      protected FileFinder buildDefault() {
145         return new BasicFileFinder(this);
146      }
147
148      //-------------------------------------------------------------------------------------------------------------
149      // Properties
150      //-------------------------------------------------------------------------------------------------------------
151
152      /**
153       * Adds a class subpackage to the lookup paths.
154       *
155       * @param c The class whose package will be added to the lookup paths.  Must not be <jk>null</jk>.
156       * @param path The absolute or relative subpath.
157       * @param recursive If <jk>true</jk>, also recursively adds all the paths of the parent classes as well.
158       * @return This object.
159       */
160      @FluentSetter
161      public Builder cp(Class<?> c, String path, boolean recursive) {
162         assertArgNotNull("c", c);
163         while (c != null) {
164            roots.add(new LocalDir(c, path));
165            c = recursive ? c.getSuperclass() : null;
166         }
167         return this;
168      }
169
170      /**
171       * Adds a file system directory to the lookup paths.
172       *
173       * @param path The path relative to the working directory.  Must not be <jk>null</jk>
174       * @return This object.
175       */
176      @FluentSetter
177      public Builder dir(String path) {
178         assertArgNotNull("path", path);
179         return path(Paths.get(".").resolve(path));
180      }
181
182      /**
183       * Adds a file system directory to the lookup paths.
184       *
185       * @param path The directory path.
186       * @return This object.
187       */
188      @FluentSetter
189      public Builder path(Path path) {
190         roots.add(new LocalDir(path));
191         return this;
192      }
193
194      /**
195       * Enables in-memory caching of files for quicker retrieval.
196       *
197       * @param cachingLimit The maximum file size in bytes.
198       * @return This object.
199       */
200      @FluentSetter
201      public Builder caching(long cachingLimit) {
202         this.cachingLimit = cachingLimit;
203         return this;
204      }
205
206      /**
207       * Specifies the regular expression file name patterns to use to include files being retrieved from the file source.
208       *
209       * @param patterns
210       *    The regular expression include patterns.
211       *    <br>The default is <js>".*"</js>.
212       * @return This object.
213       */
214      @FluentSetter
215      public Builder include(String...patterns) {
216         this.include = alist(patterns).stream().map(x->Pattern.compile(x)).toArray(Pattern[]::new);
217         return this;
218      }
219
220      /**
221       * Specifies the regular expression file name pattern to use to exclude files from being retrieved from the file source.
222       *
223       * @param patterns
224       *    The regular expression exclude patterns.
225       *    <br>If none are specified, no files will be excluded.
226       * @return This object.
227       */
228      @FluentSetter
229      public Builder exclude(String...patterns) {
230         this.exclude = alist(patterns).stream().map(x->Pattern.compile(x)).toArray(Pattern[]::new);
231         return this;
232      }
233
234      // <FluentSetters>
235
236      @Override /* GENERATED - org.apache.juneau.BeanBuilder */
237      public Builder impl(Object value) {
238         super.impl(value);
239         return this;
240      }
241
242      @Override /* GENERATED - org.apache.juneau.BeanBuilder */
243      public Builder type(Class<?> value) {
244         super.type(value);
245         return this;
246      }
247
248      // </FluentSetters>
249   }
250
251   //-----------------------------------------------------------------------------------------------------------------
252   // Instance
253   //-----------------------------------------------------------------------------------------------------------------
254
255   /**
256    * Returns the contents of the resource with the specified name.
257    *
258    * @param name The resource name.
259    *    See {@link Class#getResource(String)} for format.
260    * @param locale
261    *    The locale of the resource to retrieve.
262    *    <br>If <jk>null</jk>, won't look for localized file names.
263    * @return The resolved resource contents, or <jk>null</jk> if the resource was not found.
264    * @throws IOException Thrown by underlying stream.
265    */
266   public Optional<InputStream> getStream(String name, Locale locale) throws IOException;
267
268   /**
269    * Returns the file with the specified name as a string.
270    *
271    * @param name The file name.
272    * @param locale
273    *    The locale of the resource to retrieve.
274    *    <br>If <jk>null</jk>, won't look for localized file names.
275    * @return The contents of the file as a string.  Assumes UTF-8 encoding.
276    * @throws IOException If file could not be read.
277    */
278   public Optional<String> getString(String name, Locale locale) throws IOException;
279}
