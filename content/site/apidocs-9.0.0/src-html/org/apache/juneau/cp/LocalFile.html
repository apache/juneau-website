013package org.apache.juneau.cp;
014
015import static org.apache.juneau.internal.ArgUtils.*;
016import static org.apache.juneau.internal.IOUtils.*;
017
018import java.io.*;
019import java.nio.file.*;
020
021/**
022 * Identifies a file located either on the classpath or file system.
023 *
024 * Used to encapsulate basic resolution and retrieval of files regardless of where they are located.
025 *
026 * <ul class='seealso'>
027 *    <li class='extlink'>{@source}
028 * </ul>
029 */
030public class LocalFile {
031
032   private final Class<?> clazz;
033   private final String clazzPath;
034   private final Path path;
035   private final String name;
036   private byte[] cache;
037
038   /**
039    * Constructor for classpath file.
040    *
041    * @param clazz The class used to retrieve resources.  Must not be <jk>null</jk>.
042    * @param clazzPath The path relative to the class.  Must be a non-null normalized relative path.
043    */
044   public LocalFile(Class<?> clazz, String clazzPath) {
045      this.clazz = assertArgNotNull("clazz", clazz);
046      this.clazzPath = assertArgNotNull("clazzPath", clazzPath);
047      this.path = null;
048      int i = clazzPath.lastIndexOf('/');
049      this.name = i == -1 ? clazzPath : clazzPath.substring(i+1);
050   }
051
052   /**
053    * Constructor for file system file.
054    *
055    * @param path Filesystem file location.  Must not be <jk>null</jk>.
056    */
057   public LocalFile(Path path) {
058      this.clazz = null;
059      this.clazzPath = null;
060      this.path = assertArgNotNull("path", path);
061      this.name = path.getFileName().toString();
062   }
063
064   /**
065    * Returns the contents of this file.
066    *
067    * @return An input stream of the contents of this file.
068    * @throws IOException If file could not be read.
069    */
070   public InputStream read() throws IOException {
071      synchronized(this) {
072         if (cache != null)
073            return new ByteArrayInputStream(cache);
074      }
075      if (clazz != null)
076         return clazz.getResourceAsStream(clazzPath);
077      return Files.newInputStream(path);
078   }
079
080   /**
081    * Returns the size of this file.
082    *
083    * @return The size of this file in bytes, or <c>-1</c> if not known.
084    * @throws IOException If file size could not be determined.
085    */
086   public long size() throws IOException {
087      return (path == null ? -1 : Files.size(path));
088   }
089
090   /**
091    * Caches the contents of this file into an internal byte array for quick future retrieval.
092    *
093    * @return This object.
094    * @throws IOException If file could not be read.
095    */
096   public LocalFile cache() throws IOException {
097      synchronized(this) {
098         this.cache = readBytes(read());
099      }
100      return this;
101   }
102
103   /**
104    * Returns the name of this file.
105    *
106    * @return The name of this file.
107    */
108   public String getName() {
109      return name;
110   }
111}
