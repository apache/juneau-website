032/**
033 * Java bean store.
034 *
035 * <p>
036 * A simple storage database for beans keyed by type and name.
037 * Used to retrieve and instantiate beans using an injection-like API.
038 * It's similar in concept to the injection framework of Spring but greatly simplified in function and not intended to implement a full-fledged injection framework.
039 *
040 * <p>
041 * Beans can be stored with or without names.  Named beans are typically resolved using
042 * the <ja>@Named</ja> or <ja>@Qualified</ja> annotations on constructor or method parameters.
043 *
044 * <p>
045 * Beans are added through the following methods:
046 * <ul class='javatreec'>
047 *    <li class='jm'>{@link #add(Class,Object) add(Class,Object)}
048 *    <li class='jm'>{@link #add(Class,Object,String) add(Class,Object,String)}
049 *    <li class='jm'>{@link #addBean(Class,Object) addBean(Class,Object)}
050 *    <li class='jm'>{@link #addBean(Class,Object,String) addBean(Class,Object,String)}
051 *    <li class='jm'>{@link #addSupplier(Class,Supplier) addSupplier(Class,Supplier)}
052 *    <li class='jm'>{@link #addSupplier(Class,Supplier,String) addSupplier(Class,Supplier,String)}
053 * </ul>
054 *
055 * <p>
056 * Beans are retrieved through the following methods:
057 * <ul class='javatreec'>
058 *    <li class='jm'>{@link #getBean(Class) getBean(Class)}
059 *    <li class='jm'>{@link #getBean(Class,String) getBean(Class,String)}
060 *    <li class='jm'>{@link #stream(Class) stream(Class)}
061 * </ul>
062 *
063 * <p>
064 * Beans are created through the following methods:
065 * <ul class='javatreec'>
066 *    <li class='jm'>{@link #createBean(Class) createBean(Class)}
067 *    <li class='jm'>{@link #createMethodFinder(Class) createMethodFinder(Class)}
068 *    <li class='jm'>{@link #createMethodFinder(Class,Class) createMethodFinder(Class,Class)}
069 *    <li class='jm'>{@link #createMethodFinder(Class,Object) createMethodFinder(Class,Object)}
070 * </ul>
071 *
072 * <ul class='notes'>
073 *    <li class='note'>Bean stores can be nested using {@link Builder#parent(BeanStore)}.
074 *    <li class='note'>Bean stores can be made read-only using {@link Builder#readOnly()}.
075 *    <li class='note'>Bean stores can be made thread-safe using {@link Builder#threadSafe()}.
076 * </ul>
077 *
078 * <ul class='seealso'>
079 *    <li class='extlink'>{@source}
080 * </ul>
081 */
082public class BeanStore {

140      Object outer;
141      Class<? extends BeanStore> type;
142      BeanStore impl;

144      /**
145       * Constructor.
146       */

149      /**
150       * Instantiates this bean store.
151       *
152       * @return A new bean store.
153       */
154      public BeanStore build() {
155         if (impl != null)
156            return impl;
157         if (type == null || type == BeanStore.class)
158            return new BeanStore(this);
159
160         ClassInfo c = ClassInfo.of(type);
161
162         MethodInfo m = c.getDeclaredMethod(
163            x -> x.isPublic()
164            && x.hasNoParams()
165            && x.isStatic()
166            && x.hasName("getInstance")
167         );
168         if (m != null)
169            return m.invoke(null);
170
171         ConstructorInfo ci = c.getPublicConstructor(x -> x.canAccept(this));
172         if (ci != null)
173            return ci.invoke(this);
174
175         ci = c.getDeclaredConstructor(x -> x.isProtected() && x.canAccept(this));
176         if (ci != null)
177            return ci.accessible().invoke(this);
178
179         throw new BasicRuntimeException("Could not find a way to instantiate class {0}", type);
180      }

352      assertCanWrite();
353      BeanStoreEntry<T> e = createEntry(beanType, bean, name);
354      try (SimpleLock x = lock.write()) {
355         entries.addFirst(e);
356         if (isEmpty(name))
357            unnamedEntries.put(beanType, e);
358      }
359      return this;
360   }

413   @SuppressWarnings("unchecked")
414   public <T> Optional<T> getBean(Class<T> beanType) {
415      try (SimpleLock x = lock.read()) {
416         BeanStoreEntry<T> e = (BeanStoreEntry<T>) unnamedEntries.get(beanType);
417         if (e != null)
418            return optional(e.get());
419         if (parent.isPresent())
420            return parent.get().getBean(beanType);
421         return empty();
422      }

434   public <T> Optional<T> getBean(Class<T> beanType, String name)  {
435      try (SimpleLock x = lock.read()) {
436         BeanStoreEntry<T> e = (BeanStoreEntry<T>)entries.stream().filter(x2 -> x2.matches(beanType, name)).findFirst().orElse(null);
437         if (e != null)
438            return optional(e.get());
439         if (parent.isPresent())
440            return parent.get().getBean(beanType, name);
441         return empty();
442      }

448    * <p>

451    * <p>

462         s = Stream.concat(s, parent.get().stream(beanType));
463      return s;
464   }

483   public BeanStore removeBean(Class<?> beanType, String name) {
484      assertCanWrite();
485      try (SimpleLock x = lock.write()) {
486         if (name == null)
487            unnamedEntries.remove(beanType);
488         entries.removeIf(y -> y.matches(beanType, name));
489      }
490      return this;
491   }

583   /**
584    * Given an executable, returns a list of types that are missing from this factory.
585    *
586    * @param executable The constructor or method to get the params for.
587    * @return A comma-delimited list of types that are missing from this factory, or <jk>null</jk> if none are missing.
588    */
589   public String getMissingParams(ExecutableInfo executable) {
590      List<ParamInfo> params = executable.getParams();
591      List<String> l = list();
592      loop: for (int i = 0; i < params.size(); i++) {
593         ParamInfo pi = params.get(i);
594         ClassInfo pt = pi.getParameterType();
595         if (i == 0 && outer.isPresent() && pt.isInstance(outer.get()))
596            continue loop;
597         if (pt.is(Optional.class) || pt.is(BeanStore.class))
598            continue loop;
599         String beanName = findBeanName(pi);
600         Class<?> ptc = pt.inner();
601         if (beanName == null && !hasBean(ptc))
602            l.add(pt.getSimpleName());
603         if (beanName != null && !hasBean(ptc, beanName))
604            l.add(pt.getSimpleName() + '@' + beanName);
605      }
606      return l.isEmpty() ? null : l.stream().sorted().collect(joining(","));
607   }

609   /**
610    * Given the list of param types, returns <jk>true</jk> if this factory has all the parameters for the specified executable.
611    *
612    * @param executable The constructor or method to get the params for.
613    * @return A comma-delimited list of types that are missing from this factory.
614    */
615   public boolean hasAllParams(ExecutableInfo executable) {
616      loop: for (int i = 0; i < executable.getParamCount(); i++) {
617         ParamInfo pi = executable.getParam(i);
618         ClassInfo pt = pi.getParameterType();
619         if (i == 0 && outer.isPresent() && pt.isInstance(outer.get()))
620            continue loop;
621         if (pt.is(Optional.class) || pt.is(BeanStore.class))
622            continue loop;
623         String beanName = findBeanName(pi);
624         Class<?> ptc = pt.inner();
625         if ((beanName == null && !hasBean(ptc)) || (beanName != null && !hasBean(ptc, beanName)))
626            return false;
627      }
628      return true;
629   }

632   /**
633    * Returns the corresponding beans in this factory for the specified param types.
634    *
635    * @param executable The constructor or method to get the params for.
636    * @return The corresponding beans in this factory for the specified param types.
637    */
638   public Object[] getParams(ExecutableInfo executable) {
639      Object[] o = new Object[executable.getParamCount()];
640      for (int i = 0; i < executable.getParamCount(); i++) {
641         ParamInfo pi = executable.getParam(i);
642         ClassInfo pt = pi.getParameterType();
643         if (i == 0 && outer.isPresent() && pt.isInstance(outer.get())) {
644            o[i] = outer.get();
645         } else if (pt.is(BeanStore.class)) {
646            o[i] = this;
647         } else {
648            String beanName = findBeanName(pi);
649            Class<?> ptc = pt.unwrap(Optional.class).inner();
650            Optional<?> o2 = beanName == null ? getBean(ptc) : getBean(ptc, beanName);
651            o[i] = pt.is(Optional.class) ? o2 : o2.orElse(null);
652         }
653      }
654      return o;
655   }

689   private String findBeanName(ParamInfo pi) {
690      Annotation n = pi.getAnnotation(Annotation.class, x -> x.annotationType().getSimpleName().equals("Named"));
691      if (n != null)
692         return AnnotationInfo.of((ClassInfo)null, n).getValue(String.class, "value", NOT_EMPTY).orElse(null);
693      return null;
694   }
695
696   private void assertCanWrite() {
697      if (readOnly)
698         throw new RuntimeException("Method cannot be used because BeanStore is read-only.");
699   }
700
701   private JsonMap properties() {
702      Predicate<Boolean> nf = ObjectUtils::isTrue;
703      return filteredMap()
704         .append("identity", ObjectUtils.identity(this))
705         .append("entries", entries.stream().map(x -> x.properties()).collect(toList()))
706         .append("outer", ObjectUtils.identity(outer.orElse(null)))
707         .append("parent", parent.map(x->x.properties()).orElse(null))
708         .appendIf(nf, "readOnly", readOnly)
709         .appendIf(nf, "threadSafe", threadSafe)
710      ;
711   }
