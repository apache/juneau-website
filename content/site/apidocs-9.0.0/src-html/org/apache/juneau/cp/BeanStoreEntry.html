013package org.apache.juneau.cp;
014
015import static org.apache.juneau.collections.JsonMap.*;
016import static org.apache.juneau.internal.ArgUtils.*;
017import static org.apache.juneau.internal.ClassUtils.*;
018
019import java.util.function.*;
020
021import org.apache.juneau.collections.*;
022import org.apache.juneau.internal.*;
023
024/**
025 * Represents a bean in a {@link BeanStore}.
026 *
027 * <p>
028 * A bean entry consists of the following:
029 * <ul>
030 *    <li>A class type.
031 *    <li>A bean or bean supplier that returns an instance of the class type.  This can be a subclass of the type.
032 *    <li>An optional name.
033 * </ul>
034 *
035 * @param <T> The bean type.
036 */
037public class BeanStoreEntry<T> {
038
039   //-----------------------------------------------------------------------------------------------------------------
040   // Static
041   //-----------------------------------------------------------------------------------------------------------------
042
043   /**
044    * Static creator.
045    *
046    * @param <T> The class type to associate with the bean.
047    * @param type The class type to associate with the bean.
048    * @param bean The bean supplier.
049    * @param name Optional name to associate with the bean.  Can be <jk>null</jk>.
050    * @return A new bean store entry.
051    */
052   public static <T> BeanStoreEntry<T> create(Class<T> type, Supplier<T> bean, String name) {
053      return new BeanStoreEntry<>(type, bean, name);
054   }
055
056   //-----------------------------------------------------------------------------------------------------------------
057   // Instance
058   //-----------------------------------------------------------------------------------------------------------------
059
060   final Supplier<T> bean;
061   final Class<T> type;
062   final String name;
063
064   /**
065    * Constructor.
066    *
067    * @param type The class type to associate with the bean.
068    * @param bean The bean supplier.
069    * @param name Optional name to associate with the bean.  Can be <jk>null</jk>.
070    */
071   protected BeanStoreEntry(Class<T> type, Supplier<T> bean, String name) {
072      this.bean = assertArgNotNull("bean", bean);
073      this.type = assertArgNotNull("type", type);
074      this.name = name;
075   }
076
077   /**
078    * Returns <jk>true</jk> if this bean is exactly of the specified type.
079    *
080    * @param type The class to check.  Returns <jk>false</jk> if <jk>null</jk>.
081    * @return <jk>true</jk> if this bean is exactly of the specified type.
082    */
083   public boolean matches(Class<?> type) {
084      return this.type.equals(type);
085   }
086
087   /**
088    * Returns <jk>true</jk> if this bean is exactly of the specified type and has the specified name.
089    *
090    * @param type The class to check.  Returns <jk>false</jk> if <jk>null</jk>.
091    * @param name The name to check.  Can be <jk>null</jk> to only match if name of entry is <jk>null</jk>.
092    * @return <jk>true</jk> if this bean is exactly of the specified type and has the specified name.
093    */
094   public boolean matches(Class<?> type, String name) {
095      return matches(type) && StringUtils.eq(this.name, name);
096   }
097
098   /**
099    * Returns the bean associated with this entry.
100    *
101    * @return The bean associated with this entry.
102    */
103   public T get() {
104      return bean.get();
105   }
106
107   /**
108    * Returns the type this bean is associated with.
109    *
110    * @return The type this bean is associated with.
111    */
112   public Class<T> getType() {
113      return type;
114   }
115
116   /**
117    * Returns the name associated with this entry.
118    *
119    * @return the name associated with this entry.  <jk>null</jk> if no name is associated.
120    */
121   public String getName() {
122      return name;
123   }
124
125   /**
126    * Returns the properties in this object as a simple map for debugging purposes.
127    *
128    * @return The properties in this object as a simple map.
129    */
130   protected JsonMap properties() {
131      return filteredMap()
132         .append("type", simpleClassName(getType()))
133         .append("bean", ObjectUtils.identity(get()))
134         .append("name", getName());
135   }
136}
