256   @SuppressWarnings("rawtypes")
257   @Override /* PartSerializer */
258   public String serialize(HttpPartType partType, HttpPartSchema schema, Object value) throws SerializeException, SchemaValidationException {
259
260      ClassMeta<?> type = getClassMetaForObject(value);
261      if (type == null)
262         type = object();
263
264      // Swap if necessary
265      ObjectSwap swap = type.getSwap(this);
266      if (swap != null && ! type.isDateOrCalendarOrTemporal()) {
267         value = swap(swap, value);
268         type = swap.getSwapClassMeta(this);
269
270         // If the getSwapClass() method returns Object, we need to figure out
271         // the actual type now.
272         if (type.isObject())
273            type = getClassMetaForObject(value);
274      }
275
276      schema = ObjectUtils.firstNonNull(schema, DEFAULT_SCHEMA);
277
278      HttpPartDataType t = schema.getType(type);
279
280      HttpPartFormat f = schema.getFormat(type);
281      if (f == HttpPartFormat.NO_FORMAT)
282         f = ctx.getFormat();
283
284      HttpPartCollectionFormat cf = schema.getCollectionFormat();
285      if (cf == HttpPartCollectionFormat.NO_COLLECTION_FORMAT)
286         cf = ctx.getCollectionFormat();
287
288      String out = null;
289
290      schema.validateOutput(value, ctx.getBeanContext());
291
292      if (type.hasMutaterTo(schema.getParsedType()) || schema.getParsedType().hasMutaterFrom(type)) {
293         value = toType(value, schema.getParsedType());
294         type = schema.getParsedType();
295      }
296
297      if (type.isUri()) {
298         value = getUriResolver().resolve(value);
299         type = string();
300      }
301
302      if (value != null) {
303
304         if (t == STRING) {
305
306            if (f == BYTE) {
307               out = base64Encode(toType(value, CM_ByteArray));
308            } else if (f == BINARY) {
309               out = toHex(toType(value, CM_ByteArray));
310            } else if (f == BINARY_SPACED) {
311               out = toSpacedHex(toType(value, CM_ByteArray));
312            } else if (f == DATE) {
313               try {
314                  if (value instanceof Calendar)
315                     out = TemporalCalendarSwap.IsoDate.DEFAULT.swap(this, (Calendar)value);
316                  else if (value instanceof Date)
317                     out = TemporalDateSwap.IsoDate.DEFAULT.swap(this, (Date)value);
318                  else if (value instanceof Temporal)
319                     out = TemporalSwap.IsoDate.DEFAULT.swap(this, (Temporal)value);
320                  else
321                     out = value.toString();
322               } catch (Exception e) {
323                  throw new SerializeException(e);
324               }
325            } else if (f == DATE_TIME) {
326               try {
327                  if (value instanceof Calendar)
328                     out = TemporalCalendarSwap.IsoInstant.DEFAULT.swap(this, (Calendar)value);
329                  else if (value instanceof Date)
330                     out = TemporalDateSwap.IsoInstant.DEFAULT.swap(this, (Date)value);
331                  else if (value instanceof Temporal)
332                     out = TemporalSwap.IsoInstant.DEFAULT.swap(this, (Temporal)value);
333                  else
334                     out = value.toString();
335               } catch (Exception e) {
336                  throw new SerializeException(e);
337               }
338            } else if (f == HttpPartFormat.UON) {
339               out = super.serialize(partType, schema, value);
340            } else {
341               out = toType(value, string());
342            }
343
344         } else if (t == BOOLEAN) {
345
346            out = stringify(toType(value, CM_Boolean));
347
348         } else if (t == INTEGER) {
349
350            if (f == INT64)
351               out = stringify(toType(value, CM_Long));
352            else
353               out = stringify(toType(value, CM_Integer));
354
355         } else if (t == NUMBER) {
356
357            if (f == DOUBLE)
358               out = stringify(toType(value, CM_Double));
359            else
360               out = stringify(toType(value, CM_Float));
361
362         } else if (t == ARRAY) {
363
364            if (cf == HttpPartCollectionFormat.UONC)
365               out = super.serialize(partType, null, toList(partType, type, value, schema));
366            else {
367
368               HttpPartSchema items = schema.getItems();
369               ClassMeta<?> vt = getClassMetaForObject(value);
370               OapiStringBuilder sb = new OapiStringBuilder(cf);
371
372               if (type.isArray()) {
373                  for (int i = 0; i < Array.getLength(value); i++)
374                     sb.append(serialize(partType, items, Array.get(value, i)));
375               } else if (type.isCollection()) {
376                  ((Collection<?>)value).forEach(x -> sb.append(serialize(partType, items, x)));
377               } else if (vt.hasMutaterTo(String[].class)) {
378                  String[] ss = toType(value, CM_StringArray);
379                  for (int i = 0; i < ss.length; i++)
380                     sb.append(serialize(partType, items, ss[i]));
381               } else {
382                  throw new SerializeException("Input is not a valid array type: " + type);
383               }
384
385               out = sb.toString();
386            }
387
388         } else if (t == OBJECT) {
389
390            if (cf == HttpPartCollectionFormat.UONC) {
391               if (schema.hasProperties() && type.isMapOrBean())
392                  value = toMap(partType, type, value, schema);
393               out = super.serialize(partType, null, value);
394
395            } else if (type.isBean()) {
396               OapiStringBuilder sb = new OapiStringBuilder(cf);
397               Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
398               HttpPartSchema schema2 = schema;
399
400               toBeanMap(value).forEachValue(checkNull, (pMeta,key,val,thrown) -> {
401                  if (thrown == null)
402                     sb.append(key, serialize(partType, schema2.getProperty(key), val));
403               });
404               out = sb.toString();
405
406            } else if (type.isMap()) {
407               OapiStringBuilder sb = new OapiStringBuilder(cf);
408               HttpPartSchema schema2 = schema;
409               ((Map<?,?>)value).forEach((k,v) -> sb.append(k, serialize(partType, schema2.getProperty(stringify(k)), v)));
410               out = sb.toString();
411
412            } else {
413               throw new SerializeException("Input is not a valid object type: " + type);
414            }
415
416         } else if (t == FILE) {
417            throw new SerializeException("File part not supported.");
418
419         } else if (t == NO_TYPE) {
420            // This should never be returned by HttpPartSchema.getType(ClassMeta).
421            throw new SerializeException("Invalid type.");
422         }
423      }
424
425      schema.validateInput(out);
426      if (out == null)
427         out = schema.getDefault();
428      if (out == null)
429         out = "null";
430      return out;
431   }

433   private static class OapiStringBuilder {
434      static final AsciiSet EQ = AsciiSet.create("=\\");
435      static final AsciiSet PIPE = AsciiSet.create("|\\");
436      static final AsciiSet PIPE_OR_EQ = AsciiSet.create("|=\\");
437      static final AsciiSet COMMA = AsciiSet.create(",\\");
438      static final AsciiSet COMMA_OR_EQ = AsciiSet.create(",=\\");
439
440      private final StringBuilder sb = new StringBuilder();
441      private final HttpPartCollectionFormat cf;
442      private boolean first = true;
443
444      OapiStringBuilder(HttpPartCollectionFormat cf) {
445         this.cf = cf;
446      }
447
448      private void delim(HttpPartCollectionFormat cf) {
449         if (cf == PIPES)
450            sb.append('|');
451         else if (cf == SSV)
452            sb.append(' ');
453         else if (cf == TSV)
454            sb.append('\t');
455         else
456            sb.append(',');
457      }
458
459      OapiStringBuilder append(Object o) {
460         if (! first)
461            delim(cf);
462         first = false;
463         if (cf == PIPES)
464            sb.append(escapeChars(stringify(o), PIPE));
465         else if (cf == SSV || cf == TSV)
466            sb.append(stringify(o));
467         else
468            sb.append(escapeChars(stringify(o), COMMA));
469         return this;
470      }
471
472      OapiStringBuilder append(Object key, Object val) {
473         if (! first)
474            delim(cf);
475         first = false;
476         if (cf == PIPES)
477            sb.append(escapeChars(stringify(key), PIPE_OR_EQ)).append('=').append(escapeChars(stringify(val), PIPE_OR_EQ));
478         else if (cf == SSV || cf == TSV)
479            sb.append(escapeChars(stringify(key), EQ)).append('=').append(escapeChars(stringify(val), EQ));
480         else
481            sb.append(escapeChars(stringify(key), COMMA_OR_EQ)).append('=').append(escapeChars(stringify(val), COMMA_OR_EQ));
482         return this;
483      }
484
485      @Override
486      public String toString() {
487         return sb.toString();
488      }
489   }

491   private Map<String,Object> toMap(HttpPartType partType, ClassMeta<?> type, Object o, HttpPartSchema s) throws SerializeException, SchemaValidationException {
492      if (s == null)
493         s = DEFAULT_SCHEMA;
494      JsonMap m = new JsonMap();
495      if (type.isBean()) {
496         Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
497         HttpPartSchema s2 = s;
498         toBeanMap(o).forEachValue(checkNull, (pMeta,key,val,thrown) -> {
499            if (thrown == null)
500               m.put(key, toObject(partType, val, s2.getProperty(key)));
501         });
502      } else {
503         HttpPartSchema s2 = s;
504         ((Map<?,?>)o).forEach((k,v) -> m.put(stringify(k), toObject(partType, v, s2.getProperty(stringify(k)))));
505      }
506      if (isSortMaps())
507         return sort(m);
508      return m;
509   }

511   @SuppressWarnings("rawtypes")
512   private List toList(HttpPartType partType, ClassMeta<?> type, Object o, HttpPartSchema s) throws SerializeException, SchemaValidationException {
513      if (s == null)
514         s = DEFAULT_SCHEMA;
515      JsonList l = new JsonList();
516      HttpPartSchema items = s.getItems();
517      if (type.isArray()) {
518         for (int i = 0; i < Array.getLength(o); i++)
519            l.add(toObject(partType, Array.get(o, i), items));
520      } else if (type.isCollection()) {
521         ((Collection<?>)o).forEach(x -> l.add(toObject(partType, x, items)));
522      } else {
523         l.add(toObject(partType, o, items));
524      }
525      if (isSortCollections())
526         return sort(l);
527      return l;
528   }

530   @SuppressWarnings("rawtypes")
531   private Object toObject(HttpPartType partType, Object o, HttpPartSchema s) throws SerializeException, SchemaValidationException {
532      if (o == null)
533         return null;
534      if (s == null)
535         s = DEFAULT_SCHEMA;
536      ClassMeta cm = getClassMetaForObject(o);
537      HttpPartDataType t = s.getType(cm);
538      HttpPartFormat f = s.getFormat(cm);
539      HttpPartCollectionFormat cf = s.getCollectionFormat();
540
541      if (t == STRING) {
542         if (f == BYTE)
543            return base64Encode(toType(o, CM_ByteArray));
544         if (f == BINARY)
545            return toHex(toType(o, CM_ByteArray));
546         if (f == BINARY_SPACED)
547            return toSpacedHex(toType(o, CM_ByteArray));
548         if (f == DATE)
549            return toIsoDate(toType(o, CM_Calendar));
550         if (f == DATE_TIME)
551            return toIsoDateTime(toType(o, CM_Calendar));
552         return o;
553      } else if (t == ARRAY) {
554         List l = toList(partType, getClassMetaForObject(o), o, s);
555         if (cf == CSV)
556            return joine(l, ',');
557         if (cf == PIPES)
558            return joine(l, '|');
559         if (cf == SSV)
560            return join(l, ' ');
561         if (cf == TSV)
562            return join(l, '\t');
563         return l;
564      } else if (t == OBJECT) {
565         return toMap(partType, getClassMetaForObject(o), o, s);
566      }
567
568      return o;
569   }

571   private <T> T toType(Object in, ClassMeta<T> type) throws SerializeException {
572      try {
573         return convertToType(in, type);
574      } catch (InvalidDataConversionException e) {
575         throw new SerializeException(e);
576      }
