234   @SuppressWarnings("unchecked")
235   @Override /* HttpPartParser */
236   public <T> T parse(HttpPartType partType, HttpPartSchema schema, String in, ClassMeta<T> type) throws ParseException, SchemaValidationException {
237      if (partType == null)
238         partType = HttpPartType.OTHER;
239
240      boolean isOptional = type.isOptional();
241
242      while (type != null && type.isOptional())
243         type = (ClassMeta<T>)type.getElementType();
244
245      if (type == null)
246         type = (ClassMeta<T>)object();
247
248      schema = ObjectUtils.firstNonNull(schema, getSchema(), DEFAULT_SCHEMA);
249
250      T t = parseInner(partType, schema, in, type);
251      if (t == null && type.isPrimitive())
252         t = type.getPrimitiveDefault();
253      schema.validateOutput(t, ctx.getBeanContext());
254
255      if (isOptional)
256         t = (T)optional(t);
257
258      return t;
259   }
260
261   @Override /* ParserSession */
262   protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException, ExecutableException {
263      return parseInner(null, HttpPartSchema.DEFAULT, pipe.asString(), type);
264   }

266   @SuppressWarnings({ "unchecked" })
267   private<T> T parseInner(HttpPartType partType, HttpPartSchema schema, String in, ClassMeta<T> type) throws SchemaValidationException, ParseException {
268      schema.validateInput(in);
269      if (in == null || "null".equals(in)) {
270         if (schema.getDefault() == null)
271            return null;
272         in = schema.getDefault();
273      } else {
274
275         ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)type.getSwap(this);
276         BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)type.getBuilderSwap(this);
277         ClassMeta<?> sType = null;
278         if (builder != null)
279            sType = builder.getBuilderClassMeta(this);
280         else if (swap != null)
281            sType = swap.getSwapClassMeta(this);
282         else
283            sType = type;
284
285         if (sType.isOptional())
286            return (T)optional(parseInner(partType, schema, in, sType.getElementType()));
287
288         HttpPartDataType t = schema.getType(sType);
289         if (partType == null)
290            partType = HttpPartType.OTHER;
291
292         HttpPartFormat f = schema.getFormat(sType);
293         if (f == HttpPartFormat.NO_FORMAT)
294            f = ctx.getFormat();
295
296         if (t == STRING) {
297            if (sType.isObject()) {
298               if (f == BYTE)
299                  return toType(base64Decode(in), type);
300               if (f == DATE || f == DATE_TIME)
301                  return toType(parseIsoCalendar(in), type);
302               if (f == BINARY)
303                  return toType(fromHex(in), type);
304               if (f == BINARY_SPACED)
305                  return toType(fromSpacedHex(in), type);
306               if (f == HttpPartFormat.UON)
307                  return super.parse(partType, schema, in, type);
308               return toType(in, type);
309            }
310            if (f == BYTE)
311               return toType(base64Decode(in), type);
312            if (f == DATE) {
313               try {
314                  if (type.isCalendar())
315                     return toType(TemporalCalendarSwap.IsoDate.DEFAULT.unswap(this, in, type), type);
316                  if (type.isDate())
317                     return toType(TemporalDateSwap.IsoDate.DEFAULT.unswap(this, in, type), type);
318                  if (type.isTemporal())
319                     return toType(TemporalSwap.IsoDate.DEFAULT.unswap(this, in, type), type);
320                  return toType(in, type);
321               } catch (Exception e) {
322                  throw new ParseException(e);
323               }
324            }
325            if (f == DATE_TIME) {
326               try {
327                  if (type.isCalendar())
328                     return toType(TemporalCalendarSwap.IsoDateTime.DEFAULT.unswap(this, in, type), type);
329                  if (type.isDate())
330                     return toType(TemporalDateSwap.IsoDateTime.DEFAULT.unswap(this, in, type), type);
331                  if (type.isTemporal())
332                     return toType(TemporalSwap.IsoDateTime.DEFAULT.unswap(this, in, type), type);
333                  return toType(in, type);
334               } catch (Exception e) {
335                  throw new ParseException(e);
336               }
337            }
338            if (f == BINARY)
339               return toType(fromHex(in), type);
340            if (f == BINARY_SPACED)
341               return toType(fromSpacedHex(in), type);
342            if (f == HttpPartFormat.UON)
343               return super.parse(partType, schema, in, type);
344            return toType(in, type);
345
346         } else if (t == BOOLEAN) {
347            if (type.isObject())
348               type = (ClassMeta<T>)CM_Boolean;
349            if (type.isBoolean())
350               return super.parse(partType, schema, in, type);
351            return toType(super.parse(partType, schema, in, CM_Boolean), type);
352
353         } else if (t == INTEGER) {
354            if (type.isObject()) {
355               if (f == INT64)
356                  type = (ClassMeta<T>)CM_Long;
357               else
358                  type = (ClassMeta<T>)CM_Integer;
359            }
360            if (type.isNumber())
361               return super.parse(partType, schema, in, type);
362            return toType(super.parse(partType, schema, in, CM_Integer), type);
363
364         } else if (t == NUMBER) {
365            if (type.isObject()) {
366               if (f == DOUBLE)
367                  type = (ClassMeta<T>)CM_Double;
368               else
369                  type = (ClassMeta<T>)CM_Float;
370            }
371            if (type.isNumber())
372               return super.parse(partType, schema, in, type);
373            return toType(super.parse(partType, schema, in, CM_Double), type);
374
375         } else if (t == ARRAY) {
376
377            HttpPartCollectionFormat cf = schema.getCollectionFormat();
378            if (cf == HttpPartCollectionFormat.NO_COLLECTION_FORMAT)
379               cf = ctx.getCollectionFormat();
380
381            if (cf == HttpPartCollectionFormat.UONC)
382               return super.parse(partType, schema, in, type);
383
384            if (type.isObject())
385               type = (ClassMeta<T>)CM_JsonList;
386
387            ClassMeta<?> eType = type.isObject() ? string() : type.getElementType();
388            if (eType == null)
389               eType = schema.getParsedType().getElementType();
390            if (eType == null)
391               eType = string();
392
393            String[] ss = {};
394
395            if (cf == MULTI)
396               ss = new String[]{in};
397            else if (cf == CSV)
398               ss = split(in, ',');
399            else if (cf == PIPES)
400               ss = split(in, '|');
401            else if (cf == SSV)
402               ss = splitQuoted(in);
403            else if (cf == TSV)
404               ss = split(in, '\t');
405            else if (cf == HttpPartCollectionFormat.UONC)
406               return super.parse(partType, null, in, type);
407            else if (cf == NO_COLLECTION_FORMAT) {
408               if (firstNonWhitespaceChar(in) == '@' && lastNonWhitespaceChar(in) == ')')
409                  return super.parse(partType, null, in, type);
410               ss = split(in, ',');
411            }
412
413            HttpPartSchema items = schema.getItems();
414            if (items == null)
415               items = HttpPartSchema.DEFAULT;
416            Object o = Array.newInstance(eType.getInnerClass(), ss.length);
417            for (int i = 0; i < ss.length; i++)
418               Array.set(o, i, parse(partType, items, ss[i], eType));
419            if (type.hasMutaterFrom(schema.getParsedType()) || schema.getParsedType().hasMutaterTo(type))
420               return toType(toType(o, schema.getParsedType()), type);
421            return toType(o, type);
422
423         } else if (t == OBJECT) {
424
425            HttpPartCollectionFormat cf = schema.getCollectionFormat();
426            if (cf == HttpPartCollectionFormat.NO_COLLECTION_FORMAT)
427               cf = ctx.getCollectionFormat();
428
429            if (cf == HttpPartCollectionFormat.UONC)
430               return super.parse(partType, schema, in, type);
431
432            if (type.isObject())
433               type = (ClassMeta<T>)CM_JsonMap;
434
435            if (! type.isMapOrBean())
436               throw new ParseException("Invalid type {0} for part type OBJECT.", type);
437
438            String[] ss = {};
439
440            if (cf == MULTI)
441               ss = new String[]{in};
442            else if (cf == CSV)
443               ss = split(in, ',');
444            else if (cf == PIPES)
445               ss = split(in, '|');
446            else if (cf == SSV)
447               ss = splitQuoted(in);
448            else if (cf == TSV)
449               ss = split(in, '\t');
450            else if (cf == HttpPartCollectionFormat.UONC)
451               return super.parse(partType, null, in, type);
452            else if (cf == NO_COLLECTION_FORMAT) {
453               if (firstNonWhitespaceChar(in) == '@' && lastNonWhitespaceChar(in) == ')')
454                  return super.parse(partType, null, in, type);
455               ss = split(in, ',');
456            }
457
458            if (type.isBean()) {
459               BeanMap<T> m = ctx.getBeanContext().newBeanMap(type.getInnerClass());
460               for (String s : ss) {
461                  String[] kv = split(s, '=', 2);
462                  if (kv.length != 2)
463                     throw new ParseException("Invalid input {0} for part type OBJECT.", in);
464                  String key = kv[0], value = kv[1];
465                  BeanPropertyMeta bpm = m.getPropertyMeta(key);
466                  if (bpm == null && ! isIgnoreUnknownBeanProperties())
467                     throw new ParseException("Invalid input {0} for part type OBJECT.  Cannot find property {1}", in, key);
468                  m.put(key, parse(partType, schema.getProperty(key), value, ((ClassMeta<T>)(bpm == null ? object() : bpm.getClassMeta()))));
469               }
470               return m.getBean();
471            }
472
473            ClassMeta<?> eType = type.isObject() ? string() : type.getValueType();
474            if (eType == null)
475               eType = schema.getParsedType().getValueType();
476            if (eType == null)
477               eType = string();
478
479            try {
480               Map<String,Object> m = (Map<String,Object>)type.newInstance();
481               if (m == null)
482                  m = JsonMap.create();
483
484               for (String s : ss) {
485                  String[] kv = split(s, '=', 2);
486                  if (kv.length != 2)
487                     throw new ParseException("Invalid input {0} for part type OBJECT.", in);
488                  String key = kv[0], value = kv[1];
489                  m.put(key, parse(partType, schema.getProperty(key), value, eType));
490               }
491               return (T)m;
492            } catch (ExecutableException e) {
493               throw new ParseException(e);
494            }
495
496         } else if (t == FILE) {
497            throw new ParseException("File part not supported.");
498
499         } else if (t == NO_TYPE) {
500            // This should never be returned by HttpPartSchema.getType(ClassMeta).
501            throw new ParseException("Invalid type.");
502         }
503      }
504
505      return super.parse(partType, schema, in, type);
506   }
507
508   private <T> T toType(Object in, ClassMeta<T> type) throws ParseException {
509      try {
510         return convertToType(in, type);
511      } catch (InvalidDataConversionException e) {
512         throw new ParseException(e.getMessage());
513      }
514   }
