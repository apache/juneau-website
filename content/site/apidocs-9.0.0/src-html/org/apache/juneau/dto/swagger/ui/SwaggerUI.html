030/**
031 * Generates a Swagger-UI interface from a Swagger document.
032 *
033 * <ul class='seealso'>
034 *    <li class='link'>{@doc jd.SwaggerUi}
035 *    <li class='extlink'>{@source}
036 * </ul>

038public class SwaggerUI extends ObjectSwap<Swagger,Div> {
039
040   static final FileFinder RESOURCES = FileFinder
041      .create(BeanStore.INSTANCE)
042      .cp(SwaggerUI.class, null, true)
043      .dir(",")
044      .caching(Boolean.getBoolean("RestContext.disableClasspathResourceCaching.b") ? -1 : 1_000_000)
045      .build();
046
047   private static final Set<String> STANDARD_METHODS = set("get", "put", "post", "delete", "options");
048
049   /**
050    * This UI applies to HTML requests only.
051    */
052   @Override
053   public MediaType[] forMediaTypes() {
054      return new MediaType[] {MediaType.HTML};
055   }

057   private static final class Session {
058      final int resolveRefsMaxDepth;
059      final Swagger swagger;
060
061      Session(BeanSession bs, Swagger swagger) {
062         this.swagger = swagger.copy();
063         this.resolveRefsMaxDepth = 1;
064      }
065   }
066
067   @Override
068   public Div swap(BeanSession beanSession, Swagger swagger) throws Exception {
069
070      Session s = new Session(beanSession, swagger);
071
072      String css = RESOURCES.getString("files/htdocs/styles/SwaggerUI.css").orElse(null);
073      if (css == null)
074         css = RESOURCES.getString("SwaggerUI.css").orElse(null);
075
076      Div outer = div(
077         style(css),
078         script("text/javascript", new String[]{RESOURCES.getString("SwaggerUI.js").orElse(null)}),
079         header(s)
080      )._class("swagger-ui");
081
082      // Operations without tags are rendered first.
083      outer.child(div()._class("tag-block tag-block-open").children(tagBlockContents(s, null)));
084
085      if (s.swagger.getTags() != null) {
086         s.swagger.getTags().forEach(x -> {
087            Div tagBlock = div()._class("tag-block tag-block-open").children(
088               tagBlockSummary(x),
089               tagBlockContents(s, x)
090            );
091            outer.child(tagBlock);
092         });
093      }
094
095      if (s.swagger.getDefinitions() != null) {
096         Div modelBlock = div()._class("tag-block").children(
097            modelsBlockSummary(),
098            modelsBlockContents(s)
099         );
100         outer.child(modelBlock);
101      }
102
103      return outer;
104   }

106   // Creates the informational summary before the ops.
107   private Table header(Session s) {
108      Table table = table()._class("header");
109
110      Info info = s.swagger.getInfo();
111      if (info != null) {
112
113         if (info.getDescription() != null)
114            table.child(tr(th("Description:"),td(toBRL(info.getDescription()))));
115
116         if (info.getVersion() != null)
117            table.child(tr(th("Version:"),td(info.getVersion())));
118
119         Contact c = info.getContact();
120         if (c != null) {
121            Table t2 = table();
122
123            if (c.getName() != null)
124               t2.child(tr(th("Name:"),td(c.getName())));
125            if (c.getUrl() != null)
126               t2.child(tr(th("URL:"),td(a(c.getUrl(), c.getUrl()))));
127            if (c.getEmail() != null)
128               t2.child(tr(th("Email:"),td(a("mailto:"+ c.getEmail(), c.getEmail()))));
129
130            table.child(tr(th("Contact:"),td(t2)));
131         }
132
133         License l = info.getLicense();
134         if (l != null) {
135            Object child = l.getUrl() != null ? a(l.getUrl(), l.getName() != null ? l.getName() : l.getUrl()) : l.getName();
136            table.child(tr(th("License:"),td(child)));
137         }
138
139         ExternalDocumentation ed = s.swagger.getExternalDocs();
140         if (ed != null) {
141            Object child = ed.getUrl() != null ? a(ed.getUrl(), ed.getDescription() != null ? ed.getDescription() : ed.getUrl()) : ed.getDescription();
142            table.child(tr(th("Docs:"),td(child)));
143         }
144
145         if (info.getTermsOfService() != null) {
146            String tos = info.getTermsOfService();
147            Object child = StringUtils.isUri(tos) ? a(tos, tos) : tos;
148            table.child(tr(th("Terms of Service:"),td(child)));
149         }
150      }
151
152      return table;
153   }

155   // Creates the "pet  Everything about your Pets  ext-link" header.
156   private HtmlElement tagBlockSummary(Tag t) {
157      ExternalDocumentation ed = t.getExternalDocs();
158
159      return div()._class("tag-block-summary").children(
160         span(t.getName())._class("name"),
161         span(toBRL(t.getDescription()))._class("description"),
162         ed == null ? null : span(a(ed.getUrl(), ed.getDescription() != null ? ed.getDescription() : ed.getUrl()))._class("extdocs")
163      ).onclick("toggleTagBlock(this)");
164   }
165
166   // Creates the contents under the "pet  Everything about your Pets  ext-link" header.
167   private Div tagBlockContents(Session s, Tag t) {
168      Div tagBlockContents = div()._class("tag-block-contents");
169
170      s.swagger.getPaths().forEach((path,v) -> {
171         v.forEach((opName,op) -> {
172            if ((t == null && op.getTags() == null) || (t != null && op.getTags() != null && op.getTags() != null && op.getTags().contains(t.getName())))
173               tagBlockContents.child(opBlock(s, path, opName, op));
174         });
175      });
176
177      return tagBlockContents;
178   }
179
180   private Div opBlock(Session s, String path, String opName, Operation op) {
181
182      String opClass = op.isDeprecated() ? "deprecated" : opName.toLowerCase();
183      if (! op.isDeprecated() && ! STANDARD_METHODS.contains(opClass))
184         opClass = "other";
185
186      return div()._class("op-block op-block-closed " + opClass).children(
187         opBlockSummary(path, opName, op),
188         div(tableContainer(s, op))._class("op-block-contents")
189      );
190   }
191
192   private HtmlElement opBlockSummary(String path, String opName, Operation op) {
193      return div()._class("op-block-summary").children(
194         span(opName.toUpperCase())._class("method-button"),
195         span(path)._class("path"),
196         op.getSummary() != null ? span(op.getSummary())._class("summary") : null
197      ).onclick("toggleOpBlock(this)");
198   }

200   private Div tableContainer(Session s, Operation op) {
201      Div tableContainer = div()._class("table-container");
202
203      if (op.getDescription() != null)
204         tableContainer.child(div(toBRL(op.getDescription()))._class("op-block-description"));
205
206      if (op.getParameters() != null) {
207         tableContainer.child(div(h4("Parameters")._class("title"))._class("op-block-section-header"));
208
209         Table parameters = table(tr(th("Name")._class("parameter-key"), th("Description")._class("parameter-key")))._class("parameters");
210
211         op.getParameters().forEach(x -> {
212            String piName = "body".equals(x.getIn()) ? "body" : x.getName();
213            boolean required = x.getRequired() == null ? false : x.getRequired();
214
215            Td parameterKey = td(
216               div(piName)._class("name" + (required ? " required" : "")),
217               required ? div("required")._class("requiredlabel") : null,
218               div(x.getType())._class("type"),
219               div('(' + x.getIn() + ')')._class("in")
220            )._class("parameter-key");
221
222            Td parameterValue = td(
223               div(toBRL(x.getDescription()))._class("description"),
224               examples(s, x)
225            )._class("parameter-value");
226
227            parameters.child(tr(parameterKey, parameterValue));
228         });
229
230         tableContainer.child(parameters);
231      }
232
233      if (op.getResponses() != null) {
234         tableContainer.child(div(h4("Responses")._class("title"))._class("op-block-section-header"));
235
236         Table responses = table(tr(th("Code")._class("response-key"), th("Description")._class("response-key")))._class("responses");
237         tableContainer.child(responses);
238
239         op.getResponses().forEach((k,v) -> {
240            Td code = td(k)._class("response-key");
241
242            Td codeValue = td(
243               div(toBRL(v.getDescription()))._class("description"),
244               examples(s, v),
245               headers(s, v)
246            )._class("response-value");
247
248            responses.child(tr(code, codeValue));
249         });
250      }
251
252      return tableContainer;
253   }
254
255   private Div headers(Session s, ResponseInfo ri) {
256      if (ri.getHeaders() == null)
257         return null;
258
259      Table sectionTable = table(tr(th("Name"),th("Description"),th("Schema")))._class("section-table");
260
261      Div headers = div(
262         div("Headers:")._class("section-name"),
263         sectionTable
264      )._class("headers");
265
266      ri.getHeaders().forEach((k,v) -> {
267         sectionTable.child(
268            tr(
269               td(k)._class("name"),
270               td(toBRL(v.getDescription()))._class("description"),
271               td(v.asMap().keepAll("type","format","items","collectionFormat","default","maximum","exclusiveMaximum","minimum","exclusiveMinimum","maxLength","minLength","pattern","maxItems","minItems","uniqueItems","enum","multipleOf"))
272            )
273         );
274      });
275
276      return headers;
277   }

279   private Div examples(Session s, ParameterInfo pi) {
280      boolean isBody = "body".equals(pi.getIn());
281
282      JsonMap m = new JsonMap();
283
284      try {
285         if (isBody) {
286            SchemaInfo si = pi.getSchema();
287            if (si != null)
288               m.put("model", si.copy().resolveRefs(s.swagger, new ArrayDeque<String>(), s.resolveRefsMaxDepth));
289         } else {
290            JsonMap m2 = pi
291               .copy()
292               .resolveRefs(s.swagger, new ArrayDeque<String>(), s.resolveRefsMaxDepth)
293               .asMap()
294               .keepAll("format","pattern","collectionFormat","maximum","minimum","multipleOf","maxLength","minLength","maxItems","minItems","allowEmptyValue","exclusiveMaximum","exclusiveMinimum","uniqueItems","items","default","enum");
295            m.put("model", m2.isEmpty() ? i("none") : m2);
296         }
297
298      } catch (Exception e) {
299         e.printStackTrace();
300      }
301
302      if (m.isEmpty())
303         return null;
304
305      return examplesDiv(m);
306   }
307
308   private Div examples(Session s, ResponseInfo ri) {
309      SchemaInfo si = ri.getSchema();
310
311      JsonMap m = new JsonMap();
312      try {
313         if (si != null) {
314            si = si.copy().resolveRefs(s.swagger, new ArrayDeque<String>(), s.resolveRefsMaxDepth);
315            m.put("model", si);
316         }
317
318         Map<String,?> examples = ri.getExamples();
319         if (examples != null)
320            examples.forEach((k,v) -> m.put(k,v));
321      } catch (Exception e) {
322         e.printStackTrace();
323      }
324
325      if (m.isEmpty())
326         return null;
327
328      return examplesDiv(m);
329   }
330
331   private Div examplesDiv(JsonMap m) {
332      if (m.isEmpty())
333         return null;
334
335      Select select = null;
336      if (m.size() > 1) {
337         select = (Select)select().onchange("selectExample(this)")._class("example-select");
338      }
339
340      Div div = div(select)._class("examples");
341
342      if (select != null)
343         select.child(option("model","model"));
344      div.child(div(m.remove("model"))._class("model active").attr("data-name", "model"));
345
346      Select select2 = select;
347      m.forEach((k,v) -> {
348         if (select2 != null)
349            select2.child(option(k, k));
350         div.child(div(v.toString().replaceAll("\\n", "\n"))._class("example").attr("data-name", k));
351      });
352
353      return div;
354   }

356   // Creates the "Model" header.
357   private HtmlElement modelsBlockSummary() {
358      return div()._class("tag-block-summary").children(span("Models")._class("name")).onclick("toggleTagBlock(this)");
359   }

361   // Creates the contents under the "Model" header.
362   private Div modelsBlockContents(Session s) {
363      Div modelBlockContents = div()._class("tag-block-contents");
364      s.swagger.getDefinitions().forEach((k,v) -> modelBlockContents.child(modelBlock(k,v)));
365      return modelBlockContents;
366   }
367
368   private Div modelBlock(String modelName, JsonMap model) {
369      return div()._class("op-block op-block-closed model").children(
370         modelBlockSummary(modelName, model),
371         div(model)._class("op-block-contents")
372      );
373   }
374
375   private HtmlElement modelBlockSummary(String modelName, JsonMap model) {
376      return div()._class("op-block-summary").children(
377         span(modelName)._class("method-button"),
378         model.containsKey("description") ? span(toBRL(model.remove("description").toString()))._class("summary") : null
379      ).onclick("toggleOpBlock(this)");
380   }

382   /**
383    * Replaces newlines with <br> elements.
384    */
385   private static List<Object> toBRL(String s) {
386      if (s == null)
387         return null;
388      if (s.indexOf(',') == -1)
389         return singletonList(s);
390      List<Object> l = list();
391      String[] sa = s.split("\n");
392      for (int i = 0; i < sa.length; i++) {
393         if (i > 0)
394            l.add(br());
395         l.add(sa[i]);
396      }
397      return l;
398   }
