229   /*
230    * Returns <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.
231    */
232   private boolean shouldUseExpandedParams(BeanPropertyMeta pMeta) {
233      ClassMeta<?> cm = pMeta.getClassMeta().getSerializedClassMeta(this);
234      if (cm.isCollectionOrArray()) {
235         if (isExpandedParams() || getUrlEncodingClassMeta(pMeta.getBeanMeta().getClassMeta()).isExpandedParams())
236            return true;
237      }
238      return false;
239   }
240
241   /*
242    * Returns <jk>true</jk> if the specified value should be represented as an expanded parameter list.
243    */
244   private boolean shouldUseExpandedParams(Object value) {
245      if (value == null || ! isExpandedParams())
246         return false;
247      ClassMeta<?> cm = getClassMetaForObject(value).getSerializedClassMeta(this);
248      if (cm.isCollectionOrArray()) {
249         if (isExpandedParams())
250            return true;
251      }
252      return false;
253   }

255   @Override /* SerializerSession */
256   protected void doSerialize(SerializerPipe out, Object o) throws IOException, SerializeException {
257      serializeAnything(getUonWriter(out).i(getInitialDepth()), o);
258   }

260   /*
261    * Workhorse method. Determines the type of object, and then calls the appropriate type-specific serialization method.
262    */
263   private SerializerWriter serializeAnything(UonWriter out, Object o) throws IOException, SerializeException {
264
265      ClassMeta<?> aType;        // The actual type
266      ClassMeta<?> sType;        // The serialized type
267
268      ClassMeta<?> eType = getExpectedRootType(o);
269      aType = push2("root", o, eType);
270      indent--;
271      if (aType == null)
272         aType = object();
273
274      sType = aType;
275      String typeName = getBeanTypeName(this, eType, aType, null);
276
277      // Swap if necessary
278      ObjectSwap swap = aType.getSwap(this);
279      if (swap != null) {
280         o = swap(swap, o);
281         sType = swap.getSwapClassMeta(this);
282
283         // If the getSwapClass() method returns Object, we need to figure out
284         // the actual type now.
285         if (sType.isObject())
286            sType = getClassMetaForObject(o);
287      }
288
289      if (sType.isMap()) {
290         if (o instanceof BeanMap)
291            serializeBeanMap(out, (BeanMap)o, typeName);
292         else
293            serializeMap(out, (Map)o, sType);
294      } else if (sType.isBean()) {
295         serializeBeanMap(out, toBeanMap(o), typeName);
296      } else if (sType.isCollection() || sType.isArray()) {
297         Map m = sType.isCollection() ? getCollectionMap((Collection)o) : getCollectionMap(o);
298         serializeCollectionMap(out, m, getClassMeta(Map.class, Integer.class, Object.class));
299      } else if (sType.isReader()) {
300         pipe((Reader)o, out);
301      } else if (sType.isInputStream()) {
302         pipe((InputStream)o, out);
303      } else {
304         // All other types can't be serialized as key/value pairs, so we create a
305         // mock key/value pair with a "_value" key.
306         out.append("_value=");
307         pop();
308         super.serializeAnything(out, o, null, null, null);
309         return out;
310      }
311
312      pop();
313      return out;
314   }

316   /*
317    * Converts a Collection into an integer-indexed map.
318    */
319   private static Map<Integer,Object> getCollectionMap(Collection<?> c) {
320      Map<Integer,Object> m = new TreeMap<>();
321      IntValue i = IntValue.create();
322      c.forEach(o -> m.put(i.getAndIncrement(), o));
323      return m;
324   }
325
326   /*
327    * Converts an array into an integer-indexed map.
328    */
329   private static Map<Integer,Object> getCollectionMap(Object array) {
330      Map<Integer,Object> m = new TreeMap<>();
331      for (int i = 0; i < Array.getLength(array); i++)
332         m.put(i, Array.get(array, i));
333      return m;
334   }

336   private SerializerWriter serializeMap(UonWriter out, Map m, ClassMeta<?> type) throws SerializeException {
337
338      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
339
340      Flag addAmp = Flag.create();
341
342      forEachEntry(m, e -> {
343         Object key = generalize(e.getKey(), keyType);
344         Object value = e.getValue();
345
346         if (shouldUseExpandedParams(value)) {
347            if (value instanceof Collection) {
348               ((Collection<?>)value).forEach(x -> {
349                  addAmp.ifSet(()->out.cr(indent).append('&')).set();
350                  out.appendObject(key, true).append('=');
351                  super.serializeAnything(out, x, null, stringify(key), null);
352               });
353            } else /* array */ {
354               for (int i = 0; i < Array.getLength(value); i++) {
355                  addAmp.ifSet(()->out.cr(indent).append('&')).set();
356                  out.appendObject(key, true).append('=');
357                  super.serializeAnything(out, Array.get(value, i), null, stringify(key), null);
358               }
359            }
360         } else {
361            addAmp.ifSet(()->out.cr(indent).append('&')).set();
362            out.appendObject(key, true).append('=');
363            super.serializeAnything(out, value, valueType, (key == null ? null : key.toString()), null);
364         }
365      });
366
367      return out;
368   }
369
370   private SerializerWriter serializeCollectionMap(UonWriter out, Map<?,?> m, ClassMeta<?> type) throws SerializeException {
371
372      ClassMeta<?> valueType = type.getValueType();
373
374      Flag addAmp = Flag.create();
375
376      m.forEach((k,v) -> {
377         addAmp.ifSet(()->out.cr(indent).append('&')).set();
378         out.append(k).append('=');
379         super.serializeAnything(out, v, valueType, null, null);
380      });
381
382      return out;
383   }
384
385   private SerializerWriter serializeBeanMap(UonWriter out, BeanMap<?> m, String typeName) throws SerializeException {
386      Flag addAmp = Flag.create();
387
388      if (typeName != null) {
389         BeanPropertyMeta pm = m.getMeta().getTypeProperty();
390         out.appendObject(pm.getName(), true).append('=').appendObject(typeName, false);
391         addAmp.set();
392      }
393
394      Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
395      m.forEachValue(checkNull, (pMeta,key,value,thrown) -> {
396         ClassMeta<?> cMeta = pMeta.getClassMeta();
397         ClassMeta<?> sMeta = cMeta.getSerializedClassMeta(this);
398
399         if (thrown != null)
400            onBeanGetterException(pMeta, thrown);
401
402         if (canIgnoreValue(sMeta, key, value))
403            return;
404
405         if (value != null && shouldUseExpandedParams(pMeta)) {
406            // Transformed object array bean properties may be transformed resulting in ArrayLists,
407            // so we need to check type if we think it's an array.
408            if (sMeta.isCollection() || value instanceof Collection) {
409               ((Collection<?>)value).forEach(x -> {
410                  addAmp.ifSet(()->out.cr(indent).append('&')).set();
411                  out.appendObject(key, true).append('=');
412                  super.serializeAnything(out, x, cMeta.getElementType(), key, pMeta);
413               });
414            } else /* array */ {
415               for (int i = 0; i < Array.getLength(value); i++) {
416                  addAmp.ifSet(()->out.cr(indent).append('&')).set();
417                  out.appendObject(key, true).append('=');
418                  super.serializeAnything(out, Array.get(value, i), cMeta.getElementType(), key, pMeta);
419               }
420            }
421         } else {
422            addAmp.ifSet(()->out.cr(indent).append('&')).set();
423            out.appendObject(key, true).append('=');
424            super.serializeAnything(out, value, cMeta, key, pMeta);
425         }
426      });
427
428      return out;
429   }
