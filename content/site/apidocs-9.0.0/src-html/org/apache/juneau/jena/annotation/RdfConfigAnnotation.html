021/**
022 * Utility classes and methods for the {@link RdfConfig @RdfConfig} annotation.
023 *
024 * <ul class='seealso'>
025 *    <li class='link'>{@doc jmr.RdfDetails}
026 *    <li class='extlink'>{@source}
027 * </ul>
028 */
029public class RdfConfigAnnotation {
030
031   /**
032    * Applies {@link RdfConfig} annotations to a {@link org.apache.juneau.jena.RdfSerializer.Builder}.
033    */
034   public static class SerializerApply extends AnnotationApplier<RdfConfig,RdfSerializer.Builder> {
035
036      /**
037       * Constructor.
038       *
039       * @param vr The resolver for resolving values in annotations.
040       */
041      public SerializerApply(VarResolverSession vr) {
042         super(RdfConfig.class, RdfSerializer.Builder.class, vr);
043      }
044
045      @Override
046      public void apply(AnnotationInfo<RdfConfig> ai, RdfSerializer.Builder b) {
047         RdfConfig a = ai.inner();
048
049         string(a.language()).ifPresent(x -> b.language(x));
050         string(a.juneauNs()).map(Namespace::of).ifPresent(x -> b.juneauNs(x));
051         string(a.juneauBpNs()).map(Namespace::of).ifPresent(x -> b.juneauBpNs(x));
052         bool(a.disableUseXmlNamespaces()).ifPresent(x -> b.disableUseXmlNamespaces(x));
053         string(a.collectionFormat()).map(RdfCollectionFormat::valueOf).ifPresent(x -> b.collectionFormat(x));
054         bool(a.looseCollections()).ifPresent(x -> b.looseCollections(x));
055         bool(a.addBeanTypes()).ifPresent(x -> b.addBeanTypesRdf(x));
056         bool(a.addLiteralTypes()).ifPresent(x -> b.addLiteralTypes(x));
057         bool(a.addRootProperty()).ifPresent(x -> b.addRootProperty(x));
058         bool(a.disableAutoDetectNamespaces()).ifPresent(x -> b.disableAutoDetectNamespaces(x));
059         strings(a.namespaces()).map(Namespace::createArray).ifPresent(x -> b.namespaces(x));
060         string(a.rdfxml_iriRules()).ifPresent(x -> b.rdfxml_iriRules(x));
061         string(a.rdfxml_errorMode()).ifPresent(x -> b.rdfxml_errorMode(x));
062         bool(a.rdfxml_embedding()).ifPresent(x -> b.rdfxml_embedding(x));
063         string(a.rdfxml_xmlBase()).ifPresent(x -> b.rdfxml_xmlbase(x));
064         bool(a.rdfxml_longId()).ifPresent(x -> b.rdfxml_longId(x));
065         bool(a.rdfxml_allowBadUris()).ifPresent(x -> b.rdfxml_allowBadUris(x));
066         string(a.rdfxml_relativeUris()).ifPresent(x -> b.rdfxml_relativeUris(x));
067         string(a.rdfxml_showXmlDeclaration()).ifPresent(x -> b.rdfxml_showXmlDeclaration(x));
068         bool(a.rdfxml_disableShowDoctypeDeclaration()).ifPresent(x -> b.rdfxml_disableShowDoctypeDeclaration(x));
069         integer(a.rdfxml_tab(), "rdfxml_tab").ifPresent(x -> b.rdfxml_tab(x));
070         string(a.rdfxml_attributeQuoteChar()).map(x -> x.charAt(0)).ifPresent(x -> b.rdfxml_attributeQuoteChar(x));
071         string(a.rdfxml_blockRules()).ifPresent(x -> b.rdfxml_blockRules(x));
072         integer(a.n3_minGap(), "n3_minGap").ifPresent(x -> b.n3_minGap(x));
073         bool(a.n3_disableObjectLists()).ifPresent(x -> b.n3_disableObjectLists(x));
074         integer(a.n3_subjectColumn(), "n3_subjectColumn").ifPresent(x -> b.n3_subjectColumn(x));
075         integer(a.n3_propertyColumn(), "n3_propertyColumn").ifPresent(x -> b.n3_propertyColumn(x));
076         integer(a.n3_indentProperty(), "n3_indentProperty").ifPresent(x -> b.n3_indentProperty(x));
077         integer(a.n3_widePropertyLen(), "n3_widePropertyLen").ifPresent(x -> b.n3_widePropertyLen(x));
078         bool(a.n3_disableAbbrevBaseUri()).ifPresent(x -> b.n3_disableAbbrevBaseUri(x));
079         bool(a.n3_disableUsePropertySymbols()).ifPresent(x -> b.n3_disableUsePropertySymbols(x));
080         bool(a.n3_disableUseTripleQuotedStrings()).ifPresent(x -> b.n3_disableUseTripleQuotedStrings(x));
081         bool(a.n3_disableUseDoubles()).ifPresent(x -> b.n3_disableUseDoubles(x));
082      }
083   }
