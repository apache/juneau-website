215   protected RdfParserSession(Builder builder) {
216      super(builder);
217      ctx = builder.ctx;
218      model = ModelFactory.createDefaultModel();
219      addModelPrefix(ctx.getJuneauNs());
220      addModelPrefix(ctx.getJuneauBpNs());
221      pRoot = model.createProperty(ctx.getJuneauNs().getUri(), RDF_juneauNs_ROOT);
222      pValue = model.createProperty(ctx.getJuneauNs().getUri(), RDF_juneauNs_VALUE);
223      pType = model.createProperty(ctx.getJuneauBpNs().getUri(), RDF_juneauNs_TYPE);
224      pRdfType = model.createProperty("http://www.w3.org/1999/02/22-rdf-syntax-ns#type");
225      rdfReader = model.getReader(ctx.getLanguage());
226
227      // Note: NTripleReader throws an exception if you try to set any properties on it.
228      if (! ctx.getLanguage().equals(LANG_NTRIPLE))
229         ctx.getJenaSettings().forEach((k,v) -> rdfReader.setProperty(k, v));
230   }

232   @Override /* ReaderParserSession */
233   protected <T> T doParse(ParserPipe pipe, ClassMeta<T> type) throws IOException, ParseException, ExecutableException {
234
235      RDFReader r = rdfReader;
236      r.read(model, pipe.getBufferedReader(), null);
237
238      List<Resource> roots = getRoots(model);
239
240      // Special case where we're parsing a loose collection of resources.
241      if (isLooseCollections() && type.isCollectionOrArray()) {
242         Collection c = null;
243         if (type.isArray() || type.isArgs())
244            c = list();
245         else
246            c = (
247               type.canCreateNewInstance(getOuter())
248               ? (Collection<?>)type.newInstance(getOuter())
249               : new JsonList(this)
250            );
251
252         AtomicInteger argIndex = new AtomicInteger(0);
253         Collection c2 = c;
254         roots.forEach(x -> c2.add(parseAnything(type.isArgs() ? type.getArg(argIndex.getAndIncrement()) : type.getElementType(), x, getOuter(), null)));
255
256         if (type.isArray() || type.isArgs())
257            return (T)toArray(type, c);
258         return (T)c;
259      }
260
261      if (roots.isEmpty())
262         return type.isOptional() ? (T)empty() : null;
263
264      if (roots.size() > 1)
265         throw new ParseException(this, "Too many root nodes found in model:  {0}", roots.size());
266      Resource resource = roots.get(0);
267
268      return parseAnything(type, resource, getOuter(), null);
269   }

275   /*
276    * Decodes the specified string.
277    * If {@link RdfParser#RDF_trimWhitespace} is <jk>true</jk>, the resulting string is trimmed before decoding.
278    * If {@link #isTrimStrings()} is <jk>true</jk>, the resulting string is trimmed after decoding.
279    */
280   private String decodeString(Object o) {
281      if (o == null)
282         return null;
283      String s = o.toString();
284      if (s.isEmpty())
285         return s;
286      if (isTrimWhitespace())
287         s = s.trim();
288      s = XmlUtils.decode(s, null);
289      if (isTrimStrings())
290         s = s.trim();
291      return s;
292   }

294   /*
295    * Finds the roots in the model using either the "root" property to identify it,
296    *    or by resorting to scanning the model for all nodes with no incoming predicates.
297    */
298   private List<Resource> getRoots(Model m) {
299      List<Resource> l = new LinkedList<>();
300
301      // First try to find the root using the "http://www.apache.org/juneau/root" property.
302      Property root = m.createProperty(getJuneauNs().getUri(), RDF_juneauNs_ROOT);
303      for (ResIterator i  = m.listResourcesWithProperty(root); i.hasNext();)
304         l.add(i.next());
305
306      if (! l.isEmpty())
307         return l;
308
309      // Otherwise, we need to find all resources that aren't objects.
310      // We want to explicitly ignore statements where the subject
311      // and object are the same node.
312      Set<RDFNode> objects = new HashSet<>();
313      for (StmtIterator i = m.listStatements(); i.hasNext();) {
314         Statement st = i.next();
315         RDFNode subject = st.getSubject();
316         RDFNode object = st.getObject();
317         if (object.isResource() && ! object.equals(subject))
318            objects.add(object);
319      }
320      for (ResIterator i = m.listSubjects(); i.hasNext();) {
321         Resource r = i.next();
322         if (! objects.contains(r))
323            l.add(r);
324      }
325      return l;
326   }

328   private <T> BeanMap<T> parseIntoBeanMap(Resource r2, BeanMap<T> m) throws ParseException {
329      BeanMeta<T> bm = m.getMeta();
330      RdfBeanMeta rbm = getRdfBeanMeta(bm);
331      if (rbm.hasBeanUri() && r2.getURI() != null)
332         rbm.getBeanUriProperty().set(m, null, r2.getURI());
333      for (StmtIterator i = r2.listProperties(); i.hasNext();) {
334         Statement st = i.next();
335         Property p = st.getPredicate();
336         String key = decodeString(p.getLocalName());
337         BeanPropertyMeta pMeta = m.getPropertyMeta(key);
338         setCurrentProperty(pMeta);
339         if (pMeta != null) {
340            RDFNode o = st.getObject();
341            ClassMeta<?> cm = pMeta.getClassMeta();
342            if (cm.isCollectionOrArray() && isMultiValuedCollections(pMeta)) {
343               ClassMeta<?> et = cm.getElementType();
344               Object value = parseAnything(et, o, m.getBean(false), pMeta);
345               setName(et, value, key);
346               try {
347                  pMeta.add(m, key, value);
348               } catch (BeanRuntimeException e) {
349                  onBeanSetterException(pMeta, e);
350                  throw e;
351               }
352            } else {
353               Object value = parseAnything(cm, o, m.getBean(false), pMeta);
354               setName(cm, value, key);
355               try {
356                  pMeta.set(m, key, value);
357               } catch (BeanRuntimeException e) {
358                  onBeanSetterException(pMeta, e);
359                  throw e;
360               }
361            }
362         } else if (! (p.equals(pRoot) || p.equals(pType))) {
363            RDFNode o = st.getObject();
364            Object value = parseAnything(object(), o, m.getBean(false), null);
365            onUnknownProperty(key, m, value);
366         }
367         setCurrentProperty(null);
368      }
369      return m;
370   }

372   private boolean isMultiValuedCollections(BeanPropertyMeta pMeta) {
373      RdfBeanPropertyMeta bpRdf = (pMeta == null ? RdfBeanPropertyMeta.DEFAULT : getRdfBeanPropertyMeta(pMeta));
374
375      if (bpRdf.getCollectionFormat() != RdfCollectionFormat.DEFAULT)
376         return bpRdf.getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
377
378      return getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
379   }

381   private <T> T parseAnything(ClassMeta<?> eType, RDFNode n, Object outer, BeanPropertyMeta pMeta) throws ParseException {
382
383      if (eType == null)
384         eType = object();
385      ObjectSwap<T,Object> swap = (ObjectSwap<T,Object>)eType.getSwap(this);
386      BuilderSwap<T,Object> builder = (BuilderSwap<T,Object>)eType.getBuilderSwap(this);
387      ClassMeta<?> sType = null;
388      if (builder != null)
389         sType = builder.getBuilderClassMeta(this);
390      else if (swap != null)
391         sType = swap.getSwapClassMeta(this);
392      else
393         sType = eType;
394
395      if (sType.isOptional())
396         return (T)optional(parseAnything(eType.getElementType(), n, outer, pMeta));
397
398      setCurrentClass(sType);
399
400      if (! sType.canCreateNewInstance(outer)) {
401         if (n.isResource()) {
402            Statement st = n.asResource().getProperty(pType);
403            if (st != null) {
404               String c = st.getLiteral().getString();
405               ClassMeta tcm = getClassMeta(c, pMeta, eType);
406               if (tcm != null)
407                  sType = eType = tcm;
408            }
409         }
410      }
411
412      Object o = null;
413      if (n.isResource() && n.asResource().getURI() != null && n.asResource().getURI().equals(RDF_NIL)) {
414         // Do nothing.  Leave o == null.
415      } else if (sType.isObject()) {
416         if (n.isLiteral()) {
417            o = n.asLiteral().getValue();
418            if (o instanceof String) {
419               o = decodeString(o);
420            }
421         }
422         else if (n.isResource()) {
423            Resource r = n.asResource();
424            if (! urisVisited.add(r))
425               o = r.getURI();
426            else if (r.getProperty(pValue) != null) {
427               o = parseAnything(object(), n.asResource().getProperty(pValue).getObject(), outer, null);
428            } else if (isSeq(r)) {
429               o = new JsonList(this);
430               parseIntoCollection(r.as(Seq.class), (Collection)o, sType, pMeta);
431            } else if (isBag(r)) {
432               o = new JsonList(this);
433               parseIntoCollection(r.as(Bag.class), (Collection)o, sType, pMeta);
434            } else if (r.canAs(RDFList.class)) {
435               o = new JsonList(this);
436               parseIntoCollection(r.as(RDFList.class), (Collection)o, sType, pMeta);
437            } else {
438               // If it has a URI and no child properties, we interpret this as an
439               // external resource, and convert it to just a URL.
440               String uri = r.getURI();
441               if (uri != null && ! r.listProperties().hasNext()) {
442                  o = r.getURI();
443               } else {
444                  JsonMap m2 = new JsonMap(this);
445                  parseIntoMap(r, m2, null, null, pMeta);
446                  o = cast(m2, pMeta, eType);
447               }
448            }
449         } else {
450            throw new ParseException(this, "Unrecognized node type ''{0}'' for object", n);
451         }
452      } else if (sType.isBoolean()) {
453         o = convertToType(getValue(n, outer), boolean.class);
454      } else if (sType.isCharSequence()) {
455         o = decodeString(getValue(n, outer));
456      } else if (sType.isChar()) {
457         o = parseCharacter(decodeString(getValue(n, outer)));
458      } else if (sType.isNumber()) {
459         o = parseNumber(getValue(n, outer).toString(), (Class<? extends Number>)sType.getInnerClass());
460      } else if (sType.isMap()) {
461         Resource r = n.asResource();
462         if (! urisVisited.add(r))
463            return null;
464         Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : newGenericMap(sType));
465         o = parseIntoMap(r, m, eType.getKeyType(), eType.getValueType(), pMeta);
466      } else if (sType.isCollectionOrArray() || sType.isArgs()) {
467         if (sType.isArray() || sType.isArgs())
468            o = list();
469         else
470            o = (sType.canCreateNewInstance(outer) ? (Collection<?>)sType.newInstance(outer) : new JsonList(this));
471         Resource r = n.asResource();
472         if (! urisVisited.add(r))
473            return null;
474         if (isSeq(r)) {
475            parseIntoCollection(r.as(Seq.class), (Collection)o, sType, pMeta);
476         } else if (isBag(r)) {
477            parseIntoCollection(r.as(Bag.class), (Collection)o, sType, pMeta);
478         } else if (r.canAs(RDFList.class)) {
479            parseIntoCollection(r.as(RDFList.class), (Collection)o, sType, pMeta);
480         } else {
481            throw new ParseException(this, "Unrecognized node type ''{0}'' for collection", n);
482         }
483         if (sType.isArray() || sType.isArgs())
484            o = toArray(sType, (Collection)o);
485      } else if (builder != null) {
486         Resource r = n.asResource();
487         if (! urisVisited.add(r))
488            return null;
489         BeanMap<?> bm = toBeanMap(builder.create(this, eType));
490         o = builder.build(this, parseIntoBeanMap(r, bm).getBean(), eType);
491      } else if (sType.canCreateNewBean(outer)) {
492         Resource r = n.asResource();
493         if (! urisVisited.add(r))
494            return null;
495         BeanMap<?> bm = newBeanMap(outer, sType.getInnerClass());
496         o = parseIntoBeanMap(r, bm).getBean();
497      } else if (sType.isUri() && n.isResource()) {
498         o = sType.newInstanceFromString(outer, decodeString(n.asResource().getURI()));
499      } else if (sType.canCreateNewInstanceFromString(outer)) {
500         o = sType.newInstanceFromString(outer, decodeString(getValue(n, outer)));
501      } else if (n.isResource()) {
502         Resource r = n.asResource();
503         Map m = newGenericMap(sType);
504         parseIntoMap(r, m, sType.getKeyType(), sType.getValueType(), pMeta);
505         if (m.containsKey(getBeanTypePropertyName(eType)))
506            o = cast((JsonMap)m, pMeta, eType);
507         else if (sType.getProxyInvocationHandler() != null)
508            o = newBeanMap(outer, sType.getInnerClass()).load(m).getBean();
509         else
510            throw new ParseException(this, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
511      } else {
512         throw new ParseException(this, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
513      }
514
515      if (swap != null && o != null)
516         o = unswap(swap, o, eType);
517
518      if (outer != null)
519         setParent(eType, o, outer);
520
521      return (T)o;
522   }

524   private boolean isSeq(RDFNode n) {
525      if (n.isResource()) {
526         Statement st = n.asResource().getProperty(pRdfType);
527         if (st != null)
528            return RDF_SEQ.equals(st.getResource().getURI());
529      }
530      return false;
531   }

533   private boolean isBag(RDFNode n) {
534      if (n.isResource()) {
535         Statement st = n.asResource().getProperty(pRdfType);
536         if (st != null)
537            return RDF_BAG.equals(st.getResource().getURI());
538      }
539      return false;
540   }

542   private Object getValue(RDFNode n, Object outer) throws ParseException {
543      if (n.isLiteral())
544         return n.asLiteral().getValue();
545      if (n.isResource()) {
546         Statement st = n.asResource().getProperty(pValue);
547         if (st != null) {
548            n = st.getObject();
549            if (n.isLiteral())
550               return n.asLiteral().getValue();
551            return parseAnything(object(), st.getObject(), outer, null);
552         }
553      }
554      throw new ParseException(this, "Unknown value type for node ''{0}''", n);
555   }

557   private <K,V> Map<K,V> parseIntoMap(Resource r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType, BeanPropertyMeta pMeta) throws ParseException {
558      // Add URI as "uri" to generic maps.
559      if (r.getURI() != null) {
560         K uri = convertAttrToType(m, "uri", keyType);
561         V value = convertAttrToType(m, r.getURI(), valueType);
562         m.put(uri, value);
563      }
564      for (StmtIterator i = r.listProperties(); i.hasNext();) {
565         Statement st = i.next();
566         Property p = st.getPredicate();
567         String key = p.getLocalName();
568         if (! (key.equals("root") && p.getURI().equals(getJuneauNs().getUri()))) {
569            key = decodeString(key);
570            RDFNode o = st.getObject();
571            K key2 = convertAttrToType(m, key, keyType);
572            V value = parseAnything(valueType, o, m, pMeta);
573            setName(valueType, value, key);
574            m.put(key2, value);
575         }
576
577      }
578      return m;
579   }

581   private <E> Collection<E> parseIntoCollection(Container c, Collection<E> l, ClassMeta<?> type, BeanPropertyMeta pMeta) throws ParseException {
582      int argIndex = 0;
583      for (NodeIterator ni = c.iterator(); ni.hasNext();) {
584         E e = (E)parseAnything(type.isArgs() ? type.getArg(argIndex++) : type.getElementType(), ni.next(), l, pMeta);
585         l.add(e);
586      }
587      return l;
588   }

590   private <E> Collection<E> parseIntoCollection(RDFList list, Collection<E> l, ClassMeta<?> type, BeanPropertyMeta pMeta) throws ParseException {
591      int argIndex = 0;
592      for (ExtendedIterator<RDFNode> ni = list.iterator(); ni.hasNext();) {
593         E e = (E)parseAnything(type.isArgs() ? type.getArg(argIndex++) : type.getElementType(), ni.next(), l, pMeta);
594         l.add(e);
595      }
596      return l;
597   }
