230   /**
231    * Constructor.
232    *
233    * @param builder The builder for this object.
234    */
235   protected RdfSerializerSession(Builder builder) {
236      super(builder);
237      ctx = builder.ctx;
238
239      namespaces = ctx.namespaces;
240      model = ModelFactory.createDefaultModel();
241      addModelPrefix(ctx.getJuneauNs());
242      addModelPrefix(ctx.getJuneauBpNs());
243      for (Namespace ns : this.namespaces)
244         addModelPrefix(ns);
245      pRoot = model.createProperty(ctx.getJuneauNs().getUri(), RDF_juneauNs_ROOT);
246      pValue = model.createProperty(ctx.getJuneauNs().getUri(), RDF_juneauNs_VALUE);
247      writer = model.getWriter(ctx.getLanguage());
248
249      // Only apply properties with this prefix!
250      String propPrefix = LANG_PROP_MAP.get(ctx.getLanguage());
251      if (propPrefix == null)
252         throw runtimeException("Unknown RDF language encountered: ''{0}''", ctx.getLanguage());
253
254      // RDF/XML specific properties.
255      if (propPrefix.equals("rdfXml.")) {
256         writer.setProperty("tab", isUseWhitespace() ? 2 : 0);
257         writer.setProperty("attributeQuoteChar", Character.toString(getQuoteChar()));
258      }
259
260      ctx.getJenaSettings().forEach((k,v) -> {
261         if (k.startsWith(propPrefix, 5))
262            writer.setProperty(k.substring(5 + propPrefix.length()), v);
263      });
264   }

273   /*
274    * XML-encodes the specified string using the {@link XmlUtils#escapeText(Object)} method.
275    */
276   private String encodeTextInvalidChars(Object o) {
277      if (o == null)
278         return null;
279      String s = toString(o);
280      return XmlUtils.escapeText(s);
281   }

290   @Override /* Serializer */
291   protected void doSerialize(SerializerPipe out, Object o) throws SerializeException {
292
293      Resource r = null;
294
295      ClassMeta<?> cm = getClassMetaForObject(o);
296      if (isLooseCollections() && cm != null && cm.isCollectionOrArray()) {
297         Collection c = cm.isCollection() ? (Collection)o : toList(cm.getInnerClass(), o);
298         forEachEntry(c, x -> serializeAnything(x, false, object(), "root", null, null));
299      } else {
300         RDFNode n = serializeAnything(o, false, getExpectedRootType(o), "root", null, null);
301         if (n.isLiteral()) {
302            r = model.createResource();
303            r.addProperty(pValue, n);
304         } else {
305            r = n.asResource();
306         }
307
308         if (isAddRootProp())
309            r.addProperty(pRoot, "true");
310      }
311
312      writer.write(model, out.getWriter(), "http://unknown/");
313   }

315   private RDFNode serializeAnything(Object o, boolean isURI, ClassMeta<?> eType, String attrName, BeanPropertyMeta bpm, Resource parentResource) throws SerializeException {
316      Model m = model;
317
318      ClassMeta<?> aType = null;       // The actual type
319      ClassMeta<?> wType = null;       // The wrapped type
320      ClassMeta<?> sType = object();   // The serialized type
321
322      aType = push2(attrName, o, eType);
323
324      if (eType == null)
325         eType = object();
326
327      // Handle recursion
328      if (aType == null) {
329         o = null;
330         aType = object();
331      }
332
333      // Handle Optional<X>
334      if (isOptional(aType)) {
335         o = getOptionalValue(o);
336         eType = getOptionalType(eType);
337         aType = getClassMetaForObject(o, object());
338      }
339
340      if (o != null) {
341
342         if (aType.isDelegate()) {
343            wType = aType;
344            aType = ((Delegate)o).getClassMeta();
345         }
346
347         sType = aType;
348
349         // Swap if necessary
350         ObjectSwap swap = aType.getSwap(this);
351         if (swap != null) {
352            o = swap(swap, o);
353            sType = swap.getSwapClassMeta(this);
354
355            // If the getSwapClass() method returns Object, we need to figure out
356            // the actual type now.
357            if (sType.isObject())
358               sType = getClassMetaForObject(o);
359         }
360      } else {
361         sType = eType.getSerializedClassMeta(this);
362      }
363
364      String typeName = getBeanTypeName(this, eType, aType, bpm);
365
366      RDFNode n = null;
367
368      if (o == null || sType.isChar() && ((Character)o).charValue() == 0) {
369         if (bpm != null) {
370            if (isKeepNullProperties()) {
371               n = m.createResource(RDF_NIL);
372            }
373         } else {
374            n = m.createResource(RDF_NIL);
375         }
376
377      } else if (sType.isUri() || isURI) {
378         // Note that RDF URIs must be absolute to be valid!
379         String uri = getUri(o, null);
380         if (StringUtils.isAbsoluteUri(uri))
381            n = m.createResource(uri);
382         else
383            n = m.createLiteral(encodeTextInvalidChars(uri));
384
385      } else if (sType.isCharSequence() || sType.isChar()) {
386         n = m.createLiteral(encodeTextInvalidChars(o));
387
388      } else if (sType.isNumber() || sType.isBoolean()) {
389         if (! isAddLiteralTypes())
390            n = m.createLiteral(o.toString());
391         else
392            n = m.createTypedLiteral(o);
393
394      } else if (sType.isMap() || (wType != null && wType.isMap())) {
395         if (o instanceof BeanMap) {
396            BeanMap bm = (BeanMap)o;
397            Object uri = null;
398            RdfBeanMeta rbm = getRdfBeanMeta(bm.getMeta());
399            if (rbm.hasBeanUri())
400               uri = rbm.getBeanUriProperty().get(bm, null);
401            String uri2 = getUri(uri, null);
402            n = m.createResource(uri2);
403            serializeBeanMap(bm, (Resource)n, typeName);
404         } else {
405            Map m2 = (Map)o;
406            n = m.createResource();
407            serializeMap(m2, (Resource)n, sType);
408         }
409
410      } else if (sType.isBean()) {
411         BeanMap bm = toBeanMap(o);
412         Object uri = null;
413         RdfBeanMeta rbm = getRdfBeanMeta(bm.getMeta());
414         if (rbm.hasBeanUri())
415            uri = rbm.getBeanUriProperty().get(bm, null);
416         String uri2 = getUri(uri, null);
417         n = m.createResource(uri2);
418         serializeBeanMap(bm, (Resource)n, typeName);
419
420      } else if (sType.isCollectionOrArray() || (wType != null && wType.isCollection())) {
421
422         Collection c = sort(sType.isCollection() ? (Collection)o : toList(sType.getInnerClass(), o));
423         RdfCollectionFormat f = getCollectionFormat();
424         RdfClassMeta cRdf = getRdfClassMeta(sType);
425         RdfBeanPropertyMeta bpRdf = getRdfBeanPropertyMeta(bpm);
426
427         if (cRdf.getCollectionFormat() != RdfCollectionFormat.DEFAULT)
428            f = cRdf.getCollectionFormat();
429         if (bpRdf.getCollectionFormat() != RdfCollectionFormat.DEFAULT)
430            f = bpRdf.getCollectionFormat();
431
432         switch (f) {
433            case BAG: n = serializeToContainer(c, eType, m.createBag()); break;
434            case LIST: n = serializeToList(c, eType); break;
435            case MULTI_VALUED: serializeToMultiProperties(c, eType, bpm, attrName, parentResource); break;
436            default: n = serializeToContainer(c, eType, m.createSeq());
437         }
438
439      } else if (sType.isReader()) {
440         n = m.createLiteral(encodeTextInvalidChars(read((Reader)o, SerializerSession::handleThrown)));
441      } else if (sType.isInputStream()) {
442         n = m.createLiteral(encodeTextInvalidChars(read((InputStream)o, SerializerSession::handleThrown)));
443
444      } else {
445         n = m.createLiteral(encodeTextInvalidChars(toString(o)));
446      }
447
448      pop();
449
450      return n;
451   }

453   private String getUri(Object uri, Object uri2) {
454      String s = null;
455      if (uri != null)
456         s = uri.toString();
457      if ((s == null || s.isEmpty()) && uri2 != null)
458         s = uri2.toString();
459      if (s == null)
460         return null;
461      return getUriResolver().resolve(s);
462   }

464   private void serializeMap(Map m, Resource r, ClassMeta<?> type) throws SerializeException {
465
466      m = sort(m);
467
468      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
469
470      ArrayList<Map.Entry<Object,Object>> l = listFrom(m.entrySet());
471      Collections.reverse(l);
472      l.forEach(x -> {
473         Object value = x.getValue();
474         Object key = generalize(x.getKey(), keyType);
475         Namespace ns = getJuneauBpNs();
476         Property p = model.createProperty(ns.getUri(), encodeElementName(toString(key)));
477         RDFNode n = serializeAnything(value, false, valueType, toString(key), null, r);
478         if (n != null)
479            r.addProperty(p, n);
480      });
481   }

483   private void serializeBeanMap(BeanMap<?> m, Resource r, String typeName) throws SerializeException {
484      List<BeanPropertyValue> l = new ArrayList<>();
485
486      if (typeName != null) {
487         BeanPropertyMeta pm = m.getMeta().getTypeProperty();
488         l.add(new BeanPropertyValue(pm, pm.getName(), typeName, null));
489      }
490
491      Predicate<Object> checkNull = x -> isKeepNullProperties() || x != null;
492      m.forEachValue(checkNull, (pMeta,key,value,thrown) -> {
493         l.add(new BeanPropertyValue(pMeta, key, value, thrown));
494      });
495
496      Collections.reverse(l);
497      l.forEach(x -> {
498         BeanPropertyMeta bpMeta = x.getMeta();
499         ClassMeta<?> cMeta = bpMeta.getClassMeta();
500         RdfBeanPropertyMeta bpRdf = getRdfBeanPropertyMeta(bpMeta);
501         XmlBeanPropertyMeta bpXml = getXmlBeanPropertyMeta(bpMeta);
502
503         if (bpRdf.isBeanUri())
504            return;
505
506         String key = x.getName();
507         Object value = x.getValue();
508         Throwable t = x.getThrown();
509         if (t != null)
510            onBeanGetterException(bpMeta, t);
511
512         if (canIgnoreValue(cMeta, key, value))
513            return;
514
515         Namespace ns = bpRdf.getNamespace();
516         if (ns == null && isUseXmlNamespaces())
517            ns = bpXml.getNamespace();
518         if (ns == null)
519            ns = getJuneauBpNs();
520         else if (isAutoDetectNamespaces())
521            addModelPrefix(ns);
522
523         Property p = model.createProperty(ns.getUri(), encodeElementName(key));
524         RDFNode n = serializeAnything(value, bpMeta.isUri(), cMeta, key, bpMeta, r);
525         if (n != null)
526            r.addProperty(p, n);
527      });
528   }

531   private Container serializeToContainer(Collection c, ClassMeta<?> type, Container list) throws SerializeException {
532      ClassMeta<?> elementType = type.getElementType();
533      c.forEach(x -> list.add(serializeAnything(x, false, elementType, null, null, null)));
534      return list;
535   }
536
537   private RDFList serializeToList(Collection c, ClassMeta<?> type) throws SerializeException {
538      ClassMeta<?> elementType = type.getElementType();
539      List<RDFNode> l = list(c.size());
540      c.forEach(x -> l.add(serializeAnything(x, false, elementType, null, null, null)));
541      return model.createList(l.iterator());
542   }
543
544   private void serializeToMultiProperties(Collection c, ClassMeta<?> sType,
545         BeanPropertyMeta bpm, String attrName, Resource parentResource) throws SerializeException {
546
547      ClassMeta<?> elementType = sType.getElementType();
548      RdfBeanPropertyMeta bpRdf = getRdfBeanPropertyMeta(bpm);
549      XmlBeanPropertyMeta bpXml = getXmlBeanPropertyMeta(bpm);
550
551      c.forEach(x -> {
552         Namespace ns = bpRdf.getNamespace();
553         if (ns == null && isUseXmlNamespaces())
554            ns = bpXml.getNamespace();
555         if (ns == null)
556            ns = getJuneauBpNs();
557         else if (isAutoDetectNamespaces())
558            addModelPrefix(ns);
559         RDFNode n2 = serializeAnything(x, false, elementType, null, null, null);
560         Property p = model.createProperty(ns.getUri(), encodeElementName(attrName));
561         parentResource.addProperty(p, n2);
562      });
563   }
