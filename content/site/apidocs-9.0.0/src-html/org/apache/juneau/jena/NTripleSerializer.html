013package org.apache.juneau.jena;
014
015/**
016 * Subclass of {@link RdfSerializer} for serializing RDF in N-Triple notation.
017 *
018 * <ul class='notes'>
019 *    <li class='note'>This class is thread safe and reusable.
020 * </ul>
021 *
022 * <ul class='seealso'>
023 *    <li class='link'>{@doc jmr.RdfDetails}
024 *    <li class='extlink'>{@source}
025 * </ul>
026 */
027public class NTripleSerializer extends RdfSerializer {
028
029   //-------------------------------------------------------------------------------------------------------------------
030   // Static
031   //-------------------------------------------------------------------------------------------------------------------
032
033   /** Default N-Triple serializer, all default settings.*/
034   public static final NTripleSerializer DEFAULT = new NTripleSerializer(create());
035
036   /**
037    * Creates a new builder for this object.
038    *
039    * @return A new builder.
040    */
041   public static RdfSerializer.Builder create() {
042      return RdfSerializer.create().ntriple();
043   }
044
045   //-------------------------------------------------------------------------------------------------------------------
046   // Instance
047   //-------------------------------------------------------------------------------------------------------------------
048
049   /**
050    * Constructor.
051    *
052    * @param builder The builder for this object.
053    */
054   public NTripleSerializer(RdfSerializer.Builder builder) {
055      super(builder.ntriple());
056   }
057}
