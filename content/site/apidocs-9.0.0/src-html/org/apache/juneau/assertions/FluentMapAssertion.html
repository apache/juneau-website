106@FluentSetters(returns="FluentMapAssertion<K,V,R>")
107public class FluentMapAssertion<K,V,R> extends FluentObjectAssertion<Map<K,V>,R>  {

109   //-----------------------------------------------------------------------------------------------------------------
110   // Static
111   //-----------------------------------------------------------------------------------------------------------------
112
113   private static final Messages MESSAGES = Messages.of(FluentMapAssertion.class, "Messages");
114   private static final String
115      MSG_mapWasNotEmpty = MESSAGES.getString("mapWasNotEmpty"),
116      MSG_mapDidNotContainExpectedKey = MESSAGES.getString("mapDidNotContainExpectedKey"),
117      MSG_mapContainedUnexpectedKey = MESSAGES.getString("mapContainedUnexpectedKey"),
118      MSG_mapWasEmpty = MESSAGES.getString("mapWasEmpty"),
119      MSG_mapDidNotHaveTheExpectedSize = MESSAGES.getString("mapDidNotHaveTheExpectedSize");
120

125   /**
126    * Constructor.
127    *
128    * @param value
129    *    The object being tested.
130    *    <br>Can be <jk>null</jk>.
131    * @param returns
132    *    The object to return after a test method is called.
133    *    <br>If <jk>null</jk>, the test method returns this object allowing multiple test method calls to be
134    * used on the same assertion.
135    */
136   public FluentMapAssertion(Map<K,V> value, R returns) {
137      this(null, value, returns);
138   }
139
140   /**
141    * Chained constructor.
142    *
143    * <p>
144    * Used when transforming one assertion into another so that the assertion config can be used by the new assertion.
145    *
146    * @param creator
147    *    The assertion that created this assertion.
148    *    <br>Should be <jk>null</jk> if this is the top-level assertion.
149    * @param value
150    *    The object being tested.
151    *    <br>Can be <jk>null</jk>.
152    * @param returns
153    *    The object to return after a test method is called.
154    *    <br>If <jk>null</jk>, the test method returns this object allowing multiple test method calls to be
155    * used on the same assertion.
156    */
157   public FluentMapAssertion(Assertion creator, Map<K,V> value, R returns) {
158      super(creator, value, returns);
159   }
160
161   //-----------------------------------------------------------------------------------------------------------------
162   // Transform methods
163   //-----------------------------------------------------------------------------------------------------------------
164
165   @Override /* FluentObjectAssertion */
166   public FluentMapAssertion<K,V,R> asTransformed(Function<Map<K,V>,Map<K,V>> function) {
167      return new FluentMapAssertion<>(this, function.apply(orElse(null)), returns());
168   }
169
170   /**
171    * Returns an object assertion on the value specified at the specified key.
172    *
173    * <p>
174    * If the map is <jk>null</jk> or the map doesn't contain the specified key, the returned assertion is a null assertion
175    * (meaning {@link FluentAnyAssertion#isExists()} returns <jk>false</jk>).
176    *
177    * @param key The key of the item to retrieve from the map.
178    * @return A new assertion.
179    */
180   public FluentAnyAssertion<V,R> asValue(K key) {
181      return new FluentAnyAssertion<>(this, get(key), returns());
182   }
183
184   /**
185    * Returns a {@link FluentListAssertion} of the values of the specified keys.
186    *
187    * If the map is <jk>null</jk>, the returned assertion is a null assertion
188    * (meaning {@link FluentObjectAssertion#isExists()} returns <jk>false</jk>).
189    *
190    * @param keys The keys of the values to retrieve from the map.
191    * @return A new assertion.
192    */
193   public FluentListAssertion<Object,R> asValues(@SuppressWarnings("unchecked") K...keys) {
194      return new FluentListAssertion<>(this, valueIsNull() ? null : stream(keys).map(x -> get(x)).collect(toList()), returns());
195   }
196
197   /**
198    * Extracts a subset of this map.
199    *
200    * @param keys The entries to extract.
201    * @return This object.
202    */
203   @SuppressWarnings("unchecked")
204   public FluentMapAssertion<K,V,R> asValueMap(K...keys) {
205      if (valueIsNull())
206         return new FluentMapAssertion<>(this, null, returns());
207      Map<K,V> m1 = value(), m2 = CollectionUtils.map();
208      if (m1 != null)
209         for (K k : keys)
210            m2.put(k, m1.get(k));
211      return new FluentMapAssertion<>(this, m2, returns());
212   }
213
214   /**
215    * Returns an integer assertion on the size of this map.
216    *
217    * <p>
218    * If the map is <jk>null</jk>, the returned assertion is a null assertion
219    * (meaning {@link FluentIntegerAssertion#isExists()} returns <jk>false</jk>).
220    *
221    * @return A new assertion.
222    */
223   public FluentIntegerAssertion<R> asSize() {
224      return new FluentIntegerAssertion<>(this, valueIsNull() ? null : value().size(), returns());
225   }

227   //-----------------------------------------------------------------------------------------------------------------
228   // Test methods
229   //-----------------------------------------------------------------------------------------------------------------
230
231   /**
232    * Asserts that the map exists and is empty.
233    *
234    * @return The fluent return object.
235    * @throws AssertionError If assertion failed or value was <jk>null</jk>.
236    */
237   public R isEmpty() throws AssertionError {
238      if (! value().isEmpty())
239         throw error(MSG_mapWasNotEmpty);
240      return returns();
241   }
242
243   /**
244    * Asserts that the map exists and is not empty.
245    *
246    * @return The fluent return object.
247    * @throws AssertionError If assertion failed or value was <jk>null</jk>.
248    */
249   public R isNotEmpty() throws AssertionError {
250      if (value().isEmpty())
251         throw error(MSG_mapWasEmpty);
252      return returns();
253   }
254
255   /**
256    * Asserts that the map contains the expected key.
257    *
258    * @param name The key name to check for.
259    * @return The fluent return object.
260    * @throws AssertionError If assertion failed or value was <jk>null</jk>.
261    */
262   public R isContainsKey(String name) throws AssertionError {
263      if (value().containsKey(name))
264         return returns();
265      throw error(MSG_mapDidNotContainExpectedKey, name, value());
266   }
267
268   /**
269    * Asserts that the map contains the expected key.
270    *
271    * @param name The key name to check for.
272    * @return The fluent return object.
273    * @throws AssertionError If assertion failed or value was <jk>null</jk>.
274    */
275   public R isNotContainsKey(String name) throws AssertionError {
276      if (! value().containsKey(name))
277         return returns();
278      throw error(MSG_mapContainedUnexpectedKey, name, value());
279   }
280
281   /**
282    * Asserts that the map exists and is the specified size.
283    *
284    * @param size The expected size.
285    * @return The fluent return object.
286    * @throws AssertionError If assertion failed or value was <jk>null</jk>.
287    */
288   public R isSize(int size) throws AssertionError {
289      if (size2() != size)
290         throw error(MSG_mapDidNotHaveTheExpectedSize, size, size2());
291      return returns();
292   }
