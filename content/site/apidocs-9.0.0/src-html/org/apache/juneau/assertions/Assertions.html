026/**
027 * Main class for creation of assertions for stand-alone testing.
028 *
029 * <p>
030 * Provides assertions for various common POJO types.
031 *
032 * <h5 class='section'>Example:</h5>
033 * <p class='bjava'>
034 *    <jk>import static</jk> org.apache.juneau.assertions.Assertions.*;
035 *
036 * <jc>// Assert string is greater than 100 characters and contains "foo".</jc>
037 *    <jsm>assertString</jsm>(<jv>myString</jv>)
038 *       .length().isGt(100)
039 *       .contains(<js>"foo"</js>);
040 * </p>
041 *
042 * <p>
043 * Provides simple testing that {@link Throwable Throwables} are being thrown correctly.
044 *
045 * <h5 class='section'>Example:</h5>
046 * <p class='bjava'>
047 *    <jk>import static</jk> org.apache.juneau.assertions.Assertions.*;
048 *
049 * <jc>// Assert that calling doBadCall() causes a RuntimeException.</jc>
050 *    <jsm>assertThrown</jsm>(() -&gt; <jv>myPojo</jv>.doBadCall())
051 *       .isType(RuntimeException.<jk>class</jk>)
052 *       .message().contains(<js>"Bad thing happened."</js>);
053 * </p>
054 *
055 * <p>
056 * Provides other assertion convenience methods such as asserting non-null method arguments.
057 *
058 * <h5 class='section'>Example:</h5>
059 * <p class='bjava'>
060 *    <jk>import static</jk> org.apache.juneau.assertions.Assertions.*;
061 *
062 * <jk>public</jk> String getFoo(String <jv>bar</jv>) {
063 *    <jsm>assertArgNotNull</jsm>(<js>"bar"</js>, <jv>bar</jv>);
064 *    ...
065 * }
066 * </p>
067 *
068 * <ul class='seealso'>
069 *    <li class='link'>{@doc jm.FluentAssertions}
070 *    <li class='extlink'>{@source}
071 * </ul>
072 */

083   /**
084    * Performs an assertion on an arbitrary POJO.
085    *
086    * <p>
087    * The distinction between {@link ObjectAssertion} and {@link AnyAssertion} is that the latter supports all
088    * the operations of the former, but adds various transform methods for conversion to specific assertion types.
089    *
090    * <p>
091    * Various transform methods such as {@link FluentListAssertion#asItem(int)} and {@link FluentBeanAssertion#asProperty(String)}
092    * return generic any-assertions so that they can be easily transformed into other assertion types.
093    *
094    * <h5 class='section'>Example:</h5>
095    * <p class='bjava'>
096    *    <jk>import static</jk> org.apache.juneau.assertions.Assertions.*;
097    *
098    *    <jc>// Asserts that the property 'foo' of a bean is 'bar'.</jc>
099    *    <jsm>assertAny</jsm>(<jv>myPojo</jv>)  <jc>// Start with AnyAssertion.</jc>
100    *       .asBean(MyBean.<jk>class</jk>)  <jc>// Transform to BeanAssertion.</jc>
101    *          .property(<js>"foo"</js>).is(<js>"bar"</js>);
102    * </p>
103    *
104    * <p>
105    * See {@doc jm.FluentAssertions Fluent Assertions} for general assertion usage and {@link AnyAssertion} for supported operations on this type.
106    *
107    * @param <T> The value type.
108    * @param value
109    *    The object being tested.
110    *    <br>Can be <jk>null</jk>.
111    * @return
112    *    A new assertion object.
113    *    <br>Never <jk>null</jk>.
114    */
115   public static final <T> AnyAssertion<T> assertAny(T value) {
116      return AnyAssertion.create(value);
117   }

1038   /**
1039    * Throws an {@link IllegalArgumentException} if the specified value doesn't have all subclasses of the specified type.
1040    *
1041    * @param <E> The element type.
1042    * @param name The argument name.
1043    * @param type The expected parent class.
1044    * @param value The array value being checked.
1045    * @return The value cast to the specified array type.
1046    * @throws IllegalArgumentException Constructed exception.
1047    */
1048   @SuppressWarnings("unchecked")
1049   public static final <E> Class<E>[] assertClassArrayArgIsType(String name, Class<E> type, Class<?>[] value) throws IllegalArgumentException {
1050      for (int i = 0; i < value.length; i++)
1051         if (! type.isAssignableFrom(value[i]))
1052            throw illegalArgumentException("Arg {0} did not have arg of type {1} at index {2}: {3}", name, type.getName(), i, value[i].getName());
1053      return (Class<E>[])value;
1054   }
