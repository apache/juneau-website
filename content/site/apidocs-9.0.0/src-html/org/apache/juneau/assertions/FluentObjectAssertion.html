030/**
031 * Used for fluent assertion calls against POJOs.
032 *
033 * <h5 class='topic'>Test Methods</h5>
034 * <p>
035 * <ul class='javatree'>
036 *    <li class='jc'>{@link FluentObjectAssertion}
037 *    <ul class='javatreec'>
038 *       <li class='jm'>{@link FluentObjectAssertion#isExists() isExists()}
039 *       <li class='jm'>{@link FluentObjectAssertion#is(Object) is(Object)}
040 *       <li class='jm'>{@link FluentObjectAssertion#is(Predicate) is(Predicate)}
041 *       <li class='jm'>{@link FluentObjectAssertion#isNot(Object) isNot(Object)}
042 *       <li class='jm'>{@link FluentObjectAssertion#isAny(Object...) isAny(Object...)}
043 *       <li class='jm'>{@link FluentObjectAssertion#isNotAny(Object...) isNotAny(Object...)}
044 *       <li class='jm'>{@link FluentObjectAssertion#isNull() isNull()}
045 *       <li class='jm'>{@link FluentObjectAssertion#isNotNull() isNotNull()}
046 *       <li class='jm'>{@link FluentObjectAssertion#isString(String) isString(String)}
047 *       <li class='jm'>{@link FluentObjectAssertion#isJson(String) isJson(String)}
048 *       <li class='jm'>{@link FluentObjectAssertion#isSame(Object) isSame(Object)}
049 *       <li class='jm'>{@link FluentObjectAssertion#isSameJsonAs(Object) isSameJsonAs(Object)}
050 *       <li class='jm'>{@link FluentObjectAssertion#isSameSortedJsonAs(Object) isSameSortedJsonAs(Object)}
051 *       <li class='jm'>{@link FluentObjectAssertion#isSameSerializedAs(Object, WriterSerializer) isSameSerializedAs(Object, WriterSerializer)}
052 *       <li class='jm'>{@link FluentObjectAssertion#isType(Class) isType(Class)}

131    * Constructor.

136    * @param returns

145   /**
146    * Chained constructor.
147    *
148    * <p>
149    * Used when transforming one assertion into another so that the assertion config can be used by the new assertion.
150    *
151    * @param creator
152    *    The assertion that created this assertion.
153    *    <br>Should be <jk>null</jk> if this is the top-level assertion.
154    * @param value
155    *    The object being tested.
156    *    <br>Can be <jk>null</jk>.
157    * @param returns
158    *    The object to return after a test method is called.
159    *    <br>If <jk>null</jk>, the test method returns this object allowing multiple test method calls to be
160    * used on the same assertion.
161    */
162   public FluentObjectAssertion(Assertion creator, T value, R returns) {
163      super(creator, returns);
164      this.value = value;
165   }

171   /**
172    * Converts this object to a string using {@link Object#toString} and returns it as a new assertion.
173    *
174    * <h5 class='section'>Example:</h5>
175    * <p class='bjava'>
176    *    <jc>// Validates that the specified object is "foobar" after converting to a string.</jc>
177    *    <jsm>assertObject</jsm>(<jv>myPojo</jv>)
178    *       .asString()
179    *       .is(<js>"foobar"</js>);
180    * </p>
181    *
182    * @return A new fluent string assertion.
183    */
184   public FluentStringAssertion<R> asString() {
185      return new FluentStringAssertion<>(this, valueAsString(), returns());
186   }
187
188   /**
189    * Converts this object to text using the specified serializer and returns it as a new assertion.
190    *
191    * <h5 class='section'>Example:</h5>
192    * <p class='bjava'>
193    *    <jc>// Validates that the specified object is an instance of MyBean.</jc>
194    *    <jsm>assertObject</jsm>(<jv>myPojo</jv>)
195    *       .asString(XmlSerializer.<jsf>DEFAULT</jsf>)
196    *       .is(<js>"&lt;object&gt;&lt;foo&gt;bar&lt;/foo&gt;&lt;baz&gt;qux&lt;/baz&gt;&lt;/object&gt;"</js>);
197    * </p>
198    *
199    * @param ws The serializer to use to convert the object to text.
200    * @return A new fluent string assertion.
201    */
202   public FluentStringAssertion<R> asString(WriterSerializer ws) {
203      try {
204         return new FluentStringAssertion<>(this, ws.serialize(value), returns());
205      } catch (SerializeException e) {
206         throw runtimeException(e);
207      }
208   }

296   /**
297    * Asserts that the object is not null.
298    *
299    * <p>
300    * Equivalent to {@link #isNotNull()}.
301    *
302    * @return The fluent return object.
303    * @throws AssertionError If assertion failed.
304    */
305   public R isExists() throws AssertionError {
306      return isNotNull();
307   }
308
309   /**
310    * Asserts that the object i null.
311    *
312    * <p>
313    * Equivalent to {@link #isNotNull()}.
314    *
315    * @return The fluent return object.
316    * @throws AssertionError If assertion failed.
317    */
318   public R isNull() throws AssertionError {
319      if (value != null)
320         throw error(MSG_valueWasNotNull);
321      return returns();
322   }
323
324   /**
325    * Asserts that the object is not null.
326    *
327    * <p>
328    * Equivalent to {@link #isNotNull()}.
329    *
330    * @return The fluent return object.
331    * @throws AssertionError If assertion failed.
332    */
333   public R isNotNull() throws AssertionError {
334      if (value == null)
335         throw error(MSG_valueWasNull);
336      return returns();
337   }
338
339   /**
340    * Asserts that the value equals the specified value.
341    *
342    * @param value The value to check against.
343    * @return The fluent return object.
344    * @throws AssertionError If assertion failed.
345    */
346   public R is(T value) throws AssertionError {
347      if (this.value == value)
348         return returns();
349      if (! equals(orElse(null), value))
350         throw error(MSG_unexpectedValue, value, this.value);
351      return returns();
352   }
353
354   /**
355    * Asserts that the value converted to a string equals the specified value.
356    *
357    * @param value The value to check against.
358    * @return The fluent return object.
359    * @throws AssertionError If assertion failed.
360    */
361   public R isString(String value) {
362      return asString().is(value);
363   }
364
365   /**
366    * Asserts that the value does not equal the specified value.
367    *
368    * @param value The value to check against.
369    * @return The fluent return object.
370    * @throws AssertionError If assertion failed.
371    */
372   public R isNot(T value) throws AssertionError {
373      if (equals(orElse(null), value))
374         throw error(MSG_unexpectedValueDidNotExpect, value, orElse(null));
375      return returns();
376   }
377
378   /**
379    * Asserts that the value is one of the specified values.
380    *
381    * @param values The values to check against.
382    * @return The fluent return object.
383    * @throws AssertionError If assertion failed.
384    */
385   @SuppressWarnings("unchecked")
386   public R isAny(T...values) throws AssertionError {
387      for (T v : values)
388         if (equals(orElse(null), v))
389            return returns();
390      throw error(MSG_expectedValueNotFound, values, value);
391   }
392
393   /**
394    * Asserts that the value is not one of the specified values.
395    *
396    * @param values The values to check against.
397    * @return The fluent return object.
398    * @throws AssertionError If assertion failed.
399    */
400   @SuppressWarnings("unchecked")
401   public R isNotAny(T...values) throws AssertionError {
402      for (T v : values)
403         if (equals(orElse(null), v))
404            throw error(MSG_unexpectedValueFound, v, value);
405      return returns();
406   }
407
408   /**
409    * Asserts that the specified object is the same object as this object.
410    *
411    * @param value The value to check against.
412    * @return The fluent return object.
413    * @throws AssertionError If assertion failed.
414    */
415   public R isSame(T value) throws AssertionError {
416      if (this.value == value)
417         return returns();
418      throw error(MSG_notTheSameValue, value, ObjectUtils.identity(value), this.value, ObjectUtils.identity(this.value));
419   }
420
421   /**
422    * Verifies that two objects are equivalent after converting them both to JSON.
423    *
424    * @param o The object to compare against.
425    * @return The fluent return object.
426    * @throws AssertionError If assertion failed.
427    */
428   public R isSameJsonAs(Object o) throws AssertionError {
429      return isSameSerializedAs(o, JSON);
430   }
431
432   /**
433    * Verifies that two objects are equivalent after converting them both to sorted JSON.
434    *
435    * <p>
436    * Properties, maps, and collections are all sorted on both objects before comparison.
437    *
438    * @param o The object to compare against.
439    * @return The fluent return object.
440    * @throws AssertionError If assertion failed.
441    */
442   public R isSameSortedJsonAs(Object o) {
443      return isSameSerializedAs(o, JSON_SORTED);
444   }
445
446   /**
447    * Asserts that the specified object is the same as this object after converting both to strings using the specified serializer.
448    *
449    * @param o The object to compare against.
450    * @param serializer The serializer to use to serialize this object.
451    * @return The fluent return object.
452    * @throws AssertionError If assertion failed.
453    */
454   public R isSameSerializedAs(Object o, WriterSerializer serializer) {
455      String s1 = serializer.toString(value);
456      String s2 = serializer.toString(o);
457      if (ne(s1, s2))
458         throw error(MSG_unexpectedComparison, s2, s1);
459      return returns();
460   }
461
462   /**
463    * Asserts that the object is an instance of the specified class.
464    *
465    * <h5 class='section'>Example:</h5>
466    * <p class='bjava'>
467    *    <jc>// Validates that the specified object is an instance of MyBean.</jc>
468    *    <jsm>assertObject</jsm>(<jv>myPojo</jv>).isType(MyBean.<jk>class</jk>);
469    * </p>
470    *
471    * @param parent The value to check against.
472    * @return The fluent return object.
473    * @throws AssertionError If assertion failed.
474    */
475   public R isType(Class<?> parent) throws AssertionError {
476      assertArgNotNull("parent", parent);
477      if (! ClassInfo.of(value()).isChildOf(parent))
478         throw error(MSG_unexpectedType, className(parent), className(value));
479      return returns();
480   }
481
482   /**
483    * Asserts that the object is an instance of the specified class.
484    *
485    * <h5 class='section'>Example:</h5>
486    * <p class='bjava'>
487    *    <jc>// Validates that the specified object is an instance of MyBean.</jc>
488    *    <jsm>assertObject</jsm>(<jv>myPojo</jv>).isExactType(MyBean.<jk>class</jk>);
489    * </p>
490    *
491    * @param type The value to check against.
492    * @return The fluent return object.
493    * @throws AssertionError If assertion failed.
494    */
495   public R isExactType(Class<?> type) throws AssertionError {
496      assertArgNotNull("parent", type);
497      if (value().getClass() != type)
498         throw error(MSG_unexpectedType, className(type), className(value));
499      return returns();
500   }
501
502   /**
503    * Asserts that the value passes the specified predicate test.
504    *
505    * @param test The predicate to use to test the value.
506    * @return The fluent return object.
507    * @throws AssertionError If assertion failed.
508    */
509   public R is(Predicate<T> test) throws AssertionError {
510      if (test != null && ! test.test(value))
511         throw error(getFailureMessage(test, value));
512      return returns();
513   }

524    * </p>
525    *
526    * @param value The expected string value.
527    * @return The fluent return object.
528    */
529   public R isJson(String value) {
530      return asJson().is(value);
531   }

643   /**
644    * Returns the predicate failure message.
645    *
646    * <p>
647    * If the predicate extends from {@link AssertionPredicate}, then the message comes from {@link AssertionPredicate#getFailureMessage()}.
648    * Otherwise, returns a generic <js>"Unexpected value: x"</js> message.
649    *
650    * @param p The function to run against the value.
651    * @param value The value that failed the test.
652    * @return The result, never <jk>null</jk>.
653    */
654   protected String getFailureMessage(Predicate<?> p, Object value) {
655      if (p instanceof AssertionPredicate)
