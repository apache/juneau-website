013package org.apache.juneau.assertions;
014
015import java.io.*;
016
017import org.apache.juneau.internal.*;
018
019/**
020 * Parent class of all fluent assertion calls.
021 *
022 * <p>
023 * Defines a {@link #returns()} method that returns an original object.
024 * Assertion test methods that pass use this method to return to the origin of the call.
025 *
026 * <h5 class='figure'>Example:</h5>
027 * <p class='bjava'>
028 *    <jc>// Create a basic REST client with JSON support and download a bean.</jc>
029 *    MyPojo <jv>myPojo</jv> = ...;
030 *    MyTestedBean <jv>myTestedBean</jv> = ...;
031 *
032 *    Assertion <jv>assertion</jv> = <jk>new</jk> FluentBeanAssertion&lt;MyPojo,MyTestedBean&gt;(<jv>myPojo</jv>, <jv>myTestedBean</jv>);
033 *    <jv>myPojo</jv> = <jv>assertion</jv>.test(<jv>x</jv> -&gt; <jv>x</jv>.getMyProperty().equals(<js>"foo"</js>));  <jc>// Returns myPojo after test.</jc>
034 * </p>
035 * <p>
036 * For subclasses such as {@link IntegerAssertion}, the return object is simply itself so that multiple tests
037 * can be performed using the same assertion.
038 * <h5 class='figure'>Example:</h5>
039 * <p class='bjava'>
040 *    <jk>public class</jk> IntegerAssertion <jk>extends</jk> FluentIntegerAssertion&lt;IntegerAssertion&gt; {
041 *    ...
042 *    }
043 *
044 *    Assertion <jv>assertion</jv> = <jk>new</jk> IntegerAssertion(123);
045 *    <jv>assertion</jv>
046 *       .isNotNull()
047 *       .isGt(100)
048 *  ;
049 * </p>
050 *
051 *
052 * <h5 class='topic'>Test Methods</h5>
053 * <p>
054 * <ul class='javatree'>
055 *    <li>None
056 * </ul>
057  *
058 * <h5 class='topic'>Transform Methods</h5>
059 * <p>
060 * <ul class='javatree'>
061 *    <li>None
062 * </ul>
063 *
064 * <h5 class='topic'>Configuration Methods</h5>
065 * <p>
066 * <ul class='javatree'>
067 *    <li class='jc'>{@link Assertion}
068 *    <ul class='javatreec'>
069 *       <li class='jm'>{@link Assertion#setMsg(String, Object...) setMsg(String, Object...)}
070 *       <li class='jm'>{@link Assertion#setOut(PrintStream) setOut(PrintStream)}
071 *       <li class='jm'>{@link Assertion#setSilent() setSilent()}
072 *       <li class='jm'>{@link Assertion#setStdOut() setStdOut()}
073 *       <li class='jm'>{@link Assertion#setThrowable(Class) setThrowable(Class)}
074 *    </ul>
075 * </ul>
076 *
077 * <ul class='seealso'>
078 *    <li class='link'>{@doc ja.Overview}
079 *    <li class='extlink'>{@source}
080 * </ul>
081 *
082 * @param <R> The return type.
083 */
084@FluentSetters(returns="FluentAssertion<R>")
085public abstract class FluentAssertion<R> extends Assertion {
086
087   //-----------------------------------------------------------------------------------------------------------------
088   // Instance
089   //-----------------------------------------------------------------------------------------------------------------
090
091   private final R returns;
092
093   /**
094    * Constructor.
095    *
096    * @param creator
097    *    The assertion that created this assertion.
098    *    <br>Should be <jk>null</jk> if this is the top-level assertion.
099    * @param returns
100    *    The object to return after a test method is called.
101    *    <br>If <jk>null</jk>, the test method returns this object allowing multiple test method calls to be
102    * used on the same assertion.
103    */
104   protected FluentAssertion(Assertion creator, R returns) {
105      super(creator);
106      this.returns = returns;
107   }
108
109   //-----------------------------------------------------------------------------------------------------------------
110   // Fluent setters
111   //-----------------------------------------------------------------------------------------------------------------
112
113   // <FluentSetters>
114
115   @Override /* GENERATED - org.apache.juneau.assertions.Assertion */
116   public FluentAssertion<R> setMsg(String msg, Object...args) {
117      super.setMsg(msg, args);
118      return this;
119   }
120
121   @Override /* GENERATED - org.apache.juneau.assertions.Assertion */
122   public FluentAssertion<R> setOut(PrintStream value) {
123      super.setOut(value);
124      return this;
125   }
126
127   @Override /* GENERATED - org.apache.juneau.assertions.Assertion */
128   public FluentAssertion<R> setSilent() {
129      super.setSilent();
130      return this;
131   }
132
133   @Override /* GENERATED - org.apache.juneau.assertions.Assertion */
134   public FluentAssertion<R> setStdOut() {
135      super.setStdOut();
136      return this;
137   }
138
139   @Override /* GENERATED - org.apache.juneau.assertions.Assertion */
140   public FluentAssertion<R> setThrowable(Class<? extends java.lang.RuntimeException> value) {
141      super.setThrowable(value);
142      return this;
143   }
144
145   // </FluentSetters>
146
147   //-----------------------------------------------------------------------------------------------------------------
148   // Utility methods
149   //-----------------------------------------------------------------------------------------------------------------
150
151   /**
152    * Returns the object that the fluent methods on this class should return.
153    *
154    * @return The response object.
155    */
156   @SuppressWarnings("unchecked")
157   protected R returns() {
158      return returns != null ? returns : (R)this;
159   }
160
161}
