024/**
025 * Base class for all assertion objects.
026 *
027 * <h5 class='topic'>Test Methods</h5>
028 * <p>
029 * <ul class='javatree'>
030 *    <li>None
031 * </ul>
032 *
033 * <h5 class='topic'>Transform Methods</h5>
034 * <p>
035 * <ul class='javatree'>
036 *    <li>None
037 * </ul>
038 *
039 * <h5 class='topic'>Configuration Methods</h5>
040 * <p>
041 * <ul class='javatree'>
042 *    <li class='jc'>{@link Assertion}
043 *    <ul class='javatreec'>
044 *       <li class='jm'>{@link Assertion#setMsg(String, Object...) setMsg(String, Object...)}
045 *       <li class='jm'>{@link Assertion#setOut(PrintStream) setOut(PrintStream)}
046 *       <li class='jm'>{@link Assertion#setSilent() setSilent()}
047 *       <li class='jm'>{@link Assertion#setStdOut() setStdOut()}
048 *       <li class='jm'>{@link Assertion#setThrowable(Class) setThrowable(Class)}
049 *    </ul>
050 * </ul>
051 *
052 * <ul class='seealso'>
053 *    <li class='link'>{@doc ja.Overview}
054 *    <li class='extlink'>{@source}
055 * </ul>
056 */
057@FluentSetters
058public class Assertion {

061   // Static
062   //-----------------------------------------------------------------------------------------------------------------
063
064   private static final Messages MESSAGES = Messages.of(Assertion.class, "Messages");
065
066   static final String
067      MSG_parameterCannotBeNull = MESSAGES.getString("parameterCannotBeNull"),
068      MSG_causedBy = MESSAGES.getString("causedBy");
069
070   //-----------------------------------------------------------------------------------------------------------------
071   // Instance
072   //-----------------------------------------------------------------------------------------------------------------
073
074   private String msg;
075   private Object[] msgArgs;
076   private PrintStream out = System.err;
077   private Class<? extends RuntimeException> throwable;
078
079   /**
080    * Constructor used when this assertion is being created from within another assertion.
081    *
082    * @param creator The creator of this assertion.
083    */
084   protected Assertion(Assertion creator) {
085      if (creator != null) {
086         this.msg = creator.msg;
087         this.msgArgs = creator.msgArgs;
088         this.out = creator.out;
089         this.throwable = creator.throwable;
090      }
091   }
092
093   //-----------------------------------------------------------------------------------------------------------------
094   // Config
095   //-----------------------------------------------------------------------------------------------------------------
096
097   /**
098    * Allows you to override the assertion failure message.
099    *
100    * <p>
101    * String can contain <js>"{msg}"</js> to represent the original message.
102    *
103    * <h5 class='section'>Example:</h5>
104    * <p class='bjava'>
105    *    <jk>import static</jk> org.apache.juneau.assertions.Assertions.*;
106    *
107    * <jc>// Throws an assertion with a custom message instead of the default "Value was null."</jc>
108    *    <jsm>assertString</jsm>(<jv>myString</jv>)
109    *       .setMsg(<js>"My string was bad:  {msg}"</js>)
110    *       .isNotNull();
111    * </p>
112    *
113    * @param msg The assertion failure message.
114    * @param args Optional message arguments.
115    * @return This object.
116    */
117   @FluentSetter
118   public Assertion setMsg(String msg, Object...args) {
119      this.msg = msg.replace("{msg}", "<<<MSG>>>");
120      this.msgArgs = args;
121      return this;
122   }
123
124   /**
125    * If an error occurs, send the error message to STDOUT instead of STDERR.
126    *
127    * @return This object.
128    */
129   @FluentSetter
130   public Assertion setStdOut() {
131      return setOut(System.out);
132   }
133
134   /**
135    * If an error occurs, send the error message to the specified stream instead of STDERR.
136    *
137    * @param value
138    *    The output stream.
139    *    Can be <jk>null</jk> to suppress output.
140    * @return This object.
141    */
142   @FluentSetter
143   public Assertion setOut(PrintStream value) {
144      this.out = value;
145      return this;
146   }
147
148   /**
149    * Suppresses output to STDERR.
150    *
151    * <p>
152    * This is the equivalent to calling <c>out(<jk>null</jk>)</c>.
153    *
154    * @return This object.
155    */
156   @FluentSetter
157   public Assertion setSilent() {
158      return setOut(null);
159   }
160
161   /**
162    * If an error occurs, throw this exception instead of the standard {@link AssertionError}.
163    *
164    * <p>
165    * The throwable class must have a public constructor that takes in any of the following parameters:
166    * <ul>
167    *    <li>{@link Throwable} - The caused-by exception (if there is one).
168    *    <li>{@link String} - The assertion failure message.
169    * </ul>
170    *
171    * <p>
172    * If the throwable cannot be instantiated, a {@link RuntimeException} is thrown instead.
173    *
174    * <h5 class='section'>Example:</h5>
175    * <p class='bjava'>
176    *    <jk>import static</jk> org.apache.juneau.assertions.Assertions.*;
177    *
178    * <jc>// Throws a BadRequest instead of an AssertionError if the string is null.</jc>
179    *    <jsm>assertString</jsm>(<jv>myString</jv>)
180    *       .setThrowable(BadRequest.<jk>class</jk>)
181    *       .isNotNull();
182    * </p>
183    *
184    * @param value The new value for this setting.
185    * @return This object.
186    */
187   @FluentSetter
188   public Assertion setThrowable(Class<? extends RuntimeException> value) {
189      this.throwable = value;
190      return this;
191   }

206   /**
207    * Creates a new {@link BasicAssertionError}.
208    *
209    * @param msg The message.
210    * @param args The message arguments.
211    * @return A new {@link BasicAssertionError}.
212    */
213   protected BasicAssertionError error(String msg, Object...args) {
214      return error(null, msg, args);
215   }
216
217   /**
218    * Creates a new {@link BasicAssertionError}.
219    *
220    * @param cause Optional caused-by throwable.
221    * @param msg The message.
222    * @param args The message arguments.
223    * @return A new {@link BasicAssertionError}.
224    */
225   protected BasicAssertionError error(Throwable cause, String msg, Object...args) {
226      msg = format(msg, args);
227      if (this.msg != null)
228         msg = format(this.msg, this.msgArgs).replace("<<<MSG>>>", msg);
229      if (out != null)
230         out.println(msg);
231      if (throwable != null) {
232         try {
233            throw BeanStore
234               .create()
235               .build()
236               .addBean(Throwable.class, cause)
237               .addBean(String.class, msg)
238               .addBean(Object[].class,new Object[0])
239               .createBean(throwable)
240               .run();
241         } catch (ExecutableException e) {
242            // If we couldn't create requested exception, just throw a RuntimeException.
243            throw new BasicRuntimeException(cause, msg);
244         }
245      }
246      return new BasicAssertionError(cause, msg);
247   }

249   /**
250    * Convenience method for getting the class name for an object.
251    *
252    * @param o The object to get the class name for.
253    * @return The class name for an object.
254    */
255   protected static String className(Object o) {
256      return ClassUtils.className(o);
257   }
258
259   /**
260    * Convenience method for getting the array class of the specified element type.
261    *
262    * @param <E> The element type.
263    * @param c The object to get the class name for.
264    * @return The class name for an object.
265    */
266   @SuppressWarnings("unchecked")
267   protected static <E> Class<E[]> arrayClass(Class<E> c) {
268      return (Class<E[]>)Array.newInstance(c,0).getClass();
269   }
