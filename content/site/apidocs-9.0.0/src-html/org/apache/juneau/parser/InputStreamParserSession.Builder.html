178   private final InputStreamParser ctx;
179
180   /**
181    * Constructor.
182    *
183    * @param builder The builder for this object.
184    */
185   protected InputStreamParserSession(Builder builder) {
186      super(builder);
187      this.ctx = builder.ctx;
188   }
189
190   @Override /* ParserSession */
191   public final boolean isReaderParser() {
192      return false;
193   }
194
195   /**
196    * Wraps the specified input object into a {@link ParserPipe} object so that it can be easily converted into
197    * a stream or reader.
198    *
199    * @param input
200    *    The input.
201    *    <br>This can be any of the following types:
202    *    <ul>
203    *       <li><jk>null</jk>
204    *       <li>{@link InputStream}
205    *       <li><code><jk>byte</jk>[]</code>
206    *       <li>{@link File}
207    *       <li>{@link CharSequence} containing encoded bytes according to the {@link InputStreamParser.Builder#binaryFormat(BinaryFormat)} setting.
208    *    </ul>
209    * @return
210    *    A new {@link ParserPipe} wrapper around the specified input object.
211    */
212   @SuppressWarnings("resource")
213   @Override /* ParserSession */
214   public final ParserPipe createPipe(Object input) {
215      return setPipe(new ParserPipe(input, isDebug(), ctx.isAutoCloseStreams(), ctx.isUnbuffered(), ctx.getBinaryFormat()));
216   }

226    * @return

228    */
229   protected final BinaryFormat getBinaryFormat() {
230      return ctx.getBinaryFormat();
231   }
