420   /**
421    * Converts the specified <c>JsonMap</c> into a bean identified by the <js>"_type"</js> property in the map.
422    *
423    * @param m The map to convert to a bean.
424    * @param pMeta The current bean property being parsed.
425    * @param eType The current expected type being parsed.
426    * @return
427    *    The converted bean, or the same map if the <js>"_type"</js> entry wasn't found or didn't resolve to a bean.
428    */
429   protected final Object cast(JsonMap m, BeanPropertyMeta pMeta, ClassMeta<?> eType) {
430
431      String btpn = getBeanTypePropertyName(eType);
432
433      Object o = m.get(btpn);
434      if (o == null)
435         return m;
436      String typeName = o.toString();
437
438      ClassMeta<?> cm = getClassMeta(typeName, pMeta, eType);
439
440      if (cm != null) {
441         BeanMap<?> bm = m.getBeanSession().newBeanMap(cm.getInnerClass());
442
443         // Iterate through all the entries in the map and set the individual field values.
444         m.forEach((k,v) -> {
445            if (! k.equals(btpn)) {
446               // Attempt to recursively cast child maps.
447               if (v instanceof JsonMap)
448                  v = cast((JsonMap)v, pMeta, eType);
449               bm.put(k, v);
450            }
451         });
452         return bm.getBean();
453      }
454
455      return m;
456   }

458   /**
459    * Give the specified dictionary name, resolve it to a class.
460    *
461    * @param typeName The dictionary name to resolve.
462    * @param pMeta The bean property we're currently parsing.
463    * @param eType The expected type we're currently parsing.
464    * @return The resolved class, or <jk>null</jk> if the type name could not be resolved.
465    */
466   protected final ClassMeta<?> getClassMeta(String typeName, BeanPropertyMeta pMeta, ClassMeta<?> eType) {
467      BeanRegistry br = null;
468
469      // Resolve via @Beanp(dictionary={})
470      if (pMeta != null) {
471         br = pMeta.getBeanRegistry();
472         if (br != null && br.hasName(typeName))
473            return br.getClassMeta(typeName);
474      }
475
476      // Resolve via @Bean(dictionary={}) on the expected type where the
477      // expected type is an interface with subclasses.
478      if (eType != null) {
479         br = eType.getBeanRegistry();
480         if (br != null && br.hasName(typeName))
481            return br.getClassMeta(typeName);
482      }
483
484      // Last resort, resolve using the session registry.
485      return getBeanRegistry().getClassMeta(typeName);
486   }

502   /**
503    * Method that gets called when an unknown bean property name is encountered.
504    *
505    * @param propertyName The unknown bean property name.
506    * @param beanMap The bean that doesn't have the expected property.
507    * @param value The parsed value.
508    * @throws ParseException
509    *    Automatically thrown if {@link org.apache.juneau.BeanContext.Builder#ignoreUnknownBeanProperties()} setting on this parser is
510    *    <jk>false</jk>
511    * @param <T> The class type of the bean map that doesn't have the expected property.
512    */
513   protected final <T> void onUnknownProperty(String propertyName, BeanMap<T> beanMap, Object value) throws ParseException {
514      if (propertyName.equals(getBeanTypePropertyName(beanMap.getClassMeta())))
515         return;
516      if (! isIgnoreUnknownBeanProperties())
517         if (value != null || ! isIgnoreUnknownNullBeanProperties())
518            throw new ParseException(this,
519               "Unknown property ''{0}'' encountered while trying to parse into class ''{1}''", propertyName,
520               beanMap.getClassMeta());
521      if (listener != null)
522         listener.onUnknownBeanProperty(this, propertyName, beanMap.getClassMeta().getInnerClass(), beanMap.getBean());
523   }

525   /**
526    * Parses input into the specified object type.
527    *
528    * <p>
529    * The type can be a simple type (e.g. beans, strings, numbers) or parameterized type (collections/maps).
530    *
531    * <h5 class='section'>Examples:</h5>
532    * <p class='bjava'>
533    *    ReaderParser <jv>parser</jv> = JsonParser.<jsf>DEFAULT</jsf>;
534    *
535    *    <jc>// Parse into a linked-list of strings.</jc>
536    *    List <jv>list1</jv> = <jv>parser</jv>.parse(<jv>json</jv>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
537    *
538    *    <jc>// Parse into a linked-list of beans.</jc>
539    *    List <jv>list2</jv> = <jv>parser</jv>.parse(<jv>json</jv>, LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
540    *
541    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
542    *    List <jv>list3</jv> = <jv>parser</jv>.parse(<jv>json</jv>, LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
543    *
544    *    <jc>// Parse into a map of string keys/values.</jc>
545    *    Map <jv>map1</jv> = <jv>parser</jv>.parse(<jv>json</jv>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
546    *
547    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
548    *    Map <jv>map2</jv> = <jv>parser</jv>.parse(<jv>json</jv>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
549    * </p>
550    *
551    * <p>
552    * <c>Collection</c> classes are assumed to be followed by zero or one objects indicating the element type.
553    *
554    * <p>
555    * <c>Map</c> classes are assumed to be followed by zero or two meta objects indicating the key and value types.
556    *
557    * <p>
558    * The array can be arbitrarily long to indicate arbitrarily complex data structures.
559    *
560    * <ul class='notes'>
561    *    <li class='note'>
562    *       Use the {@link #parse(Object, Class)} method instead if you don't need a parameterized map/collection.
563    * </ul>
564    *
565    * @param <T> The class type of the object to create.
566    * @param input
567    *    The input.
568    *    <br>Character-based parsers can handle the following input class types:
569    *    <ul>
570    *       <li><jk>null</jk>
571    *       <li>{@link Reader}
572    *       <li>{@link CharSequence}
573    *       <li>{@link InputStream} containing UTF-8 encoded text (or charset defined by
574    *          {@link ReaderParser.Builder#streamCharset(Charset)} property value).
575    *       <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or charset defined by
576    *          {@link ReaderParser.Builder#streamCharset(Charset)} property value).
577    *       <li>{@link File} containing system encoded text (or charset defined by
578    *          {@link ReaderParser.Builder#fileCharset(Charset)} property value).
579    *    </ul>
580    *    <br>Stream-based parsers can handle the following input class types:
581    *    <ul>
582    *       <li><jk>null</jk>
583    *       <li>{@link InputStream}
584    *       <li><code><jk>byte</jk>[]</code>
585    *       <li>{@link File}
586    *    </ul>
587    * @param type
588    *    The object type to create.
589    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
590    * @param args
591    *    The type arguments of the class if it's a collection or map.
592    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
593    *    <br>Ignored if the main type is not a map or collection.
594    * @return The parsed object.
595    * @throws ParseException Malformed input encountered.
596    * @see BeanSession#getClassMeta(Type,Type...) for argument syntax for maps and collections.
597    * @throws IOException Thrown by the underlying stream.
598    */
599   @SuppressWarnings("unchecked")
600   public final <T> T parse(Object input, Type type, Type...args) throws ParseException, IOException {
601      try (ParserPipe pipe = createPipe(input)) {
602         return (T)parseInner(pipe, getClassMeta(type, args));
603      }
604   }

606   /**
607    * Same as {@link #parse(Object,Type,Type...)} but parses from a string and doesn't throw an {@link IOException}.
608    *
609    * @param <T> The class type of the object to create.
610    * @param input
611    *    The input.
612    *    <br>Character-based parsers can handle the following input class types:
613    *    <ul>
614    *       <li><jk>null</jk>
615    *       <li>{@link Reader}
616    *       <li>{@link CharSequence}
617    *       <li>{@link InputStream} containing UTF-8 encoded text (or charset defined by
618    *          {@link ReaderParser.Builder#streamCharset(Charset)} property value).
619    *       <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or charset defined by
620    *          {@link ReaderParser.Builder#streamCharset(Charset)} property value).
621    *       <li>{@link File} containing system encoded text (or charset defined by
622    *          {@link ReaderParser.Builder#fileCharset(Charset)} property value).
623    *    </ul>
624    *    <br>Stream-based parsers can handle the following input class types:
625    *    <ul>
626    *       <li><jk>null</jk>
627    *       <li>{@link InputStream}
628    *       <li><code><jk>byte</jk>[]</code>
629    *       <li>{@link File}
630    *    </ul>
631    * @param type
632    *    The object type to create.
633    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
634    * @param args
635    *    The type arguments of the class if it's a collection or map.
636    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
637    *    <br>Ignored if the main type is not a map or collection.
638    * @return The parsed object.
639    * @throws ParseException Malformed input encountered.
640    * @see BeanSession#getClassMeta(Type,Type...) for argument syntax for maps and collections.
641    */
642   @SuppressWarnings("unchecked")
643   public final <T> T parse(String input, Type type, Type...args) throws ParseException {
644      try (ParserPipe pipe = createPipe(input)) {
645         return (T)parseInner(pipe, getClassMeta(type, args));
646      } catch (IOException e) {
647         throw new ParseException(e); // Shouldn't happen.
648      }
649   }

653    *
654    * <p>

956   /**
957    * Converts the specified string to the specified type.
958    *
959    * @param outer
960    *    The outer object if we're converting to an inner object that needs to be created within the context
961    *    of an outer object.
962    * @param s The string to convert.
963    * @param type The class type to convert the string to.
964    * @return The string converted as an object of the specified type.
965    * @param <T> The class type to convert the string to.
966    * @throws ParseException Malformed input encountered.
967    * @throws ExecutableException Exception occurred on invoked constructor/method/field.
968    */
969   @SuppressWarnings({ "unchecked", "rawtypes" })
970   protected final <T> T convertAttrToType(Object outer, String s, ClassMeta<T> type) throws ParseException {
971      if (s == null)
972         return null;
973
974      if (type == null)
975         type = (ClassMeta<T>)object();
976      ObjectSwap swap = type.getSwap(this);
977      ClassMeta<?> sType = swap == null ? type : swap.getSwapClassMeta(this);
978
979      Object o = s;
980      if (sType.isChar())
981         o = parseCharacter(s);
982      else if (sType.isNumber())
983         o = parseNumber(s, (Class<? extends Number>)sType.getInnerClass());
984      else if (sType.isBoolean())
985         o = Boolean.parseBoolean(s);
986      else if (! (sType.isCharSequence() || sType.isObject())) {
987         if (sType.canCreateNewInstanceFromString(outer))
988            o = sType.newInstanceFromString(outer, s);
989         else
990            throw new ParseException(this, "Invalid conversion from string to class ''{0}''", type);
991      }
992
993      if (swap != null)
994         o = unswap(swap, o, type);
995
996      return (T)o;
997   }
