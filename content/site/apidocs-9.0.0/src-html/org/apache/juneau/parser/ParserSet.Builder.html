030/**
031 * Represents a group of {@link Parser Parsers} that can be looked up by media type.
032 *
033 * <h5 class='topic'>Description</h5>
034 *
035 * Provides the following features:
036 * <ul class='spaced-list'>
037 *    <li>
038 *       Finds parsers based on HTTP <c>Content-Type</c> header values.
039 *    <li>
040 *       Sets common properties on all parsers in a single method call.
041 *    <li>
042 *       Locks all parsers in a single method call.
043 *    <li>
044 *       Clones existing groups and all parsers within the group in a single method call.
045 * </ul>
046 *
047 * <h5 class='topic'>Match ordering</h5>
048 *
049 * Parsers are matched against <c>Content-Type</c> strings in the order they exist in this group.
050 *
051 * <p>
052 * Adding new entries will cause the entries to be prepended to the group.
053 * This allows for previous parsers to be overridden through subsequent calls.
054 *
055 * <p>
056 * For example, calling <code>g.append(P1.<jk>class</jk>,P2.<jk>class</jk>).append(P3.<jk>class</jk>,P4.<jk>class</jk>)</code>
057 * will result in the order <c>P3, P4, P1, P2</c>.
058 *
059 * <h5 class='section'>Example:</h5>
060 * <p class='bjava'>
061 *    <jc>// Construct a new parser group builder</jc>
062 *    ParserSet <jv>parsers</jv> = ParserSet.<jsm>create</jsm>();
063 *       .add(JsonParser.<jk>class</jk>, XmlParser.<jk>class</jk>); <jc>// Add some parsers to it</jc>
064 *    .forEach(<jv>x</jv> -&gt; <jv>x</jv>.swaps(CalendarSwap.IsoLocalDateTime.<jk>class</jk>))
065 *    .forEach(<jv>x</jv> -&gt; <jv>x</jv>.beansRequireSerializable())
066 *       .build();
067 *
068 *    <jc>// Find the appropriate parser by Content-Type</jc>
069 *    ReaderParser <jv>parser</jv> = (ReaderParser)<jv>parsers</jv>.getParser(<js>"text/json"</js>);
070 *
071 *    <jc>// Parse a bean from JSON</jc>
072 *    String <jv>json</jv> = <js>"{...}"</js>;
073 *    AddressBook <jv>addressBook</jv> = <jv>parser</jv>.parse(<jv>json</jv>, AddressBook.<jk>class</jk>);
074 * </p>
075 *
076 * <ul class='notes'>
077 *    <li class='note'>This class is thread safe and reusable.
078 * </ul>
079 *
080 * <ul class='seealso'>
081 *    <li class='link'>{@doc jm.SerializersAndParsers}
082 *    <li class='extlink'>{@source}
083 * </ul>
084 */
085public final class ParserSet {

167      /**
168       * Clone an existing parser group builder.
169       *
170       * <p>
171       * Parser builders will be cloned during this process.
172       *
173       * @param copyFrom The parser group that we're copying settings and parsers from.
174       */
175      protected Builder(Builder copyFrom) {
176         super(copyFrom);
177         bcBuilder = copyFrom.bcBuilder == null ? null : copyFrom.bcBuilder.copy();
178         entries = list();
179         copyFrom.entries.stream().map(x -> copyBuilder(x)).forEach(x -> entries.add(x));
180      }
181
182      private Object copyBuilder(Object o) {
183         if (o instanceof Parser.Builder) {
184            Parser.Builder x = (Parser.Builder)o;
185            Parser.Builder x2 = x.copy();
186            if (ne(x.getClass(), x2.getClass()))
187               throw runtimeException("Copy method not implemented on class " + x.getClass().getName());
188            x = x2;
189            if (bcBuilder != null)
190               x.beanContext(bcBuilder);
191            return x;
192         }
193         return o;
194      }

238      /**
239       * Adds the specified parsers to this group.
240       *
241       * <p>
242       * Entries are added in-order to the beginning of the list in the group.
243       *
244       * <p>
245       * The {@link NoInherit} class can be used to clear out the existing list of parsers before adding the new entries.
246       *
247       * <h5 class='section'>Example:</h5>
248       * <p class='bjava'>
249       *    ParserSet.Builder <jv>builder</jv> = ParserSet.<jsm>create</jsm>();  <jc>// Create an empty builder.</jc>
250       *
251       *    <jv>builder</jv>.add(FooParser.<jk>class</jk>);  <jc>// Now contains:  [FooParser]</jc>
252       *
253       *    <jv>builder</jv>.add(BarParser.<jk>class</jk>, BazParser.<jk>class</jk>);  <jc>// Now contains:  [BarParser,BazParser,FooParser]</jc>
254       *
255       *    <jv>builder</jv>.add(NoInherit.<jk>class</jk>, QuxParser.<jk>class</jk>);  <jc>// Now contains:  [QuxParser]</jc>
256       * </p>
257       *
258       * @param values The parsers to add to this group.
259       * @return This object.
260       * @throws IllegalArgumentException If one or more values do not extend from {@link Parser}.
261       */
262      public Builder add(Class<?>...values) {
263         List<Object> l = list();
264         for (Class<?> v : values)
265            if (v.getSimpleName().equals("NoInherit"))
266               clear();
267         for (Class<?> v : values) {
268            if (Parser.class.isAssignableFrom(v)) {
269               l.add(createBuilder(v));
270            } else if (! v.getSimpleName().equals("NoInherit")) {
271               throw runtimeException("Invalid type passed to ParserSet.Builder.add(): " + v.getName());
272            }
273         }
274         entries.addAll(0, l);
275         return this;
276      }
277
278      /**
279       * Sets the specified parsers for this group.
280       *
281       * <p>
282       * Existing values are overwritten.
283       *
284       * <p>
285       * The {@link Inherit} class can be used to insert existing entries in this group into the position specified.
286       *
287       * <h5 class='section'>Example:</h5>
288       * <p class='bjava'>
289       *    ParserSet.Builder <jv>builder</jv> = ParserSet.<jsm>create</jsm>();  <jc>// Create an empty builder.</jc>
290       *
291       *    <jv>builder</jv>.set(FooParser.<jk>class</jk>);  <jc>// Now contains:  [FooParser]</jc>
292       *
293       *    <jv>builder</jv>.set(BarParser.<jk>class</jk>, BazParser.<jk>class</jk>);  <jc>// Now contains:  [BarParser,BazParser]</jc>
294       *
295       *    <jv>builder</jv>.set(Inherit.<jk>class</jk>, QuxParser.<jk>class</jk>);  <jc>// Now contains:  [BarParser,BazParser,QuxParser]</jc>
296       * </p>
297       *
298       * @param values The parsers to set in this group.
299       * @return This object.
300       * @throws IllegalArgumentException If one or more values do not extend from {@link Parser} or named <js>"Inherit"</js>.
301       */
302      public Builder set(Class<?>...values) {
303         List<Object> l = list();
304         for (Class<?> v : values) {
305            if (v.getSimpleName().equals("Inherit")) {
306               l.addAll(entries);
307            } else if (Parser.class.isAssignableFrom(v)) {
308               l.add(createBuilder(v));
309            } else {
310               throw runtimeException("Invalid type passed to ParserGrouup.Builder.set(): " + v.getName());
311            }
312         }
313         entries = l;
314         return this;
315      }
316
317      private Object createBuilder(Object o) {
318         if (o instanceof Class) {
319
320            // Check for no-arg constructor.
321            ConstructorInfo ci = ClassInfo.of((Class<?>)o).getPublicConstructor(x -> x.hasNoParams());
322            if (ci != null)
323               return ci.invoke();
324
325            // Check for builder.
326            @SuppressWarnings("unchecked")
327            Parser.Builder b = Parser.createParserBuilder((Class<? extends Parser>)o);
328            if (bcBuilder != null)
329               b.beanContext(bcBuilder);
330            o = b;
331         }
332         return o;
333      }

482   // Maps Content-Type headers to matches.
483   private final ConcurrentHashMap<String,ParserMatch> cache = new ConcurrentHashMap<>();
484
485   private final MediaType[] mediaTypes;
486   private final Parser[] mediaTypeParsers;
487
488   final Parser[] entries;
489
490   /**
491    * Constructor.
492    *
493    * @param builder The builder for this bean.
494    */
495   public ParserSet(Builder builder) {
496
497      this.entries = builder.entries.stream().map(x -> build(x)).toArray(Parser[]::new);
498
499      List<MediaType> lmt = list();
500      List<Parser> l = list();
501      for (Parser e : entries) {
502         e.getMediaTypes().forEach(x -> {
503            lmt.add(x);
504            l.add(e);
505         });
506      }
507
508      this.mediaTypes = array(lmt, MediaType.class);
509      this.mediaTypeParsers = array(l, Parser.class);
510   }
511
512   private Parser build(Object o) {
513      if (o instanceof Parser)
514         return (Parser)o;
515      return ((Parser.Builder)o).build();
516   }

527   /**
528    * Searches the group for a parser that can handle the specified <l>Content-Type</l> header value.
529    *
530    * <p>
531    * The returned object includes both the parser and media type that matched.
532    *
533    * @param contentTypeHeader The HTTP <l>Content-Type</l> header value.
534    * @return The parser and media type that matched the content type header, or <jk>null</jk> if no match was made.
535    */
536   public ParserMatch getParserMatch(String contentTypeHeader) {
537      ParserMatch pm = cache.get(contentTypeHeader);
538      if (pm != null)
539         return pm;
540
541      MediaType ct = MediaType.of(contentTypeHeader);
542      int match = ct.match(ulist(mediaTypes));
543
544      if (match >= 0) {
545         pm = new ParserMatch(mediaTypes[match], mediaTypeParsers[match]);
546         cache.putIfAbsent(contentTypeHeader, pm);
547      }
548
549      return cache.get(contentTypeHeader);
550   }

552   /**
553    * Same as {@link #getParserMatch(String)} but matches using a {@link MediaType} instance.
554    *

556    * @return The parser and media type that matched the media type, or <jk>null</jk> if no match was made.
557    */
558   public ParserMatch getParserMatch(MediaType mediaType) {
559      return getParserMatch(mediaType.toString());
560   }

567    */

569      ParserMatch pm = getParserMatch(contentTypeHeader);
570      return pm == null ? null : pm.getParser();

578    */
579   public Parser getParser(MediaType mediaType) {
580      ParserMatch pm = getParserMatch(mediaType);
