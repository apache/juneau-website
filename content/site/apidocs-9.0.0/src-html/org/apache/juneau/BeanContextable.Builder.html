029/**
030 * Context class for classes that use {@link BeanContext} objects.
031 *
032 * <p>
033 * This abstraction exists to allow different kinds of subclasses (e.g. JsonSerilalizer, XmlParser...) to share bean context objects since
034 * bean context objects are heavyweight objects that cache metadata about encountered beans.
035 *
036 * <ul class='notes'>
037 *    <li class='note'>This class is thread safe and reusable.
038 * </ul>
039 *
040 * <ul class='seealso'>
041 *    <li class='extlink'>{@source}
042 * </ul>
043 */
044public abstract class BeanContextable extends Context {
045
046   //-----------------------------------------------------------------------------------------------------------------
047   // Builder
048   //-----------------------------------------------------------------------------------------------------------------
049
050   /**
051    * Builder class.
052    */
053   @FluentSetters(ignore={"annotations","debug"})
054   public abstract static class Builder extends Context.Builder {
055
056      BeanContext.Builder bcBuilder;
057      BeanContext bc;
058
059      /**
060       * Constructor.
061       *
062       * All default settings.
063       */
064      protected Builder() {
065         super();
066         this.bcBuilder = BeanContext.create();
067         registerBuilders(bcBuilder);
068      }
069
070      /**
071       * Copy constructor.
072       *
073       * @param copyFrom The bean to copy from.
074       */
075      protected Builder(BeanContextable copyFrom) {
076         super(copyFrom);
077         this.bcBuilder = copyFrom.getBeanContext().copy();
078         registerBuilders(bcBuilder);
079      }
080
081      /**
082       * Copy constructor.
083       *
084       * @param copyFrom The builder to copy from.
085       */
086      protected Builder(Builder copyFrom) {
087         super(copyFrom);
088         this.bcBuilder = copyFrom.bcBuilder.copy();
089         this.bc = copyFrom.bc;
090         registerBuilders(bcBuilder);
091      }
092
093      @Override /* Context.Builder */
094      public abstract Builder copy();
095
096      @Override /* Context.Builder */
097      public HashKey hashKey() {
098         return HashKey.of(
099            super.hashKey(),
100            bcBuilder.hashKey(),
101            bc == null ? 0 : bc.hashKey
102         );
103      }
104
105      /**
106       * Returns the inner bean context builder.
107       *
108       * @return The inner bean context builder.
109       */
110      public BeanContext.Builder beanContext() {
111         return bcBuilder;
112      }
113
114      /**
115       * Applies an operation to the inner bean context builder.
116       *
117       * @param operation The operation to apply.
118       * @return This object.
119       */
120      public final Builder beanContext(Consumer<BeanContext.Builder> operation) {
121         operation.accept(beanContext());
122         return this;
123      }

163      /**
164       * Minimum bean class visibility.
165       *
166       * <p>
167       * Classes are not considered beans unless they meet the minimum visibility requirements.
168       * For example, if the visibility is <jsf>PUBLIC</jsf> and the bean class is <jk>protected</jk>, then the class
169       * will not be interpreted as a bean class and be serialized as a string.
170       * Use this setting to reduce the visibility requirement.
171       *
172       * <h5 class='section'>Example:</h5>
173       * <p class='bjava'>
174       *    <jc>// A bean with a protected class and one field.</jc>
175       *    <jk>protected class</jk> MyBean {
176       *       <jk>public</jk> String <jf>foo</jf> = <js>"bar"</js>;
177       *    }
178       *
179       *    <jc>// Create a serializer that's capable of serializing the class.</jc>
180       *    WriterSerializer <jv>serializer</jv> = JsonSerializer
181       *       .<jsm>create</jsm>()
182       *       .beanClassVisibility(<jsf>PROTECTED</jsf>)
183       *       .build();
184       *
185       *    <jc>// Produces:  {"foo","bar"}</jc>
186       *    String <jv>json</jv> = <jv>serializer</jv>.serialize(<jk>new</jk> MyBean());
187       * </p>
188       *
189       * <ul class='notes'>
190       *    <li class='note'>The {@link Bean @Bean} annotation can be used on a non-public bean class to override this setting.
191       *    <li class='note'>The {@link BeanIgnore @BeanIgnore} annotation can also be used on a public bean class to ignore it as a bean.
192       * </ul>
193       *
194       * <ul class='seealso'>
195       *    <li class='ja'>{@link BeanConfig#beanClassVisibility()}
196       * </ul>
197       *
198       * @param value
199       *    The new value for this setting.
200       *    <br>The default is {@link Visibility#PUBLIC}.
201       * @return This object.
202       */
203      @FluentSetter
204      public Builder beanClassVisibility(Visibility value) {
205         bcBuilder.beanClassVisibility(value);
206         return this;
207      }

316      /**
317       * Bean interceptor.
318       *
319       * <p>
320       * Bean interceptors can be used to intercept calls to getters and setters and alter their values in transit.
321       *
322       * <h5 class='section'>Example:</h5>
323       * <p class='bjava'>
324       *    <jc>// Interceptor that strips out sensitive information.</jc>
325       *    <jk>public class</jk> AddressInterceptor <jk>extends</jk> BeanInterceptor&lt;Address&gt; {
326       *
327       *       <jk>public</jk> Object readProperty(Address <jv>bean</jv>, String <jv>name</jv>, Object <jv>value</jv>) {
328       *          <jk>if</jk> (<js>"taxInfo"</js>.equals(<jv>name</jv>))
329       *             <jk>return</jk> <js>"redacted"</js>;
330       *          <jk>return</jk> <jv>value</jv>;
331       *       }
332       *
333       *       <jk>public</jk> Object writeProperty(Address <jv>bean</jv>, String <jv>name</jv>, Object <jv>value</jv>) {
334       *          <jk>if</jk> (<js>"taxInfo"</js>.equals(<jv>name</jv>) &amp;&amp; <js>"redacted"</js>.equals(<jv>value</jv>))
335       *             <jk>return</jk> TaxInfoUtils.<jsm>lookup</jsm>(<jv>bean</jv>.getStreet(), <jv>bean</jv>.getCity(), <jv>bean</jv>.getState());
336       *          <jk>return</jk> <jv>value</jv>;
337       *       }
338       *    }
339       *
340       *    <jc>// Our bean class.</jc>
341       *    <jk>public class</jk> Address {
342       *       <jk>public</jk> String getTaxInfo() {...}
343       *       <jk>public void</jk> setTaxInfo(String <jv>value</jv>) {...}
344       *    }
345       *
346       *    <jc>// Register filter on serializer or parser.</jc>
347       *    WriterSerializer <jv>serializer</jv> = JsonSerializer
348       *       .<jsm>create</jsm>()
349       *       .beanInterceptor(Address.<jk>class</jk>, AddressInterceptor.<jk>class</jk>)
350       *       .build();
351       *
352       *    <jc>// Produces:  {"taxInfo":"redacted"}</jc>
353       *    String <jv>json</jv> = <jv>serializer</jv>.serialize(<jk>new</jk> Address());
354       * </p>
355       *
356       * <ul class='seealso'>
357       *    <li class='jc'>{@link BeanInterceptor}
358       *    <li class='ja'>{@link Bean#interceptor() Bean(interceptor)}
359       * </ul>
360       *
361       * @param on The bean that the filter applies to.
362       * @param value
363       *    The new value for this setting.
364       * @return This object.
365       */
366      @FluentSetter
367      public Builder beanInterceptor(Class<?> on, Class<? extends BeanInterceptor<?>> value) {
368         bcBuilder.beanInterceptor(on, value);
369         return this;
370      }
