031   /**
032    * Appends one or more elements to an array.
033    *
034    * @param <T> The element type.
035    * @param array The array to append to.
036    * @param newElements The new elements to append to the array.
037    * @return A new array with the specified elements appended.
038    */
039   @SuppressWarnings("unchecked")
040   public static <T> T[] append(T[] array, T...newElements) {
041      if (array == null)
042         return newElements;
043      if (newElements.length == 0)
044         return array;
045      T[] a = (T[])Array.newInstance(array.getClass().getComponentType(), array.length + newElements.length);
046      for (int i = 0; i < array.length; i++)
047         a[i] = array[i];
048      for (int i = 0; i < newElements.length; i++)
049         a[i+array.length] = newElements[i];
050      return a;
051   }

053   /**
054    * Combine an arbitrary number of arrays into a single array.
055    *
056    * @param <E> The element type.
057    * @param arrays Collection of arrays to combine.
058    * @return A new combined array, or <jk>null</jk> if all arrays are <jk>null</jk>.
059    */
060   @SuppressWarnings("unchecked")
061   public static <E> E[] combine(E[]...arrays) {
062      assertArgNotNull("arrays", arrays);
063      int l = 0;
064      E[] a1 = null;
065      for (E[] a : arrays) {
066         if (a1 == null && a != null)
067            a1 = a;
068         l += (a == null ? 0 : a.length);
069      }
070      if (a1 == null)
071         return null;
072      E[] a = (E[])Array.newInstance(a1.getClass().getComponentType(), l);
073      int i = 0;
074      for (E[] aa : arrays)
075         if (aa != null)
076            for (E t : aa)
077               a[i++] = t;
078      return a;
079   }

081   /**
082    * Converts the specified array to a <c>Set</c>.
083    *
084    * <p>
085    * The order of the entries in the set are the same as the array.
086    *
087    * @param <T> The entry type of the array.
088    * @param array The array being wrapped in a <c>Set</c> interface.
089    * @return The new set.
090    */
091   public static <T> Set<T> asSet(final T[] array) {
092      assertArgNotNull("array", array);
093      return new AbstractSet<T>() {
094
095         @Override /* Set */
096         public Iterator<T> iterator() {
097            return new Iterator<T>() {
098               int i = 0;
099
100               @Override /* Iterator */
101               public boolean hasNext() {
102                  return i < array.length;
103               }
104
105               @Override /* Iterator */
106               public T next() {
107                  if (i >= array.length)
108                     throw new NoSuchElementException();
109                  T t = array[i];
110                  i++;
111                  return t;
112               }
113
114               @Override /* Iterator */
115               public void remove() {
116                  throw unsupportedOperationException("Not supported.");
117               }
118            };
119         }
120
121         @Override /* Set */
122         public int size() {
123            return array.length;
124         }
125      };
126   }

128   /**
129    * Converts the specified collection to an array.
130    *
131    * <p>
132    * Works on both object and primitive arrays.
133    *
134    * @param <E> The element type.
135    * @param c The collection to convert to an array.
136    * @param elementType The component type of the collection.
137    * @return A new array.
138    */
139   public static <E> Object toArray(Collection<?> c, Class<E> elementType) {
140      Object a = Array.newInstance(elementType, c.size());
141      Iterator<?> it = c.iterator();
142      int i = 0;
143      while (it.hasNext())
144         Array.set(a, i++, it.next());
145      return a;
146   }

158   /**
159    * Converts the specified array to an <c>ArrayList</c>
160    *
161    * @param <E> The element type.
162    * @param array The array to convert.
163    * @param elementType
164    *    The type of objects in the array.
165    *    It must match the actual component type in the array.
166    * @return A new {@link ArrayList}
167    */
168   @SuppressWarnings("unchecked")
169   public static <E> List<E> toList(Object array, Class<E> elementType) {
170      List<E> l = new ArrayList<>(Array.getLength(array));
171      for (int i = 0; i < Array.getLength(array); i++)
172         l.add((E)Array.get(array, i));
173      return l;
174   }
175
176   /**
177    * Recursively converts the specified array into a list of objects.
178    *
179    * @param array The array to convert.
180    * @return A new {@link ArrayList}
181    */
182   public static List<Object> toObjectList(Object array) {
183      List<Object> l = new ArrayList<>(Array.getLength(array));
184      for (int i = 0; i < Array.getLength(array); i++) {
185         Object o = Array.get(array, i);
186         if (o != null && o.getClass().isArray())
187            o = toObjectList(o);
188         l.add(o);
189      }
190      return l;
191   }

193   /**
194    * Copies the specified array into the specified list.
195    *
196    * <p>
197    * Works on both object and primitive arrays.
198    *
199    * @param array The array to copy into a list.
200    * @param list The list to copy the values into.
201    * @return The same list passed in.
202    */
203   @SuppressWarnings({"unchecked","rawtypes"})
204   public static List copyToList(Object array, List list) {
205      if (array != null) {
206         int length = Array.getLength(array);
207         for (int i = 0; i < length; i++)
208            list.add(Array.get(array, i));
209      }
210      return list;
211   }

213   /**
214    * Returns <jk>true</jk> if the specified array contains the specified element using the {@link String#equals(Object)}
215    * method.
216    *
217    * @param element The element to check for.
218    * @param array The array to check.
219    * @return
220    *    <jk>true</jk> if the specified array contains the specified element,
221    *    <jk>false</jk> if the array or element is <jk>null</jk>.
222    */
223   public static boolean contains(String element, String[] array) {
224      return indexOf(element, array) != -1;
225   }
226
227   /**
228    * Returns the index position of the element in the specified array using the {@link String#equals(Object)} method.
229    *
230    * @param element The element to check for.
231    * @param array The array to check.
232    * @return
233    *    The index position of the element in the specified array, or
234    *    <c>-1</c> if the array doesn't contain the element, or the array or element is <jk>null</jk>.
235    */
236   public static int indexOf(String element, String[] array) {
237      if (element == null || array == null)
238         return -1;
239      for (int i = 0; i < array.length; i++)
240         if (element.equals(array[i]))
241            return i;
242      return -1;
243   }

245   /**
246    * Converts the specified collection to an array of strings.
247    *
248    * <p>
249    * Entries are converted to strings using {@link #toString()}.
250    * <jk>null</jk> values remain <jk>null</jk>.
251    *
252    * @param c The collection to convert.
253    * @return The collection as a string array.
254    */
255   public static String[] toStringArray(Collection<?> c) {
256      String[] r = new String[c.size()];
257      int i = 0;
258      for (Object o : c)
259         r[i++] = stringify(o);
260      return r;
261   }

263   /**
264    * Returns <jk>true</jk> if the following sorted arrays are equals.
265    *
266    * @param a1 Array #1.
267    * @param a2 Array #2.
268    * @return <jk>true</jk> if the following sorted arrays are equals.
269    */
270   public static boolean equals(String[] a1, String[] a2) {
271      if (a1.length != a2.length)
272         return false;
273      for (int i = 0; i < a1.length; i++)
274         if (! StringUtils.eq(a1[i], a2[i]))
275            return false;
276      return true;
277   }

321   /**
322    * Reverses the entries in an array.
323    *
324    * @param <E> The element type.
325    * @param array The array to reverse.
326    * @return The same array.
327    */
328   public static final <E> E[] reverse(E[] array) {
329      for (int i = 0; i < array.length / 2; i++) {
330         E temp = array[i];
331         array[i] = array[array.length - i - 1];
332         array[array.length - i - 1] = temp;
333      }
334      return array;
335   }
