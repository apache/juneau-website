013package org.apache.juneau.internal;
014
015import static org.apache.juneau.internal.ConverterUtils.*;
016import static org.apache.juneau.internal.ThrowableUtils.*;
017import static org.apache.juneau.internal.StringUtils.*;
018import static java.util.Collections.*;
019
020import java.lang.reflect.*;
021import java.util.*;
022
023import org.apache.juneau.collections.*;
024import org.apache.juneau.parser.*;
025
026/**
027 * Builder for lists.
028 *
029 * <ul class='seealso'>
030 *    <li class='extlink'>{@source}
031 * </ul>
032 *
033 * @param <E> Element type.
034 */
035public final class ListBuilder<E> {
036
037   //-----------------------------------------------------------------------------------------------------------------
038   // Static
039   //-----------------------------------------------------------------------------------------------------------------
040
041   /**
042    * Static creator.
043    *
044    * @param <E> The element type.
045    * @param elementType The element type.
046    * @param elementTypeArgs Optional element type arguments.
047    * @return A new builder.
048    */
049   public static <E> ListBuilder<E> create(Class<E> elementType, Type...elementTypeArgs) {
050      return new ListBuilder<>(elementType, elementTypeArgs);
051   }
052
053   //-----------------------------------------------------------------------------------------------------------------
054   // Instance
055   //-----------------------------------------------------------------------------------------------------------------
056
057   private List<E> list;
058   private boolean unmodifiable = false, sparse = false;
059   private Comparator<E> comparator;
060
061   private Class<E> elementType;
062   private Type[] elementTypeArgs;
063
064   /**
065    * Constructor.
066    *
067    * @param elementType The element type.
068    * @param elementTypeArgs The element type generic arguments if there are any.
069    */
070   public ListBuilder(Class<E> elementType, Type...elementTypeArgs) {
071      this.elementType = elementType;
072      this.elementTypeArgs = elementTypeArgs;
073   }
074
075   /**
076    * Constructor.
077    *
078    * @param addTo The list to add to.
079    */
080   public ListBuilder(List<E> addTo) {
081      this.list = addTo;
082   }
083
084   /**
085    * Builds the list.
086    *
087    * @return A list conforming to the settings on this builder.
088    */
089   public List<E> build() {
090      if (sparse) {
091         if (list != null && list.isEmpty())
092            list = null;
093      } else {
094         if (list == null)
095            list = new ArrayList<>(0);
096      }
097      if (list != null) {
098         if (comparator != null)
099            Collections.sort(list, comparator);
100         if (unmodifiable)
101            list = unmodifiableList(list);
102      }
103      return list;
104   }
105
106   /**
107    * When specified, the {@link #build()} method will return <jk>null</jk> if the list is empty.
108    *
109    * <p>
110    * Otherwise {@link #build()} will never return <jk>null</jk>.
111    *
112    * @return This object.
113    */
114   public ListBuilder<E> sparse() {
115      this.sparse = true;
116      return this;
117   }
118
119   /**
120    * When specified, {@link #build()} will return an unmodifiable list.
121    *
122    * @return This object.
123    */
124   public ListBuilder<E> unmodifiable() {
125      this.unmodifiable = true;
126      return this;
127   }
128
129   /**
130    * Forces the existing list to be copied instead of appended to.
131    *
132    * @return This object.
133    */
134   public ListBuilder<E> copy() {
135      if (list != null)
136         list = new ArrayList<>(list);
137      return this;
138   }
139
140   /**
141    * Sorts the contents of the list.
142    *
143    * @return This object.
144    */
145   @SuppressWarnings("unchecked")
146   public ListBuilder<E> sorted() {
147      return sorted((Comparator<E>)Comparator.naturalOrder());
148   }
149
150   /**
151    * Sorts the contents of the list using the specified comparator.
152    *
153    * @param comparator The comparator to use for sorting.
154    * @return This object.
155    */
156   public ListBuilder<E> sorted(Comparator<E> comparator) {
157      this.comparator = comparator;
158      return this;
159   }
160
161   /**
162    * Appends the contents of the specified collection into this list.
163    *
164    * <p>
165    * This is a no-op if the value is <jk>null</jk>.
166    *
167    * @param value The collection to add to this list.
168    * @return This object.
169    */
170   public ListBuilder<E> addAll(Collection<E> value) {
171      if (value != null) {
172         if (list == null)
173            list = new LinkedList<>(value);
174         else
175            list.addAll(value);
176      }
177      return this;
178   }
179
180   /**
181    * Adds a single value to this list.
182    *
183    * @param value The value to add to this list.
184    * @return This object.
185    */
186   public ListBuilder<E> add(E value) {
187      if (list == null)
188         list = new ArrayList<>();
189      list.add(value);
190      return this;
191   }
192
193   /**
194    * Adds multiple values to this list.
195    *
196    * @param values The values to add to this list.
197    * @return This object.
198    */
199   @SuppressWarnings("unchecked")
200   public ListBuilder<E> add(E...values) {
201      for (E v : values)
202         add(v);
203      return this;
204   }
205
206   /**
207    * Adds entries to this list via JSON array strings.
208    *
209    * @param values The JSON array strings to parse and add to this list.
210    * @return This object.
211    */
212   public ListBuilder<E> addJson(String...values) {
213      return addAny((Object[])values);
214   }
215
216   /**
217    * Adds arbitrary values to this list.
218    *
219    * <p>
220    * Objects can be any of the following:
221    * <ul>
222    *    <li>The same type or convertible to the element type of this list.
223    *    <li>Collections or arrays of anything on this list.
224    *    <li>JSON array strings parsed and convertible to the element type of this list.
225    * </ul>
226    *
227    * @param values The values to add.
228    * @return This object.
229    */
230   public ListBuilder<E> addAny(Object...values) {
231      if (elementType == null)
232         throw new RuntimeException("Unknown element type.  Cannot use this method.");
233      try {
234         if (values != null) {
235            for (Object o : values) {
236               if (o != null) {
237                  if (o instanceof Collection) {
238                     ((Collection<?>)o).forEach(x -> addAny(x));
239                  } else if (o.getClass().isArray()) {
240                     for (int i = 0; i < Array.getLength(o); i++)
241                        addAny(Array.get(o, i));
242                  } else if (isJsonArray(o, false)) {
243                     new JsonList(o.toString()).forEach(x -> addAny(x));
244                  } else if (elementType.isInstance(o)) {
245                     add(elementType.cast(o));
246                  } else {
247                     add(toType(o, elementType, elementTypeArgs));
248                  }
249               }
250            }
251         }
252      } catch (ParseException e) {
253         throw asRuntimeException(e);
254      }
255      return this;
256   }
257
258   /**
259    * Appends a value to this list of the flag is true.
260    *
261    * @param flag The flag.
262    * @param value The value.
263    * @return This object.
264    */
265   public ListBuilder<E> addIf(boolean flag, E value) {
266      if (flag)
267         add(value);
268      return this;
269   }
270}
