013package org.apache.juneau.internal;
014
015import static org.apache.juneau.internal.ClassUtils.*;
016import static org.apache.juneau.internal.ConverterUtils.*;
017import static org.apache.juneau.internal.ThrowableUtils.*;
018import static org.apache.juneau.internal.StringUtils.*;
019
020import java.lang.reflect.*;
021import java.util.*;
022
023import org.apache.juneau.*;
024import org.apache.juneau.collections.*;
025import org.apache.juneau.parser.*;
026
027/**
028 * Builder for maps.
029 *
030 * <ul class='seealso'>
031 *    <li class='extlink'>{@source}
032 * </ul>
033 *
034 * @param <K> Key type.
035 * @param <V> Value type.
036 */
037public final class MapBuilder<K,V> {
038
039   private Map<K,V> map;
040   private boolean unmodifiable = false, sparse = false;
041   private Comparator<K> comparator = null;
042
043   private Class<K> keyType;
044   private Class<V> valueType;
045   private Type[] valueTypeArgs;
046
047   /**
048    * Constructor.
049    *
050    * @param keyType The key type.
051    * @param valueType The value type.
052    * @param valueTypeArgs The value type generic arguments if there are any.
053    */
054   public MapBuilder(Class<K> keyType, Class<V> valueType, Type...valueTypeArgs) {
055      this.keyType = keyType;
056      this.valueType = valueType;
057      this.valueTypeArgs = valueTypeArgs;
058   }
059
060   /**
061    * Constructor.
062    *
063    * @param addTo The map to add to.
064    */
065   public MapBuilder(Map<K,V> addTo) {
066      this.map = addTo;
067   }
068
069   /**
070    * Builds the map.
071    *
072    * @return A map conforming to the settings on this builder.
073    */
074   public Map<K,V> build() {
075      if (sparse) {
076         if (map != null && map.isEmpty())
077            map = null;
078      } else {
079         if (map == null)
080            map = new LinkedHashMap<>();
081      }
082      if (map != null) {
083         if (comparator != null) {
084            Map<K,V> m2 = new TreeMap<>(comparator);
085            m2.putAll(map);
086            map = m2;
087         }
088         if (unmodifiable)
089            map = Collections.unmodifiableMap(map);
090      }
091      return map;
092   }
093
094   /**
095    * When specified, the {@link #build()} method will return <jk>null</jk> if the map is empty.
096    *
097    * <p>
098    * Otherwise {@link #build()} will never return <jk>null</jk>.
099    *
100    * @return This object.
101    */
102   public MapBuilder<K,V> sparse() {
103      this.sparse = true;
104      return this;
105   }
106
107   /**
108    * When specified, {@link #build()} will return an unmodifiable map.
109    *
110    * @return This object.
111    */
112   public MapBuilder<K,V> unmodifiable() {
113      this.unmodifiable = true;
114      return this;
115   }
116
117   /**
118    * Forces the existing set to be copied instead of appended to.
119    *
120    * @return This object.
121    */
122   public MapBuilder<K,V> copy() {
123      if (map != null)
124         map = new LinkedHashMap<>(map);
125      return this;
126   }
127
128   /**
129    * Converts the set into a {@link SortedMap}.
130    *
131    * @return This object.
132    */
133   @SuppressWarnings("unchecked")
134   public MapBuilder<K,V> sorted() {
135      return sorted((Comparator<K>)Comparator.naturalOrder());
136   }
137
138   /**
139    * Converts the set into a {@link SortedMap} using the specified comparator.
140    *
141    * @param comparator The comparator to use for sorting.
142    * @return This object.
143    */
144   public MapBuilder<K,V> sorted(Comparator<K> comparator) {
145      this.comparator = comparator;
146      return this;
147   }
148
149   /**
150    * Appends the contents of the specified map into this map.
151    *
152    * <p>
153    * This is a no-op if the value is <jk>null</jk>.
154    *
155    * @param value The map to add to this map.
156    * @return This object.
157    */
158   public MapBuilder<K,V> addAll(Map<K,V> value) {
159      if (value != null) {
160         if (map == null)
161            map = new LinkedHashMap<>(value);
162         else
163            map.putAll(value);
164      }
165      return this;
166   }
167
168   /**
169    * Adds a single entry to this map.
170    *
171    * @param key The map key.
172    * @param value The map value.
173    * @return This object.
174    */
175   public MapBuilder<K,V> add(K key, V value) {
176      if (map == null)
177         map = new LinkedHashMap<>();
178      map.put(key, value);
179      return this;
180   }
181
182   /**
183    * Adds entries to this list via JSON object strings.
184    *
185    * @param values The JSON object strings to parse and add to this list.
186    * @return This object.
187    */
188   public MapBuilder<K,V> addJson(String...values) {
189      return addAny((Object[])values);
190   }
191
192   /**
193    * Adds arbitrary values to this list.
194    *
195    * <p>
196    * Objects can be any of the following:
197    * <ul>
198    *    <li>Maps of key/value types convertible to the key/value types of this map.
199    *    <li>JSON object strings parsed and convertible to the key/value types of this map.
200    * </ul>
201    *
202    * @param values The values to add.
203    * @return This object.
204    */
205   @SuppressWarnings("unchecked")
206   public MapBuilder<K,V> addAny(Object...values) {
207      if (keyType == null || valueType == null)
208         throw new RuntimeException("Unknown key and value types.  Cannot use this method.");
209      try {
210         for (Object o : values) {
211            if (o != null) {
212               if (o instanceof Map) {
213                  ((Map<Object,Object>)o).forEach((k,v) -> add(toType(k, keyType), toType(v, valueType, valueTypeArgs)));
214               } else if (isJsonObject(o, false)) {
215                  JsonMap.ofJson(o.toString()).forEach((k,v) -> add(toType(k, keyType), toType(v, valueType, valueTypeArgs)));
216               } else {
217                  throw new BasicRuntimeException("Invalid object type {0} passed to addAny()", className(o));
218               }
219            }
220         }
221      } catch (ParseException e) {
222         throw asRuntimeException(e);
223      }
224      return this;
225   }
226
227   /**
228    * Adds a list of key/value pairs to this map.
229    *
230    * @param pairs The pairs to add.
231    * @return This object.
232    */
233   @SuppressWarnings("unchecked")
234   public MapBuilder<K,V> addPairs(Object...pairs) {
235      if (pairs.length % 2 != 0)
236         throw new RuntimeException("Odd number of parameters passed into AMap.ofPairs()");
237      for (int i = 0; i < pairs.length; i+=2)
238         add((K)pairs[i], (V)pairs[i+1]);
239      return this;
240   }
241}
