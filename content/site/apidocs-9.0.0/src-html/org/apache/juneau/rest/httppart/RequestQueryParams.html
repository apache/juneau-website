035/**
036 * Represents the query parameters in an HTTP request.
037 *
038 * <p>
039 *    The {@link RequestQueryParams} object is the API for accessing the GET query parameters of an HTTP request.
040 *    It can be accessed by passing it as a parameter on your REST Java method:
041 * </p>
042 * <p class='bjava'>
043 *    <ja>@RestPost</ja>(...)
044 *    <jk>public</jk> Object myMethod(RequestQueryParams <jv>query</jv>) {...}
045 * </p>
046 *
047 * <h5 class='figure'>Example:</h5>
048 * <p class='bjava'>
049 *    <ja>@RestPost</ja>(...)
050 *    <jk>public</jk> Object myMethod(RequestQueryParams <jv>query</jv>) {
051 *
052 *       <jc>// Get query parameters converted to various types.</jc>
053 *       <jk>int</jk> <jv>p1</jv> = <jv>query</jv>.get(<js>"p1"</js>).asInteger().orElse(0);
054 *       String <jv>p2</jv> = <jv>query</jv>.get(<js>"p2"</js>).orElse(<jk>null</jk>);
055 *       UUID <jv>p3</jv> = <jv>query</jv>.get(<js>"p3"</js>).as(UUID.<jk>class</jk>).orElse(<jk>null</jk>);
056 *     }
057 * </p>
058 *
059 * <p>
060 *    An important distinction between the behavior of this object and <l>HttpServletRequest.getParameter(String)</l> is
061 *    that the former will NOT load the content of the request on FORM POSTS and will only look at parameters
062 *    found in the query string.
063 *    This can be useful in cases where you're mixing GET parameters and FORM POSTS and you don't want to
064 *    inadvertently read the content of the request to get a query parameter.
065 * </p>
066 *
067 * <p>
068 *    Some important methods on this class are:
069 * </p>
070 * <ul class='javatree'>
071 *    <li class='jc'>{@link RequestQueryParams}
072 *    <ul class='spaced-list'>
073 *       <li>Methods for retrieving query parameters:
074 *       <ul class='javatreec'>
075 *          <li class='jm'>{@link RequestQueryParams#contains(String...) contains(String...)}
076 *          <li class='jm'>{@link RequestQueryParams#containsAny(String...) containsAny(String...)}
077 *          <li class='jm'>{@link RequestQueryParams#get(Class) get(Class)}
078 *          <li class='jm'>{@link RequestQueryParams#get(String) get(String)}
079 *          <li class='jm'>{@link RequestQueryParams#getAll() getAll()}
080 *          <li class='jm'>{@link RequestQueryParams#getAll(String) getAll(String)}
081 *          <li class='jm'>{@link RequestQueryParams#getFirst(String) getFirst(String)}
082 *          <li class='jm'>{@link RequestQueryParams#getLast(String) getLast(String)}
083 *          <li class='jm'>{@link RequestQueryParams#getSearchArgs() getSearchArgs()}
084 *          <li class='jm'>{@link RequestQueryParams#getViewArgs() getViewArgs()}
085 *          <li class='jm'>{@link RequestQueryParams#getSortArgs() getSortArgs()}
086 *          <li class='jm'>{@link RequestQueryParams#getPageArgs() getPageArgs()}
087 *       </ul>
088 *       <li>Methods overridding query parameters:
089 *       <ul class='javatreec'>
090 *          <li class='jm'>{@link RequestQueryParams#add(NameValuePair...) add(NameValuePair...)}
091 *          <li class='jm'>{@link RequestQueryParams#add(String,Object) add(String,Object)}
092 *          <li class='jm'>{@link RequestQueryParams#addDefault(List) addDefault(List)}
093 *          <li class='jm'>{@link RequestQueryParams#addDefault(NameValuePair...) addDefault(NameValuePair...)}
094 *          <li class='jm'>{@link RequestQueryParams#addDefault(String,String) addDefault(String,String)}
095 *          <li class='jm'>{@link RequestQueryParams#remove(NameValuePair...) remove(NameValuePair...)}
096 *          <li class='jm'>{@link RequestQueryParams#remove(String...) remove(String...)}
097 *          <li class='jm'>{@link RequestQueryParams#set(NameValuePair...) set(NameValuePair...)}
098 *          <li class='jm'>{@link RequestQueryParams#set(String,Object) set(String,Object)}
099 *       </ul>
100 *       <li>Other methods:
101 *       <ul class='javatreec'>
102 *          <li class='jm'>{@link RequestQueryParams#asQueryString() asQueryString()}
103 *          <li class='jm'>{@link RequestQueryParams#copy() copy()}
104 *          <li class='jm'>{@link RequestQueryParams#isEmpty() isEmpty()}
105 *       </ul>
106 *    </ul>
107 * </ul>
108 *
109 * <p>
110 * Entries are stored in a case-sensitive map unless overridden via the constructor.
111 *
112 * <ul class='seealso'>
113 *    <li class='jc'>{@link RequestQueryParam}
114 *    <li class='ja'>{@link org.apache.juneau.http.annotation.Query}
115 *    <li class='ja'>{@link org.apache.juneau.http.annotation.HasQuery}
116 *    <li class='link'>{@doc jrs.HttpParts}
117 *    <li class='extlink'>{@source}
118 * </ul>
119 */
120public class RequestQueryParams {
121
122   private final RestRequest req;
123   private final boolean caseSensitive;
124   private final VarResolverSession vs;
125   private HttpPartParserSession parser;
126
127   private List<RequestQueryParam> list = new LinkedList<>();
128   private Map<String,List<RequestQueryParam>> map = new TreeMap<>();
129
130   /**
131    * Constructor.
132    *
133    * @param req The request creating this bean.
134    * @param query The raw parsed query parameter values.
135    * @param caseSensitive Whether case-sensitive name matching is enabled.
136    */
137   public RequestQueryParams(RestRequest req, Map<String,String[]> query, boolean caseSensitive) {
138      this.req = req;
139      this.caseSensitive = caseSensitive;
140      this.vs = req.getVarResolverSession();
141
142      for (Map.Entry<String,String[]> e : query.entrySet()) {
143         String name = e.getKey();
144         String key = key(name);
145         List<RequestQueryParam> l = list();
146
147         String[] values = e.getValue();
148         if (values == null)
149            values = new String[0];
150
151         // Fix for behavior difference between Tomcat and WAS.
152         // getParameter("foo") on "&foo" in Tomcat returns "".
153         // getParameter("foo") on "&foo" in WAS returns null.
154         if (values.length == 1 && values[0] == null)
155            values[0] = "";
156
157         for (String value : values) {
158            RequestQueryParam p = new RequestQueryParam(req, name, value);
159            list.add(p);
160            l.add(p);
161         }
162         map.put(key, l);
163      }
164   }
165
166   /**
167    * Copy constructor.
168    */
169   private RequestQueryParams(RequestQueryParams copyFrom) {
170      req = copyFrom.req;
171      caseSensitive = copyFrom.caseSensitive;
172      parser = copyFrom.parser;
173      list.addAll(copyFrom.list);
174      map.putAll(copyFrom.map);
175      vs = copyFrom.vs;
176   }
177
178   /**
179    * Sets the parser to use for part values.
180    *
181    * @param value The new value for this setting.
182    * @return This object.
183    */
184   public RequestQueryParams parser(HttpPartParserSession value) {
185      this.parser = value;
186      for (RequestQueryParam p : list)
187         p.parser(parser);
188      return this;
189   }

195   /**
196    * Adds default entries to these parameters.
197    *
198    * <p>
199    * Similar to {@link #set(String, Object)} but doesn't override existing values.
200    *
201    * @param pairs
202    *    The default entries.
203    *    <br>Can be <jk>null</jk>.
204    * @return This object.
205    */
206   public RequestQueryParams addDefault(List<? extends NameValuePair> pairs) {
207      for (NameValuePair p : pairs) {
208         String name = p.getName();
209         String key = key(name);
210         List<RequestQueryParam> l = map.get(key);
211         boolean hasAllBlanks = l != null && l.stream().allMatch(x -> StringUtils.isEmpty(x.getValue()));
212         if (l == null || hasAllBlanks) {
213            if (hasAllBlanks)
214               list.removeAll(l);
215            RequestQueryParam x = new RequestQueryParam(req, name, vs.resolve(p.getValue()));
216            list.add(x);
217            map.put(key, list(x));
218         }
219      }
220      return this;
221   }

249   /**
250    * Returns all the parameters with the specified name.
251    *
252    * @param name The parameter name.
253    * @return The list of all parameters with the specified name, or an empty list if none are found.
254    */
255   public List<RequestQueryParam> getAll(String name) {
256      assertArgNotNull("name", name);
257      List<RequestQueryParam> l = map.get(key(name));
258      return l == null ? emptyList() : unmodifiable(l);

270   /**
271    * Returns <jk>true</jk> if the parameters with the specified names are present.
272    *
273    * @param names The parameter names.  Must not be <jk>null</jk>.
274    * @return <jk>true</jk> if the parameters with the specified names are present.
275    */
276   public boolean contains(String...names) {
277      assertArgNotNull("names", names);
278      for (String n : names)
279         if (! map.containsKey(key(n)))
280            return false;
281      return true;
282   }
283
284   /**
285    * Returns <jk>true</jk> if the parameter with any of the specified names are present.
286    *
287    * @param names The parameter names.  Must not be <jk>null</jk>.
288    * @return <jk>true</jk> if the parameter with any of the specified names are present.
289    */
290   public boolean containsAny(String...names) {
291      assertArgNotNull("names", names);
292      for (String n : names)
293         if (map.containsKey(key(n)))
294            return true;
295      return false;
296   }

307   /**
308    * Adds a parameter value.
309    *
310    * <p>
311    * Parameter is added to the end.
312    * <br>Existing parameter with the same name are not changed.
313    *
314    * @param name The parameter name.  Must not be <jk>null</jk>.
315    * @param value The parameter value.
316    * @return This object.
317    */
318   public RequestQueryParams add(String name, Object value) {
319      assertArgNotNull("name", name);
320      String key = key(name);
321      RequestQueryParam h = new RequestQueryParam(req, name, stringify(value)).parser(parser);
322      if (map.containsKey(key))
323         map.get(key).add(h);
324      else
325         map.put(key, list(h));
326      list.add(h);
327      return this;
328   }
329
330   /**
331    * Adds request parameter values.
332    *
333    * <p>
334    * Parameters are added to the end.
335    * <br>Existing parameters with the same name are not changed.
336    *
337    * @param parameters The parameter objects.  Must not be <jk>null</jk>.
338    * @return This object.
339    */
340   public RequestQueryParams add(NameValuePair...parameters) {
341      assertArgNotNull("parameters", parameters);
342      for (NameValuePair p : parameters) {
343         if (p != null)
344            add(p.getName(), p.getValue());
345      }
346      return this;
347   }
348
349   /**
350    * Sets a parameter value.
351    *
352    * <p>
353    * Parameter is added to the end.
354    * <br>Any previous parameters with the same name are removed.
355    *
356    * @param name The parameter name.  Must not be <jk>null</jk>.
357    * @param value
358    *    The parameter value.
359    *    <br>Converted to a string using {@link Object#toString()}.
360    *    <br>Can be <jk>null</jk>.
361    * @return This object.
362    */
363   public RequestQueryParams set(String name, Object value) {
364      assertArgNotNull("name", name);
365      String key = key(name);
366      RequestQueryParam p = new RequestQueryParam(req, name, stringify(value)).parser(parser);
367      if (map.containsKey(key))
368         list.removeIf(x->caseSensitive?x.getName().equals(name):x.getName().equalsIgnoreCase(name));
369      list.add(p);
370      map.put(key, list(p));
371      return this;
372   }
373
374
375   /**
376    * Sets request header values.
377    *
378    * <p>
379    * Parameters are added to the end of the headers.
380    * <br>Any previous parameters with the same name are removed.
381    *
382    * @param parameters The parameters to set.  Must not be <jk>null</jk> or contain <jk>null</jk>.
383    * @return This object.
384    */
385   public RequestQueryParams set(NameValuePair...parameters) {
386      assertArgNotNull("headers", parameters);
387      for (NameValuePair p : parameters)
388         remove(p);
389      for (NameValuePair p : parameters)
390         add(p);
391      return this;
392   }
393
394   /**
395    * Remove parameters.
396    *
397    * @param name The parameter names.  Must not be <jk>null</jk>.
398    * @return This object.
399    */
400   public RequestQueryParams remove(String...name) {
401      assertArgNotNull("name", name);
402      for (String n : name) {
403         String key = key(n);
404         if (map.containsKey(key))
405            list.removeAll(map.get(key));
406         map.remove(key);
407      }
408      return this;
409   }
410
411   /**
412    * Remove parameters.
413    *
414    * @param parameters The parameters to remove.  Must not be <jk>null</jk>.
415    * @return This object.
416    */
417   public RequestQueryParams remove(NameValuePair...parameters) {
418      for (NameValuePair p : parameters)
419         remove(p.getName());
420      return this;
421   }

427   /**
428    * Returns the first parameter with the specified name.
429    *
430    * <p>
431    * Note that this method never returns <jk>null</jk> and that {@link RequestQueryParam#isPresent()} can be used
432    * to test for the existence of the parameter.
433    *
434    * @param name The parameter name.
435    * @return The parameter.  Never <jk>null</jk>.
436    */
437   public RequestQueryParam getFirst(String name) {
438      assertArgNotNull("name", name);
439      List<RequestQueryParam> l = map.get(key(name));
440      return (l == null || l.isEmpty() ? new RequestQueryParam(req, name, null).parser(parser) : l.get(0));
441   }

443   /**
444    * Returns the last parameter with the specified name.
445    *
446    * <p>
447    * Note that this method never returns <jk>null</jk> and that {@link RequestQueryParam#isPresent()} can be used
448    * to test for the existence of the parameter.
449    *
450    * @param name The parameter name.
451    * @return The parameter.  Never <jk>null</jk>.
452    */
453   public RequestQueryParam getLast(String name) {
454      assertArgNotNull("name", name);
455      List<RequestQueryParam> l = map.get(key(name));
456      return (l == null || l.isEmpty() ? new RequestQueryParam(req, name, null).parser(parser) : l.get(l.size()-1));
457   }

460    * Returns the last parameter with the specified name.
461    *
462    * <p>
463    * This is equivalent to {@link #getLast(String)}.
464    *
465    * @param name The parameter name.

467    */

469      return getLast(name);
470   }

472   /**
473    * Returns the query parameter as the specified bean type.
474    *
475    * <p>
476    * Type must have a name specified via the {@link org.apache.juneau.http.annotation.Query} annotation
477    * and a public constructor that takes in either <c>value</c> or <c>name,value</c> as strings.
478    *
479    * @param <T> The bean type to create.
480    * @param type The bean type to create.
481    * @return The bean, never <jk>null</jk>.
482    */
483   public <T> Optional<T> get(Class<T> type) {
484      ClassMeta<T> cm = req.getBeanSession().getClassMeta(type);
485      String name = HttpParts.getName(QUERY, cm).orElseThrow(()->new BasicRuntimeException("@Query(name) not found on class {0}", className(type)));
486      return get(name).as(type);
487   }

493   /**
494    * Converts this object to a query string.
495    *
496    * <p>
497    * Returned query string does not start with <js>'?'</js>.
498    *
499    * @return A new query string, or an empty string if this object is empty.
500    */
501   public String asQueryString() {
502      StringBuilder sb = new StringBuilder();
503      for (RequestQueryParam e : list) {
504         if (sb.length() > 0)
505            sb.append("&");
506         sb.append(urlEncode(e.getName())).append('=').append(urlEncode(e.getValue()));
507      }
508      return sb.toString();
509   }

560   /**
561    * Converts the parameters to a readable string.
562    *
563    * @param sorted Sort the parameters by name.
564    * @return A JSON string containing the contents of the parameters.
565    */
566   public String toString(boolean sorted) {
567      JsonMap m = JsonMap.create();
568      if (sorted) {
569         for (List<RequestQueryParam> p1 : map.values())
570            for (RequestQueryParam p2 : p1)
571               m.append(p2.getName(), p2.getValue());
572      } else {
573         for (RequestQueryParam p : list)
574            m.append(p.getName(), p.getValue());
575      }
576      return m.toString();
577   }
578
579   private String key(String name) {
580      return caseSensitive ? name : name.toLowerCase();
581   }
582
583   @Override /* Object */
584   public String toString() {
585      return toString(false);
586   }
