159   private final ConcurrentHashMap<Long,ThrownStats> db = new ConcurrentHashMap<>();
160   private final Optional<ThrownStore> parent;
161   private final BeanStore beanStore;
162   private final Class<? extends ThrownStats> statsImplClass;
163   private final Set<String> ignoreClasses;
164
165   /**
166    * Constructor.
167    */
168   public ThrownStore() {
169      this(create(BeanStore.INSTANCE));
170   }
171
172   /**
173    * Constructor.
174    *
175    * @param builder The builder for this object.
176    */
177   public ThrownStore(Builder builder) {
178      this.parent = optional(builder.parent);
179      this.beanStore = builder.beanStore();
180
181      this.statsImplClass = firstNonNull(builder.statsImplClass, parent.isPresent() ? parent.get().statsImplClass : null, null);
182
183      Set<String> s = null;
184      if (builder.ignoreClasses != null)
185         s = builder.ignoreClasses.stream().map(x->x.getName()).collect(toSet());
186      if (s == null && parent.isPresent())
187         s = parent.get().ignoreClasses;
188      if (s == null)
189         s = Collections.emptySet();
190      this.ignoreClasses = unmodifiable(s);
191   }

194   /**
195    * Adds the specified thrown exception to this database.
196    *
197    * @param e The exception to add.
198    * @return This object.
199    */
200   public ThrownStats add(Throwable e) {
201      ThrownStats s = find(e);
202      s.increment();
203      parent.ifPresent(x->x.add(e));
204      return s;
205   }

207   /**
208    * Retrieves the stats for the specified thrown exception.
209    *
210    * @param e The exception.
211    * @return A clone of the stats, never <jk>null</jk>.
212    */
213   public Optional<ThrownStats> getStats(Throwable e) {
214      return getStats(hash(e));
215   }
216
217   /**
218    * Retrieves the stack trace information for the exception with the specified hash as calculated by {@link #hash(Throwable)}.
219    *
220    * @param hash The hash of the exception.
221    * @return A clone of the stack trace info, never <jk>null</jk>.
222    */
223   public Optional<ThrownStats> getStats(long hash) {
224      ThrownStats s = db.get(hash);
225      return optional(s == null ? null : s.clone());
226   }
227
228   /**
229    * Returns the list of all stack traces in this database.
230    *
231    * @return The list of all stack traces in this database, cloned and sorted by count descending.
232    */
233   public List<ThrownStats> getStats() {
234      return db.values().stream().map(x -> x.clone()).sorted(comparingInt(ThrownStats::getCount).reversed()).collect(toList());
235   }

237   /**
238    * Clears out the stack trace cache.

242   }

244   /**
245    * Calculates a 32-bit hash for the specified throwable based on the stack trace generated by {@link #createStackTrace(Throwable)}.
246    *
247    * <p>
248    * Subclasses can override this method to provide their own implementation.
249    *
250    * @param t The throwable to calculate the stack trace on.
251    * @return A calculated hash.
252    */
253   protected long hash(Throwable t) {
254      long h = 1125899906842597L; // prime
255      for (String s : createStackTrace(t)) {
256         int len = s.length();
257         for (int i = 0; i < len; i++)
258            h = 31*h + s.charAt(i);
259      }
260      return h;
261   }
262
263   /**
264    * Converts the stack trace for the specified throwable into a simple list of strings.
265    *
266    * <p>
267    * The stack trace elements for the throwable are sent through {@link #normalize(StackTraceElement)} to convert
268    * them to simple strings.
269    *
270    *
271    * @param t The throwable to create the stack trace for.
272    * @return A modifiable list of strings.
273    */
274   protected List<String> createStackTrace(Throwable t) {
275      return alist(t.getStackTrace()).stream().filter(x -> include(x)).map(x -> normalize(x)).collect(toList());
276   }
277
278   /**
279    * Returns <jk>true</jk> if the specified stack trace element should be included in {@link #createStackTrace(Throwable)}.
280    *
281    * @param e The stack trace element.
282    * @return <jk>true</jk> if the specified stack trace element should be included in {@link #createStackTrace(Throwable)}.
283    */
284   protected boolean include(StackTraceElement e) {
285      return true;
286   }
287
288   /**
289    * Converts the specified stack trace element into a normalized string.
290    *
291    * <p>
292    * The default implementation simply replaces <js>"\\$.*"</js> with <js>"..."</js> which should take care of stuff like stack
293    * trace elements of lambda expressions.
294    *
295    * @param e The stack trace element to convert.
296    * @return The converted stack trace element.
297    */
298   protected String normalize(StackTraceElement e) {
299      if (ignoreClasses.contains(e.getClassName()))
300         return "<ignored>";
301      String s = e.toString();
302      int i = s.indexOf('$');
303      if (i == -1)
304         return s;
305      int j = s.indexOf('(', i);
306      if (j == -1)
307         return s;  // Probably can't happen.
308      String s2 = s.substring(0, i), s3 = s.substring(j);
309      if (ignoreClasses.contains(s2))
310         return "<ignored>";
311      return s2 + "..." + s3;
312   }
313
314   private ThrownStats find(final Throwable t) {
315
316      if (t == null)
317         return null;
318
319      long hash = hash(t);
320
321      ThrownStats stc = db.get(hash);
322      if (stc == null) {
323         stc = ThrownStats
324            .create(beanStore)
325            .type(statsImplClass)
326            .throwable(t)
327            .hash(hash)
328            .stackTrace(createStackTrace(t))
329            .causedBy(find(t.getCause()))
330            .build();
331
332         db.putIfAbsent(hash, stc);
333         stc = db.get(hash);
334      }
335
336      return stc;
337   }
