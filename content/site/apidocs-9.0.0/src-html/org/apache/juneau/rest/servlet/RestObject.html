013package org.apache.juneau.rest.servlet;
014
015import static org.apache.juneau.internal.ClassUtils.*;
016
017import java.text.*;
018import java.util.concurrent.atomic.*;
019import java.util.function.*;
020import java.util.logging.*;
021
022import javax.servlet.*;
023import javax.servlet.http.*;
024
025import org.apache.juneau.http.response.*;
026import org.apache.juneau.internal.*;
027import org.apache.juneau.rest.*;
028
029/**
030 * Identical to {@link RestServlet} but doesn't extend from {@link HttpServlet}.
031 *
032 * <p>
033 * This is particularly useful in Spring Boot environments that auto-detect servlets to deploy in servlet containers,
034 * but you want this resource to be deployed as a child instead.
035 *
036 * <ul class='seealso'>
037 *    <li class='link'>{@doc jrs.AnnotatedClasses}
038 *    <li class='extlink'>{@source}
039 * </ul>
040 */
041public abstract class RestObject {
042
043   private AtomicReference<RestContext> context = new AtomicReference<>();
044
045   //-----------------------------------------------------------------------------------------------------------------
046   // Context methods.
047   //-----------------------------------------------------------------------------------------------------------------
048
049   /**
050    * Sets the context object for this servlet.
051    *
052    * @param context Sets the context object on this servlet.
053    * @throws ServletException If error occurred during post-initialiation.
054    */
055   protected void setContext(RestContext context) throws ServletException {
056      this.context.set(context);
057   }
058
059   /**
060    * Returns the read-only context object that contains all the configuration information about this resource.
061    *
062    * @return The context information on this servlet.
063    */
064   protected RestContext getContext() {
065      if (context.get() == null)
066         throw new InternalServerError("RestContext object not set on resource.");
067      return context.get();
068   }
069
070   //-----------------------------------------------------------------------------------------------------------------
071   // Convenience logger methods
072   //-----------------------------------------------------------------------------------------------------------------
073
074   /**
075    * Log a message.
076    *
077    * <p>
078    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
079    *
080    * @param level The log level.
081    * @param msg The message to log.
082    * @param args Optional {@link MessageFormat}-style arguments.
083    */
084   public void log(Level level, String msg, Object...args) {
085      doLog(level, null, () -> StringUtils.format(msg, args));
086   }
087
088   /**
089    * Log a message.
090    *
091    * <p>
092    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
093    *
094    * @param level The log level.
095    * @param cause The cause.
096    * @param msg The message to log.
097    * @param args Optional {@link MessageFormat}-style arguments.
098    */
099   public void log(Level level, Throwable cause, String msg, Object...args) {
100      doLog(level, cause, () -> StringUtils.format(msg, args));
101   }
102
103   /**
104    * Main logger method.
105    *
106    * <p>
107    * The default behavior logs a message to the Java logger of the class name.
108    *
109    * <p>
110    * Subclasses can override this method to implement their own logger handling.
111    *
112    * @param level The log level.
113    * @param cause Optional throwable.
114    * @param msg The message to log.
115    */
116   protected void doLog(Level level, Throwable cause, Supplier<String> msg) {
117      RestContext c = context.get();
118      Logger logger = c == null ? null : c.getLogger();
119      if (logger == null)
120         logger = Logger.getLogger(className(this));
121      logger.log(level, cause, msg);
122   }
123
124   //-----------------------------------------------------------------------------------------------------------------
125   // Other methods
126   //-----------------------------------------------------------------------------------------------------------------
127
128   /**
129    * Returns the current thread-local HTTP request.
130    *
131    * @return The current thread-local HTTP request, or <jk>null</jk> if it wasn't created.
132    */
133   public synchronized RestRequest getRequest() {
134      return getContext().getLocalSession().getOpSession().getRequest();
135   }
136
137   /**
138    * Returns the current thread-local HTTP response.
139    *
140    * @return The current thread-local HTTP response, or <jk>null</jk> if it wasn't created.
141    */
142   public synchronized RestResponse getResponse() {
143      return getContext().getLocalSession().getOpSession().getResponse();
144   }
145}
