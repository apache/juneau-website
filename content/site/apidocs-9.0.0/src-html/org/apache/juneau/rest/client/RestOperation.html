013package org.apache.juneau.rest.client;
014
015import java.net.*;
016import java.util.*;
017
018import org.apache.http.client.utils.*;
019import org.apache.juneau.http.*;
020
021/**
022 * Aggregates the HTTP method, URL, and optional body into a single bean.
023 *
024 * <ul class='seealso'>
025 *    <li class='link'>{@doc juneau-rest-client}
026 *    <li class='extlink'>{@source}
027 * </ul>
028 */
029public class RestOperation {
030
031   /**
032    * A placeholder for a non-existent body.
033    * Used to identify when form-data should be used in a request body.
034    * Note that this is different than a <jk>null</jk> body since a <jk>null</jk> can be a serialized request.
035    */
036   public static final Object NO_BODY = "NO_BODY";
037
038   private final Object url;
039   private final String method;
040   private final Object content;
041   private boolean hasContent;
042
043   /**
044    * Creator.
045    *
046    * @param method The HTTP method.
047    * @param url
048    *    The URI of the remote REST resource.
049    *    <br>Can be any of the following types:
050    *    <ul>
051    *       <li>{@link URIBuilder}
052    *       <li>{@link URI}
053    *       <li>{@link URL}
054    *       <li>{@link String}
055    *       <li>{@link Object} - Converted to <c>String</c> using <c>toString()</c>
056    *    </ul>
057    * @return A new {@link RestOperation} object.
058    */
059   public static RestOperation of(String method, Object url) {
060      return new RestOperation(method, url, NO_BODY);
061   }
062
063   /**
064    * Creator.
065    *
066    * @param method The HTTP method.
067    * @param url
068    *    The URI of the remote REST resource.
069    *    <br>Can be any of the following types:
070    *    <ul>
071    *       <li>{@link URIBuilder}
072    *       <li>{@link URI}
073    *       <li>{@link URL}
074    *       <li>{@link String}
075    *       <li>{@link Object} - Converted to <c>String</c> using <c>toString()</c>
076    *    </ul>
077    * @param body The HTTP body.
078    * @return A new {@link RestOperation} object.
079    */
080   public static RestOperation of(String method, Object url, Object body) {
081      return new RestOperation(method, url, body);
082   }
083
084   /**
085    * Constructor.
086    *
087    * @param method The HTTP method.
088    * @param url
089    *    The URI of the remote REST resource.
090    *    <br>Can be any of the following types:
091    *    <ul>
092    *       <li>{@link URIBuilder}
093    *       <li>{@link URI}
094    *       <li>{@link URL}
095    *       <li>{@link String}
096    *       <li>{@link Object} - Converted to <c>String</c> using <c>toString()</c>
097    *    </ul>
098    * @param body The HTTP body.
099    */
100   public RestOperation(String method, Object url, Object body) {
101      this.url = url;
102      this.method = method.toUpperCase(Locale.ENGLISH);
103      this.content = body;
104      this.hasContent = HttpMethod.hasContent(method);
105   }
106
107   /**
108    * Bean property getter:  <property>url</property>.
109    *
110    * @return The value of the <property>url</property> property on this bean, or <jk>null</jk> if it is not set.
111    */
112   public Object getUri() {
113      return url;
114   }
115
116   /**
117    * Bean property getter:  <property>method</property>.
118    *
119    * @return The value of the <property>method</property> property on this bean, or <jk>null</jk> if it is not set.
120    */
121   public String getMethod() {
122      return method;
123   }
124
125   /**
126    * Bean property getter:  <property>content</property>.
127    *
128    * @return
129    *    The value of the <property>content</property> property on this bean.
130    *    <br>Returns {@link #NO_BODY} if the request does not have a body set.
131    *    <br>A <jk>null</jk> value means <jk>null</jk> should be the serialized response.
132    */
133   public Object getContent() {
134      return content;
135   }
136
137   /**
138    * Identifies whether this HTTP method typically has content.
139    *
140    * @return <jk>true</jk> if this HTTP method typically has content.
141    */
142   public boolean hasContent() {
143      return hasContent;
144   }
145
146   /**
147    * Overrides the default value for the {@link #hasContent()} method.
148    *
149    * @param value The new value.
150    * @return This object.
151    */
152   public RestOperation hasContent(boolean value) {
153      this.hasContent = value;
154      return this;
155   }
156}
