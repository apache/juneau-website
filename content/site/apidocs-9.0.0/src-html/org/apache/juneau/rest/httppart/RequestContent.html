037/**
038 * Contains the content of the HTTP request.
039 *
040 * <p>
041 *    The {@link RequestContent} object is the API for accessing the content of an HTTP request.
042 *    It can be accessed by passing it as a parameter on your REST Java method:
043 * </p>
044 * <p class='bjava'>
045 *    <ja>@RestPost</ja>(...)
046 *    <jk>public</jk> Object myMethod(RequestContent <jv>content</jv>) {...}
047 * </p>
048 *
049 * <h5 class='figure'>Example:</h5>
050 * <p class='bjava'>
051 *    <ja>@RestPost</ja>(...)
052 *    <jk>public void</jk> doPost(RequestContent <jv>content</jv>) {
053 *       <jc>// Convert content to a linked list of Person objects.</jc>
054 *       List&lt;Person&gt; <jv>list</jv> = <jv>content</jv>.as(LinkedList.<jk>class</jk>, Person.<jk>class</jk>);
055 *       ...
056 *    }
057 * </p>
058 *
059 * <p>
060 *    Some important methods on this class are:
061 * </p>
062 * <ul class='javatree'>
063 *    <li class='jc'>{@link RequestContent}
064 *    <ul class='spaced-list'>
065 *       <li>Methods for accessing the raw contents of the request content:
066 *       <ul class='javatreec'>
067 *          <li class='jm'>{@link RequestContent#asBytes() asBytes()}
068 *          <li class='jm'>{@link RequestContent#asHex() asHex()}
069 *          <li class='jm'>{@link RequestContent#asSpacedHex() asSpacedHex()}
070 *          <li class='jm'>{@link RequestContent#asString() asString()}
071 *          <li class='jm'>{@link RequestContent#getInputStream() getInputStream()}
072 *          <li class='jm'>{@link RequestContent#getReader() getReader()}
073 *       </ul>
074 *       <li>Methods for parsing the contents of the request content:
075 *       <ul class='javatreec'>
076 *          <li class='jm'>{@link RequestContent#as(Class) as(Class)}
077 *          <li class='jm'>{@link RequestContent#as(Type, Type...) as(Type, Type...)}
078 *          <li class='jm'>{@link RequestContent#setSchema(HttpPartSchema) setSchema(HttpPartSchema)}
079 *       </ul>
080 *       <li>Other methods:
081 *       <ul class='javatreec'>
082 *          <li class='jm'>{@link RequestContent#cache() cache()}
083 *          <li class='jm'>{@link RequestContent#getParserMatch() getParserMatch()}
084 *       </ul>
085 *    </ul>
086 * </ul>
087 *
088 * <ul class='seealso'>
089 *    <li class='link'>{@doc jrs.HttpParts}
090 *    <li class='extlink'>{@source}
091 * </ul>
092 */

197   /**
198    * Reads the input from the HTTP request parsed into a POJO.
199    *
200    * <p>
201    * The parser used is determined by the matching <c>Content-Type</c> header on the request.
202    *
203    * <p>
204    * If type is <jk>null</jk> or <code>Object.<jk>class</jk></code>, then the actual type will be determined
205    * automatically based on the following input:
206    * <table class='styled'>
207    *    <tr><th>Type</th><th>JSON input</th><th>XML input</th><th>Return type</th></tr>
208    *    <tr>
209    *       <td>object</td>
210    *       <td><js>"{...}"</js></td>
211    *       <td><code><xt>&lt;object&gt;</xt>...<xt>&lt;/object&gt;</xt></code><br><code><xt>&lt;x</xt> <xa>type</xa>=<xs>'object'</xs><xt>&gt;</xt>...<xt>&lt;/x&gt;</xt></code></td>
212    *       <td>{@link JsonMap}</td>
213    *    </tr>
214    *    <tr>
215    *       <td>array</td>
216    *       <td><js>"[...]"</js></td>
217    *       <td><code><xt>&lt;array&gt;</xt>...<xt>&lt;/array&gt;</xt></code><br><code><xt>&lt;x</xt> <xa>type</xa>=<xs>'array'</xs><xt>&gt;</xt>...<xt>&lt;/x&gt;</xt></code></td>
218    *       <td>{@link JsonList}</td>
219    *    </tr>
220    *    <tr>
221    *       <td>string</td>
222    *       <td><js>"'...'"</js></td>
223    *       <td><code><xt>&lt;string&gt;</xt>...<xt>&lt;/string&gt;</xt></code><br><code><xt>&lt;x</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>...<xt>&lt;/x&gt;</xt></code></td>
224    *       <td>{@link String}</td>
225    *    </tr>
226    *    <tr>
227    *       <td>number</td>
228    *       <td><c>123</c></td>
229    *       <td><code><xt>&lt;number&gt;</xt>123<xt>&lt;/number&gt;</xt></code><br><code><xt>&lt;x</xt> <xa>type</xa>=<xs>'number'</xs><xt>&gt;</xt>...<xt>&lt;/x&gt;</xt></code></td>
230    *       <td>{@link Number}</td>
231    *    </tr>
232    *    <tr>
233    *       <td>boolean</td>
234    *       <td><jk>true</jk></td>
235    *       <td><code><xt>&lt;boolean&gt;</xt>true<xt>&lt;/boolean&gt;</xt></code><br><code><xt>&lt;x</xt> <xa>type</xa>=<xs>'boolean'</xs><xt>&gt;</xt>...<xt>&lt;/x&gt;</xt></code></td>
236    *       <td>{@link Boolean}</td>
237    *    </tr>
238    *    <tr>
239    *       <td>null</td>
240    *       <td><jk>null</jk> or blank</td>
241    *       <td><code><xt>&lt;null/&gt;</xt></code> or blank<br><code><xt>&lt;x</xt> <xa>type</xa>=<xs>'null'</xs><xt>/&gt;</xt></code></td>
242    *       <td><jk>null</jk></td>
243    *    </tr>
244    * </table>
245    *
246    * <p>
247    * Refer to {@doc jm.PojoCategories} for a complete definition of supported POJOs.
248    *
249    * <h5 class='section'>Examples:</h5>
250    * <p class='bjava'>
251    *    <jc>// Parse into an integer.</jc>
252    *    <jk>int</jk> <jv>content1</jv> = <jv>req</jv>.getContent().as(<jk>int</jk>.<jk>class</jk>);
253    *
254    *    <jc>// Parse into an int array.</jc>
255    *    <jk>int</jk>[] <jv>content2</jv> = <jv>req</jv>.getContent().as(<jk>int</jk>[].<jk>class</jk>);
256
257    *    <jc>// Parse into a bean.</jc>
258    *    MyBean <jv>content3</jv> = <jv>req</jv>.getContent().as(MyBean.<jk>class</jk>);
259    *
260    *    <jc>// Parse into a linked-list of objects.</jc>
261    *    List <jv>content4</jv> = <jv>req</jv>.getContent().as(LinkedList.<jk>class</jk>);
262    *
263    *    <jc>// Parse into a map of object keys/values.</jc>
264    *    Map <jv>content5</jv> = <jv>req</jv>.getContent().as(TreeMap.<jk>class</jk>);
265    * </p>
266    *
267    * <ul class='notes'>
268    *    <li class='note'>
269    *       If {@code allowContentParam} init parameter is true, then first looks for {@code &content=xxx} in the URL query string.
270    * </ul>
271    *
272    * @param type The class type to instantiate.
273    * @param <T> The class type to instantiate.
274    * @return The input parsed to a POJO.
275    * @throws BadRequest Thrown if input could not be parsed or fails schema validation.
276    * @throws UnsupportedMediaType Thrown if the Content-Type header value is not supported by one of the parsers.
277    * @throws InternalServerError Thrown if an {@link IOException} occurs.
278    */
279   public <T> T as(Class<T> type) throws BadRequest, UnsupportedMediaType, InternalServerError {
280      return getInner(getClassMeta(type));
281   }

283   /**
284    * Reads the input from the HTTP request parsed into a POJO.
285    *
286    * <p>
287    * This is similar to {@link #as(Class)} but allows for complex collections of POJOs to be created.
288    *
289    * <h5 class='section'>Examples:</h5>
290    * <p class='bjava'>
291    *    <jc>// Parse into a linked-list of strings.</jc>
292    *    List&lt;String&gt; <jv>content1</jv> = <jv>req</jv>.getContent().as(LinkedList.<jk>class</jk>, String.<jk>class</jk>);
293    *
294    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
295    *    List&lt;List&lt;String&gt;&gt; <jv>content2</jv> = <jv>req</jv>.getContent().as(LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
296    *
297    *    <jc>// Parse into a map of string keys/values.</jc>
298    *    Map&lt;String,String&gt; <jv>content3</jv> = <jv>req</jv>.getContent().as(TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
299    *
300    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
301    *    Map&lt;String,List&lt;MyBean&gt;&gt; <jv>content4</jv> = <jv>req</jv>.getContent().as(TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
302    * </p>
303    *
304    * <ul class='notes'>
305    *    <li class='note'>
306    *       <c>Collections</c> must be followed by zero or one parameter representing the value type.
307    *    <li class='note'>
308    *       <c>Maps</c> must be followed by zero or two parameters representing the key and value types.
309    *    <li class='note'>
310    *       If {@code allowContentParam} init parameter is true, then first looks for {@code &content=xxx} in the URL query string.
311    * </ul>
312    *
313    * @param type
314    *    The type of object to create.
315    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
316    * @param args
317    *    The type arguments of the class if it's a collection or map.
318    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
319    *    <br>Ignored if the main type is not a map or collection.
320    * @param <T> The class type to instantiate.
321    * @return The input parsed to a POJO.
322    * @throws BadRequest Thrown if input could not be parsed or fails schema validation.
323    * @throws UnsupportedMediaType Thrown if the Content-Type header value is not supported by one of the parsers.
324    * @throws InternalServerError Thrown if an {@link IOException} occurs.
325    */
326   public <T> T as(Type type, Type...args) throws BadRequest, UnsupportedMediaType, InternalServerError {
327      return getInner(this.<T>getClassMeta(type, args));
328   }

394   /**
395    * Returns the HTTP content content as a {@link Reader}.
396    *
397    * <ul class='notes'>
398    *    <li class='note'>
399    *       If {@code allowContentParam} init parameter is true, then first looks for {@code &content=xxx} in the URL query string.
400    *    <li class='note'>
401    *       Automatically handles GZipped input streams.
402    * </ul>
403    *
404    * @return The content contents as a reader.
405    * @throws IOException Thrown by underlying stream.
406    */
407   public BufferedReader getReader() throws IOException {
408      Reader r = getUnbufferedReader();
409      if (r instanceof BufferedReader)
410         return (BufferedReader)r;
411      int len = req.getHttpServletRequest().getContentLength();
412      int buffSize = len <= 0 ? 8192 : Math.max(len, 8192);
413      return new BufferedReader(r, buffSize);
414   }

416   /**
417    * Same as {@link #getReader()}, but doesn't encapsulate the result in a {@link BufferedReader};
418    *
419    * @return An unbuffered reader.
420    * @throws IOException Thrown by underlying stream.
421    */
422   protected Reader getUnbufferedReader() throws IOException {
423      if (content != null)
424         return new CharSequenceReader(new String(content, UTF8));
425      return new InputStreamReader(getInputStream(), req.getCharset());
426   }

428   /**
429    * Returns the HTTP content content as an {@link InputStream}.
430    *
431    * @return The negotiated input stream.
432    * @throws IOException If any error occurred while trying to get the input stream or wrap it in the GZIP wrapper.
433    */
434   public ServletInputStream getInputStream() throws IOException {
435
436      if (content != null)
437         return new BoundedServletInputStream(content);
438
439      Encoder enc = getEncoder();
440
441      InputStream is = req.getHttpServletRequest().getInputStream();
442
443      if (enc == null)
444         return new BoundedServletInputStream(is, maxInput);
445
446      return new BoundedServletInputStream(enc.getInputStream(is), maxInput);
447   }

449   /**
450    * Returns the parser and media type matching the request <c>Content-Type</c> header.
451    *
452    * @return
453    *    The parser matching the request <c>Content-Type</c> header, or {@link Optional#empty()} if no matching parser was
454    *    found.
455    *    Includes the matching media type.
456    */
457   public Optional<ParserMatch> getParserMatch() {
458      if (mediaType != null && parser != null)
459         return optional(new ParserMatch(mediaType, parser));
460      MediaType mt = getMediaType();
461      return optional(mt).map(x -> parsers.getParserMatch(x));
462   }

464   private MediaType getMediaType() {
465      if (mediaType != null)
466         return mediaType;
467      Optional<ContentType> ct = req.getHeader(ContentType.class);
468      if (!ct.isPresent() && content != null)
469         return MediaType.UON;
470      return ct.isPresent() ? ct.get().asMediaType().orElse(null) : null;
471   }

473   private <T> T getInner(ClassMeta<T> cm) throws BadRequest, UnsupportedMediaType, InternalServerError {
474      try {
475         return parse(cm);
476      } catch (UnsupportedMediaType e) {
477         throw e;
478      } catch (SchemaValidationException e) {
479         throw new BadRequest("Validation failed on request content. " + e.getLocalizedMessage());
480      } catch (ParseException e) {
481         throw new BadRequest(e, "Could not convert request content content to class type ''{0}''.", cm);
482      } catch (IOException e) {
483         throw new InternalServerError(e, "I/O exception occurred while parsing request content.");
484      } catch (Exception e) {
485         throw new InternalServerError(e, "Exception occurred while parsing request content.");
486      }
487   }
488
489   /* Workhorse method */
490   private <T> T parse(ClassMeta<T> cm) throws SchemaValidationException, ParseException, UnsupportedMediaType, IOException {
491
492      if (cm.isReader())
493         return (T)getReader();
494
495      if (cm.isInputStream())
496         return (T)getInputStream();
497
498      Optional<TimeZone> timeZone = req.getTimeZone();
499      Locale locale = req.getLocale();
500      ParserMatch pm = getParserMatch().orElse(null);
501
502      if (schema == null)
503         schema = HttpPartSchema.DEFAULT;
504
505      if (pm != null) {
506         Parser p = pm.getParser();
507         MediaType mediaType = pm.getMediaType();
508         ParserSession session = p
509            .createSession()
510            .properties(req.getAttributes().asMap())
511            .javaMethod(req.getOpContext().getJavaMethod())
512            .locale(locale)
513            .timeZone(timeZone.orElse(null))
514            .mediaType(mediaType)
515            .apply(ReaderParser.Builder.class, x -> x.streamCharset(req.getCharset()))
516            .schema(schema)
517            .debug(req.isDebug() ? true : null)
518            .outer(req.getContext().getResource())
519            .build();
520         ;
521         try (Closeable in = session.isReaderParser() ? getUnbufferedReader() : getInputStream()) {
522            T o = session.parse(in, cm);
523            if (schema != null)
524               schema.validateOutput(o, cm.getBeanContext());
525            return o;
526         }
527      }
528
529      if (cm.hasReaderMutater())
530         return cm.getReaderMutater().mutate(getReader());
531
532      if (cm.hasInputStreamMutater())
533         return cm.getInputStreamMutater().mutate(getInputStream());
534
535      MediaType mt = getMediaType();
536
537      if ((isEmpty(stringify(mt)) || mt.toString().startsWith("text/plain")) && cm.hasStringMutater())
538         return cm.getStringMutater().mutate(asString());
539
540      Optional<ContentType> ct = req.getHeader(ContentType.class);
541      throw new UnsupportedMediaType(
542         "Unsupported media-type in request header ''Content-Type'': ''{0}''\n\tSupported media-types: {1}",
543         ct.isPresent() ? ct.get().asMediaType().orElse(null) : "not-specified", SimpleJson.of(req.getOpContext().getParsers().getSupportedMediaTypes())
544      );
545   }
546
547   private Encoder getEncoder() throws UnsupportedMediaType {
548      if (encoder == null) {
549         String ce = req.getHeaderParam("content-encoding").orElse(null);
550         if (isNotEmpty(ce)) {
551            ce = ce.trim();
552            encoder = encoders.getEncoder(ce);
553            if (encoder == null)
554               throw new UnsupportedMediaType(
555                  "Unsupported encoding in request header ''Content-Encoding'': ''{0}''\n\tSupported codings: {1}",
556                  req.getHeaderParam("content-encoding").orElse(null), SimpleJson.of(encoders.getSupportedEncodings())
557               );
558         }
559
560         if (encoder != null)
561            contentLength = -1;
562      }
563      // Note that if this is the identity encoder, we want to return null
564      // so that we don't needlessly wrap the input stream.
565      if (encoder == IdentityEncoder.INSTANCE)
566         return null;
567      return encoder;
568   }

579   /**
580    * Caches the content in memory for reuse.
581    *
582    * @return This object.
583    * @throws IOException If error occurs while reading stream.

585   public RequestContent cache() throws IOException {
586      if (content == null)
587         content = readBytes(getInputStream());
588      return this;
