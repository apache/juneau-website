036/**
037 * Represents the parsed form-data parameters in an HTTP request.
038 *
039 * <p>
040 *    The {@link RequestFormParams} object is the API for accessing the HTTP request content as form data.
041 *    It can be accessed by passing it as a parameter on your REST Java method:
042 * </p>
043 * <p class='bjava'>
044 *    <ja>@RestPost</ja>(...)
045 *    <jk>public</jk> Object myMethod(RequestFormParams <jv>formData</jv>) {...}
046 * </p>
047 *
048 * <h5 class='figure'>Example:</h5>
049 * <p class='bjava'>
050 *    <ja>@RestPost</ja>(...)
051 *    <jk>public</jk> Object myMethod(RequestFormParams <jv>formData</jv>) {
052 *
053 *       <jc>// Get query parameters converted to various types.</jc>
054 *       <jk>int</jk> <jv>p1</jv> = <jv>formData</jv>.get(<js>"p1"</js>).asInteger().orElse(0);
055 *       String <jv>p2</jv> = <jv>formData</jv>.get(<js>"p2"</js>).orElse(<jk>null</jk>);
056 *       UUID <jv>p3</jv> = <jv>formData</jv>.get(<js>"p3"</js>).as(UUID.<jk>class</jk>).orElse(<jk>null</jk>);
057 *     }
058 * </p>
059 *
060 * <p>
061 *    Note that this object does NOT take GET parameters into account and only returns values found in the content of the request.
062 * </p>
063 *
064 * <p>
065 *    Some important methods on this class are:
066 * </p>
067 * <ul class='javatree'>
068 *    <li class='jc'>{@link RequestFormParams}
069 *    <ul class='spaced-list'>
070 *       <li>Methods for retrieving form data parameters:
071 *       <ul class='javatreec'>
072 *          <li class='jm'>{@link RequestFormParams#contains(String...) contains(String...)}
073 *          <li class='jm'>{@link RequestFormParams#containsAny(String...) containsAny(String...)}
074 *          <li class='jm'>{@link RequestFormParams#get(Class) get(Class)}
075 *          <li class='jm'>{@link RequestFormParams#get(String) get(String)}
076 *          <li class='jm'>{@link RequestFormParams#getAll() getAll()}
077 *          <li class='jm'>{@link RequestFormParams#getAll(String) getAll(String)}
078 *          <li class='jm'>{@link RequestFormParams#getFirst(String) getFirst(String)}
079 *          <li class='jm'>{@link RequestFormParams#getLast(String) getLast(String)}
080 *       </ul>
081 *       <li>Methods overridding form data parameters:
082 *       <ul class='javatreec'>
083 *          <li class='jm'>{@link RequestFormParams#add(NameValuePair...) add(NameValuePair...)}
084 *          <li class='jm'>{@link RequestFormParams#add(Part) add(Part)}
085 *          <li class='jm'>{@link RequestFormParams#add(String,Object) add(String,Object)}
086 *          <li class='jm'>{@link RequestFormParams#addDefault(List) addDefault(List)}
087 *          <li class='jm'>{@link RequestFormParams#addDefault(NameValuePair...) addDefault(NameValuePair...)}
088 *          <li class='jm'>{@link RequestFormParams#addDefault(String,String) addDefault(String,String)}
089 *          <li class='jm'>{@link RequestFormParams#remove(NameValuePair...) remove(NameValuePair...)}
090 *          <li class='jm'>{@link RequestFormParams#remove(String...) remove(String...)}
091 *          <li class='jm'>{@link RequestFormParams#set(NameValuePair...) set(NameValuePair...)}
092 *          <li class='jm'>{@link RequestFormParams#set(String,Object) set(String,Object)}
093 *       </ul>
094 *       <li>Other methods:
095 *       <ul class='javatreec'>
096 *          <li class='jm'>{@link RequestFormParams#asQueryString() asQueryString()}
097 *          <li class='jm'>{@link RequestFormParams#copy() copy()}
098 *          <li class='jm'>{@link RequestFormParams#isEmpty() isEmpty()}
099 *       </ul>
100 *    </ul>
101 * </ul>
102 *
103 * <ul class='seealso'>
104 *    <li class='jc'>{@link RequestFormParam}
105 *    <li class='ja'>{@link org.apache.juneau.http.annotation.FormData}
106 *    <li class='ja'>{@link org.apache.juneau.http.annotation.HasFormData}
107 *    <li class='link'>{@doc jrs.HttpParts}
108 *    <li class='extlink'>{@source}
109 * </ul>
110 */
111public class RequestFormParams {
112
113   private final RestRequest req;
114   private final boolean caseSensitive;
115   private HttpPartParserSession parser;
116   private final VarResolverSession vs ;
117
118   private List<RequestFormParam> list = new LinkedList<>();
119   private Map<String,List<RequestFormParam>> map = new TreeMap<>();
120
121   /**
122    * Constructor.
123    *
124    * @param req The request creating this bean.
125    * @param caseSensitive Whether case-sensitive name matching is enabled.
126    * @throws Exception Any exception can be thrown.
127    */
128   public RequestFormParams(RestRequest req, boolean caseSensitive) throws Exception {
129      this.req = req;
130      this.caseSensitive = caseSensitive;
131      this.vs = req.getVarResolverSession();
132
133      Map<String,String[]> m = null;
134      Collection<Part> c = null;
135
136      RequestContent content = req.getContent();
137      if (content.isLoaded() || ! req.getHeader(ContentType.class).orElse(ContentType.NULL).equalsIgnoreCase("multipart/form-data"))
138         m = RestUtils.parseQuery(content.getReader());
139      else {
140         c = req.getHttpServletRequest().getParts();
141         if (c == null || c.isEmpty())
142            m = req.getHttpServletRequest().getParameterMap();
143      }
144
145      if (m != null) {
146         for (Map.Entry<String,String[]> e : m.entrySet()) {
147            String name = e.getKey();
148            String key = key(name);
149            List<RequestFormParam> l = list();
150
151            String[] values = e.getValue();
152            if (values == null)
153               values = new String[0];
154
155            // Fix for behavior difference between Tomcat and WAS.
156            // getParameter("foo") on "&foo" in Tomcat returns "".
157            // getParameter("foo") on "&foo" in WAS returns null.
158            if (values.length == 1 && values[0] == null)
159               values[0] = "";
160
161            for (String value : values) {
162               RequestFormParam p = new RequestFormParam(req, name, value);
163               list.add(p);
164               l.add(p);
165            }
166            map.put(key, l);
167         }
168      } else if (c != null) {
169         c.stream().forEach(x->add(x));
170      }
171   }

174   /**
175    * Copy constructor.
176    */
177   private RequestFormParams(RequestFormParams copyFrom) {
178      req = copyFrom.req;
179      caseSensitive = copyFrom.caseSensitive;
180      parser = copyFrom.parser;
181      list.addAll(copyFrom.list);
182      map.putAll(copyFrom.map);
183      vs = copyFrom.vs;
184   }

186   /**
187    * Sets the parser to use for part values.
188    *
189    * @param value The new value for this setting.
190    * @return This object.
191    */
192   public RequestFormParams parser(HttpPartParserSession value) {
193      this.parser = value;
194      for (RequestFormParam p : list)
195         p.parser(parser);
196      return this;
197   }

203   /**
204    * Adds default entries to these parameters.
205    *
206    * <p>
207    * Similar to {@link #set(String, Object)} but doesn't override existing values.
208    *
209    * @param pairs
210    *    The default entries.
211    *    <br>Can be <jk>null</jk>.
212    * @return This object.
213    */
214   public RequestFormParams addDefault(List<? extends NameValuePair> pairs) {
215      for (NameValuePair p : pairs) {
216         String name = p.getName();
217         String key = key(name);
218         List<RequestFormParam> l = map.get(key);
219         boolean hasAllBlanks = l != null && l.stream().allMatch(x -> StringUtils.isEmpty(x.getValue()));
220         if (l == null || hasAllBlanks) {
221            if (hasAllBlanks)
222               list.removeAll(l);
223            RequestFormParam x = new RequestFormParam(req, name, vs.resolve(p.getValue()));
224            list.add(x);
225            map.put(key, list(x));
226         }
227      }
228      return this;
229   }

257   /**
258    * Returns all the parameters with the specified name.
259    *
260    * @param name The parameter name.
261    * @return The list of all parameters with the specified name, or an empty list if none are found.

263   public List<RequestFormParam> getAll(String name) {
264      assertArgNotNull("name", name);
265      List<RequestFormParam> l = map.get(key(name));
266      return l == null ? emptyList() : unmodifiable(l);

278   /**
279    * Returns <jk>true</jk> if the parameters with the specified names are present.
280    *
281    * @param names The parameter names.  Must not be <jk>null</jk>.
282    * @return <jk>true</jk> if the parameters with the specified names are present.
283    */
284   public boolean contains(String...names) {
285      assertArgNotNull("names", names);
286      for (String n : names)
287         if (! map.containsKey(key(n)))
288            return false;
289      return true;
290   }

292   /**
293    * Returns <jk>true</jk> if the parameter with any of the specified names are present.
294    *
295    * @param names The parameter names.  Must not be <jk>null</jk>.
296    * @return <jk>true</jk> if the parameter with any of the specified names are present.
297    */
298   public boolean containsAny(String...names) {
299      assertArgNotNull("names", names);
300      for (String n : names)
301         if (map.containsKey(key(n)))
302            return true;
303      return false;
304   }

315   /**
316    * Adds a parameter value.
317    *
318    * <p>
319    * Parameter is added to the end.
320    * <br>Existing parameter with the same name are not changed.
321    *
322    * @param name The parameter name.  Must not be <jk>null</jk>.
323    * @param value The parameter value.
324    * @return This object.
325    */
326   public RequestFormParams add(String name, Object value) {
327      assertArgNotNull("name", name);
328      String key = key(name);
329      RequestFormParam h = new RequestFormParam(req, name, stringify(value)).parser(parser);
330      if (map.containsKey(key))
331         map.get(key).add(h);
332      else
333         map.put(key, list(h));
334      list.add(h);
335      return this;
336   }

339    * Adds a parameter value.
340    *
341    * <p>
342    * Parameter is added to the end.
343    * <br>Existing parameter with the same name are not changed.
344    *
345    * @param part The parameter part.  Must not be <jk>null</jk>.
346    * @return This object.
347    */
348   public RequestFormParams add(Part part) {
349      assertArgNotNull("part", part);
350      String key = key(part.getName());
351      RequestFormParam h = new RequestFormParam(req, part).parser(parser);
352      if (map.containsKey(key))
353         map.get(key).add(h);
354      else
355         map.put(key, list(h));
356      list.add(h);
357      return this;

360   /**
361    * Adds request parameter values.
362    *
363    * <p>
364    * Parameters are added to the end.
365    * <br>Existing parameters with the same name are not changed.
366    *
367    * @param parameters The parameter objects.  Must not be <jk>null</jk>.
368    * @return This object.
369    */
370   public RequestFormParams add(NameValuePair...parameters) {
371      assertArgNotNull("parameters", parameters);
372      for (NameValuePair p : parameters) {
373         if (p != null)
374            add(p.getName(), p.getValue());
375      }
376      return this;
377   }

379   /**
380    * Sets a parameter value.
381    *
382    * <p>
383    * Parameter is added to the end.
384    * <br>Any previous parameters with the same name are removed.
385    *
386    * @param name The parameter name.  Must not be <jk>null</jk>.
387    * @param value
388    *    The parameter value.
389    *    <br>Converted to a string using {@link Object#toString()}.
390    *    <br>Can be <jk>null</jk>.
391    * @return This object.
392    */
393   public RequestFormParams set(String name, Object value) {
394      assertArgNotNull("name", name);
395      String key = key(name);
396      RequestFormParam p = new RequestFormParam(req, name, stringify(value)).parser(parser);
397      if (map.containsKey(key))
398         list.removeIf(x->caseSensitive?x.getName().equals(name):x.getName().equalsIgnoreCase(name));
399      list.add(p);
400      map.put(key, list(p));
401      return this;
402   }

405   /**
406    * Sets request header values.
407    *
408    * <p>
409    * Parameters are added to the end of the headers.
410    * <br>Any previous parameters with the same name are removed.
411    *
412    * @param parameters The parameters to set.  Must not be <jk>null</jk> or contain <jk>null</jk>.
413    * @return This object.
414    */
415   public RequestFormParams set(NameValuePair...parameters) {
416      assertArgNotNull("headers", parameters);
417      for (NameValuePair p : parameters)
418         remove(p);
419      for (NameValuePair p : parameters)
420         add(p);
421      return this;
422   }
423
424   /**
425    * Remove parameters.
426    *
427    * @param name The parameter names.  Must not be <jk>null</jk>.
428    * @return This object.
429    */
430   public RequestFormParams remove(String...name) {
431      assertArgNotNull("name", name);
432      for (String n : name) {
433         String key = key(n);
434         if (map.containsKey(key))
435            list.removeAll(map.get(key));
436         map.remove(key);
437      }
438      return this;
439   }

441   /**
442    * Remove parameters.
443    *
444    * @param parameters The parameters to remove.  Must not be <jk>null</jk>.
445    * @return This object.
446    */
447   public RequestFormParams remove(NameValuePair...parameters) {
448      for (NameValuePair p : parameters)
449         remove(p.getName());
450      return this;
451   }

457   /**
458    * Returns the first parameter with the specified name.
459    *
460    * <p>
461    * Note that this method never returns <jk>null</jk> and that {@link RequestFormParam#isPresent()} can be used
462    * to test for the existence of the parameter.
463    *
464    * @param name The parameter name.
465    * @return The parameter.  Never <jk>null</jk>.
466    */
467   public RequestFormParam getFirst(String name) {
468      assertArgNotNull("name", name);
469      List<RequestFormParam> l = map.get(key(name));
470      return (l == null || l.isEmpty() ? new RequestFormParam(req, name, null).parser(parser) : l.get(0));
471   }

473   /**
474    * Returns the last parameter with the specified name.
475    *
476    * <p>
477    * Note that this method never returns <jk>null</jk> and that {@link RequestFormParam#isPresent()} can be used
478    * to test for the existence of the parameter.
479    *
480    * @param name The parameter name.
481    * @return The parameter.  Never <jk>null</jk>.
482    */
483   public RequestFormParam getLast(String name) {
484      assertArgNotNull("name", name);
485      List<RequestFormParam> l = map.get(key(name));
486      return (l == null || l.isEmpty() ? new RequestFormParam(req, name, null).parser(parser) : l.get(l.size()-1));
487   }

502   /**
503    * Returns the form data parameter as the specified bean type.
504    *
505    * <p>
506    * Type must have a name specified via the {@link org.apache.juneau.http.annotation.FormData} annotation
507    * and a public constructor that takes in either <c>value</c> or <c>name,value</c> as strings.
508    *
509    * @param <T> The bean type to create.
510    * @param type The bean type to create.
511    * @return The bean, never <jk>null</jk>.
512    */
513   public <T> Optional<T> get(Class<T> type) {
514      ClassMeta<T> cm = req.getBeanSession().getClassMeta(type);
515      String name = HttpParts.getName(FORMDATA, cm).orElseThrow(()->new BasicRuntimeException("@FormData(name) not found on class {0}", className(type)));
516      return get(name).as(type);
517   }

523   /**
524    * Converts this object to a query string.
525    *
526    * <p>
527    * Returned query string does not start with <js>'?'</js>.
528    *
529    * @return A new query string, or an empty string if this object is empty.
530    */
531   public String asQueryString() {
532      StringBuilder sb = new StringBuilder();
533      for (RequestFormParam e : list) {
534         if (sb.length() > 0)
535            sb.append("&");
536         sb.append(urlEncode(e.getName())).append('=').append(urlEncode(e.getValue()));
537      }
538      return sb.toString();
539   }

550   /**
551    * Converts the parameters to a readable string.
552    *
553    * @param sorted Sort the parameters by name.
554    * @return A JSON string containing the contents of the parameters.
555    */
556   public String toString(boolean sorted) {
557      JsonMap m = JsonMap.create();
558      if (sorted) {
559         for (List<RequestFormParam> p1 : map.values())
560            for (RequestFormParam p2 : p1)
561               m.append(p2.getName(), p2.getValue());
562      } else {
563         for (RequestFormParam p : list)
564            m.append(p.getName(), p.getValue());
565      }
566      return m.toString();
567   }
568
569   private String key(String name) {
570      return caseSensitive ? name : name.toLowerCase();
571   }
572
573   @Override /* Object */
574   public String toString() {
575      return toString(false);
576   }
