020import org.apache.juneau.http.annotation.*;
021import org.apache.juneau.httppart.*;
022import org.apache.juneau.internal.*;
023import org.apache.juneau.reflect.*;

027import org.apache.juneau.rest.util.*;

029/**
030 * Resolves method parameters and parameter types annotated with {@link Path} on {@link RestOp}-annotated Java methods.
031 *
032 * <p>
033 * The parameter value is resolved using:
034 * <p class='bjava'>
035 *    <jv>opSession</jv>
036 *       .{@link RestOpSession#getRequest() getRequest}()
037 *       .{@link RestRequest#getPathParams() getPathParams}()
038 *       .{@link RequestPathParams#get(String) get}(<jv>name</jv>)
039 *       .{@link RequestPathParam#as(Class) as}(<jv>type</jv>);
040 * </p>
041 *
042 * <p>
043 * {@link HttpPartSchema schema} is derived from the {@link Path} annotation.
044 *
045 * <ul class='seealso'>
046 *    <li class='link'>{@doc jrs.JavaMethodParameters}
047 *    <li class='extlink'>{@source}
048 * </ul>
049 */
050public class PathArg implements RestOpArg {
051   private final HttpPartParser partParser;
052   private final HttpPartSchema schema;
053   private final String name, def;
054   private final Type type;
055
056   /**
057    * Static creator.
058    *
059    * @param paramInfo The Java method parameter being resolved.
060    * @param annotations The annotations to apply to any new part parsers.
061    * @param pathMatcher Path matcher for the specified method.
062    * @return A new {@link PathArg}, or <jk>null</jk> if the parameter is not annotated with {@link Path}.
063    */
064   public static PathArg create(ParamInfo paramInfo, AnnotationWorkList annotations, UrlPathMatcher pathMatcher) {
065      if (paramInfo.hasAnnotation(Path.class) || paramInfo.getParameterType().hasAnnotation(Path.class))
066         return new PathArg(paramInfo, annotations, pathMatcher);
067      return null;
068   }
069
070   /**
071    * Constructor.
072    *
073    * @param paramInfo The Java method parameter being resolved.
074    * @param annotations The annotations to apply to any new part parsers.
075    * @param pathMatcher Path matcher for the specified method.
076    */
077   protected PathArg(ParamInfo paramInfo, AnnotationWorkList annotations, UrlPathMatcher pathMatcher) {
078      this.name = getName(paramInfo, pathMatcher);
079      this.def = findDef(paramInfo).orElse(null);
080      this.type = paramInfo.getParameterType().innerType();
081      this.schema = HttpPartSchema.create(Path.class, paramInfo);
082      Class<? extends HttpPartParser> pp = schema.getParser();
083      this.partParser = pp != null ? HttpPartParser.creator().type(pp).apply(annotations).create() : null;
084   }
085
086   private String getName(ParamInfo pi, UrlPathMatcher pathMatcher) {
087      String p = findName(pi).orElse(null);
088      if (p != null)
089         return p;
090      if (pathMatcher != null) {
091         int idx = 0;
092         int i = pi.getIndex();
093         MethodInfo mi = pi.getMethod();
094
095         for (int j = 0; j < i; j++)
096            if (mi.getParam(i).getAnnotation(Path.class) != null)
097               idx++;
098
099         String[] vars = pathMatcher.getVars();
100         if (vars.length <= idx)
101            throw new ArgException(pi, "Number of attribute parameters exceeds the number of URL pattern variables");
102
103         // Check for {#} variables.
104         String idxs = String.valueOf(idx);
105         for (int j = 0; j < vars.length; j++)
106            if (StringUtils.isNumeric(vars[j]) && vars[j].equals(idxs))
107               return vars[j];
108
109         return pathMatcher.getVars()[idx];
110      }
111      throw new ArgException(pi, "@Path used without name or value");
112   }
113
114   @Override /* RestOpArg */
115   public Object resolve(RestOpSession opSession) throws Exception {
116      RestRequest req = opSession.getRequest();
117      if (name.equals("*")) {
118         JsonMap m = new JsonMap();
119         req.getPathParams().getAll().stream().forEach(x -> m.put(x.getName(), x.getValue()));
120         return req.getBeanSession().convertToType(m, type);
121      }
122      HttpPartParserSession ps = partParser == null ? req.getPartParserSession() : partParser.getPartSession();
123      return req.getPathParams().get(name).parser(ps).schema(schema).def(def).as(type).orElse(null);
124   }
125}
