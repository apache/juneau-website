031/**
032 * Basic configuration for a REST resource that supports all languages and provides common default configuration values.
033 *
034 * <p>
035 *    Default settings defined:
036 * </p>
037 * <ul class='spaced-list'>
038 *    <li class='ja'>{@link Rest}:
039 *       <ul>
040 *          <li class='jma'>{@link Rest#serializers() serializers}:
041 *             <ul class='javatree'>
042 *                <li class='jc'>{@link HtmlDocSerializer}
043 *                <li class='jc'>{@link HtmlStrippedDocSerializer}
044 *                <li class='jc'>{@link HtmlSchemaDocSerializer}
045 *                <li class='jc'>{@link JsonSerializer}
046 *                <li class='jc'>{@link SimpleJsonSerializer}
047 *                <li class='jc'>{@link JsonSchemaSerializer}
048 *                <li class='jc'>{@link XmlDocSerializer}
049 *                <li class='jc'>{@link UonSerializer}
050 *                <li class='jc'>{@link UrlEncodingSerializer}
051 *                <li class='jc'>{@link OpenApiSerializer}
052 *                <li class='jc'>{@link MsgPackSerializer}
053 *                <li class='jc'>{@link SoapXmlSerializer}
054 *                <li class='jc'>{@link PlainTextSerializer}
055 *             </ul>
056 *          </li>
057 *          <li class='jma'>{@link Rest#parsers() parsers}:
058 *             <ul class='javatree'>
059 *                <li class='jc'>{@link JsonParser}
060 *                <li class='jc'>{@link SimpleJsonParser}
061 *                <li class='jc'>{@link XmlParser}
062 *                <li class='jc'>{@link HtmlParser}
063 *                <li class='jc'>{@link UonParser}
064 *                <li class='jc'>{@link UrlEncodingParser}
065 *                <li class='jc'>{@link OpenApiParser}
066 *                <li class='jc'>{@link MsgPackParser}
067 *                <li class='jc'>{@link PlainTextParser}
068 *             </ul>
069 *          </li>
070 *          <li class='jma'>{@link Rest#config() config}:  <js>"$S{juneau.configFile,SYSTEM_DEFAULT}"</js>
071 *    </ul>
072 * </li>
073 *    <li class='ja'>{@link BeanConfig}:
074 *       <ul>
075 *          <li class='jma'>{@link BeanConfig#ignoreUnknownBeanProperties() ignoreUnknownBeanProperties}:  <js>"true"</js>
076 *       </ul>
077 *    </li>
078 *    <li class='ja'>{@link SerializerConfig}:
079 *       <ul>
080 *          <li class='jma'>{@link SerializerConfig#uriResolution() uriResolution}:  <js>"ROOT_RELATIVE"</js>
081 *       </ul>
082 *    </li>
083 *    <li class='ja'>{@link HtmlDocConfig}:
084 *       <ul>
085 *          <li class='jma'>{@link HtmlDocConfig#header() header}:  <js>"&lt;h1&gt;$RS{title}&lt;/h1&gt;&lt;h2&gt;$RS{operationSummary,description}&lt;/h2&gt;$C{REST/header}"</js>
086 *          <li class='jma'>{@link HtmlDocConfig#navlinks() navlinks}:  <js>"up: request:/.."</js>
087 *          <li class='jma'>{@link HtmlDocConfig#stylesheet() stylesheet}:  <js>"$C{REST/theme,servlet:/htdocs/themes/devops.css}"</js>
088 *          <li class='jma'>{@link HtmlDocConfig#head() head}:  <js>"$C{REST/head}"</js>
089 *          <li class='jma'>{@link HtmlDocConfig#footer() footer}:  <js>"$C{REST/footer}"</js>
090 *          <li class='jma'>{@link HtmlDocConfig#nowrap() nowrap}:  <js>"true"</js>
091 *       </ul>
092 *    </li>
093 * </ul>
094 *
095 * <p>
096 *    This annotation can be applied to REST resource classes to define common default configurations:
097 * </p>
098 * <p class='bjava'>
099 *    <jc>// Used on a top-level resource.</jc>
100 *    <ja>@Rest</ja>
101 *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet <jk>implements</jk> BasicUniversalConfig { ... }
102 * </p>
103 * <p class='bjava'>
104 *    <jc>// Used on a child resource.</jc>
105 *    <ja>@Rest</ja>
106 *    <jk>public class</jk> MyResource <jk>extends</jk> RestObject <jk>implements</jk> BasicUniversalConfig { ... }
107 * </p>
108 *
109 * <p>
110 *    Note that the framework will aggregate annotations defined on all classes in the class hierarchy with
111 *    values defined on child classes overriding values defined on parent classes.  That allows any values defined
112 *    on this interface to be overridden by annotations defined on the implemented class.
113 * </p>
114 *
115 * <ul class='seealso'>
116 *    <ul class='javatreec'>
117 *       <li class='jc'>{@link BasicRestServlet}
118 *       <li class='jc'>{@link BasicRestServletGroup}
119 *       <li class='jc'>{@link BasicRestObject}
120 *       <li class='jc'>{@link BasicRestObjectGroup}
121 * </ul>
122 *    <li class='link'>{@doc jrs.AnnotatedClasses}
123 *    <li class='extlink'>{@source}
124 * </ul>
125 */
126@Rest(
127
128   // Default serializers for all Java methods in the class.
129   serializers={
130      HtmlDocSerializer.class, // HTML must be listed first because Internet Explore does not include text/html in their Accept header.
131      HtmlStrippedDocSerializer.class,
132      HtmlSchemaDocSerializer.class,
133      JsonSerializer.class,
134      SimpleJsonSerializer.class,
135      JsonSchemaSerializer.class,
136      XmlDocSerializer.class,
137      UonSerializer.class,
138      UrlEncodingSerializer.class,
139      OpenApiSerializer.class,
140      MsgPackSerializer.class,
141      SoapXmlSerializer.class,
142      PlainTextSerializer.class,
143      CsvSerializer.class
144   },
145
146   // Default parsers for all Java methods in the class.
147   parsers={
148      JsonParser.class,
149      SimpleJsonParser.class,
150      XmlParser.class,
151      HtmlParser.class,
152      UonParser.class,
153      UrlEncodingParser.class,
154      OpenApiParser.class,
155      MsgPackParser.class,
156      PlainTextParser.class,
157      CsvParser.class
158   },
159
160   // Optional external configuration file.
161   config="$S{juneau.configFile,SYSTEM_DEFAULT}"
162)
163@BeanConfig(
164   // When parsing generated beans, ignore unknown properties that may only exist as getters and not setters.
165   ignoreUnknownBeanProperties="true",
166   ignoreUnknownEnumValues="true"
167)
168@SerializerConfig(
169   // Enable automatic resolution of URI objects to root-relative values.
170   uriResolution="ROOT_RELATIVE"
171)
172@HtmlDocConfig(
173
174   // Default page header contents.
175   header={
176      "<h1>$RS{title}</h1>",  // Use @Rest(title)
177      "<h2>$RS{operationSummary,description}</h2>", // Use either @RestOp(summary) or @Rest(description)
178      "$C{REST/header}"  // Extra header HTML defined in external config file.
179   },
180
181   // Basic page navigation links.
182   navlinks={
183      "up: request:/.."
184   },
185
186   // Default stylesheet to use for the page.
187   // Can be overridden from external config file.
188   // Default is DevOps look-and-feel (aka Depression look-and-feel).
189   stylesheet="$C{REST/theme,servlet:/htdocs/themes/devops.css}",
190
191   // Default contents to add to the <head> section of the HTML page.
192   // Use it to add a favicon link to the page.
193   head="$C{REST/head}",
194
195   // No default page footer contents.
196   // Can be overridden from external config file.
197   footer="$C{REST/footer}",
198
199   // By default, table cell contents should not wrap.
200   nowrap="true"
201)
202public interface BasicUniversalConfig {}
