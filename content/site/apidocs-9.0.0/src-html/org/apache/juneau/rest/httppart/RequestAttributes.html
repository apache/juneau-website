013package org.apache.juneau.rest.httppart;
014
015import static org.apache.juneau.internal.ArgUtils.*;
016import static org.apache.juneau.internal.CollectionUtils.*;
017
018import java.util.*;
019
020import javax.servlet.http.*;
021
022import org.apache.juneau.collections.*;
023import org.apache.juneau.rest.*;
024import org.apache.juneau.svl.*;

026/**
027 * Represents the attributes in an HTTP request.
028 *
029 * <p>
030 *    The {@link RequestAttributes} object is the API for accessing the standard servlet attributes on an HTTP request
031 *    (i.e. {@link javax.servlet.ServletRequest#getAttribute(String)}).
032 * </p>
033 *
034 * <p class='bjava'>
035 *    <ja>@RestPost</ja>(...)
036 *    <jk>public</jk> Object myMethod(RequestAttributes <jv>attributes</jv>) {...}
037 * </p>
038 *
039 * <h5 class='figure'>Example:</h5>
040 * <p class='bjava'>
041 *    <ja>@RestPost</ja>(...)
042 *    <jk>public</jk> Object myMethod(RequestAttributes <jv>attributes</jv>) {
043 *
044 *       <jc>// Add a default value.</jc>
045 *       <jv>attributes</jv>.addDefault(<js>"Foo"</js>, 123);
046 *
047 *       <jc>// Get an attribute value as a POJO.</jc>
048 *       UUID <jv>etag</jv> = <jv>attributes</jv>.get(<js>"ETag"</js>).as(UUID.<jk>class</jk>).orElse(<jk>null</jk>);
049 *    }
050 * </p>
051 *
052 * <p>
053 *    Some important methods on this class are:
054 * </p>
055 * <ul class='javatree'>
056 *    <li class='jc'>{@link RequestHeaders}
057 *    <ul class='spaced-list'>
058 *       <li>Methods for retrieving request attributes:
059 *       <ul class='javatreec'>
060 *          <li class='jm'>{@link RequestAttributes#contains(String...) contains(String...)}
061 *          <li class='jm'>{@link RequestAttributes#containsAny(String...) containsAny(String...)}
062 *          <li class='jm'>{@link RequestAttributes#get(String) get(String)}
063 *          <li class='jm'>{@link RequestAttributes#getAll() getAll()}
064 *       </ul>
065 *       <li>Methods for overriding request attributes:
066 *       <ul class='javatreec'>
067 *          <li class='jm'>{@link RequestAttributes#addDefault(List) addDefault(List)}
068 *          <li class='jm'>{@link RequestAttributes#addDefault(NamedAttribute...) addDefault(NamedAttribute...)}
069 *          <li class='jm'>{@link RequestAttributes#addDefault(NamedAttributeList) addDefault(NamedAttributeList)}
070 *          <li class='jm'>{@link RequestAttributes#addDefault(String,Object) addDefault(String,Object)}
071 *          <li class='jm'>{@link RequestAttributes#remove(NamedAttribute...) remove(NamedAttribute...)}
072 *          <li class='jm'>{@link RequestAttributes#remove(String...) remove(String...)}
073 *          <li class='jm'>{@link RequestAttributes#set(NamedAttribute...) set(NamedAttribute...)}
074 *          <li class='jm'>{@link RequestAttributes#set(String,Object) set(String,Object)}
075 *       </ul>
076 *       <li>Other methods:
077 *       <ul class='javatreec'>
078 *          <li class='jm'>{@link RequestAttributes#asMap() asMap()}
079 *       </ul>
080 *    </ul>
081 * </ul>
082 *
083 * <p>
084 *    Modifications made to request attributes through the <c>RequestAttributes</c> bean are automatically reflected in
085 *    the underlying servlet request attributes making it possible to mix the usage of both APIs.
086 * </p>
087 *
088 * <ul class='seealso'>
089 *    <li class='link'>{@doc jrs.HttpParts}
090 *    <li class='extlink'>{@source}
091 * </ul>
092 */
093public class RequestAttributes {
094
095   final RestRequest req;
096   final HttpServletRequest sreq;
097   final VarResolverSession vs;
098
099   /**
100    * Constructor.
101    *
102    * @param req The request creating this bean.
103    */
104   public RequestAttributes(RestRequest req) {
105      super();
106      this.req = req;
107      this.sreq = req.getHttpServletRequest();
108      this.vs = req.getVarResolverSession();
109   }
110
111   /**
112    * Adds default entries to the request attributes.
113    *
114    * @param pairs
115    *    The default entries.
116    *    <br>Can be <jk>null</jk>.
117    * @return This object.
118    */
119   public RequestAttributes addDefault(List<NamedAttribute> pairs) {
120      for (NamedAttribute p : pairs)
121         if (sreq.getAttribute(p.getName()) == null) {
122            Object o = p.getValue();
123            sreq.setAttribute(p.getName(), o instanceof String ? vs.resolve(o) : o);
124         }
125      return this;
126   }
127
128   /**
129    * Adds default entries to the request attributes.
130    *
131    * @param pairs
132    *    The default entries.
133    *    <br>Can be <jk>null</jk>.
134    * @return This object.
135    */
136   public RequestAttributes addDefault(NamedAttributeList pairs) {
137      for (NamedAttribute p : pairs.entries)
138         if (sreq.getAttribute(p.getName()) == null) {
139            Object o = p.getValue();
140            sreq.setAttribute(p.getName(), o instanceof String ? vs.resolve(o) : o);
141         }
142      return this;
143   }
144
145   /**
146    * Adds default entries to the request attributes.
147    *
148    * @param pairs
149    *    The default entries.
150    *    <br>Can be <jk>null</jk>.
151    * @return This object.
152    */
153   public RequestAttributes addDefault(NamedAttribute...pairs) {
154      return addDefault(alist(pairs));
155   }
156
157   /**
158    * Adds a default entry to the request attributes.
159    *
160    * @param name The name.
161    * @param value The value.
162    * @return This object.
163    */
164   public RequestAttributes addDefault(String name, Object value) {
165      return addDefault(BasicNamedAttribute.of(name, value));
166   }
167
168   /**
169    * Returns the request attribute with the specified name.
170    *
171    * @param name The attribute name.
172    * @return The parameter value, or {@link Optional#empty()} if it doesn't exist.
173    */
174   public RequestAttribute get(String name) {
175      return new RequestAttribute(req, name, sreq.getAttribute(name));
176   }
177
178   /**
179    * Returns all the attribute on this request.
180    *
181    * @return All the attribute on this request.
182    */
183   public List<RequestAttribute> getAll() {
184      List<RequestAttribute> l = list();
185      Enumeration<String> e = sreq.getAttributeNames();
186      while (e.hasMoreElements()) {
187         String n = e.nextElement();
188         l.add(new RequestAttribute(req, n, sreq.getAttribute(n)));
189      }
190      return l;
191   }
192
193   /**
194    * Returns <jk>true</jk> if the attributes with the specified names are present.
195    *
196    * @param names The attribute names.  Must not be <jk>null</jk>.
197    * @return <jk>true</jk> if the parameters with the specified names are present.
198    */
199   public boolean contains(String...names) {
200      assertArgNotNull("names", names);
201      for (String n : names)
202         if (sreq.getAttribute(n) == null)
203            return false;
204      return true;
205   }
206
207   /**
208    * Returns <jk>true</jk> if the attribute with any of the specified names are present.
209    *
210    * @param names The attribute names.  Must not be <jk>null</jk>.
211    * @return <jk>true</jk> if the attribute with any of the specified names are present.
212    */
213   public boolean containsAny(String...names) {
214      assertArgNotNull("names", names);
215      for (String n : names)
216         if (sreq.getAttribute(n) != null)
217            return true;
218      return false;
219   }
220
221   /**
222    * Sets a request attribute.
223    *
224    * @param name The attribute name.  Must not be <jk>null</jk>.
225    * @param value
226    *    The attribute value.
227    *    <br>Can be <jk>null</jk>.
228    * @return This object.
229    */
230   public RequestAttributes set(String name, Object value) {
231      assertArgNotNull("name", name);
232      sreq.setAttribute(name, value);
233      return this;
234   }
235
236   /**
237    * Sets request attributes.
238    *
239    * @param attributes The parameters to set.  Must not be <jk>null</jk> or contain <jk>null</jk>.
240    * @return This object.
241    */
242   public RequestAttributes set(NamedAttribute...attributes) {
243      assertArgNotNull("attributes", attributes);
244      for (NamedAttribute p : attributes)
245         set(p);
246      return this;
247   }
248
249   /**
250    * Remove request attributes.
251    *
252    * @param name The attribute names.  Must not be <jk>null</jk>.
253    * @return This object.
254    */
255   public RequestAttributes remove(String...name) {
256      assertArgNotNull("name", name);
257      for (String n : name) {
258         sreq.removeAttribute(n);
259      }
260      return this;
261   }
262
263   /**
264    * Remove request attributes.
265    *
266    * @param attributes The attributes to remove.  Must not be <jk>null</jk>.
267    * @return This object.
268    */
269   public RequestAttributes remove(NamedAttribute...attributes) {
270      for (NamedAttribute p : attributes)
271         remove(p.getName());
272      return this;
273   }
274
275   /**
276    * Returns the request attributes as a map.
277    *
278    * @return The request attributes as a map.  Never <jk>null</jk>.
279    */
280   public Map<String,Object> asMap() {
281      JsonMap m = new JsonMap();
282      Enumeration<String> e = sreq.getAttributeNames();
283      while (e.hasMoreElements()) {
284         String n = e.nextElement();
285         m.put(n, sreq.getAttribute(n));
286      }
287      return m;
288   }
289
290   @Override /* Object */
291   public String toString() {
292      return asMap().toString();
293   }
294}
