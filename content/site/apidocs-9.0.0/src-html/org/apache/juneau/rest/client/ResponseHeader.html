035/**
036 * Represents a single header on an HTTP response.
037 *
038 * <p>
039 * An extension of an HttpClient {@link Header} that provides various support for converting the header to POJOs and
040 * other convenience methods.
041 *
042 * <ul class='seealso'>
043 *    <li class='link'>{@doc juneau-rest-client}
044 *    <li class='extlink'>{@source}
045 * </ul>
046 */
047public class ResponseHeader extends BasicHeader {
048
049   private static final long serialVersionUID = 1L;

051   static final Header NULL_HEADER = new Header() {
052
053      @Override /* Header */
054      public String getName() {
055         return null;
056      }
057
058      @Override /* Header */
059      public String getValue() {
060         return null;
061      }
062
063      @Override /* Header */
064      public HeaderElement[] getElements() throws org.apache.http.ParseException {
065         return new HeaderElement[0];
066      }
067   };
068
069   private final HeaderElement[] elements;
070   private final RestRequest request;
071   private final RestResponse response;
072   private HttpPartParserSession parser;
073   private HttpPartSchema schema;

075   /**
076    * Constructor.
077    * @param name The header name.
078    * @param request The request object.
079    * @param response The response object.
080    * @param header The wrapped header.  Can be <jk>null</jk>.
081    * @throws IllegalArgumentException If name is <jk>null</jk> or empty.
082    */
083   public ResponseHeader(String name, RestRequest request, RestResponse response, Header header) {
084      super(name, header == null ? null : header.getValue());
085      this.request = request;
086      this.response = response;
087      this.elements = header == null ? new HeaderElement[0] : header.getElements();
088      parser(null);
089   }

175   /**
176    * Returns the value of this header as a {@link BasicHeader}.
177    *
178    * @param c The subclass of {@link BasicHeader} to instantiate.
179    * @param <T> The subclass of {@link BasicHeader} to instantiate.
180    * @return The value of this header as a string, never <jk>null</jk>.
181    */
182   public <T extends BasicHeader> T asHeader(Class<T> c) {
183      try {
184         ClassInfo ci = ClassInfo.of(c);
185         ConstructorInfo cc = ci.getPublicConstructor(x -> x.hasParamTypes(String.class));
186         if (cc != null)
187            return cc.invoke(getValue());
188         cc = ci.getPublicConstructor(x -> x.hasParamTypes(String.class, String.class));
189         if (cc != null)
190            return cc.invoke(getName(), getValue());
191      } catch (Throwable e) {
192         if (e instanceof ExecutableException)
193            e = ((ExecutableException)e).getCause();
194         throw runtimeException(e);
195      }
196      throw runtimeException("Could not determine a method to construct type {0}", className(c));
197   }

362    * @return The converted type, or <jk>null</jk> if header is not present.

364   public <T> Optional<T> as(ClassMeta<T> type) {
365      try {
366         return optional(parser.parse(HEADER, schema, getValue(), type));
367      } catch (ParseException e) {
368         throw BasicRuntimeException.create().causedBy(e).message("Could not parse response header {0}.", getName()).build();
369      }

386    * Matches the specified pattern against this header value.
387    *
388    * <h5 class='section'>Example:</h5>
389    * <p class='bjava'>
390    *    <jc>// Parse header using a regular expression.</jc>
391    *    Matcher <jv>matcher</jv> = <jv>client</jv>
392    *       .get(<jsf>URI</jsf>)
393    *       .run()
394    *       .getResponseHeader(<js>"Content-Type"</js>).asMatcher(Pattern.<jsm>compile</jsm>(<js>"application/(.*)"</js>));

396    *    <jk>if</jk> (<jv>matcher</jv>.matches()) {

459   /**
460    * Provides the ability to perform fluent-style assertions on this response header.
461    *
462    * <h5 class='section'>Examples:</h5>
463    * <p class='bjava'>
464    *    <jc>// Validates the content type header is provided.</jc>
465    *    <jv>client</jv>
466    *       .get(<jsf>URI</jsf>)
467    *       .run()
468    *       .getHeader(<js>"Content-Type"</js>).assertValue().exists();
469    *
470    *    <jc>// Validates the content type is JSON.</jc>
471    *    <jv>client</jv>
472    *       .get(<jsf>URI</jsf>)
473    *       .run()
474    *       .getHeader(<js>"Content-Type"</js>).assertValue().equals(<js>"application/json"</js>);
475    *
476    *    <jc>// Validates the content type is JSON using test predicate.</jc>
477    *    <jv>client</jv>
478    *       .get(<jsf>URI</jsf>)
479    *       .run()
480    *       .getHeader(<js>"Content-Type"</js>).assertValue().is(<jv>x</jv> -&gt; <jv>x</jv>.equals(<js>"application/json"</js>));
481    *
482    *    <jc>// Validates the content type is JSON by just checking for substring.</jc>
483    *    <jv>client</jv>
484    *       .get(<jsf>URI</jsf>)
485    *       .run()
486    *       .getHeader(<js>"Content-Type"</js>).assertValue().contains(<js>"json"</js>);
487    *
488    *    <jc>// Validates the content type is JSON using regular expression.</jc>
489    *    <jv>client</jv>
490    *       .get(<jsf>URI</jsf>)
491    *       .run()
492    *       .getHeader(<js>"Content-Type"</js>).assertValue().isPattern(<js>".*json.*"</js>);
493    *
494    *    <jc>// Validates the content type is JSON using case-insensitive regular expression.</jc>
495    *    <jv>client</jv>
496    *       .get(<jsf>URI</jsf>)
497    *       .run()
498    *       .getHeader(<js>"Content-Type"</js>).assertValue().isPattern(<js>".*json.*"</js>, <jsf>CASE_INSENSITIVE</jsf>);
499    * </p>
500    *
501    * <p>
502    * The assertion test returns the original response object allowing you to chain multiple requests like so:
503    * <p class='bjava'>
504    *    <jc>// Validates the header and converts it to a bean.</jc>
505    *    MediaType <jv>mediaType</jv> = <jv>client</jv>
506    *       .get(<jsf>URI</jsf>)
507    *       .run()
508    *       .getHeader(<js>"Content-Type"</js>).assertValue().isNotEmpty()
509    *       .getHeader(<js>"Content-Type"</js>).assertValue().isPattern(<js>".*json.*"</js>)
510    *       .getHeader(<js>"Content-Type"</js>).as(MediaType.<jk>class</jk>);
511    * </p>
512    *
513    * @return A new fluent assertion object.
514    */
515   public FluentResponseHeaderAssertion<ResponseHeader> assertValue() {
516      return new FluentResponseHeaderAssertion<>(this, this);
517   }
