035/**
036 * Servlet implementation of a REST resource.
037 *
038 * <p>
039 *    The {@link RestServlet} class is the entry point for your REST resources.
040 *    It extends directly from <l>HttpServlet</l> and is deployed like any other servlet.
041 * </p>
042 * <p>
043 *    When the servlet <l>init()</l> method is called, it triggers the code to find and process the <l>@Rest</l>
044 *    annotations on that class and all child classes.
045 *    These get constructed into a {@link RestContext} object that holds all the configuration
046 *    information about your resource in a read-only object.
047 * </p>
048 *
049 * <ul class='notes'>
050 *    <li class='note'>
051 *       Users will typically extend from {@link BasicRestServlet} or {@link BasicRestServletGroup}
052 *       instead of this class directly.
053 * </ul>
054 *
055 * <ul class='seealso'>
056 *    <li class='link'>{@doc jrs.AnnotatedClasses}
057 *    <li class='extlink'>{@source}
058 * </ul>
059 *
060 * @serial exclude
061 */
062public abstract class RestServlet extends HttpServlet {
063
064   private static final long serialVersionUID = 1L;
065
066   private AtomicReference<RestContext> context = new AtomicReference<>();
067   private AtomicReference<Exception> initException = new AtomicReference<>();
068
069   @Override /* Servlet */
070   public synchronized void init(ServletConfig servletConfig) throws ServletException {
071      try {
072         if (context.get() != null)
073            return;
074         super.init(servletConfig);
075         context.set(RestContext.create(this.getClass(), null, servletConfig).init(()->this).build());
076         context.get().postInit();
077         context.get().postInitChildFirst();
078      } catch (ServletException e) {
079         initException.set(e);
080         log(SEVERE, e, "Servlet init error on class ''{0}''", className(this));
081         throw e;
082      } catch (BasicHttpException e) {
083         initException.set(e);
084         log(SEVERE, e, "Servlet init error on class ''{0}''", className(this));
085      } catch (Throwable e) {
086         initException.set(new InternalServerError(e));
087         log(SEVERE, e, "Servlet init error on class ''{0}''", className(this));
088      }
089   }
090
091   /**
092    * Sets the context object for this servlet.
093    *
094    * <p>
095    * This method is effectively a no-op if {@link #init(ServletConfig)} has already been called.
096    *
097    * @param context Sets the context object on this servlet.
098    * @throws ServletException If error occurred during initialization.
099    */
100   protected void setContext(RestContext context) throws ServletException {
101      if (this.context.get() == null) {
102         super.init(context.getBuilder());
103         this.context.set(context);
104      }
105   }
106
107   /**
108    * Returns the path defined on this servlet if it's defined via {@link Rest#path()}.
109    *
110    * @return The path defined on this servlet, or an empty string if not specified.
111    */
112   public synchronized String getPath() {
113      RestContext context = this.context.get();
114      if (context != null)
115         return context.getFullPath();
116      ClassInfo ci = ClassInfo.of(getClass());
117      Value<String> path = Value.empty();
118      ci.forEachAnnotation(Rest.class, x -> isNotEmpty(x.path()), x -> path.set(trimSlashes(x.path())));
119      return path.orElse("");
120   }
121
122   //-----------------------------------------------------------------------------------------------------------------
123   // Lifecycle methods
124   //-----------------------------------------------------------------------------------------------------------------
125
126   /**
127    * The main service method.
128    *
129    * <p>
130    * Subclasses can optionally override this method if they want to tailor the behavior of requests.
131    */
132   @Override /* Servlet */
133   public void service(HttpServletRequest r1, HttpServletResponse r2) throws ServletException, InternalServerError, IOException {
134      try {
135         if (initException.get() != null)
136            throw initException.get();
137         if (context.get() == null)
138            throw new InternalServerError("Servlet {0} not initialized.  init(ServletConfig) was not called.  This can occur if you've overridden this method but didn't call super.init(RestConfig).", className(this));
139         getContext().execute(this, r1, r2);
140
141      } catch (Throwable e) {
142         r2.sendError(SC_INTERNAL_SERVER_ERROR, e.getLocalizedMessage());
143      }
144   }
145
146   @Override /* GenericServlet */
147   public synchronized void destroy() {
148      if (context.get() != null)
149         context.get().destroy();
150      super.destroy();
151   }
152
153   //-----------------------------------------------------------------------------------------------------------------
154   // Context methods.
155   //-----------------------------------------------------------------------------------------------------------------
156
157   /**
158    * Returns the read-only context object that contains all the configuration information about this resource.
159    *
160    * <p>
161    * This object is <jk>null</jk> during the call to {@link #init(ServletConfig)} but is populated by the time
162    * {@link #init()} is called.
163    *
164    * <p>
165    * Resource classes that don't extend from {@link RestServlet} can add the following method to their class to get
166    * access to this context object:
167    * <p class='bjava'>
168    *    <jk>public void</jk> init(RestServletContext <jv>context</jv>) <jk>throws</jk> Exception;
169    * </p>
170    *
171    * @return The context information on this servlet.
172    */
173   public synchronized RestContext getContext() {
174      RestContext rc = context.get();
175      if (rc == null)
176         throw new InternalServerError("RestContext object not set on resource.");
177      return rc;
178   }
179
180   //-----------------------------------------------------------------------------------------------------------------
181   // Convenience logger methods
182   //-----------------------------------------------------------------------------------------------------------------
183
184   /**
185    * Log a message.
186    *
187    * <p>
188    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
189    *
190    * @param level The log level.
191    * @param msg The message to log.
192    * @param args Optional {@link MessageFormat}-style arguments.
193    */
194   public void log(Level level, String msg, Object...args) {
195      doLog(level, null, () -> StringUtils.format(msg, args));
196   }
197
198   /**
199    * Log a message.
200    *
201    * <p>
202    * Subclasses can intercept the handling of these messages by overriding {@link #doLog(Level, Throwable, Supplier)}.
203    *
204    * @param level The log level.
205    * @param cause The cause.
206    * @param msg The message to log.
207    * @param args Optional {@link MessageFormat}-style arguments.
208    */
209   public void log(Level level, Throwable cause, String msg, Object...args) {
210      doLog(level, cause, () -> StringUtils.format(msg, args));
211   }
212
213   /**
214    * Main logger method.
215    *
216    * <p>
217    * The default behavior logs a message to the Java logger of the class name.
218    *
219    * <p>
220    * Subclasses can override this method to implement their own logger handling.
221    *
222    * @param level The log level.
223    * @param cause Optional throwable.
224    * @param msg The message to log.
225    */
226   protected void doLog(Level level, Throwable cause, Supplier<String> msg) {
227      RestContext c = context.get();
228      Logger logger = c == null ? null : c.getLogger();
229      if (logger == null)
230         logger = Logger.getLogger(className(this));
231      logger.log(level, cause, msg);
232   }

238   /**
239    * Returns the current thread-local HTTP request.
240    *
241    * @return The current thread-local HTTP request, or <jk>null</jk> if it wasn't created.
242    */
243   public synchronized RestRequest getRequest() {
244      return getContext().getLocalSession().getOpSession().getRequest();
245   }
246
247   /**
248    * Returns the current thread-local HTTP response.
249    *
250    * @return The current thread-local HTTP response, or <jk>null</jk> if it wasn't created.
251    */
252   public synchronized RestResponse getResponse() {
253      return getContext().getLocalSession().getOpSession().getResponse();
254   }
