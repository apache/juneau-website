013package org.apache.juneau.rest.logger;
014
015import static java.util.logging.Level.*;
016import static org.apache.juneau.rest.logger.CallLoggingDetail.*;
017
018import java.util.concurrent.atomic.*;
019import java.util.logging.*;
020
021import org.apache.juneau.assertions.*;
022import org.apache.juneau.cp.*;
023
024/**
025 *
026 * Implementation of a {@link CallLogger} that captures log entries for testing logging itself.
027 *
028 * <p>
029 * Instead of logging messages to a log file, messages are simply kept in an internal atomic string reference.
030 * Once a message has been logged, you can use the {@link #getMessage()} or {@link #assertMessage()} methods
031 * to access and evaluate it.
032 *
033 * <p>
034 * The following example shows how the capture logger can be associated with a REST class so that logged messages can
035 * be examined.
036 *
037 * <p class='bjava'>
038 *    <jk>public class</jk> MyTests {
039 *
040 *       <jk>private static final</jk> CaptureLogger <jsf>LOGGER</jsf> = <jk>new</jk> CaptureLogger();
041 *
042 *       <jk>public static class</jk> CaptureLogger <jk>extends</jk> BasicTestCaptureCallLogger {
043 *          <jc>// How our REST class will get the logger.</jc>
044 *          <jk>public static</jk> CaptureLogger <jsm>getInstance</jsm>() {
045 *             <jk>return</jk> <jsf>LOGGER</jsf>;
046 *          }
047 *       }
048 *
049 *       <ja>@Rest</ja>(callLogger=CaptureLogger.<jk>class</jk>)
050 *       <jk>public static class</jk> TestRest <jk>implements</jk> BasicRestServlet {
051 *          <ja>@RestGet</ja>
052 *          <jk>public boolean</jk> bad() <jk>throws</jk> InternalServerError {
053 *             <jk>throw new</jk> InternalServerError(<js>"foo"</js>);
054 *          }
055 *       }
056 *
057 *       <ja>@Test</ja>
058 *       <jk>public void</jk> testBadRequestLogging() <jk>throws</jk> Exception {
059 *          <jc>// Create client that won't throw exceptions.</jc>
060 *          RestClient <jv>client</jv> = MockRestClient.<jsm>create</jsm>(TestRest.<jk>class</jk>).ignoreErrors().build();
061 *          <jc>// Make the REST call.</jc>
062 *          <jv>client</jv>.get(<js>"/bad"</js>).run().assertStatusCode(500).assertContent().contains(<js>"foo"</js>);
063 *          <jc>// Make sure the message was logged in our expected format.</jc>
064 *          <jsf>LOGGER</jsf>.assertMessageAndReset().contains(<js>"[500] HTTP GET /bad"</js>);
065 *       }
066 * }
067 * </p>
068 *
069 * <ul class='seealso'>
070 *    <li class='link'>{@doc jrs.LoggingAndDebugging}
071 *    <li class='extlink'>{@source}
072 * </ul>
073 */
074public class BasicTestCaptureCallLogger extends CallLogger {
075
076   private AtomicReference<LogRecord> lastRecord = new AtomicReference<>();
077
078   /**
079    * Constructor using specific settings.
080    *
081    * @param beanStore The bean store containing injectable beans for this logger.
082    */
083   public BasicTestCaptureCallLogger(BeanStore beanStore) {
084      super(beanStore);
085   }
086
087   /**
088    * Constructor using default settings.
089    * <p>
090    * Uses the same settings as {@link CallLogger}.
091    */
092   public BasicTestCaptureCallLogger() {
093      super(BeanStore.INSTANCE);
094   }
095
096   @Override
097   protected Builder init(BeanStore beanStore) {
098      return super.init(beanStore)
099         .normalRules(  // Rules when debugging is not enabled.
100            CallLoggerRule.create(beanStore)  // Log 500+ errors with status-line and header information.
101               .statusFilter(x -> x >= 500)
102               .level(SEVERE)
103               .requestDetail(HEADER)
104               .responseDetail(HEADER)
105               .build(),
106            CallLoggerRule.create(beanStore)  // Log 400-500 errors with just status-line information.
107               .statusFilter(x -> x >= 400)
108               .level(WARNING)
109               .requestDetail(STATUS_LINE)
110               .responseDetail(STATUS_LINE)
111               .build()
112         )
113         .debugRules(  // Rules when debugging is enabled.
114            CallLoggerRule.create(beanStore)  // Log everything with full details.
115               .level(SEVERE)
116               .requestDetail(ENTITY)
117               .responseDetail(ENTITY)
118               .build()
119         )
120      ;
121   }
122
123   @Override
124   protected void log(Level level, String msg, Throwable e) {
125      LogRecord r = new LogRecord(level, msg);
126      r.setThrown(e);
127      this.lastRecord.set(r);
128   }
129
130   /**
131    * Returns the last logged message.
132    *
133    * @return The last logged message, or <jk>null</jk> if nothing was logged.
134    */
135   public String getMessage() {
136      LogRecord r = lastRecord.get();
137      return r == null ? null : r.getMessage();
138   }
139
140   /**
141    * Returns the last logged message and then deletes it internally.
142    *
143    * @return The last logged message.
144    */
145   public String getMessageAndReset() {
146      String msg = getMessage();
147      reset();
148      return msg;
149   }
150
151   /**
152    * Returns an assertion of the last logged message.
153    *
154    * @return The last logged message as an assertion object.  Never <jk>null</jk>.
155    */
156   public StringAssertion assertMessage() {
157      return new StringAssertion(getMessage());
158   }
159
160   /**
161    * Returns an assertion of the last logged message and then deletes it internally.
162    *
163    * @return The last logged message as an assertion object.  Never <jk>null</jk>.
164    */
165   public StringAssertion assertMessageAndReset() {
166      return new StringAssertion(getMessageAndReset());
167   }
168
169   /**
170    * Returns the last logged message level.
171    *
172    * @return The last logged message level, or <jk>null</jk> if nothing was logged.
173    */
174   public Level getLevel() {
175      LogRecord r = lastRecord.get();
176      return r == null ? null : r.getLevel();
177   }
178
179   /**
180    * Returns the last logged message level.
181    *
182    * @return The last logged message level, or <jk>null</jk> if nothing was logged.
183    */
184   public Throwable getThrown() {
185      LogRecord r = lastRecord.get();
186      return r == null ? null : r.getThrown();
187   }
188
189   /**
190    * Returns the last logged message level.
191    *
192    * @return The last logged message level, or <jk>null</jk> if nothing was logged.
193    */
194   public ThrowableAssertion<Throwable> assertThrown() {
195      return new ThrowableAssertion<>(getThrown());
196   }
197
198   /**
199    * Resets the internal message buffer.
200    *
201    * @return This object.
202    */
203   public BasicTestCaptureCallLogger reset() {
204      this.lastRecord.set(null);
205      return this;
206   }
207}
