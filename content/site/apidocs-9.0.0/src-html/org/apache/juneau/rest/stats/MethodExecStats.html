013package org.apache.juneau.rest.stats;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017import java.lang.reflect.*;
018import java.util.*;
019import java.util.concurrent.atomic.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.cp.*;
023import org.apache.juneau.internal.*;

025/**
026 * Method execution statistics.
027 *
028 * Keeps track of number of starts/finishes on tasks and keeps an average run time.
029 *
030 * <ul class='seealso'>
031 *    <li class='link'>{@doc jrs.ExecutionStatistics}
032 *    <li class='extlink'>{@source}
033 * </ul>
034 */
035public class MethodExecStats {
036
037   //-----------------------------------------------------------------------------------------------------------------
038   // Static
039   //-----------------------------------------------------------------------------------------------------------------
040
041   /**
042    * Static creator.
043    *
044    * @param beanStore The bean store to use for creating beans.
045    * @return A new builder for this object.
046    */
047   public static Builder create(BeanStore beanStore) {
048      return new Builder(beanStore);
049   }
050
051   //-----------------------------------------------------------------------------------------------------------------
052   // Builder
053   //-----------------------------------------------------------------------------------------------------------------
054
055   /**
056    * Builder class.
057    */
058   @FluentSetters
059   public static class Builder extends BeanBuilder<MethodExecStats> {
060
061      Method method;
062      ThrownStore thrownStore;
063
064      /**
065       * Constructor.
066       *
067       * @param beanStore The bean store to use for creating beans.
068       */
069      protected Builder(BeanStore beanStore) {
070         super(MethodExecStats.class, beanStore);
071      }
072
073      @Override /* BeanBuilder */
074      protected MethodExecStats buildDefault() {
075         return new MethodExecStats(this);
076      }
077
078      //-------------------------------------------------------------------------------------------------------------
079      // Properties
080      //-------------------------------------------------------------------------------------------------------------
081
082      /**
083       * Specifies the Java method.
084       *
085       * @param value The new value for this setting.
086       * @return This object.
087       */
088      @FluentSetter
089      public Builder method(Method value) {
090         method = value;
091         return this;
092      }
093
094      /**
095       * Specifies the thrown store for tracking exceptions.
096       *
097       * @param value The new value for this setting.
098       * @return This object.
099       */
100      @FluentSetter
101      public Builder thrownStore(ThrownStore value) {
102         thrownStore = value;
103         return this;
104      }
105
106      // <FluentSetters>
107
108      @Override /* GENERATED - org.apache.juneau.BeanBuilder */
109      public Builder impl(Object value) {
110         super.impl(value);
111         return this;
112      }
113
114      @Override /* GENERATED - org.apache.juneau.BeanBuilder */
115      public Builder type(Class<?> value) {
116         super.type(value);
117         return this;
118      }
119
120      // </FluentSetters>
121   }
122
123   //-----------------------------------------------------------------------------------------------------------------
124   // Instance
125   //-----------------------------------------------------------------------------------------------------------------
126
127   private final long guid;
128   private final Method method;
129   private final ThrownStore thrownStore;
130
131   private volatile int minTime = -1, maxTime;
132
133   private AtomicInteger
134      starts = new AtomicInteger(),
135      finishes = new AtomicInteger(),
136      errors = new AtomicInteger();
137
138   private AtomicLong
139      totalTime = new AtomicLong();
140
141   /**
142    * Constructor.
143    *
144    * @param builder The builder for this object.
145    */
146   protected MethodExecStats(Builder builder) {
147      this.guid = new Random().nextLong();
148      this.method = builder.method;
149      this.thrownStore = builder.thrownStore != null ? builder.thrownStore : new ThrownStore();
150   }
151
152
153   /**
154    * Call when task is started.
155    *
156    * @return This object.
157    */
158   public MethodExecStats started() {
159      starts.incrementAndGet();
160      return this;
161   }
162
163   /**
164    * Call when task is finished.
165    *
166    * @param nanoTime The execution time of the task in nanoseconds.
167    * @return This object.
168    */
169   public MethodExecStats finished(long nanoTime) {
170      finishes.incrementAndGet();
171      int milliTime = (int)(nanoTime/1_000_000);
172      totalTime.addAndGet(nanoTime);
173      minTime = minTime == -1 ? milliTime : Math.min(minTime, milliTime);
174      maxTime = Math.max(maxTime, milliTime);
175      return this;
176   }
177
178   /**
179    * Call when an error occurs.
180    *
181    * @param e The exception thrown.  Can be <jk>null</jk>.
182    * @return This object.
183    */
184   public MethodExecStats error(Throwable e) {
185      errors.incrementAndGet();
186      thrownStore.add(e);
187      return this;
188   }
189
190   /**
191    * Returns a globally unique ID for this object.
192    *
193    * <p>
194    * A random long generated during the creation of this object.
195    * Allows this object to be differentiated from other similar objects in multi-node environments so that
196    * statistics can be reliably stored in a centralized location.
197    *
198    * @return The globally unique ID for this object.
199    */
200   public long getGuid() {
201      return guid;
202   }
203
204   /**
205    * Returns the method name of these stats.
206    *
207    * @return The method name of these stats.
208    */
209   public Method getMethod() {
210      return method;
211   }
212
213   /**
214    * Returns the number of times the {@link #started()} method was called.
215    *
216    * @return The number of times the {@link #started()} method was called.
217    */
218   public int getRuns() {
219      return starts.get();
220   }
221
222   /**
223    * Returns the number currently running method invocations.
224    *
225    * @return The number of currently running method invocations.
226    */
227   public int getRunning() {
228      return starts.get() - finishes.get();
229   }
230
231   /**
232    * Returns the number of times the {@link #error(Throwable)} method was called.
233    *
234    * @return The number of times the {@link #error(Throwable)} method was called.
235    */
236   public int getErrors() {
237      return errors.get();
238   }
239
240   /**
241    * Returns the max execution time.
242    *
243    * @return The average execution time in milliseconds.
244    */
245   public int getMinTime() {
246      return minTime == -1 ? 0 : minTime;
247   }
248
249   /**
250    * Returns the max execution time.
251    *
252    * @return The average execution time in milliseconds.
253    */
254   public int getMaxTime() {
255      return maxTime;
256   }
257
258   /**
259    * Returns the average execution time.
260    *
261    * @return The average execution time in milliseconds.
262    */
263   public int getAvgTime() {
264      int runs = finishes.get();
265      return runs == 0 ? 0 : (int)(getTotalTime() / runs);
266   }
267
268   /**
269    * Returns the total execution time.
270    *
271    * @return The total execution time in milliseconds.
272    */
273   public long getTotalTime() {
274      return totalTime.get() / 1_000_000;
275   }
276
277   /**
278    * Returns information on all stack traces of all exceptions encountered.
279    *
280    * @return Information on all stack traces of all exceptions encountered.
281    */
282   public ThrownStore getThrownStore() {
283      return thrownStore;
284   }
285
286   @Override /* Object */
287   public String toString() {
288      return json(this);
289   }
290}
