227   /**
228    * Constructor.
229    */
230   RestRequest(RestOpContext opContext, RestSession session) throws Exception {
231      this.session = session;
232      this.opContext = opContext;
233
234      inner = session.getRequest();
235      context = session.getContext();
236
237      attrs = new RequestAttributes(this);
238
239      queryParams = new RequestQueryParams(this, session.getQueryParams(), true);
240
241      headers = new RequestHeaders(this, queryParams, false);
242
243      content = new RequestContent(this);
244
245      if (context.isAllowContentParam()) {
246         String b = queryParams.get("content").asString().orElse(null);
247         if (b != null) {
248            headers.set("Content-Type", UonSerializer.DEFAULT.getResponseContentType());
249            content.mediaType(MediaType.UON).parser(UonParser.DEFAULT).content(b.getBytes(UTF8));
250         }
251      }
252
253      pathParams = new RequestPathParams(session, this, true);
254
255      beanSession = opContext.getBeanContext().getSession();
256
257      partParserSession = opContext.getPartParser().getPartSession();
258
259      pathParams.parser(partParserSession);
260
261      queryParams
262         .addDefault(opContext.getDefaultRequestQueryData().getAll())
263         .parser(partParserSession);
264
265      headers
266         .addDefault(opContext.getDefaultRequestHeaders().getAll())
267         .addDefault(context.getDefaultRequestHeaders().getAll())
268         .parser(partParserSession);
269
270      content
271         .encoders(opContext.getEncoders())
272         .parsers(opContext.getParsers())
273         .maxInput(opContext.getMaxInput());
274
275      attrs
276         .addDefault(opContext.getDefaultRequestAttributes())
277         .addDefault(context.getDefaultRequestAttributes());
278
279      if (isDebug())
280         inner = CachingHttpServletRequest.wrap(inner);
281   }

288    * Returns the request line of this request.
289    *
290    * @return The request line of this request.
291    */
292   public RequestLine getRequestLine() {
293      String x = inner.getProtocol();
294      int i = x.indexOf('/');
295      int j = x.indexOf('.', i);
296      ProtocolVersion pv = new ProtocolVersion(x.substring(0,i), parseInt(x.substring(i+1,j)), parseInt(x.substring(j+1)));
297      return new BasicRequestLine(inner.getMethod(), inner.getRequestURI(), pv);
298   }

508   /**
509    * Returns the charset specified on the <c>Content-Type</c> header, or <js>"UTF-8"</js> if not specified.
510    *
511    * @return The charset to use to decode the request content.
512    */
513   public Charset getCharset() {
514      if (charset == null) {
515         // Determine charset
516         // NOTE:  Don't use super.getCharacterEncoding() because the spec is implemented inconsistently.
517         // Jetty returns the default charset instead of null if the character is not specified on the request.
518         String h = getHeader("Content-Type").orElse(null);
519         if (h != null) {
520            int i = h.indexOf(";charset=");
521            if (i > 0)
522               charset = Charset.forName(h.substring(i+9).trim());
523         }
524         if (charset == null)
525            charset = opContext.getDefaultCharset();
526         if (charset == null)
527            charset = Charset.forName("UTF-8");
528      }
529      return charset;
530   }

532   /**
533    * Returns the preferred Locale that the client will accept content in, based on the Accept-Language header.
534    *
535    * <p>
536    * If the client request doesn't provide an <c>Accept-Language</c> header, this method returns the default locale for the server.
537    *
538    * @return The preferred Locale that the client will accept content in.  Never <jk>null</jk>.
539    */
540   public Locale getLocale() {
541      Locale best = inner.getLocale();
542      String h = headers.get("Accept-Language").asString().orElse(null);
543      if (h != null) {
544         StringRanges sr = StringRanges.of(h);
545         float qValue = 0;
546         for (StringRange r : sr.toList()) {
547            if (r.getQValue() > qValue) {
548               best = toLocale(r.getName());
549               qValue = r.getQValue();
550            }
551         }
552      }
553      return best;
554   }

1127      String uri = inner.getRequestURI();
1128      if (includeQuery || addQueryParams != null) {
1129         StringBuilder sb = new StringBuilder(uri);
1130         RequestQueryParams rq = this.queryParams.copy();
1131         if (addQueryParams != null)
1132            for (Map.Entry<String,?> e : addQueryParams.entrySet())

1134         if (! rq.isEmpty())
1135            sb.append('?').append(rq.asQueryString());
1136         uri = sb.toString();
1137      }
1138      try {
1139         return new URI(uri);
1140      } catch (URISyntaxException e) {
1141         // Shouldn't happen.

1598    * Same as {@link #getRequest(Class)} but used on pre-instantiated {@link RequestBeanMeta} objects.
1599    *
1600    * @param <T> The request bean interface to instantiate.
1601    * @param rbm The metadata about the request bean interface to create.
1602    * @return A new request bean proxy for this REST request.
1603    */
1604   public <T> T getRequest(final RequestBeanMeta rbm) {
1605      try {
1606         Class<T> c = (Class<T>)rbm.getClassMeta().getInnerClass();
1607         final BeanSession bs = getBeanSession();
1608         final BeanMeta<T> bm = bs.getBeanMeta(c);
1609         return (T)Proxy.newProxyInstance(
1610            c.getClassLoader(),
1611            new Class[] { c },
1612            new InvocationHandler() {
1613               @Override /* InvocationHandler */
1614               public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
1615                  RequestBeanPropertyMeta pm = rbm.getProperty(method.getName());
1616                  if (pm != null) {
1617                     HttpPartParserSession pp = pm.getParser(getPartParserSession());
1618                     HttpPartSchema schema = pm.getSchema();
1619                     String name = pm.getPartName();
1620                     ClassMeta<?> type = bs.getClassMeta(method.getGenericReturnType());
1621                     HttpPartType pt = pm.getPartType();
1622                     if (pt == HttpPartType.BODY)
1623                        return getContent().setSchema(schema).as(type);
1624                     if (pt == QUERY)
1625                        return getQueryParam(name).parser(pp).schema(schema).as(type).orElse(null);
1626                     if (pt == FORMDATA)
1627                        return getFormParam(name).parser(pp).schema(schema).as(type).orElse(null);
1628                     if (pt == HEADER)
1629                        return getHeader(name).parser(pp).schema(schema).as(type).orElse(null);
1630                     if (pt == PATH)
1631                        return getPathParam(name).parser(pp).schema(schema).as(type).orElse(null);
1632                  }
1633                  return null;
1634               }
1635
1636         });
1637      } catch (Exception e) {
1638         throw asRuntimeException(e);
1639      }
