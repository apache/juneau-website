035/**
036 * Represents the headers in an HTTP request.
037 *
038 * <p>
039 *    The {@link RequestHeaders} object is the API for accessing the headers of an HTTP request.
040 *    It can be accessed by passing it as a parameter on your REST Java method:
041 * </p>
042 * <p class='bjava'>
043 *    <ja>@RestPost</ja>(...)
044 *    <jk>public</jk> Object myMethod(RequestHeaders <jv>headers</jv>) {...}
045 * </p>
046 *
047 * <h5 class='figure'>Example:</h5>
048 * <p class='bjava'>
049 *    <ja>@RestPost</ja>(...)
050 *    <jk>public</jk> Object myMethod(RequestHeaders <jv>headers</jv>) {
051 *
052 *       <jc>// Add a default value.</jc>
053 *       <jv>headers</jv>.addDefault(<js>"ETag"</js>, <jsf>DEFAULT_UUID</jsf>);
054 *
055 *       <jc>// Get a header value as a POJO.</jc>
056 *       UUID <jv>etag</jv> = <jv>headers</jv>.get(<js>"ETag"</js>).as(UUID.<jk>class</jk>).get();
057 *
058 *       <jc>// Get a header as a standard HTTP part.</jc>
059 *       ContentType <jv>contentType</jv> = <jv>headers</jv>.get(ContentType.<jk>class</jk>).orElse(ContentType.<jsf>TEXT_XML</jsf>);
060 *    }
061 * </p>
062 *
063 * <p>
064 *    Some important methods on this class are:
065 * </p>
066 * <ul class='javatree'>
067 *    <li class='jc'>{@link RequestHeaders}
068 *    <ul class='spaced-list'>
069 *       <li>Methods for retrieving headers:
070 *       <ul class='javatreec'>
071 *          <li class='jm'>{@link RequestHeaders#contains(String...) contains(String...)}
072 *          <li class='jm'>{@link RequestHeaders#containsAny(String...) containsAny(String...)}
073 *          <li class='jm'>{@link RequestHeaders#get(Class) get(Class)}
074 *          <li class='jm'>{@link RequestHeaders#get(String) get(String)}
075 *          <li class='jm'>{@link RequestHeaders#getAll() getAll()}
076 *          <li class='jm'>{@link RequestHeaders#getAll(String) getAll(String)}
077 *          <li class='jm'>{@link RequestHeaders#getFirst(String) getFirst(String)}
078 *          <li class='jm'>{@link RequestHeaders#getLast(String) getLast(String)}
079 *       </ul>
080 *       <li>Methods overridding headers:
081 *       <ul class='javatreec'>
082 *          <li class='jm'>{@link RequestHeaders#add(Header...) add(Header...)}
083 *          <li class='jm'>{@link RequestHeaders#add(String, Object) add(String, Object)}
084 *          <li class='jm'>{@link RequestHeaders#addDefault(Header...) addDefault(Header...)}
085 *          <li class='jm'>{@link RequestHeaders#addDefault(List) addDefault(List)}
086 *          <li class='jm'>{@link RequestHeaders#addDefault(String,String) addDefault(String,String)}
087 *          <li class='jm'>{@link RequestHeaders#remove(Header...) remove(Header...)}
088 *          <li class='jm'>{@link RequestHeaders#remove(String...) remove(String...)}
089 *          <li class='jm'>{@link RequestHeaders#set(Header...) set(Header...)}
090 *          <li class='jm'>{@link RequestHeaders#set(String,Object) set(String,Object)}
091 *       </ul>
092 *       <li>Other methods:
093 *       <ul class='javatreec'>
094 *          <li class='jm'>{@link RequestHeaders#copy() copy()}
095 *          <li class='jm'>{@link RequestHeaders#isEmpty() isEmpty()}
096 *          <li class='jm'>{@link RequestHeaders#subset(String...) subset(String...)}
097 *       </ul>
098 *    </ul>
099 * </ul>
100 *
101 * <p>
102 * Entries are stored in a case-insensitive map unless overridden via the constructor.
103 *
104 * <ul class='seealso'>
105 *    <li class='jc'>{@link RequestHeader}
106 *    <li class='ja'>{@link org.apache.juneau.http.annotation.Header}
107 *    <li class='link'>{@doc jrs.HttpParts}
108 *    <li class='extlink'>{@source}
109 * </ul>
110 */
111public class RequestHeaders {
112
113   private final RestRequest req;
114   private final boolean caseSensitive;
115   private final VarResolverSession vs ;
116
117   private HttpPartParserSession parser;
118
119   private List<RequestHeader> list = new LinkedList<>();
120   private Map<String,List<RequestHeader>> map = new TreeMap<>();
121
122   /**
123    * TODO
124    *
125    * @param req The request creating this bean.
126    * @param query The query parameters on the request (used for overloaded header values).
127    * @param caseSensitive Whether case-sensitive name matching is enabled.
128    */
129   public RequestHeaders(RestRequest req, RequestQueryParams query, boolean caseSensitive) {
130      this.req = req;
131      this.caseSensitive = caseSensitive;
132      this.vs = req.getVarResolverSession();
133
134      for (Enumeration<String> e = req.getHttpServletRequest().getHeaderNames(); e.hasMoreElements();) {
135         String name = e.nextElement();
136         String key = key(name);
137         List<RequestHeader> l = list();
138         for (Enumeration<String> ve = req.getHttpServletRequest().getHeaders(name); ve.hasMoreElements();) {
139            RequestHeader h = new RequestHeader(req, name, ve.nextElement());
140            list.add(h);
141            l.add(h);
142         }
143         map.put(key, l);
144      }
145
146      // Parameters defined on the request URL overwrite existing headers.
147      Set<String> allowedHeaderParams = req.getContext().getAllowedHeaderParams();
148      for (RequestQueryParam p : query.getAll()) {
149         String name = p.getName();
150         String key = key(name);
151         if (allowedHeaderParams.contains(key) || allowedHeaderParams.contains("*")) {
152            set(name, p.getValue());
153         }
154      }
155   }
156
157   /**
158    * Copy constructor.
159    */
160   private RequestHeaders(RequestHeaders copyFrom) {
161      req = copyFrom.req;
162      caseSensitive = copyFrom.caseSensitive;
163      parser = copyFrom.parser;
164      list.addAll(copyFrom.list);
165      map.putAll(copyFrom.map);
166      vs = copyFrom.vs;
167   }
168
169   /**
170    * Subset constructor.
171    */
172   private RequestHeaders(RequestHeaders copyFrom, Map<String,List<RequestHeader>> headerMap) {
173      this.req = copyFrom.req;
174      map.putAll(headerMap);
175      list = headerMap.values().stream().flatMap(List::stream).collect(toList());
176      parser = copyFrom.parser;
177      caseSensitive = copyFrom.caseSensitive;
178      vs = copyFrom.vs;
179   }
180
181   /**
182    * Sets the parser to use for part values.
183    *
184    * @param value The new value for this setting.
185    * @return This object.
186    */
187   public RequestHeaders parser(HttpPartParserSession value) {
188      this.parser = value;
189      for (RequestHeader h : list)
190         h.parser(parser);
191      return this;
192   }

198   /**
199    * Adds default entries to these headers.
200    *
201    * <p>
202    * Similar to {@link #set(String, Object)} but doesn't override existing values.
203    *
204    * @param pairs The default entries.  Must not be <jk>null</jk>.
205    * @return This object.
206    */
207   public RequestHeaders addDefault(List<Header> pairs) {
208      assertArgNotNull("pairs", pairs);
209      for (Header p : pairs) {
210         String name = p.getName();
211         String key = key(name);
212         List<RequestHeader> l = map.get(key);
213         boolean hasAllBlanks = l != null && l.stream().allMatch(x -> StringUtils.isEmpty(x.getValue()));
214         if (l == null || hasAllBlanks) {
215            if (hasAllBlanks)
216               list.removeAll(l);
217            RequestHeader x = new RequestHeader(req, name, vs.resolve(p.getValue()));
218            list.add(x);
219            map.put(key, list(x));
220         }
221      }
222      return this;
223   }

249   /**
250    * Returns all the headers with the specified name.
251    *
252    * @param name The header name.  Must not be <jk>null</jk>.
253    * @return The list of all headers with the specified name, or an empty list if none are found.
254    */
255   public List<RequestHeader> getAll(String name) {
256      assertArgNotNull("name", name);
257      List<RequestHeader> l = map.get(key(name));
258      return l == null ? emptyList() : unmodifiable(l);
259   }

270   /**
271    * Returns <jk>true</jk> if the headers with the specified names are present.
272    *
273    * @param names The header names.  Must not be <jk>null</jk>.
274    * @return <jk>true</jk> if the headers with the specified names are present.
275    */
276   public boolean contains(String...names) {
277      assertArgNotNull("names", names);
278      for (String n : names)
279         if (! map.containsKey(key(n)))
280            return false;
281      return true;
282   }
283
284   /**
285    * Returns <jk>true</jk> if the header with any of the specified names are present.
286    *
287    * @param names The header names.  Must not be <jk>null</jk>.
288    * @return <jk>true</jk> if the header with any of the specified names are present.
289    */
290   public boolean containsAny(String...names) {
291      assertArgNotNull("names", names);
292      for (String n : names)
293         if (map.containsKey(key(n)))
294            return true;
295      return false;
296   }

299    * Returns <jk>true</jk> if these headers are empty.

301    * @return <jk>true</jk> if these headers are empty.
302    */
303   public boolean isEmpty() {
304      return list.isEmpty();

307   /**
308    * Adds a request header value.
309    *
310    * <p>
311    * Header is added to the end.
312    * <br>Existing headers with the same name are not changed.
313    *
314    * @param name The header name.  Must not be <jk>null</jk>.
315    * @param value The header value.  Can be <jk>null</jk>.
316    * @return This object.
317    */
318   public RequestHeaders add(String name, Object value) {
319      assertArgNotNull("name", name);
320      String key = key(name);
321      RequestHeader h = new RequestHeader(req, name, stringify(value)).parser(parser);
322      if (map.containsKey(key))
323         map.get(key).add(h);
324      else
325         map.put(key, list(h));
326      list.add(h);
327      return this;
328   }
329
330   /**
331    * Adds request header values.
332    *
333    * <p>
334    * Headers are added to the end.
335    * <br>Existing headers with the same name are not changed.
336    *
337    * @param headers The header objects.  Must not be <jk>null</jk>.
338    * @return This object.
339    */
340   public RequestHeaders add(Header...headers) {
341      assertArgNotNull("headers", headers);
342      for (Header h : headers) {
343         if (h != null)
344            add(h.getName(), h.getValue());
345      }
346      return this;
347   }
348
349   /**
350    * Sets a request header value.
351    *
352    * <p>
353    * Header is added to the end.
354    * <br>Any previous headers with the same name are removed.
355    *
356    * @param name The header name.  Must not be <jk>null</jk>.
357    * @param value
358    *    The header value.
359    *    <br>Converted to a string using {@link Object#toString()}.
360    *    <br>Can be <jk>null</jk>.
361    * @return This object.
362    */
363   public RequestHeaders set(String name, Object value) {
364      assertArgNotNull("name", name);
365      String key = key(name);
366      remove(key);
367      RequestHeader h = new RequestHeader(req, name, stringify(value)).parser(parser);
368      map.put(key, list(h));
369      list.add(h);
370      return this;
371   }
372
373   /**
374    * Sets request header values.
375    *
376    * <p>
377    * Headers are added to the end.
378    * <br>Any previous headers with the same name are removed.
379    *
380    * @param headers The header to set.  Must not be <jk>null</jk> or contain <jk>null</jk>.
381    * @return This object.
382    */
383   public RequestHeaders set(Header...headers) {
384      assertArgNotNull("headers", headers);
385      for (Header h : headers)
386         remove(h);
387      for (Header h : headers)
388         add(h);
389      return this;
390   }
391
392   /**
393    * Remove headers.
394    *
395    * @param name The header names.  Must not be <jk>null</jk>.
396    * @return This object.
397    */
398   public RequestHeaders remove(String...name) {
399      assertArgNotNull("name", name);
400      for (String n : name) {
401         String key = key(n);
402         if (map.containsKey(key))
403            list.removeAll(map.get(key));
404         map.remove(key);
405      }
406      return this;
407   }
408
409   /**
410    * Remove headers.
411    *
412    * @param headers The headers to remove.  Must not be <jk>null</jk>.
413    * @return This object.
414    */
415   public RequestHeaders remove(Header...headers) {
416      for (Header h : headers)
417         remove(h.getName());
418      return this;
419   }

421   /**
422    * Returns a copy of this object but only with the specified header names copied.
423    *
424    * @param headers The list to include in the copy.
425    * @return A new list object.
426    */
427   public RequestHeaders subset(String...headers) {
428      Map<String,List<RequestHeader>> m = alist(headers)
429         .stream()
430         .map(x -> key(x))
431         .filter(map::containsKey)
432         .collect(toMap(Function.identity(),map::get));
433
434      return new RequestHeaders(this, m);
435   }

441   /**
442    * Returns the first header with the specified name.
443    *
444    * <p>
445    * Note that this method never returns <jk>null</jk> and that {@link RequestHeader#isPresent()} can be used
446    * to test for the existence of the header.
447    *
448    * @param name The header name.  Must not be <jk>null</jk>.
449    * @return The header.  Never <jk>null</jk>.
450    */
451   public RequestHeader getFirst(String name) {
452      assertArgNotNull("name", name);
453      List<RequestHeader> l = map.get(key(name));
454      return (l == null || l.isEmpty() ? new RequestHeader(req, name, null).parser(parser) : l.get(0));
455   }
456
457   /**
458    * Returns the last header with the specified name.
459    *
460    * <p>
461    * Note that this method never returns <jk>null</jk> and that {@link RequestHeader#isPresent()} can be used
462    * to test for the existence of the header.
463    *
464    * @param name The header name.  Must not be <jk>null</jk>.
465    * @return The header.  Never <jk>null</jk>.
466    */
467   public RequestHeader getLast(String name) {
468      assertArgNotNull("name", name);
469      List<RequestHeader> l = map.get(key(name));
470      return (l == null || l.isEmpty() ? new RequestHeader(req, name, null).parser(parser) : l.get(l.size()-1));
471   }
472
473   /**
474    * Returns the condensed header with the specified name.
475    *
476    * @param name The header name.
477    * @return The header, never <jk>null</jk>.
478    */
479   public RequestHeader get(String name) {
480      List<RequestHeader> l = getAll(name);
481      if (l.isEmpty())
482         return new RequestHeader(req, name, null).parser(parser);
483      if (l.size() == 1)
484         return l.get(0);
485      StringBuilder sb = new StringBuilder(128);
486      for (int i = 0, j = l.size(); i < j; i++) {
487         if (i > 0)
488            sb.append(", ");
489         sb.append(l.get(i).getValue());
490      }
491      return new RequestHeader(req, name, sb.toString()).parser(parser);
492   }

494   /**
495    * Returns the header as the specified bean type.
496    *
497    * <p>
498    * Type must have a name specified via the {@link org.apache.juneau.http.annotation.Header} annotation
499    * and a public constructor that takes in either <c>value</c> or <c>name,value</c> as strings.
500    *
501    * @param <T> The bean type to create.
502    * @param type The bean type to create.
503    * @return The bean, never <jk>null</jk>.
504    */
505   public <T> Optional<T> get(Class<T> type) {
506      ClassMeta<T> cm = req.getBeanSession().getClassMeta(type);
507      String name = HttpParts.getName(HEADER, cm).orElseThrow(()->new BasicRuntimeException("@Header(name) not found on class {0}", className(type)));
508      return get(name).as(type);
509   }

524   /**
525    * Converts the headers to a readable string.
526    *
527    * @param sorted Sort the headers by name.
528    * @return A JSON string containing the contents of the headers.
529    */
530   public String toString(boolean sorted) {
531      JsonMap m = JsonMap.create();
532      if (sorted) {
533         for (List<RequestHeader> h1 : map.values())
534            for (RequestHeader h2 : h1)
535               m.append(h2.getName(), h2.getValue());
536      } else {
537         for (RequestHeader h : list)
538            m.append(h.getName(), h.getValue());
539      }
540      return m.toString();
541   }
542
543   private String key(String name) {
544      return caseSensitive ? name : name.toLowerCase();
545   }
546
547   @Override /* Object */
548   public String toString() {
549      return toString(false);
550   }
