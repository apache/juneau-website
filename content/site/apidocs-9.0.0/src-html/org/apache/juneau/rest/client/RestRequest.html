067/**
068 * Represents a request to a remote REST resource.
069 *
070 * <p>
071 * Instances of this class are created by the various creator methods on the {@link RestClient} class.

073 * <ul class='notes'>

077 * <ul class='seealso'>
078 *    <li class='link'>{@doc juneau-rest-client}
079 *    <li class='extlink'>{@source}
080 * </ul>

110   /**
111    * Constructs a REST call with the specified method name.
112    *
113    * @param client The client that created this request.
114    * @param uri The target URI.
115    * @param method The HTTP method name (uppercase).
116    * @param hasBody Whether this method has a body.
117    * @throws RestCallException If an exception or non-200 response code occurred during the connection attempt.
118    */
119   protected RestRequest(RestClient client, URI uri, String method, boolean hasBody) throws RestCallException {
120      super(client.getBeanContext().createSession());
121      this.client = client;
122      this.request = createInnerRequest(method, uri, hasBody);
123      this.errorCodes = client.errorCodes;
124      this.uriBuilder = new URIBuilder(request.getURI());
125      this.ignoreErrors = client.ignoreErrors;
126      this.headerData = client.createHeaderData();
127      this.queryData = client.createQueryData();
128      this.formData = client.createFormData();
129      this.pathData = client.createPathData();
130   }
131
132   /**
133    * Constructs the {@link HttpRequestBase} object that ends up being passed to the client execute method.
134    *
135    * <p>
136    * Subclasses can override this method to create their own request base objects.
137    *
138    * @param method The HTTP method.
139    * @param uri The HTTP URI.
140    * @param hasBody Whether the HTTP request has a body.
141    * @return A new {@link HttpRequestBase} object.
142    */
143   protected HttpRequestBase createInnerRequest(String method, URI uri, boolean hasBody) {
144      HttpRequestBase req = hasBody ? new BasicHttpEntityRequestBase(this, method) : new BasicHttpRequestBase(this, method);
145      req.setURI(uri);
146      return req;
147   }
