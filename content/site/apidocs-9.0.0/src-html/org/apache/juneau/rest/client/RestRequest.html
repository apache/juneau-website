066/**

111   /**
112    * Constructs a REST call with the specified method name.
113    *
114    * @param client The client that created this request.
115    * @param uri The target URI.
116    * @param method The HTTP method name (uppercase).
117    * @param hasBody Whether this method has a body.
118    * @throws RestCallException If an exception or non-200 response code occurred during the connection attempt.
119    */
120   protected RestRequest(RestClient client, URI uri, String method, boolean hasBody) throws RestCallException {
121      super(client.getBeanContext().createSession());
122      this.client = client;
123      this.request = createInnerRequest(method, uri, hasBody);
124      this.errorCodes = client.errorCodes;
125      this.uriBuilder = new URIBuilder(request.getURI());
126      this.ignoreErrors = client.ignoreErrors;
127      this.headerDataBuilder = client.createHeaderDataBuilder();
128      this.queryDataBuilder = client.createQueryDataBuilder();
129      this.formDataBuilder = client.createFormDataBuilder();
130      this.pathDataBuilder = client.createPathDataBuilder();
131   }
132
133   /**
134    * Constructs the {@link HttpRequestBase} object that ends up being passed to the client execute method.
135    *
136    * <p>
137    * Subclasses can override this method to create their own request base objects.
138    *
139    * @param method The HTTP method.
140    * @param uri The HTTP URI.
141    * @param hasBody Whether the HTTP request has a body.
142    * @return A new {@link HttpRequestBase} object.
143    */
144   protected HttpRequestBase createInnerRequest(String method, URI uri, boolean hasBody) {
145      HttpRequestBase req = hasBody ? new BasicHttpEntityRequestBase(this, method) : new BasicHttpRequestBase(this, method);
146      req.setURI(uri);
147      return req;
148   }
