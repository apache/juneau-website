013package org.apache.juneau.rest.util;
014
015import java.io.*;
016
017import javax.servlet.*;
018
019/**
020 * ServletInputStream wrapper around a normal input stream with support for limiting input.
021 *
022 * <ul class='seealso'>
023 *    <li class='extlink'>{@source}
024 * </ul>
025 */
026public final class BoundedServletInputStream extends ServletInputStream {
027
028   private final InputStream is;
029   private final ServletInputStream sis;
030   private long remain;
031
032   /**
033    * Wraps the specified input stream.
034    *
035    * @param is The input stream to wrap.
036    * @param max The maximum number of bytes to read from the stream.
037    */
038   public BoundedServletInputStream(InputStream is, long max) {
039      this.is = is;
040      this.sis = null;
041      this.remain = max;
042   }
043
044   /**
045    * Wraps the specified input stream.
046    *
047    * @param sis The input stream to wrap.
048    * @param max The maximum number of bytes to read from the stream.
049    */
050   public BoundedServletInputStream(ServletInputStream sis, long max) {
051      this.sis = sis;
052      this.is = sis;
053      this.remain = max;
054   }
055
056   /**
057    * Wraps the specified byte array.
058    *
059    * @param b The byte contents of the stream.
060    */
061   public BoundedServletInputStream(byte[] b) {
062      this(new ByteArrayInputStream(b), Long.MAX_VALUE);
063   }
064
065   @Override /* InputStream */
066   public final int read() throws IOException {
067      decrement();
068      return is.read();
069   }
070
071   @Override /* InputStream */
072   public int read(byte[] b) throws IOException {
073      return read(b, 0, b.length);
074   }
075
076   @Override /* InputStream */
077   public int read(final byte[] b, final int off, final int len) throws IOException {
078      long numBytes = Math.min(len, remain);
079      int r = is.read(b, off, (int) numBytes);
080      if (r == -1)
081         return -1;
082      decrement(numBytes);
083      return r;
084   }
085
086   @Override /* InputStream */
087   public long skip(final long n) throws IOException {
088      long toSkip = Math.min(n, remain);
089      long r = is.skip(toSkip);
090      decrement(r);
091      return r;
092   }
093
094   @Override /* InputStream */
095   public int available() throws IOException {
096      if (remain <= 0)
097         return 0;
098      return is.available();
099   }
100
101   @Override /* InputStream */
102   public synchronized void reset() throws IOException {
103      is.reset();
104   }
105
106   @Override /* InputStream */
107   public synchronized void mark(int limit) {
108      is.mark(limit);
109   }
110
111   @Override /* InputStream */
112   public boolean markSupported() {
113      return is.markSupported();
114   }
115
116   @Override /* InputStream */
117   public final void close() throws IOException {
118      is.close();
119   }
120
121   @Override /* ServletInputStream */
122   public boolean isFinished() {
123      return sis == null ? false : sis.isFinished();
124   }
125
126   @Override /* ServletInputStream */
127   public boolean isReady() {
128      return sis == null ? true : sis.isReady();
129   }
130
131   @Override /* ServletInputStream */
132   public void setReadListener(ReadListener arg0) {
133      if (sis != null)
134         sis.setReadListener(arg0);
135   }
136
137   private void decrement() throws IOException {
138      remain--;
139      if (remain < 0)
140         throw new IOException("Input limit exceeded.  See @Rest(maxInput).");
141   }
142
143   private void decrement(long count) throws IOException {
144      remain -= count;
145      if (remain < 0)
146         throw new IOException("Input limit exceeded.  See @Rest(maxInput).");
147   }
148}
