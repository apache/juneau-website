013package org.apache.juneau.rest.client;
014
015import org.apache.http.*;
016import org.apache.juneau.rest.client.assertion.*;
017
018/**
019 * An implementation of {@link StatusLine} that adds assertions methods.
020 *
021 * <ul class='seealso'>
022 *    <li class='link'>{@doc juneau-rest-client}
023 *    <li class='extlink'>{@source}
024 * </ul>
025 */
026public class ResponseStatusLine implements StatusLine {
027
028   private final RestResponse response;
029   private final StatusLine inner;
030
031   /**
032    * Constructor.
033    *
034    * @param response The response that created this object.
035    * @param inner The status line returned by the underlying client.
036    */
037   protected ResponseStatusLine(RestResponse response, StatusLine inner) {
038      this.response = response;
039      this.inner = inner;
040   }
041
042   @Override /* StatusLine */
043   public ProtocolVersion getProtocolVersion() {
044      return inner.getProtocolVersion();
045   }
046
047   @Override /* StatusLine */
048   public int getStatusCode() {
049      return inner.getStatusCode();
050   }
051
052   @Override /* StatusLine */
053   public String getReasonPhrase() {
054      return inner.getReasonPhrase();
055   }
056
057   //------------------------------------------------------------------------------------------------------------------
058   // Assertions
059   //------------------------------------------------------------------------------------------------------------------
060
061   /**
062    * Provides the ability to perform fluent-style assertions on this response status line.
063    *
064    * <h5 class='section'>Examples:</h5>
065    * <p class='bjava'>
066    *    <jc>// Validates the content type header is provided.</jc>
067    *    <jv>client</jv>
068    *       .get(<jsf>URI</jsf>)
069    *       .run()
070    *       .getStatusLine().assertValue().asCode().is(200);
071    * </p>
072    *
073    * @return A new fluent assertion object.
074    */
075   public FluentResponseStatusLineAssertion<ResponseStatusLine> assertValue() {
076      return new FluentResponseStatusLineAssertion<>(this, this);
077   }
078
079   /**
080    * Returns the response that created this object.
081    *
082    * @return The response that created this object.
083    */
084   public RestResponse response() {
085      return response;
086   }
087
088   //------------------------------------------------------------------------------------------------------------------
089   // Other methods
090   //------------------------------------------------------------------------------------------------------------------
091
092   @Override /* Object */
093   public String toString() {
094      return inner.toString();
095   }
096}
