069/**
070 * Mocked {@link RestClient}.
071 *
072 * <p>
073 *    This class is used for performing serverless unit testing of {@link Rest @Rest}-annotated and {@link Remote @Remote}-annotated classes.
074 *
075 * <p>
076 *    The class itself extends from {@link RestClient} providing it with the rich feature set of that API and combines
077 *    it with the Apache HttpClient {@link HttpClientConnection} interface for processing requests.
078 *  The class converts {@link HttpRequest} objects to instances of {@link MockServletRequest} and {@link MockServletResponse} which are passed directly
079 *  to the call handler on the resource class {@link RestContext#execute(Object,HttpServletRequest,HttpServletResponse)}.
080 *  In effect, you're fully testing your REST API as if it were running in a live servlet container, yet not
081 *  actually having to run in a servlet container.
082 *  All aspects of the client and server side code are tested, yet no servlet container is required.  The actual
083 *  over-the-wire transmission is the only aspect being bypassed.
084 *
085 * <p>
086 * The following shows a simple example of invoking a PUT method on a simple REST interface and asserting the correct status code and response body:
087 *
088 * <h5 class='figure'>Example:</h5>
089 * <p class='bjava'>
090 *    <jk>public class</jk> MockTest {
091 *
092 *       <jc>// A simple bean with one field.</jc>
093 *       <jk>public static class</jk> MyBean {
094 *          <jk>public int</jk> <jf>foo</jf> = 1;
095 *       }
096 *
097 *       <jc>// Our REST resource to test.</jc>
098 *       <jc>// Simply echos the response.</jc>
099 *       <ja>@Rest</ja>(
100 *          serializers=SimpleJsonSerializer.<jk>class</jk>,
101 *          parsers=JsonParser.<jk>class</jk>
102 *       )
103 *       <jk>public static class</jk> EchoRest {
104 *
105 *          <ja>@RestPut</ja>(
106 *             path=<js>"/echo"</js>
107 *          )
108 *          <jk>public</jk> MyBean echo(<ja>@Content</ja> MyBean <jv>bean</jv>) {
109 *             <jk>return</jk> <jv>bean</jv>;
110 *          }
111 *       }
112 *
113 *       <jc>// Our JUnit test.</jc>
114 *       <ja>@Test</ja>
115 *       <jk>public void</jk> testEcho() <jk>throws</jk> Exception {
116 *
117 *          MyBean <jv>myBean</jv> = <jk>new</jk> MyBean();
118 *
119 *          <jc>// Do a round-trip on the bean through the REST interface</jc>
120 *          <jv>myBean</jv> = MockRestClient
121 *             .<jsm>create</jsm>(EchoRest.<jk>class</jk>)
122 *             .simpleJson()
123 *             .build()
124 *             .put(<js>"/echo"</js>, <jv>myBean</jv>)
125 *             .run()
126 *             .assertStatus().is(200)
127 *             .assertContent().is(<js>"{foo:1}"</js>)
128 *             .getContent().as(MyBean.<jk>class</jk>);
129 *
130 *          <jsm>assertEquals</jsm>(1, <jv>myBean</jv>.<jf>foo</jf>);
131 *       }
132 *    }
133 * </p>
134 * <p>
135 *    Breaking apart the fluent method call above will help you understand how this works.
136 *
137 * <p class='bjava'>
138 *    <ja>@Test</ja>
139 *    <jk>public void</jk> testEcho() <jk>throws</jk> Exception {
140 *
141 *       <jc>// Instantiate our mock client.</jc>
142 *       MockRestClient <jv>client</jv> = MockRestClient
143 *          .<jsm>create</jsm>(EchoRest.<jk>class</jk>)
144 *          .simpleJson()
145 *          .build();
146 *
147 *       <jc>// Create a request.</jc>
148 *       RestRequest <jv>req</jv> = <jv>client</jv>.put(<js>"/echo"</js>, <jv>bean</jv>);
149 *
150 *       <jc>// Execute it (by calling RestCallHandler.service(...) and then returning the response object).</jc>
151 *       RestResponse <jv>res</jv> = <jv>req</jv>.run();
152 *
153 *       <jc>// Run assertion tests on the results.</jc>
154 *       <jv>res</jv>.assertStatus().is(200);
155 *       <jv>res</jv>.assertContent().is(<js>"'foo'"</js>);
156 *
157 *       <jc>// Convert the content of the response to a bean.</jc>
158 *       <jv>bean</jv> = <jv>res</jv>.getContent().as(MyBean.<jk>class</jk>);
159 *    }
160 * </p>
161 *
162 * <p>
163 *    The <c>create(Object)</c> method can take in either <c>Class</c> objects or pre-instantiated beans.
164 *    The latter is particularly useful for testing Spring beans.
165 *
166 * <p>
167 *    The {@link MockRestRequest} object has convenience methods provided to allow you to set any properties
168 *    directly on the underlying {@link HttpServletRequest} object.  The following example shows how
169 *    this can be used to directly set roles on the request object to perform security testing.
170 *
171 * <h5 class='figure'>Example:</h5>
172 * <p class='bjava'>
173 *    <ja>@Rest</ja>(roleGuard=<js>"ADMIN"</js>)
174 *    <jk>public class</jk> A {
175 *       <ja>@RestGet</ja>
176 *       <jk>public</jk> String get() {
177 *          <jk>return</jk> <js>"OK"</js>;
178 *       }
179 *    }
180 *
181 *    <ja>@Test</ja>
182 *    <jk>public void</jk> mytest() <jk>throws</jk> Exception {
183 *       MockRestClient <jv>client</jv> = MockRestClient.<jsm>build</jsm>(A.<jk>class</jk>);
184 *
185 *       <jc>// Admin user should get 200, but anyone else should get 403-Unauthorized.</jc>
186 *       <jv>client</jv>.get().roles(<js>"ADMIN"</js>).run().assertStatus().is(200);
187 *       <jv>client</jv>.get().roles(<js>"USER"</js>).run().assertStatus().is(403);
188 *    }
189 * </p>
190 *
191 * <p>
192 *    Debug mode is provided that will cause your HTTP requests and responses to be sent to the console:
193 *
194 * <h5 class='figure'>Example:</h5>
195 * <p class='bjava'>
196 *    MockRestClient <jv>client</jv> = MockRestClient
197 *       .<jsm>create</jsm>(MyRest.<jk>class</jk>)
198 *       .debug()
199 *       .simpleJson()
200 *       .build();
201 * </p>
202 *
203 * <p>
204 *    The class can also be used for testing of {@link Remote @Remote}-annotated interfaces against {@link Rest @Rest}-annotated resources.
205 *
206 * <h5 class='figure'>Example:</h5>
207 * <p class='bjava'>
208 *    <jc>// Our remote resource to test.</jc>
209 *    <ja>@Remote</ja>
210 *    <jk>public interface</jk> MyRemoteInterface {
211 *
212 *       <ja>@RemoteGet</ja>(<js>"/echoQuery"</js>)
213 *       <jk>public int</jk> echoQuery(<ja>@Query</ja>(name=<js>"id"</js>) <jk>int</jk> <jv>id</jv>);
214 *    }
215 *
216 *    <jc>// Our mocked-up REST interface to test against.</jc>
217 *    <ja>@Rest</ja>
218 *    <jk>public class</jk> MyRest {
219 *
220 *       <ja>@RestGet</ja>(path=<js>"/echoQuery"</js>)
221 *       <jk>public int</jk> echoQuery(<ja>@Query</ja>(<js>"id"</js>) String <jv>id</jv>) {
222 *          <jk>return</jk> <jv>id</jv>;
223 *       }
224 *    }
225 *
226 *    <ja>@Test</ja>
227 *    <jk>public void</jk> testProxy() {
228 *       MyRemoteInterface <jv>client</jv> = MockRestClient
229 *          .create(MyRest.<jk>class</jk>)
230 *          .json()
231 *          .build()
232 *          .getRemote(MyRemoteInterface.<jk>class</jk>);
233 *
234 *       <jsm>assertEquals</jsm>(123, <jv>client</jv>.echoQuery(123));
235 *    }
236 * </p>
237 *
238 * <ul class='notes'>
239 *    <li class='note'>This class is thread safe and reusable.
240 * </ul>
241 *
242 * <ul class='seealso'>
243 *    <li class='link'>{@doc juneau-rest-mock}
244 *    <li class='extlink'>{@source}
245 * </ul>
246 */
247public class MockRestClient extends RestClient implements HttpClientConnection {
