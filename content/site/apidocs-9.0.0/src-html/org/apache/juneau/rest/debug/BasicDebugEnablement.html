013package org.apache.juneau.rest.debug;
014
015import static org.apache.juneau.internal.ObjectUtils.*;
016import static org.apache.juneau.Enablement.*;
017import static org.apache.juneau.collections.JsonMap.*;
018
019import java.lang.reflect.*;
020import java.util.function.*;
021
022import javax.servlet.http.*;
023
024import org.apache.juneau.*;
025import org.apache.juneau.rest.*;
026import org.apache.juneau.utils.*;
027
028/**
029 * Default implementation of the {@link DebugEnablement} interface.
030 *
031 * <ul class='seealso'>
032 *    <li class='link'>{@doc jrs.LoggingAndDebugging}
033 *    <li class='extlink'>{@source}
034 * </ul>
035 */
036public class BasicDebugEnablement implements DebugEnablement {
037
038   private final Enablement defaultEnablement;
039   private final ReflectionMap<Enablement> enablementMap;
040   private final Predicate<HttpServletRequest> conditionalPredicate;
041
042   /**
043    * Constructor.
044    *
045    * @param builder The builder containing the settings for this bean.
046    */
047   public BasicDebugEnablement(DebugEnablement.Builder builder) {
048      this.defaultEnablement = firstNonNull(builder.defaultEnablement, NEVER);
049      this.enablementMap = builder.mapBuilder.build();
050      this.conditionalPredicate = firstNonNull(builder.conditional, x -> "true".equalsIgnoreCase(x.getHeader("Debug")));
051   }
052
053   @Override
054   public boolean isDebug(RestOpContext context, HttpServletRequest req) {
055      Method m = context.getJavaMethod();
056      Enablement e = enablementMap.find(m).orElse(enablementMap.find(m.getDeclaringClass()).orElse(defaultEnablement));
057      return e == ALWAYS || (e == CONDITIONAL && isConditionallyEnabled(req));
058   }
059
060   @Override
061   public boolean isDebug(RestContext context, HttpServletRequest req) {
062      Class<?> c = context.getResourceClass();
063      Enablement e = enablementMap.find(c).orElse(defaultEnablement);
064      return e == ALWAYS || (e == CONDITIONAL && isConditionallyEnabled(req));
065   }
066
067   /**
068    * Returns <jk>true</jk> if debugging is conditionally enabled on the specified request.
069    *
070    * <p>
071    * This method only gets called when the enablement value resolves to {@link Enablement#CONDITIONAL CONDITIONAL}.
072    *
073    * <p>
074    * Subclasses can override this method to provide their own implementation.
075    * The default implementation is provided by {@link DebugEnablement.Builder#conditional(Predicate)}
076    * which has a default predicate of <c><jv>x</jv> -&gt; <js>"true"</js>.equalsIgnoreCase(<jv>x</jv>.getHeader(<js>"Debug"</js>)</c>.
077    *
078    * @param req The incoming HTTP request.
079    * @return <jk>true</jk> if debugging is conditionally enabled on the specified request.
080    */
081   protected boolean isConditionallyEnabled(HttpServletRequest req) {
082      return conditionalPredicate.test(req);
083   }
084
085   @Override /* Object */
086   public String toString() {
087      return filteredMap()
088         .append("defaultEnablement", defaultEnablement)
089         .append("enablementMap", enablementMap)
090         .append("conditionalPredicate", conditionalPredicate)
091         .asString();
092   }
093}
