013package org.apache.juneau.rest.swagger;
014
015import java.util.*;
016
017import org.apache.juneau.cp.*;
018import org.apache.juneau.dto.swagger.Swagger;
019import org.apache.juneau.jsonschema.*;
020import org.apache.juneau.rest.*;
021import org.apache.juneau.rest.annotation.*;
022import org.apache.juneau.svl.*;
023
024/**
025 * Basic implementation of a {@link SwaggerProvider}.
026 *
027 * <ul class='seealso'>
028 *    <li class='link'>{@doc jrs.Swagger}
029 *    <li class='extlink'>{@source}
030 * </ul>
031 */
032public class BasicSwaggerProvider implements SwaggerProvider {
033
034   private final VarResolver vr;
035   private final JsonSchemaGenerator js;
036   private final Messages messages;
037   private final FileFinder fileFinder;
038   private final BeanStore beanStore;
039
040   /**
041    * Constructor.
042    *
043    * @param beanStore The bean store containing injectable beans for this logger.
044    */
045   public BasicSwaggerProvider(BeanStore beanStore) {
046      this(
047         SwaggerProvider
048            .create(beanStore)
049            .varResolver(()->beanStore.getBean(VarResolver.class).get())
050            .fileFinder(()->beanStore.getBean(FileFinder.class).get())
051            .messages(()->beanStore.getBean(Messages.class).get())
052            .jsonSchemaGenerator(()->beanStore.getBean(JsonSchemaGenerator.class).get())
053      );
054   }
055
056   /**
057    * Constructor.
058    *
059    * @param builder The builder containing the settings for this Swagger provider.
060    */
061   public BasicSwaggerProvider(SwaggerProvider.Builder builder) {
062      this.beanStore = builder.beanStore;
063      this.vr = builder.varResolver().orElse(beanStore.getBean(VarResolver.class).orElse(VarResolver.DEFAULT));
064      this.js = builder.jsonSchemaGenerator().orElse(beanStore.getBean(JsonSchemaGenerator.class).orElse(JsonSchemaGenerator.DEFAULT));
065      this.messages = builder.messages().orElse(null);
066      this.fileFinder = builder.fileFinder().orElse(null);
067   }
068
069   /**
070    * Returns the Swagger associated with the specified context of a {@link Rest}-annotated class.
071    *
072    * <p>
073    * Subclasses can override this to provide their own method for generating Swagger.
074    *
075    * @param context The context of the {@link Rest}-annotated class.
076    * @param locale The request locale.
077    * @return A new {@link Swagger} object.
078    * @throws Exception If an error occurred producing the Swagger.
079    */
080   @Override /* SwaggerProvider */
081   public Swagger getSwagger(RestContext context, Locale locale) throws Exception {
082
083      Class<?> c = context.getResourceClass();
084      FileFinder ff = fileFinder != null ? fileFinder : FileFinder.create(beanStore).cp(c,null,false).build();
085      Messages mb = messages != null ? messages.forLocale(locale) : Messages.create(c).build().forLocale(locale);
086      VarResolverSession vrs = vr.createSession().bean(Messages.class, mb);
087      BasicSwaggerProviderSession session = new BasicSwaggerProviderSession(context, locale, ff, messages, vrs, js.getSession());
088
089      return session.getSwagger();
090   }
091}
