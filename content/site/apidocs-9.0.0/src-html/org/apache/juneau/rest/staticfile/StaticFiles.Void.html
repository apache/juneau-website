013package org.apache.juneau.rest.staticfile;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016
017import java.nio.file.*;
018import java.util.*;
019
020import javax.activation.*;
021
022import org.apache.http.*;
023import org.apache.juneau.*;
024import org.apache.juneau.cp.*;
025import org.apache.juneau.http.resource.*;
026import org.apache.juneau.internal.*;
027import org.apache.juneau.utils.*;
028
029/**
030 * API for retrieving localized static files from either the classpath or file system.
031 *
032 * <ul class='seealso'>
033 *    <li class='link'>{@doc jrs.StaticFiles}
034 *    <li class='extlink'>{@source}
035 * </ul>
036 */
037public interface StaticFiles extends FileFinder {
038
039   //-----------------------------------------------------------------------------------------------------------------
040   // Static
041   //-----------------------------------------------------------------------------------------------------------------
042
043   /** Represents no static files */
044   public abstract class Void implements StaticFiles {}
045
046   /**
047    * Static creator.
048    *
049    * @param beanStore The bean store to use for creating beans.
050    * @return A new builder for this object.
051    */
052   public static Builder create(BeanStore beanStore) {
053      return new Builder(beanStore);
054   }
055
056   //-----------------------------------------------------------------------------------------------------------------
057   // Builder
058   //-----------------------------------------------------------------------------------------------------------------
059
060   /**
061    * Builder class.
062    */
063   @FluentSetters
064   public static class Builder extends BeanBuilder<StaticFiles> {
065
066      List<Header> headers;
067      MimetypesFileTypeMap mimeTypes;
068      FileFinder.Builder fileFinder;
069
070      /**
071       * Constructor.
072       *
073       * @param beanStore The bean store to use for creating beans.
074       */
075      protected Builder(BeanStore beanStore) {
076         super(BasicStaticFiles.class, beanStore);
077         headers = list();
078         fileFinder = FileFinder.create(beanStore);
079         mimeTypes = new ExtendedMimetypesFileTypeMap();
080      }
081
082      @Override /* BeanBuilder */
083      protected StaticFiles buildDefault() {
084         return new BasicStaticFiles(this);
085      }
086
087      //-------------------------------------------------------------------------------------------------------------
088      // Properties
089      //-------------------------------------------------------------------------------------------------------------
090
091      /**
092       * Appends headers to add to HTTP responses.
093       *
094       * <p>
095       * Can be called multiple times to add multiple headers.
096       *
097       * @param headers The headers to add.
098       * @return This object.
099       */
100      @FluentSetter
101      public Builder headers(Header...headers) {
102         addAll(this.headers, headers);
103         return this;
104      }
105
106      /**
107       * Prepend the MIME type values to the MIME types registry.
108       *
109       * @param mimeTypes A .mime.types formatted string of entries.  See {@link MimetypesFileTypeMap#addMimeTypes(String)}.
110       * @return This object.
111       */
112      @FluentSetter
113      public Builder addMimeTypes(String mimeTypes) {
114         this.mimeTypes.addMimeTypes(mimeTypes);
115         return this;
116      }
117
118      /**
119       * Replaces the MIME types registry used for determining content types.
120       *
121       * @param mimeTypes The new MIME types registry.
122       * @return This object.
123       */
124      @FluentSetter
125      public Builder mimeTypes(MimetypesFileTypeMap mimeTypes) {
126         this.mimeTypes = mimeTypes;
127         return this;
128      }
129
130      /**
131       * Enables in-memory caching of files for quicker retrieval.
132       *
133       * @param cachingLimit The maximum file size in bytes.
134       * @return This object.
135       */
136      @FluentSetter
137      public Builder caching(long cachingLimit) {
138         fileFinder.caching(cachingLimit);
139         return this;
140      }
141
142      /**
143       * Adds a class subpackage to the lookup paths.
144       *
145       * @param c The class whose package will be added to the lookup paths.  Must not be <jk>null</jk>.
146       * @param path The absolute or relative subpath.
147       * @param recursive If <jk>true</jk>, also recursively adds all the paths of the parent classes as well.
148       * @return This object.
149       */
150      @FluentSetter
151      public Builder cp(Class<?> c, String path, boolean recursive) {
152         fileFinder.cp(c, path, recursive);
153         return this;
154      }
155
156      /**
157       * Adds a file system directory to the lookup paths.
158       *
159       * @param path The path relative to the working directory.  Must not be <jk>null</jk>
160       * @return This object.
161       */
162      @FluentSetter
163      public Builder dir(String path) {
164         fileFinder.dir(path);
165         return this;
166      }
167
168      /**
169       * Specifies the regular expression file name pattern to use to exclude files from being retrieved from the file source.
170       *
171       * @param patterns
172       *    The regular expression exclude patterns.
173       *    <br>If none are specified, no files will be excluded.
174       * @return This object.
175       */
176      @FluentSetter
177      public Builder exclude(String...patterns) {
178         fileFinder.exclude(patterns);
179         return this;
180      }
181
182      /**
183       * Specifies the regular expression file name patterns to use to include files being retrieved from the file source.
184       *
185       * @param patterns
186       *    The regular expression include patterns.
187       *    <br>The default is <js>".*"</js>.
188       * @return This object.
189       */
190      @FluentSetter
191      public Builder include(String...patterns) {
192         fileFinder.include(patterns);
193         return this;
194      }
195
196      /**
197       * Adds a file system directory to the lookup paths.
198       *
199       * @param path The directory path.
200       * @return This object.
201       */
202      @FluentSetter
203      public Builder path(Path path) {
204         fileFinder.path(path);
205         return this;
206      }
207
208      // <FluentSetters>
209
210      @Override /* GENERATED - org.apache.juneau.BeanBuilder */
211      public Builder impl(Object value) {
212         super.impl(value);
213         return this;
214      }
215
216      @Override /* GENERATED - org.apache.juneau.BeanBuilder */
217      public Builder type(Class<?> value) {
218         super.type(value);
219         return this;
220      }
221
222      // </FluentSetters>
223   }
224
225   //-----------------------------------------------------------------------------------------------------------------
226   // Instance
227   //-----------------------------------------------------------------------------------------------------------------
228
229   /**
230    * Resolve the specified path.
231    *
232    * @param path The path to resolve to a static file.
233    * @param locale Optional locale.
234    * @return The resource, or <jk>null</jk> if not found.
235    */
236   public Optional<HttpResource> resolve(String path, Locale locale);
237}
