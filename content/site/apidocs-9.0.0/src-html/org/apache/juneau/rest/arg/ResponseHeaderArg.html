013package org.apache.juneau.rest.arg;
014
015import static org.apache.juneau.http.annotation.HeaderAnnotation.*;
016import java.lang.reflect.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.http.annotation.*;
020import org.apache.juneau.http.header.*;
021import org.apache.juneau.httppart.*;
022import org.apache.juneau.reflect.*;
023import org.apache.juneau.rest.*;
024import org.apache.juneau.rest.annotation.*;
025import org.apache.juneau.rest.httppart.*;
026
027/**
028 * Resolves method parameters annotated with {@link Header} of type {@link Value} representing response headers
029 * on {@link RestOp}-annotated Java methods.
030 *
031 * <p>
032 * The parameter value must be of type {@link Value} that accepts a value that is then set via:
033 * <p class='bjava'>
034 *    <jv>opSession</jv>
035 *       .{@link RestOpSession#getResponse() getResponse}()
036 *       .{@link RestResponse#setHeader(String,String) setOutput}(<jv>name</jv>,<jv>value</jv>);
037 * </p>
038 *
039 * <ul class='seealso'>
040 *    <li class='link'>{@doc jrs.JavaMethodParameters}
041 *    <li class='extlink'>{@source}
042 * </ul>
043 */
044public class ResponseHeaderArg implements RestOpArg {
045   final ResponsePartMeta meta;
046   final String name;
047   private final Type type;
048
049   /**
050    * Static creator.
051    *
052    * @param paramInfo The Java method parameter being resolved.
053    * @param annotations The annotations to apply to any new part parsers.
054    * @return A new {@link ResponseHeaderArg}, or <jk>null</jk> if the parameter is not annotated with {@link Header}.
055    */
056   public static ResponseHeaderArg create(ParamInfo paramInfo, AnnotationWorkList annotations) {
057      if (paramInfo.getParameterType().is(Value.class) && (paramInfo.hasAnnotation(Header.class) || paramInfo.getParameterType().hasAnnotation(Header.class)))
058         return new ResponseHeaderArg(paramInfo, annotations);
059      return null;
060   }
061
062   /**
063    * Constructor.
064    *
065    * @param pi The Java method parameter being resolved.
066    * @param annotations The annotations to apply to any new part parsers.
067    */
068   protected ResponseHeaderArg(ParamInfo pi, AnnotationWorkList annotations) {
069      this.name = findName(pi).orElseThrow(() -> new ArgException(pi, "@Header used without name or value"));
070      this.type = pi.getParameterType().innerType();
071      HttpPartSchema schema = HttpPartSchema.create(Header.class, pi);
072
073      Class<? extends HttpPartSerializer> ps = schema.getSerializer();
074      this.meta = new ResponsePartMeta(HttpPartType.HEADER, schema, ps != null ? HttpPartSerializer.creator().type(ps).apply(annotations).create() : null);
075
076      Class<?> c = type instanceof Class ? (Class<?>)type : type instanceof ParameterizedType ? (Class<?>)((ParameterizedType)type).getRawType() : null;
077      if (c != Value.class)
078         throw new ArgException(pi, "Type must be Value<?> on parameter annotated with @Header annotation");
079   }
080
081   @SuppressWarnings({ "unchecked", "rawtypes" })
082   @Override /* RestOpArg */
083   public Object resolve(final RestOpSession opSession) throws Exception {
084      Value<Object> v = new Value();
085      v.listener(new ValueListener() {
086         @Override
087         public void onSet(Object o) {
088            RestRequest req = opSession.getRequest();
089            RestResponse res = opSession.getResponse();
090            ResponsePartMeta rpm = req.getOpContext().getResponseHeaderMeta(o);
091            if (rpm == null)
092               rpm = ResponseHeaderArg.this.meta;
093            HttpPartSerializerSession pss = rpm.getSerializer() == null ? req.getPartSerializerSession() : rpm.getSerializer().getPartSession();
094            res.setHeader(new SerializedHeader(name, o, pss, rpm.getSchema(), false));
095         }
096      });
097      return v;
098   }
099}
