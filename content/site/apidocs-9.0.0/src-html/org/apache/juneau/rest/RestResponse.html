042/**

044 *
045 * <p>
046 *    The {@link RestResponse} object is an extension of the <l>HttpServletResponse</l> class

126   /**
127    * Constructor.
128    */
129   RestResponse(RestOpContext opContext, RestSession session, RestRequest req) throws Exception {
130
131      inner = session.getResponse();
132      request = req;
133
134      this.opContext = opContext;
135      responseBeanMeta = opContext.getResponseMeta();
136
137      RestContext context = session.getContext();
138
139      try {
140         String passThroughHeaders = request.getHeader("x-response-headers").orElse(null);
141         if (passThroughHeaders != null) {
142            JsonMap m = context.getPartParser().getPartSession().parse(HEADER, null, passThroughHeaders, BeanContext.DEFAULT.getClassMeta(JsonMap.class));
143            for (Map.Entry<String,Object> e : m.entrySet())
144               addHeader(e.getKey(), resolveUris(e.getValue()));
145         }
146      } catch (Exception e1) {
147         throw new BadRequest(e1, "Invalid format for header 'x-response-headers'.  Must be in URL-encoded format.");
148      }
149
150      // Find acceptable charset
151      String h = request.getHeader("accept-charset").orElse(null);
152      Charset charset = null;
153      if (h == null)
154         charset = opContext.getDefaultCharset();
155      else for (StringRange r : StringRanges.of(h).toList()) {
156         if (r.getQValue() > 0) {
157            if (r.getName().equals("*"))
158               charset = opContext.getDefaultCharset();
159            else if (Charset.isSupported(r.getName()))
160               charset = Charset.forName(r.getName());
161            if (charset != null)
162               break;
163         }
164      }
165
166      request.getContext().getDefaultResponseHeaders().forEach(x->addHeader(x.getValue(), resolveUris(x.getValue())));  // Done this way to avoid list/array copy.
167
168      opContext.getDefaultResponseHeaders().forEach(x->addHeader(x.getName(), resolveUris(x.getValue())));
169
170      if (charset == null)
171         throw new NotAcceptable("No supported charsets in header ''Accept-Charset'': ''{0}''", request.getHeader("Accept-Charset").orElse(null));
172      inner.setCharacterEncoding(charset.name());
173
174   }

225    * @return This object.

315   /**
316    * Equivalent to {@link HttpServletResponse#getOutputStream()}, except wraps the output stream if an {@link Encoder}
317    * was found that matched the <c>Accept-Encoding</c> header.
318    *
319    * @return A negotiated output stream.
320    * @throws NotAcceptable If unsupported Accept-Encoding value specified.
321    * @throws IOException Thrown by underlying stream.
322    */
323   public FinishableServletOutputStream getNegotiatedOutputStream() throws NotAcceptable, IOException {
324      if (os == null) {
325         Encoder encoder = null;
326         EncoderSet encoders = request.getOpContext().getEncoders();
327
328         String ae = request.getHeader("Accept-Encoding").orElse(null);
329         if (! (ae == null || ae.isEmpty())) {
330            EncoderMatch match = encoders.getEncoderMatch(ae);
331            if (match == null) {
332               // Identity should always match unless "identity;q=0" or "*;q=0" is specified.
333               if (ae.matches(".*(identity|\\*)\\s*;\\s*q\\s*=\\s*(0(?!\\.)|0\\.0).*")) {
334                  throw new NotAcceptable(
335                     "Unsupported encoding in request header ''Accept-Encoding'': ''{0}''\n\tSupported codings: {1}",
336                     ae, SimpleJson.of(encoders.getSupportedEncodings())
337                  );
338               }
339            } else {
340               encoder = match.getEncoder();
341               String encoding = match.getEncoding().toString();
342
343               // Some clients don't recognize identity as an encoding, so don't set it.
344               if (! encoding.equals("identity"))
345                  setHeader("content-encoding", encoding);
346            }
347         }
348         @SuppressWarnings("resource")
349         ServletOutputStream sos = getOutputStream();
350         os = new FinishableServletOutputStream(encoder == null ? sos : encoder.getOutputStream(sos));
351      }
352      return os;
353   }

431   @SuppressWarnings("resource")
432   private FinishablePrintWriter getWriter(boolean raw, boolean autoflush) throws NotAcceptable, IOException {
433      if (w != null)
434         return w;
435
436      // If plain text requested, override it now.
437      if (request.isPlainText())
438         setHeader("Content-Type", "text/plain");
439
440      try {
441         OutputStream out = (raw ? getOutputStream() : getNegotiatedOutputStream());
442         w = new FinishablePrintWriter(out, getCharacterEncoding(), autoflush);
443         return w;
444      } catch (UnsupportedEncodingException e) {
445         String ce = getCharacterEncoding();
446         setCharacterEncoding("UTF-8");
447         throw new NotAcceptable("Unsupported charset in request header ''Accept-Charset'': ''{0}''", ce);
448      }
449   }

525   /**
526    * Redirects to the specified URI.
527    *
528    * <p>
529    * Relative URIs are always interpreted as relative to the context root.
530    * This is similar to how WAS handles redirect requests, and is different from how Tomcat handles redirect requests.
531    *
532    * @param uri The redirection URL.
533    * @throws IOException If an input or output exception occurs
534    */
535   public void sendRedirect(String uri) throws IOException {
536      char c = (uri.length() > 0 ? uri.charAt(0) : 0);
537      if (c != '/' && uri.indexOf("://") == -1)
538         uri = request.getContextPath() + '/' + uri;
539      inner.sendRedirect(uri);
540   }

542   /**
543    * Sets a response header with the given name and value.
544    *
545    * <p>
546    * If the header had already been set, the new value overwrites the previous one.
547    *
548    * <p>
549    * The {@link #containsHeader(String)} method can be used to test for the presence of a header before setting its value.
550    *
551    * @param name The header name.
552    * @param value The header value.
553    */
554   public void setHeader(String name, String value) {
555
556      // Jetty doesn't set the content type correctly if set through this method.
557      // Tomcat/WAS does.
558      if (name.equalsIgnoreCase("Content-Type")) {
559         inner.setContentType(value);
560         ContentType ct = contentType(value);
561         if (ct != null && ct.getParameter("charset") != null)
562            inner.setCharacterEncoding(ct.getParameter("charset"));
563      } else {
564         if (safeHeaders)
565            value = stripInvalidHttpHeaderChars(value);
566         value = abbreviate(value, maxHeaderLength);
567         inner.setHeader(name, value);
568      }
569   }

833   /**
834    * Adds a response header with the given name and value.
835    *
836    * <p>
837    * This method allows response headers to have multiple values.
838    *
839    * <p>
840    * A no-op of either the name or value is <jk>null</jk>.
841    *
842    * <p>
843    * Note that per <a class='doclink' href='https://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html#sec4.2'>RFC2616</a>,
844    * only headers defined as comma-delimited lists [i.e., #(values)] should be defined as multiple message header fields.
845    *
846    * @param name The header name.
847    * @param value The header value.
848    * @return This object.
849    */
850   public RestResponse addHeader(String name, String value) {
851      if (name != null && value != null) {
852         if (name.equalsIgnoreCase("Content-Type"))
853            setHeader(name, value);
854         else {
855            if (safeHeaders)
856               value = stripInvalidHttpHeaderChars(value);
857            value = abbreviate(value, maxHeaderLength);
858            inner.addHeader(name, value);
859         }
860      }
861      return this;
862   }

864   /**
865    * Sets a response header.
866    *
867    * <p>
868    * Any previous header values are removed.
869    *
870    * <p>
871    * Value is added at the end of the headers.
872    *
873    * @param header The header.
874    * @return This object.
875    */
876   public RestResponse setHeader(Header header) {
877      if (header == null) {
878         // Do nothing.
879      } else if (header instanceof BasicUriHeader) {
880         BasicUriHeader x = (BasicUriHeader)header;
881         setHeader(x.getName(), resolveUris(x.getValue()));
882      } else if (header instanceof SerializedHeader) {
883         SerializedHeader x = ((SerializedHeader)header).copyWith(request.getPartSerializerSession(), null);
884         String v = x.getValue();
885         if (v != null && v.indexOf("://") != -1)
886            v = resolveUris(v);
887         setHeader(x.getName(), v);
888      } else {
889         setHeader(header.getName(), header.getValue());
890      }
891      return this;
892   }

894   /**
895    * Adds a response header.
896    *
897    * <p>
898    * Any previous header values are preserved.
899    *
900    * <p>
901    * Value is added at the end of the headers.
902    *
903    * <p>
904    * If the header is a {@link BasicUriHeader}, the URI will be resolved using the {@link RestRequest#getUriResolver()} object.
905    *
906    * <p>
907    * If the header is a {@link SerializedHeader} and the serializer session is not set, it will be set to the one returned by {@link RestRequest#getPartSerializerSession()} before serialization.
908    *
909    * <p>
910    * Note that per <a class='doclink' href='https://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html#sec4.2'>RFC2616</a>,
911    * only headers defined as comma-delimited lists [i.e., #(values)] should be defined as multiple message header fields.
912    *
913    * @param header The header.
914    * @return This object.
915    */
916   public RestResponse addHeader(Header header) {
917      if (header == null) {
918         // Do nothing.
919      } else if (header instanceof BasicUriHeader) {
920         BasicUriHeader x = (BasicUriHeader)header;
921         addHeader(x.getName(), resolveUris(x.getValue()));
922      } else if (header instanceof SerializedHeader) {
923         SerializedHeader x = ((SerializedHeader)header).copyWith(request.getPartSerializerSession(), null);
924         addHeader(x.getName(), resolveUris(x.getValue()));
925      } else {
926         addHeader(header.getName(), header.getValue());
927      }
928      return this;
929   }

932      String s = stringify(value);

934   }
