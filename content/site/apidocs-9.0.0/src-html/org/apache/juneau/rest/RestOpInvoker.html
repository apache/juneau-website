013package org.apache.juneau.rest;
014
015import java.lang.reflect.*;
016
017import org.apache.juneau.http.response.*;
018import org.apache.juneau.reflect.*;
019import org.apache.juneau.rest.arg.*;
020import org.apache.juneau.rest.stats.*;
021
022/**
023 * A specialized invoker for methods that are called during a servlet request.
024 *
025 * <ul class='seealso'>
026 *    <li class='link'>{@doc jrs.RestContext}
027 *    <li class='extlink'>{@source}
028 * </ul>
029 */
030public class RestOpInvoker extends MethodInvoker {
031
032   private final RestOpArg[] opArgs;
033
034   /**
035    * Constructor.
036    *
037    * @param m The method being wrapped.
038    * @param opArgs The parameter resolvers.
039    * @param stats The instrumentor.
040    */
041   public RestOpInvoker(Method m, RestOpArg[] opArgs, MethodExecStats stats) {
042      super(m, stats);
043      this.opArgs = opArgs;
044   }
045
046   /**
047    * Invokes this method from the specified {@link RestSession}.
048    *
049    * @param opSession The REST call.
050    * @throws Throwable If an error occurred during either parameter resolution or method invocation.
051    */
052   public void invoke(RestOpSession opSession) throws Throwable {
053      Object[] args = new Object[opArgs.length];
054      for (int i = 0; i < opArgs.length; i++) {
055         ParamInfo pi = inner().getParam(i);
056         try {
057            args[i] = opArgs[i].resolve(opSession);
058         } catch (BasicHttpException e) {
059            throw e;
060         } catch (Exception e) {
061            throw new BadRequest(e, "Could not resolve parameter {0} of type ''{1}'' on method ''{2}''.", i, pi.getParameterType(), getFullName());
062         }
063      }
064      try {
065         RestSession session = opSession.getRestSession();
066         RestRequest req = opSession.getRequest();
067         RestResponse res = opSession.getResponse();
068
069         Object output = super.invoke(session.getResource(), args);
070
071         // Handle manual call to req.setDebug().
072         Boolean debug = req.getAttribute("Debug").as(Boolean.class).orElse(null);
073         if (debug == Boolean.TRUE) {
074            session.debug(true);
075         } else if (debug == Boolean.FALSE) {
076            session.debug(false);
077         }
078
079         if (! inner().hasReturnType(Void.TYPE))
080            if (output != null || ! res.getOutputStreamCalled())
081               res.setContent(output);
082
083      } catch (IllegalAccessException|IllegalArgumentException e) {
084         throw new InternalServerError(e, "Error occurred invoking method ''{0}''.", inner().getFullName());
085      } catch (InvocationTargetException e) {
086         RestResponse res = opSession.getResponse();
087         Throwable e2 = e.getTargetException();
088         res.setStatus(500);  // May be overridden later.
089         res.setContent(opSession.getRestContext().convertThrowable(e2));
090      }
091   }
092}
