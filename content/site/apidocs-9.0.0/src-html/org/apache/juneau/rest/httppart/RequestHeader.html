027/**
028 * Represents a single header on an HTTP request.
029 *
030 * <p>
031 * Typically accessed through the {@link RequestHeaders} class.
032 *
033 * <p>
034 *    Some important methods on this class are:
035 * </p>
036 * <ul class='javatree'>
037 *    <li class='jc'>{@link RequestHeader}
038 *    <ul class='spaced-list'>
039 *       <li>Methods for retrieving simple string values:
040 *       <ul class='javatreec'>
041 *          <li class='jm'>{@link RequestHeader#asString() asString()}
042 *          <li class='jm'>{@link RequestHeader#get() get()}
043 *          <li class='jm'>{@link RequestHeader#isPresent() isPresent()}
044 *          <li class='jm'>{@link RequestHeader#orElse(String) orElse(String)}
045 *       </ul>
046 *       <li>Methods for retrieving as other common types:
047 *       <ul class='javatreec'>
048 *          <li class='jm'>{@link RequestHeader#asBoolean() asBoolean()}
049 *          <li class='jm'>{@link RequestHeader#asBooleanHeader() asBooleanHeader()}
050 *          <li class='jm'>{@link RequestHeader#asCsvArray() asCsvArray()}

063 *          <li class='jm'>{@link RequestHeader#asStringHeader() asStringHeader()}
064 *          <li class='jm'>{@link RequestHeader#asStringRangesHeader() asStringRangeArrayHeader()}
065 *          <li class='jm'>{@link RequestHeader#asUriHeader() asUriHeader()}
066 *       </ul>

068 *       <ul class='javatreec'>
069 *          <li class='jm'>{@link RequestHeader#as(Class) as(Class)}

079 *          <li class='jm'>{@link RequestHeader#assertInteger() assertInteger()}
080 *          <li class='jm'>{@link RequestHeader#assertLong() assertLong()}
081 *          <li class='jm'>{@link RequestHeader#assertString() assertString()}
082 *       </ul>

084 *       <ul class='javatreec'>
085 *          <li class='jm'>{@link RequestHeader#getName() getName()}

090 * <ul class='seealso'>
091 *    <li class='link'>{@doc jrs.HttpParts}
092 *    <li class='extlink'>{@source}
093 * </ul>
094 */
095public class RequestHeader extends RequestHttpPart implements Header {
096
097   /**
098    * Constructor.
099    *
100    * @param request The request object.
101    * @param name The header name.
102    * @param value The header value.
103    */
104   public RequestHeader(RestRequest request, String name, String value) {
105      super(HEADER, request, name, value);
106   }
107
108   /**
109    * Returns the value of this header as a {@link BasicCsvHeader}.
110    *
111    * @return The value of this header as a  {@link BasicCsvHeader}, never <jk>null</jk>.
112    */
113   public BasicCsvHeader asCsvHeader() {
114      return new BasicCsvHeader(getName(), getValue());
115   }
116
117   /**
118    * Returns the value of this header as a {@link BasicDateHeader}.
119    *
120    * @return The value of this header as a {@link BasicDateHeader}, never <jk>null</jk>.
121    */
122   public BasicDateHeader asDateHeader() {
123      return new BasicDateHeader(getName(), getValue());
124   }
125
126   /**
127    * Returns the value of this header as a {@link BasicEntityTagsHeader}.
128    *
129    * @return The value of this header as a {@link BasicEntityTagsHeader}, never <jk>null</jk>.
130    */
131   public BasicEntityTagsHeader asEntityTagsHeader() {
132      return new BasicEntityTagsHeader(getName(), getValue());
133   }
134
135   /**
136    * Returns the value of this header as a {@link BasicEntityTagHeader}.
137    *
138    * @return The value of this header as a {@link BasicEntityTagHeader}, never <jk>null</jk>.
139    */
140   public BasicEntityTagHeader asEntityTagHeader() {
141      return new BasicEntityTagHeader(getName(), getValue());
142   }
143
144   /**
145    * Returns the value of this header as a {@link BasicIntegerHeader}.
146    *
147    * @return The value of this header as a {@link BasicIntegerHeader}, never <jk>null</jk>.
148    */
149   public BasicIntegerHeader asIntegerHeader() {
150      return new BasicIntegerHeader(getName(), getValue());
151   }
152
153   /**
154    * Returns the value of this header as a {@link BasicBooleanHeader}.
155    *
156    * @return The value of this header as a {@link BasicBooleanHeader}, never <jk>null</jk>.
157    */
158   public BasicBooleanHeader asBooleanHeader() {
159      return new BasicBooleanHeader(getName(), getValue());
160   }
161
162   /**
163    * Returns the value of this header as a {@link BasicLongHeader}.
164    *
165    * @return The value of this header as a {@link BasicLongHeader}, never <jk>null</jk>.
166    */
167   public BasicLongHeader asLongHeader() {
168      return new BasicLongHeader(getName(), getValue());
169   }
170
171   /**
172    * Returns the value of this header as a {@link BasicStringRangesHeader}.
173    *
174    * @return The value of this header as a {@link BasicStringRangesHeader}, never <jk>null</jk>.
175    */
176   public BasicStringRangesHeader asStringRangesHeader() {
177      return new BasicStringRangesHeader(getName(), getValue());
178   }
179
180   /**
181    * Returns the value of this header as a {@link BasicStringHeader}.
182    *
183    * @return The value of this header as a {@link BasicStringHeader}, never <jk>null</jk>.
184    */
185   public BasicStringHeader asStringHeader() {
186      return new BasicStringHeader(getName(), getValue());
187   }
188
189   /**
190    * Returns the value of this header as a {@link BasicUriHeader}.
191    *
192    * @return The value of this header as a {@link BasicUriHeader}, never <jk>null</jk>.
193    */
194   public BasicUriHeader asUriHeader() {
195      return new BasicUriHeader(getName(), getValue());
196   }
197
198   /**
199    * Parses the value.
200    *
201    * @return An array of {@link HeaderElement} entries, may be empty, but is never <jk>null</jk>.
202    * @throws BasicHttpException In case of a parsing error.
203    */
204   @Override /* Header */
205   public HeaderElement[] getElements() throws BasicHttpException {
206      return new HeaderElement[0];
207   }
208
209   @Override /* Object */
210   public String toString() {
211      return getName() + ": " + getValue();
212   }
213
214   // <FluentSetters>
215
216   @Override /* GENERATED */
217   public RequestHeader schema(HttpPartSchema value) {
218      super.schema(value);
219      return this;
220   }
221
222   @Override /* GENERATED */
223   public RequestHeader parser(HttpPartParserSession value) {
224      super.parser(value);
225      return this;
226   }
227
228   // </FluentSetters>
229}
