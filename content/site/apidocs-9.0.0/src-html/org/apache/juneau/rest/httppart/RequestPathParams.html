036/**
037 * Represents the path parameters in an HTTP request.
038 *
039 *  <p>
040 *    The {@link RequestPathParams} object is the API for accessing the matched variables
041 *    and remainder on the URL path.
042 * </p>
043 * <p class='bjava'>
044 *    <ja>@RestPost</ja>(...)
045 *    <jk>public</jk> Object myMethod(RequestPathParams <jv>path</jv>) {...}
046 * </p>
047 *
048 * <h5 class='figure'>Example:</h5>
049 * <p class='bjava'>
050 *    <ja>@RestPost</ja>(..., path=<js>"/{foo}/{bar}/{baz}/*"</js>)
051 *    <jk>public void</jk> doGet(RequestPathParams <jv>path</jv>) {
052 *       <jc>// Example URL:  /123/qux/true/quux</jc>
053 *
054 *       <jk>int</jk> <jv>foo</jv> = <jv>path</jv>.get(<js>"foo"</js>).asInteger().orElse(0);  <jc>// =123</jc>
055 *       String <jv>bar</jv> = <jv>path</jv>.get(<js>"bar"</js>).orElse(<jk>null</jk>);  <jc>// =qux</jc>
056 *       <jk>boolean</jk> <jv>baz</jv> = <jv>path</jv>.get(<js>"baz"</js>).asBoolean().orElse(<jk>false</jk>);  <jc>// =true</jc>
057 *       String <jv>remainder</jv> = <jv>path</jv>.getRemainder();  <jc>// =quux</jc>
058 *    }
059 * </p>
060 *
061 * <p>
062 *    Some important methods on this class are:
063 * </p>
064 * <ul class='javatree'>
065 *    <li class='jc'>{@link RequestPathParams}
066 *    <ul class='spaced-list'>
067 *       <li>Methods for retrieving path parameters:
068 *       <ul class='javatreec'>
069 *          <li class='jm'>{@link RequestPathParams#contains(String) contains(String)}
070 *          <li class='jm'>{@link RequestPathParams#containsAny(String...) containsAny(String...)}
071 *          <li class='jm'>{@link RequestPathParams#get(Class) get(Class)}
072 *          <li class='jm'>{@link RequestPathParams#get(String) get(String)}
073 *          <li class='jm'>{@link RequestPathParams#getAll(String) getAll(String)}
074 *          <li class='jm'>{@link RequestPathParams#getFirst(String) getFirst(String)}
075 *          <li class='jm'>{@link RequestPathParams#getLast(String) getLast(String)}
076 *          <li class='jm'>{@link RequestPathParams#getRemainder() getRemainder()}
077 *          <li class='jm'>{@link RequestPathParams#getRemainderUndecoded() getRemainderUndecoded()}
078 *       </ul>
079 *       <li>Methods overridding path parameters:
080 *       <ul class='javatreec'>
081 *          <li class='jm'>{@link RequestPathParams#add(NameValuePair...) add(NameValuePair...)}
082 *          <li class='jm'>{@link RequestPathParams#add(String,Object) add(String,Object)}
083 *          <li class='jm'>{@link RequestPathParams#addDefault(List) addDefault(List)}
084 *          <li class='jm'>{@link RequestPathParams#addDefault(NameValuePair...) addDefault(NameValuePair...)}
085 *          <li class='jm'>{@link RequestPathParams#remove(String) remove(String)}
086 *          <li class='jm'>{@link RequestPathParams#set(NameValuePair...) set(NameValuePair...)}
087 *          <li class='jm'>{@link RequestPathParams#set(String,Object) set(String,Object)}
088 *       </ul>
089 *       <li>Other methods:
090 *       <ul class='javatreec'>
091 *          <li class='jm'>{@link RequestPathParams#copy() copy()}
092 *          <li class='jm'>{@link RequestPathParams#isEmpty() isEmpty()}
093 *       </ul>
094 *    </ul>
095 * </ul>
096 *
097 * <ul class='seealso'>
098 *    <li class='jc'>{@link RequestPathParam}
099 *    <li class='ja'>{@link org.apache.juneau.http.annotation.Path}
100 *    <li class='link'>{@doc jrs.HttpParts}
101 *    <li class='extlink'>{@source}
102 * </ul>
103*/
104public class RequestPathParams extends ArrayList<RequestPathParam> {
105
106   private static final long serialVersionUID = 1L;
107
108   private final RestRequest req;
109   private boolean caseSensitive;
110   private HttpPartParserSession parser;
111   private final VarResolverSession vs;
112
113   /**
114    * Constructor.
115    *
116    * @param session The current HTTP request session.
117    * @param req The current HTTP request.
118    * @param caseSensitive Whether case-sensitive name matching is enabled.
119    */
120   public RequestPathParams(RestSession session, RestRequest req, boolean caseSensitive) {
121      this.req = req;
122      this.caseSensitive = caseSensitive;
123      this.vs = req.getVarResolverSession();
124
125      // Add parameters from parent context if any.
126      @SuppressWarnings("unchecked")
127      Map<String,String> parentVars = (Map<String,String>)req.getAttribute("juneau.pathVars").orElse(Collections.emptyMap());
128      for (Map.Entry<String,String> e : parentVars.entrySet())
129         add(e.getKey(), e.getValue());
130
131      UrlPathMatch pm = session.getUrlPathMatch();
132      if (pm != null) {
133         for (Map.Entry<String,String> e : pm.getVars().entrySet())
134            add(e.getKey(), e.getValue());
135         String r = pm.getRemainder();
136         if (r != null) {
137            add("/**", r);
138            add("/*", urlDecode(r));
139         }
140      }
141   }
142
143   /**
144    * Copy constructor.
145    */
146   private RequestPathParams(RequestPathParams copyFrom) {
147      req = copyFrom.req;
148      caseSensitive = copyFrom.caseSensitive;
149      parser = copyFrom.parser;
150      addAll(copyFrom);
151      vs = copyFrom.vs;
152   }
153
154   /**
155    * Subset constructor.
156    */
157   private RequestPathParams(RequestPathParams copyFrom, String...names) {
158      this.req = copyFrom.req;
159      caseSensitive = copyFrom.caseSensitive;
160      parser = copyFrom.parser;
161      vs = copyFrom.vs;
162      for (String n : names)
163         copyFrom.stream().filter(x -> eq(x.getName(), n)).forEach(x -> add(x));
164   }

166   /**
167    * Sets the parser to use for part values.
168    *
169    * @param value The new value for this setting.
170    * @return This object.

173      this.parser = value;
174      forEach(x -> x.parser(parser));
175      return this;

193   /**
194    * Adds default entries to these parameters.
195    *
196    * <p>
197    * Similar to {@link #set(String, Object)} but doesn't override existing values.
198    *
199    * @param pairs
200    *    The default entries.
201    *    <br>Can be <jk>null</jk>.
202    * @return This object.
203    */
204   public RequestPathParams addDefault(List<NameValuePair> pairs) {
205      for (NameValuePair p : pairs) {
206         String name = p.getName();
207         Stream<RequestPathParam> l = stream(name);
208         boolean hasAllBlanks = l.allMatch(x -> StringUtils.isEmpty(x.getValue()));
209         if (hasAllBlanks) {
210            removeAll(getAll(name));
211            add(new RequestPathParam(req, name, vs.resolve(p.getValue())));
212         }
213      }
214      return this;
215   }

228   public RequestPathParams addDefault(NameValuePair...pairs) {
229      return addDefault(alist(pairs));

243   /**
244    * Adds a parameter value.
245    *
246    * <p>
247    * Parameter is added to the end.
248    * <br>Existing parameter with the same name are not changed.
249    *
250    * @param name The parameter name.  Must not be <jk>null</jk>.
251    * @param value The parameter value.
252    * @return This object.
253    */
254   public RequestPathParams add(String name, Object value) {
255      assertArgNotNull("name", name);
256      add(new RequestPathParam(req, name, stringify(value)).parser(parser));
257      return this;
258   }

260   /**
261    * Adds request parameter values.
262    *
263    * <p>
264    * Parameters are added to the end.
265    * <br>Existing parameters with the same name are not changed.
266    *
267    * @param parameters The parameter objects.  Must not be <jk>null</jk>.
268    * @return This object.
269    */
270   public RequestPathParams add(NameValuePair...parameters) {
271      assertArgNotNull("parameters", parameters);
272      for (NameValuePair p : parameters)
273         if (p != null)
274            add(p.getName(), p.getValue());
275      return this;
276   }

278   /**
279    * Sets a parameter value.
280    *
281    * <p>
282    * Parameter is added to the end.
283    * <br>Any previous parameters with the same name are removed.
284    *
285    * @param name The parameter name.  Must not be <jk>null</jk>.
286    * @param value
287    *    The parameter value.
288    *    <br>Converted to a string using {@link Object#toString()}.
289    *    <br>Can be <jk>null</jk>.
290    * @return This object.

292   public RequestPathParams set(String name, Object value) {
293      assertArgNotNull("name", name);
294      set(new RequestPathParam(req, name, stringify(value)).parser(parser));
295      return this;

299   /**
300    * Sets request header values.
301    *
302    * <p>
303    * Parameters are added to the end of the headers.
304    * <br>Any previous parameters with the same name are removed.
305    *
306    * @param parameters The parameters to set.  Must not be <jk>null</jk> or contain <jk>null</jk>.
307    * @return This object.
308    */
309   public RequestPathParams set(NameValuePair...parameters) {
310      assertArgNotNull("headers", parameters);
311      for (NameValuePair p : parameters)
312         remove(p);
313      for (NameValuePair p : parameters)
314         add(p);
315      return this;
316   }

369   /**
370    * Returns all the parameters with the specified name.

372    * @param name The parameter name.

376      assertArgNotNull("name", name);

378   }

380   /**
381    * Returns all headers with the specified name.
382    *
383    * @param name The header name.
384    * @return The stream of all headers with matching names.  Never <jk>null</jk>.
385    */
386   public Stream<RequestPathParam> stream(String name) {
387      return stream().filter(x -> eq(x.getName(), name));
388   }

390   /**
391    * Returns all headers in sorted order.
392    *
393    * @return The stream of all headers in sorted order.
394    */
395   public Stream<RequestPathParam> getSorted() {
396      Comparator<RequestPathParam> x;
397      if (caseSensitive)
398         x = (x1,x2) -> x1.getName().compareTo(x2.getName());
399      else
400         x = (x1,x2) -> String.CASE_INSENSITIVE_ORDER.compare(x1.getName(), x2.getName());
401      return stream().sorted(x);
402   }

404   /**
405    * Returns all the unique header names in this list.
406    * @return The list of all unique header names in this list.
407    */
408   public List<String> getNames() {
409      return stream().map(x -> x.getName()).map(x -> caseSensitive ? x : x.toLowerCase()).distinct().collect(toList());
410   }

412   /**
413    * Returns the first parameter with the specified name.
414    *
415    * <p>
416    * Note that this method never returns <jk>null</jk> and that {@link RequestPathParam#isPresent()} can be used
417    * to test for the existence of the parameter.
418    *
419    * @param name The parameter name.
420    * @return The parameter.  Never <jk>null</jk>.
421    */
422   public RequestPathParam getFirst(String name) {
423      assertArgNotNull("name", name);
424      return stream(name).findFirst().orElseGet(()->new RequestPathParam(req, name, null).parser(parser));
425   }

427   /**
428    * Returns the last parameter with the specified name.
429    *
430    * <p>
431    * Note that this method never returns <jk>null</jk> and that {@link RequestPathParam#isPresent()} can be used
432    * to test for the existence of the parameter.
433    *
434    * @param name The parameter name.
435    * @return The parameter.  Never <jk>null</jk>.
436    */
437   public RequestPathParam getLast(String name) {
438      assertArgNotNull("name", name);
439      Value<RequestPathParam> v = Value.empty();
440      stream(name).forEach(x -> v.set(x));
441      return v.orElseGet(() -> new RequestPathParam(req, name, null).parser(parser));
442   }
443
444   /**
445    * Returns the last parameter with the specified name.
446    *
447    * <p>
448    * This is equivalent to {@link #getLast(String)}.
449    *
450    * @param name The parameter name.
451    * @return The parameter value, or {@link Optional#empty()} if it doesn't exist.
452    */
453   public RequestPathParam get(String name) {
454      List<RequestPathParam> l = getAll(name);
455      if (l.isEmpty())
456         return new RequestPathParam(req, name, null).parser(parser);
457      if (l.size() == 1)
458         return l.get(0);
459      StringBuilder sb = new StringBuilder(128);
460      for (int i = 0, j = l.size(); i < j; i++) {
461         if (i > 0)
462            sb.append(", ");
463         sb.append(l.get(i).getValue());
464      }
465      return new RequestPathParam(req, name, sb.toString()).parser(parser);
466   }

468   /**
469    * Returns the path parameter as the specified bean type.
470    *
471    * <p>
472    * Type must have a name specified via the {@link org.apache.juneau.http.annotation.Path} annotation
473    * and a public constructor that takes in either <c>value</c> or <c>name,value</c> as strings.
474    *
475    * @param <T> The bean type to create.
476    * @param type The bean type to create.
477    * @return The bean, never <jk>null</jk>.
478    */
479   public <T> Optional<T> get(Class<T> type) {
480      ClassMeta<T> cm = req.getBeanSession().getClassMeta(type);
481      String name = HttpParts.getName(PATH, cm).orElseThrow(()->new BasicRuntimeException("@Path(name) not found on class {0}", className(type)));
482      return get(name).as(type);
483   }

498   /**
499    * Returns the decoded remainder of the URL following any path pattern matches.
500    *
501    * <p>
502    * The behavior of path remainder is shown below given the path pattern "/foo/*":
503    * <table class='styled'>
504    *    <tr>
505    *       <th>URL</th>
506    *       <th>Path Remainder</th>
507    *    </tr>
508    *    <tr>
509    *       <td><c>/foo</c></td>
510    *       <td><jk>null</jk></td>
511    *    </tr>
512    *    <tr>
513    *       <td><c>/foo/</c></td>
514    *       <td><js>""</js></td>
515    *    </tr>
516    *    <tr>
517    *       <td><c>/foo//</c></td>
518    *       <td><js>"/"</js></td>
519    *    </tr>
520    *    <tr>
521    *       <td><c>/foo///</c></td>
522    *       <td><js>"//"</js></td>
523    *    </tr>
524    *    <tr>
525    *       <td><c>/foo/a/b</c></td>
526    *       <td><js>"a/b"</js></td>
527    *    </tr>
528    *    <tr>
529    *       <td><c>/foo//a/b/</c></td>
530    *       <td><js>"/a/b/"</js></td>
531    *    </tr>
532    *    <tr>
533    *       <td><c>/foo/a%2Fb</c></td>
534    *       <td><js>"a/b"</js></td>
535    *    </tr>
536    * </table>
537    *
538    * <h5 class='section'>Example:</h5>
539    * <p class='bjava'>
540    *    <jc>// REST method</jc>
541    *    <ja>@RestGet</ja>(<js>"/foo/{bar}/*"</js>)
542    *    <jk>public</jk> String doGetById(RequestPathParams <jv>path</jv>, <jk>int</jk> <jv>bar</jv>) {
543    *       <jk>return</jk> <jv>path</jv>.remainder().orElse(<jk>null</jk>);
544    *    }
545    * </p>
546    *
547    * <p>
548    * The remainder can also be retrieved by calling <code>get(<js>"/**"</js>)</code>.
549    *
550    * @return The path remainder string.
551    */
552   public RequestPathParam getRemainder() {
553      return get("/*");
554
555   }

562    *
563    * @return The un-decoded path remainder.
564    */
565   public RequestPathParam getRemainderUndecoded() {
566      return get("/**");

569   private boolean eq(String s1, String s2) {
570      if (caseSensitive)
571         return StringUtils.eq(s1, s2);
572      return StringUtils.eqic(s1, s2);
573   }

575   @Override /* Object */
576   public String toString() {
577      JsonMap m = new JsonMap();
578      for (String n : getNames())
579         m.put(n, get(n).asString().orElse(null));
580      return m.asJson();
581   }
