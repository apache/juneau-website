034 *
035 * <p>
036 *    Some important methods on this class are:
037 * </p>
038 * <ul class='javatree'>

040 *    <ul class='spaced-list'>

091 *    <li class='extlink'>{@source}
092 * </ul>
093 */
094public class RequestFormParam extends RequestHttpPart implements NameValuePair {
095
096   private final javax.servlet.http.Part part;
097
098   /**
099    * Constructor.
100    *
101    * @param request The request object.
102    * @param part The HTTP part.
103    */
104   public RequestFormParam(RestRequest request, javax.servlet.http.Part part) {
105      super(FORMDATA, request, part.getName(), null);
106      this.part = part;
107   }
108
109   /**
110    * Constructor.
111    *
112    * @param request The request object.
113    * @param name The parameter name.
114    * @param value The parameter value.
115    */
116   public RequestFormParam(RestRequest request, String name, String value) {
117      super(FORMDATA, request, name, value);
118      this.part = null;
119   }
120
121   //------------------------------------------------------------------------------------------------------------------
122   // Retrievers
123   //------------------------------------------------------------------------------------------------------------------
124
125   @Override /* RequestHttpPart */
126   public String getValue() {
127      if (value == null && part != null)
128         try {
129            value = IOUtils.read(part.getInputStream());
130         } catch (IOException e) {
131            throw asRuntimeException(e);
132         }
133      return value;
134   }
135
136   /**
137    * Returns this part value as an input stream.
138    *
139    * @return This part value as an input stream.
140    * @throws IOException If an error occurs in retrieving the content.
141    */
142   public InputStream getStream() throws IOException {
143      if (value != null)
144         return new ByteArrayInputStream(value.getBytes(IOUtils.UTF8));
145      return part.getInputStream();
146   }
147
148   /**
149    * Returns the content type of this part.
150    *
151    * @return The content type of this part, or <jk>null</jk> if not known.
152    */
153   public String getContentType() {
154      return (part == null ? null : part.getContentType());
155   }
156
157   /**
158    * Returns the value of the specified mime header as a String.
159    *
160    * <p>
161    * If the Part did not include a header of the specified name, this method returns null.
162    * If there are multiple headers with the same name, this method returns the first header in the part.
163    * The header name is case insensitive.
164    * You can use this method with any request header.
165    *
166    * @param name The header name.
167    * @return The value of the specified mime header as a String.
168    */
169   public String getHeader(String name) {
170      return part.getHeader(name);
171   }
172
173   /**
174    * Returns the header names of this param.
175    *
176    * @return The header names of this param.
177    */
178   public Collection<String> getHeaderNames() {
179      return part.getHeaderNames();
180   }
181
182   /**
183    * Returns the values of the param header with the given name.
184    *
185    * @param name The param name.
186    * @return The values of the param header with the given name.
187    */
188   public Collection<String> getHeaders(String name) {
189      return part.getHeaders(name);
190   }
191
192   /**
193    * Returns the size of this file.
194    *
195    * @return A long specifying the size of this part, in bytes.
196    */
197   public long getSize() {
198      return part.getSize();
199   }
200
201   /**
202    * Returns the file name specified by the client.
203    *
204    * @return The file name specified by the client.
205    */
206   public String getSubmittedFileName() {
207      return part.getSubmittedFileName();
208   }
209
210   // <FluentSetters>
211
212   @Override /* GENERATED */
213   public RequestFormParam schema(HttpPartSchema value) {
214      super.schema(value);
215      return this;
216   }
217
218   @Override /* GENERATED */
219   public RequestFormParam parser(HttpPartParserSession value) {
220      super.parser(value);
221      return this;
222   }
223
224   // </FluentSetters>
225}
