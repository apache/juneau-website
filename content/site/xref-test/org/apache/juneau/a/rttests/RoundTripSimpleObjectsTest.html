650 	// testLinkedHashMap
651 	//====================================================================================================
652 	@Test
653 	public void testLinkedHashMap() throws Exception {
654 		LinkedHashMap t = new LinkedHashMap();
655 		t.put("a", true);
656 		t.put("b", false);
657 		t.put("c", null);
658 		t.put("d", "foo");
659 		t.put(null, "bar");
660 		t.put("null", "null");
661 		t.put("true", "true");
662 		t.put("false", "false");
663 		t.put("a\"a", "a\"a");
664 		t.put("b'b", "b'b");
665 		t.put("\"cc\"", "\"cc\"");
666 		t.put("'dd'", "'dd'");
667 		t.put("<ee>", "<ee>");
668 		t.put("<ff/>", "<ff/>");
669 		t.put("</gg>", "</gg>");
670 		t.put("<>", "<>");
671 		t.put("{}", "{}");
672 		t.put("[]", "[]");
673 		t.put("&", "&");
674 		t.put("?", "?");
675 		t.put("/", "/");
676 		t.put("\b", "\b");
677 		t.put("\\b", "\\b");
678 		t.put("\n", "\n");
679 		t.put("\\n", "\\n");
680 		t.put("\t", "\t");
681 		t.put("\\t", "\\t");
682 		t.put("\f", "\f");
683 		t.put("\\f", "\\f");
684 		t.put("\\", "\\");
685 		t.put("\\\\", "\\\\");
686 		t.put("\u2345", "\u2345");
687 		t.put("\\u2345", "\\u2345");
688 		t.put("\\\u2345", "\\\u2345");
689 		t.put("<>{}[]&?/\b\n\t\f\\\\\u2345", "<>{}[]&?/\b\n\t\f\\\\\u2345");
690 		t = roundTrip(t, LinkedHashMap.class);
691 		assertEquals(true, t.get("a"));
692 		assertEquals(false, t.get("b"));
693 		assertNull(t.get("c"));
