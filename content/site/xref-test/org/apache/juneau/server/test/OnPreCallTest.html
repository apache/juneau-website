13  package org.apache.juneau.server.test;
14  
15  import static org.junit.Assert.*;
16  
17  import java.io.*;
18  
19  import org.apache.juneau.client.*;
20  import org.junit.*;
21  
22  public class OnPreCallTest {
23  
24  	private static String URL = "/testOnPreCall";
25  
26  	//====================================================================================================
27  	// Properties overridden via properties annotation.
28  	//====================================================================================================
29  	@Test
30  	public void testPropertiesOverriddenByAnnotation() throws Exception {
31  		RestClient client = new TestRestClient().setContentType("text/a1").setAccept("text/plain");
32  		String url = URL + "/testPropertiesOverriddenByAnnotation";
33  		String r;
34  
35  		r = client.doPut(url, new StringReader("")).getResponseAsString();
36  		assertEquals("p1=sp1,p2=xp2,p3=mp3,p4=xp4,p5=xp5,contentType=text/a1", r);
37  
38  		r = client.doPut(url, new StringReader("")).setHeader("Override-Content-Type", "text/a2").getResponseAsString();
39  		assertEquals("p1=sp1,p2=xp2,p3=mp3,p4=xp4,p5=xp5,contentType=text/a2", r);
40  
41  		client.closeQuietly();
42  	}
43  
44  	//====================================================================================================
45  	// Properties overridden programmatically.
46  	//====================================================================================================
47  	@Test
48  	public void testPropertiesOverriddenProgrammatically() throws Exception {
49  		RestClient client = new TestRestClient().setContentType("text/a1").setAccept("text/plain");
50  		String url = URL + "/testPropertiesOverriddenProgrammatically";
51  		String r;
52  
53  		r = client.doPut(url, new StringReader("")).getResponseAsString();
54  		assertEquals("p1=sp1,p2=xp2,p3=pp3,p4=pp4,p5=xp5,contentType=text/a1", r);
55  
56  		r = client.doPut(url, new StringReader("")).setHeader("Override-Content-Type", "text/a2").getResponseAsString();
57  		assertEquals("p1=sp1,p2=xp2,p3=pp3,p4=pp4,p5=xp5,contentType=text/a2", r);
58  
59  		client.closeQuietly();
60  	}
61  }
