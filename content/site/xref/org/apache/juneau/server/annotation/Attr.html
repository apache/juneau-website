13  package org.apache.juneau.server.annotation;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  
18  import java.lang.annotation.*;
19  
20  /**
21   * Annotation that can be applied to a parameter of a {@link RestMethod} annotated method
22   * 	to identify it as a variable in a URL path pattern converted to a POJO.
23   *
24   * <h6 class='topic'>Example</h6>
25   * <p class='bcode'>
26   * 	<ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/myurl/{foo}/{bar}/{baz}/*"</js>)
27   * 	<jk>public void</jk> doGet(RestRequest req, RestResponse res,
28   * 			<ja>@Attr</ja> String foo, <ja>@Attr</ja> <jk>int</jk> bar, <ja>@Attr</ja> UUID baz) {
29   * 		...
30   * 	}
31   * </p>
32   * <p>
33   * 	The <ja>@Attr</ja> annotation is optional if the parameters are specified immediately
34   * 	following the <code>RestRequest</code> and <code>RestResponse</code> parameters,
35   * 	and are specified in the same order as the variables in the URL path pattern.
36   * 	The following example is equivalent to the previous example.
37   * </p>
38   * <p class='bcode'>
39   * 	<ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/myurl/{foo}/{bar}/{baz}/*"</js>)
40   * 	<jk>public void</jk> doGet(RestRequest req, RestResponse res,
41   * 			String foo, <jk>int</jk> bar, UUID baz) {
42   * 		...
43   * 	}
44   * </p>
45   * <p>
46   * 	If the order of parameters is not the default order shown above, the
47   * 	attribute names must be specified (since parameter names are lost during compilation).
48   * 	The following example is equivalent to the previous example, except
49   * 	the parameter order has been switched, requiring the use of the <ja>@Attr</ja>
50   * 	annotations.
51   * <p>
52   * <p class='bcode'>
53   * 	<ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/myurl/{foo}/{bar}/{baz}/*"</js>)
54   * 	<jk>public void</jk> doGet(RestRequest req, RestResponse res,
55   * 			<ja>@Attr</ja>(<js>"baz"</js>) UUID baz, <ja>@Attr</ja>(<js>"foo"</js>) String foo, <ja>@Attr</ja>(<js>"bar"</js>) <jk>int</jk> bar) {
56   * 		...
57   * 	}
58   * </p>
59   *
60   * @author James Bognar (james.bognar@salesforce.com)
61   */
62  @Documented
63  @Target(PARAMETER)
64  @Retention(RUNTIME)
65  @Inherited
66  public @interface Attr {
67  
68  	/**
69  	 * URL variable name.
70  	 * <p>
71  	 * 	Optional if the attributes are specified in the same order as in the URL path pattern.
72  	 */
73  	String value() default "";
74  }
