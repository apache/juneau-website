18  import java.awt.image.*;
19  import java.io.*;
20  import java.net.*;
21  import java.net.URI;
22  import java.util.*;
23  
24  import javax.imageio.*;
25  
26  import org.apache.juneau.*;
27  import org.apache.juneau.annotation.*;
28  import org.apache.juneau.microservice.*;

34  /**
35   * Sample resource that allows images to be uploaded and retrieved.
36   */
37  @RestResource(
38  	path="/photos",
39  	messages="nls/PhotosResource",
40  	properties={
41  		@Property(name=HTMLDOC_title, value="Photo REST service"),
42  		@Property(name=HTMLDOC_description, value="Use a tool like Poster to upload and retrieve jpeg and png images."),
43  		@Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.PhotosResource)'}"),
44  		// Resolve all relative URIs so that they're relative to this servlet!
45  		@Property(name=SERIALIZER_relativeUriBase, value="$R{servletURI}"),
46  	}
47  )
48  public class PhotosResource extends Resource {
49  	private static final long serialVersionUID = 1L;
50  
51  	// Our cache of photos
52  	private Map<Integer,Photo> photos = new TreeMap<Integer,Photo>();
53  
54  	@Override /* Servlet */
55  	public void init() {
56  		try {
57  			// Preload an image.
58  			InputStream is = getClass().getResourceAsStream("averycutedog.jpg");
59  			BufferedImage image = ImageIO.read(is);
60  			Photo photo = new Photo(0, image);
61  			photos.put(photo.id, photo);
62  		} catch (IOException e) {
63  			throw new RuntimeException(e);
64  		}
65  	}
66  
67  	/** Our bean class for storing photos */
68  	public static class Photo {
69  		int id;
70  		BufferedImage image;
71  
72  		Photo(int id, BufferedImage image) {
73  			this.id = id;
74  			this.image = image;
75  		}
76  
77  		public URI getURI() throws URISyntaxException {
78  			return new URI(""+id);
79  		}
80  	}
81  
82  	/** GET request handler for list of all photos */
83  	@RestMethod(name="GET", path="/")
84  	public Collection<Photo> getAllPhotos() throws Exception {
85  		return photos.values();
86  	}
87  
88  	/** GET request handler for single photo */
89  	@RestMethod(name="GET", path="/{id}", serializers=ImageSerializer.class)
90  	public BufferedImage getPhoto(@Attr int id) throws Exception {
91  		Photo p = photos.get(id);
92  		if (p == null)
93  			throw new RestException(SC_NOT_FOUND, "Photo not found");
94  		return p.image;
95  	}
96  
97  	/** PUT request handler */
98  	@RestMethod(name="PUT", path="/{id}", parsers=ImageParser.class)
99  	public String addPhoto(@Attr int id, @Content BufferedImage image) throws Exception {
100 		photos.put(id, new Photo(id, image));
101 		return "OK";
102 	}
103 
104 	/** POST request handler */
105 	@RestMethod(name="POST", path="/", parsers=ImageParser.class)
106 	public Photo setPhoto(@Content BufferedImage image) throws Exception {
107 		int id = photos.size();
108 		Photo p = new Photo(id, image);
109 		photos.put(id, p);
110 		return p;
111 	}
112 
113 	/** DELETE request handler */
114 	@RestMethod(name="DELETE", path="/{id}")
115 	public String deletePhoto(@Attr int id) throws Exception {
116 		Photo p = photos.remove(id);
117 		if (p == null)
118 			throw new RestException(SC_NOT_FOUND, "Photo not found");
119 		return "OK";
120 	}
121 
122 	/** Serializer for converting images to byte streams */
123 	@Produces({"image/png","image/jpeg"})
124 	public static class ImageSerializer extends OutputStreamSerializer {
125 		@Override /* Serializer */
126 		protected void doSerialize(SerializerSession session, Object o) throws Exception {
127 			RenderedImage image = (RenderedImage)o;
128 			String mediaType = session.getProperties().getString("mediaType");
129 			ImageIO.write(image, mediaType.substring(mediaType.indexOf('/')+1), session.getOutputStream());
130 		}
131 	}
132 
133 	/** Parser for converting byte streams to images */
134 	@Consumes({"image/png","image/jpeg"})
135 	public static class ImageParser extends InputStreamParser {
136 		@Override /* Parser */
137 		@SuppressWarnings("unchecked")
138 		protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
139 			return (T)ImageIO.read(session.getInputStream());
140 		}
141 	}
142 }
