15  import static javax.servlet.http.HttpServletResponse.*;
16  
17  import java.io.*;
18  import java.lang.annotation.*;
19  import java.lang.reflect.*;
20  
21  import javax.ws.rs.*;
22  import javax.ws.rs.core.*;
23  import javax.ws.rs.ext.*;
24  
25  import org.apache.juneau.*;
26  import org.apache.juneau.internal.*;
27  import org.apache.juneau.parser.*;
28  import org.apache.juneau.serializer.*;
29  import org.apache.juneau.server.annotation.*;

31  /**
32   * Base class for defining JAX-RS providers based on Juneau serializers and parsers.
33   *
34   * @author James Bognar (james.bognar@salesforce.com)
35   */
36  public class BaseProvider implements MessageBodyReader<Object>, MessageBodyWriter<Object> {
37  
38  	private SerializerGroup serializers = new SerializerGroup();
39  	private ParserGroup parsers = new ParserGroup();
40  	private ObjectMap properties = new ObjectMap();
41  
42  	/**
43  	 * Constructor.
44  	 */
45  	protected BaseProvider() {
46  		try {
47  			properties = new ObjectMap();
48  			JuneauProvider jp = getClass().getAnnotation(JuneauProvider.class);
49  			serializers.append(jp.serializers());
50  			parsers.append(jp.parsers());
51  			for (Property p : jp.properties())
52  				properties.put(p.name(), p.value());
53  			serializers.addBeanFilters(jp.beanFilters());
54  			parsers.addBeanFilters(jp.beanFilters());
55  			serializers.addPojoSwaps(jp.pojoSwaps());
56  			parsers.addPojoSwaps(jp.pojoSwaps());
57  		} catch (Exception e) {
58  			throw new RuntimeException(e);
59  		}
60  	}
61  
62  	/**
63  	 * Returns properties defined on the specified method through the {@link RestMethod#properties()}
64  	 * 	annotation specified on the method and the {@link JuneauProvider#properties()} annotation
65  	 * 	specified on the provider class.
66  	 *
67  	 * @param a All annotations defined on the method.
68  	 * @return A map of all properties define on the method.
69  	 */
70  	protected ObjectMap getMethodProperties(Annotation[] a) {
71  		ObjectMap m = new ObjectMap().setInner(properties);
72  		for (Annotation aa : a) {
73  			if (aa instanceof RestMethod) {
74  				for (Property p : ((RestMethod)aa).properties())
75  					m.put(p.name(), p.value());
76  			}
77  		}
78  		return m;
79  	}
80  
81  	@Override /* MessageBodyWriter */
82  	public long getSize(Object o, Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
83  		return -1;
84  	}
85  
86  	@Override /* MessageBodyWriter */
87  	public boolean isWriteable(Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
88  		return serializers.findMatch(mediaType.toString()) != null;
89  	}
90  
91  	@Override /* MessageBodyWriter */
92  	public void writeTo(Object o, Class<?> type, Type gType, Annotation[] a, MediaType mediaType,
93  			MultivaluedMap<String,Object> headers, OutputStream out) throws IOException, WebApplicationException {
94  		try {
95  			String mt = serializers.findMatch(mediaType.toString());
96  			if (mt == null)
97  				throw new WebApplicationException(SC_NOT_ACCEPTABLE);
98  			Serializer s = serializers.getSerializer(mt);
99  			ObjectMap mp = getMethodProperties(a);
100 			mp.append("mediaType", mediaType.toString());
101 			if (s.isWriterSerializer()) {
102 				WriterSerializer s2 = (WriterSerializer)s;
103 				OutputStreamWriter w = new OutputStreamWriter(out, IOUtils.UTF8);
104 				SerializerSession session = s.createSession(w, mp, null);
105 				s2.serialize(session, o);
106 				w.flush();
107 				w.close();
108 			} else {
109 				OutputStreamSerializer s2 = (OutputStreamSerializer)s;
110 				SerializerSession session = s.createSession(s2, mp, null);
111 				s2.serialize(session, o);
112 				out.flush();
113 				out.close();
114 			}
115 		} catch (SerializeException e) {
116 			throw new IOException(e);
117 		}
118 	}
119 
120 	@Override /* MessageBodyReader */
121 	public boolean isReadable(Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
122 		return parsers.findMatch(mediaType.toString()) != null;
123 	}
124 
125 	@Override /* MessageBodyReader */
126 	public Object readFrom(Class<Object> type, Type gType, Annotation[] a, MediaType mediaType,
127 			MultivaluedMap<String,String> headers, InputStream in) throws IOException, WebApplicationException {
128 		try {
129 			String mt = parsers.findMatch(mediaType.toString());
130 			if (mt == null)
131 				throw new WebApplicationException(SC_UNSUPPORTED_MEDIA_TYPE);
132 			Parser p = parsers.getParser(mt);
133 			BeanContext bc = p.getBeanContext();
134 			ClassMeta<?> cm = bc.getClassMeta(gType);
135 			ObjectMap mp = getMethodProperties(a);
136 			mp.put("mediaType", mediaType.toString());
137 			if (p.isReaderParser()) {
138 				ReaderParser p2 = (ReaderParser)p;
139 				InputStreamReader r = new InputStreamReader(in, IOUtils.UTF8);
140 				ParserSession session = p2.createSession(r, mp, null, null);
141 				return p2.parse(session, cm);
142 			}
143 			InputStreamParser p2 = (InputStreamParser)p;
144 			ParserSession session = p2.createSession(in, mp, null, null);
145 			return p2.parse(session, cm);
146 		} catch (ParseException e) {
147 			throw new IOException(e);
148 		}
149 	}
150 }
