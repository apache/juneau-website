13  package org.apache.juneau.server.response;
14  
15  import java.io.*;
16  
17  import org.apache.juneau.internal.*;
18  import org.apache.juneau.server.*;
19  
20  /**
21   * Response handler for {@link Redirect} objects.
22   * <p>
23   * This handler is registered by default on {@link RestServlet RestServlets} via the
24   * 	default implementation of the {@link RestServlet#createResponseHandlers} method.
25   *
26   * @author James Bognar (james.bognar@salesforce.com)
27   */
28  public final class RedirectHandler implements ResponseHandler {
29  
30  	@Override /* ResponseHandler */
31  	public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
32  		if (output instanceof Redirect) {
33  			Redirect r = (Redirect)output;
34  			String uri = r.toUrl(res.getUrlEncodingSerializer());
35  			if (StringUtils.isEmpty(uri))
36  				uri = req.getServletURI();
37  			else {
38  				char c = (uri.length() > 0 ? uri.charAt(0) : 0);
39  				if (c != '/' && uri.indexOf("://") == -1)
40  					uri = req.getServletURIBuilder().append('/').append(uri).toString();
41  			}
42  			int rc = r.getHttpResponseCode();
43  			if (rc != 0)
44  				res.setStatus(rc);   // TODO - This may get ignored by the call below.
45  			res.sendRedirect(uri);
46  			return true;
47  		}
48  		return false;
49  	}
50  }
