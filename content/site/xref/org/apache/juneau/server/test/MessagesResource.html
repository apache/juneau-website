13  package org.apache.juneau.server.test;
14  
15  import java.util.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.serializer.*;
19  import org.apache.juneau.server.*;
20  import org.apache.juneau.server.annotation.*;
21  import org.apache.juneau.transform.*;
22  
23  /**
24   * JUnit automated testcase resource.
25   * Validates that resource bundles can be defined on both parent and child classes.
26   */
27  @RestResource(
28  	path="/testMessages",
29  	messages="MessagesResource",
30  	pojoSwaps={
31  		MessagesResource.ResourceBundleSwap.class
32  	}
33  )
34  public class MessagesResource extends RestServletDefault {
35  	private static final long serialVersionUID = 1L;
36  
37  	//====================================================================================================
38  	// Return contents of resource bundle.
39  	//====================================================================================================
40  	@RestMethod(name="GET", path="/test")
41  	public Object test(@Messages ResourceBundle nls) {
42  		return nls;
43  	}
44  
45  
46  	@SuppressWarnings("serial")
47  	@RestResource(
48  		path="/testMessages2",
49  		messages="Messages2Resource"
50  	)
51  	public static class Messages2Resource extends MessagesResource {}
52  
53  	public static class ResourceBundleSwap extends PojoSwap<ResourceBundle,ObjectMap> {
54  		@Override /* Transform */
55  		public ObjectMap swap(ResourceBundle o) throws SerializeException {
56  			ObjectMap m = new ObjectMap();
57  			for (String k : o.keySet())
58  				m.put(k, o.getString(k));
59  			return m;
60  		}
61  	}
62  }
