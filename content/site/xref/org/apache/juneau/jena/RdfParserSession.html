28  /**
29   * Session object that lives for the duration of a single use of {@link RdfParser}.
30   * <p>
31   * This class is NOT thread safe.  It is meant to be discarded after one-time use.
32   *
33   * @author James Bognar (james.bognar@salesforce.com)
34   */
35  public class RdfParserSession extends ParserSession {
36  
37  	private final String rdfLanguage;
38  	private final Namespace juneauNs, juneauBpNs;
39  	private final Property pRoot, pValue, pClass, pType;
40  	private final Model model;
41  	private final boolean trimWhitespace, looseCollections;
42  	private final RDFReader rdfReader;
43  	private final Set<Resource> urisVisited = new HashSet<Resource>();
44  	private final RdfCollectionFormat collectionFormat;
45  
46  	/**
47  	 * Create a new session using properties specified in the context.
48  	 *
49  	 * @param ctx The context creating this session object.
50  	 * 	The context contains all the configuration settings for this object.
51  	 * @param beanContext The bean context being used.
52  	 * @param input The input.  Can be any of the following types:
53  	 * 	<ul>
54  	 * 		<li><jk>null</jk>
55  	 * 		<li>{@link Reader}
56  	 * 		<li>{@link CharSequence}
57  	 * 		<li>{@link InputStream} containing UTF-8 encoded text.
58  	 * 		<li>{@link File} containing system encoded text.
59  	 * 	</ul>
60  	 * @param op The override properties.
61  	 * 	These override any context properties defined in the context.
62  	 * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
63  	 * @param outer The outer object for instantiating top-level non-static inner classes.
64  	 */
65  	protected RdfParserSession(RdfParserContext ctx, BeanContext beanContext, Object input, ObjectMap op, Method javaMethod, Object outer) {
66  		super(ctx, beanContext, input, op, javaMethod, outer);
67  		ObjectMap jenaSettings = new ObjectMap();
68  		jenaSettings.putAll(ctx.jenaSettings);
69  		if (op == null || op.isEmpty()) {
70  			this.rdfLanguage = ctx.rdfLanguage;
71  			this.juneauNs = ctx.juneauNs;
72  			this.juneauBpNs = ctx.juneauBpNs;
73  			this.trimWhitespace = ctx.trimWhitespace;
74  			this.collectionFormat = ctx.collectionFormat;
75  			this.looseCollections = ctx.looseCollections;
76  		} else {
77  			this.rdfLanguage = op.getString(RDF_language, ctx.rdfLanguage);
78  			this.juneauNs = (op.containsKey(RDF_juneauNs) ? NamespaceFactory.parseNamespace(op.get(RDF_juneauNs)) : ctx.juneauNs);
79  			this.juneauBpNs = (op.containsKey(RDF_juneauBpNs) ? NamespaceFactory.parseNamespace(op.get(RDF_juneauBpNs)) : ctx.juneauBpNs);
80  			this.trimWhitespace = op.getBoolean(RdfParserContext.RDF_trimWhitespace, ctx.trimWhitespace);
81  			this.collectionFormat = RdfCollectionFormat.valueOf(op.getString(RDF_collectionFormat, "DEFAULT"));
82  			this.looseCollections = op.getBoolean(RDF_looseCollections, ctx.looseCollections);
83  		}
84  		this.model = ModelFactory.createDefaultModel();
85  		addModelPrefix(juneauNs);
86  		addModelPrefix(juneauBpNs);
87  		this.pRoot = model.createProperty(juneauNs.getUri(), RDF_juneauNs_ROOT);
88  		this.pValue = model.createProperty(juneauNs.getUri(), RDF_juneauNs_VALUE);
89  		this.pClass = model.createProperty(juneauNs.getUri(), RDF_juneauNs_CLASS);
90  		this.pType = model.createProperty("http://www.w3.org/1999/02/22-rdf-syntax-ns#type");
91  		rdfReader = model.getReader(rdfLanguage);
92  
93  		// Note: NTripleReader throws an exception if you try to set any properties on it.
94  		if (! rdfLanguage.equals(LANG_NTRIPLE)) {
95  			for (Map.Entry<String,Object> e : jenaSettings.entrySet())
96  				rdfReader.setProperty(e.getKey(), e.getValue());
97  		}
98  	}

100 	/**
101 	 * Returns <jk>true</jk> if this resource was already visited.
102 	 *
103 	 * @param r The resource to check.
104 	 * @return <jk>true</jk> if this resource was already visited.
105 	 */
106 	public final boolean wasAlreadyProcessed(Resource r) {
107 		return ! urisVisited.add(r);
108 	}

210 	/**
211 	 * Decodes the specified string.
212 	 * <p>
213 	 * If {@link RdfParserContext#RDF_trimWhitespace} is <jk>true</jk>, the resulting string is trimmed before decoding.
214 	 * <p>
215 	 * If {@link #isTrimStrings()} is <jk>true</jk>, the resulting string is trimmed after decoding.
216 	 *
217 	 * @param o The string to trim.
218 	 * @return The trimmed string, or <jk>null</jk> if the string was <jk>null</jk>.
219 	 */
220 	public final String decodeString(Object o) {
221 		if (o == null)
222 			return null;
223 		String s = o.toString();
224 		if (s.isEmpty())
225 			return s;
226 		if (trimWhitespace)
227 			s = s.trim();
228 		s = XmlUtils.decode(s);
229 		if (isTrimStrings())
230 			s = s.trim();
231 		return s;
232 	}
