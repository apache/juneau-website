29  /**
30   * Serializes POJO models to URL-encoded notation with UON-encoded values (a notation for URL-encoded query paramter values).
31   *
32   *
33   * <h6 class='topic'>Media types</h6>
34   * <p>
35   * 	Handles <code>Accept</code> types: <code>application/x-www-form-urlencoded</code>
36   * <p>
37   * 	Produces <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
38   *
39   *
40   * <h6 class='topic'>Description</h6>
41   * <p>
42   * 	This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
43   * 	However, custom serializers can be constructed to fine-tune behavior.
44   *
45   *
46   * <h6 class='topic'>Configurable properties</h6>
47   * <p>
48   * 	This class has the following properties associated with it:
49   * <ul>
50   * 	<li>{@link UonSerializerContext}
51   * 	<li>{@link BeanContext}
52   * </ul>
53   * <p>
54   * 	The following shows a sample object defined in Javascript:
55   * </p>
56   * <p class='bcode'>
57   * 	{
58   * 		id: 1,
59   * 		name: <js>'John Smith'</js>,
60   * 		uri: <js>'http://sample/addressBook/person/1'</js>,
61   * 		addressBookUri: <js>'http://sample/addressBook'</js>,
62   * 		birthDate: <js>'1946-08-12T00:00:00Z'</js>,
63   * 		otherIds: <jk>null</jk>,
64   * 		addresses: [
65   * 			{
66   * 				uri: <js>'http://sample/addressBook/address/1'</js>,
67   * 				personUri: <js>'http://sample/addressBook/person/1'</js>,
68   * 				id: 1,
69   * 				street: <js>'100 Main Street'</js>,
70   * 				city: <js>'Anywhereville'</js>,
71   * 				state: <js>'NY'</js>,
72   * 				zip: 12345,
73   * 				isCurrent: <jk>true</jk>,
74   * 			}
75   * 		]
76   * 	}
77   * </p>
78   * <p>
79   * 	Using the "strict" syntax defined in this document, the equivalent
80   * 		URL-encoded notation would be as follows:
81   * </p>
82   * <p class='bcode'>
83   * 	<xa>id</xa>=$n(<xs>1</xs>)
84   * 	&amp;<xa>name</xa>=<xs>John+Smith</xs>,
85   * 	&amp;<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
86   * 	&amp;<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
87   * 	&amp;<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
88   * 	&amp;<xa>otherIds</xa>=<xs>%00</xs>,
89   * 	&amp;<xa>addresses</xa>=$a(
90   * 		$o(
91   * 			<xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
92   * 			<xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
93   * 			<xa>id</xa>=$n(<xs>1</xs>),
94   * 			<xa>street</xa>=<xs>100+Main+Street</xs>,
95   * 			<xa>city</xa>=<xs>Anywhereville</xs>,
96   * 			<xa>state</xa>=<xs>NY</xs>,
97   * 			<xa>zip</xa>=$n(<xs>12345</xs>),
98   * 			<xa>isCurrent</xa>=$b(<xs>true</xs>)
99   * 		)
100  * 	)
101  * </p>
102  * <p>
103  * 	A secondary "lax" syntax is available when the data type of the
104  * 		values are already known on the receiving end of the transmission:
105  * </p>
106  * <p class='bcode'>
107  * 	<xa>id</xa>=<xs>1</xs>,
108  * 	&amp;<xa>name</xa>=<xs>John+Smith</xs>,
109  * 	&amp;<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
110  * 	&amp;<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
111  * 	&amp;<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
112  * 	&amp;<xa>otherIds</xa>=<xs>%00</xs>,
113  * 	&amp;<xa>addresses</xa>=(
114  * 		(
115  * 			<xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
116  * 			<xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
117  * 			<xa>id</xa>=<xs>1</xs>,
118  * 			<xa>street</xa>=<xs>100+Main+Street</xs>,
119  * 			<xa>city</xa>=<xs>Anywhereville</xs>,
120  * 			<xa>state</xa>=<xs>NY</xs>,
121  * 			<xa>zip</xa>=<xs>12345</xs>,
122  * 			<xa>isCurrent</xa>=<xs>true</xs>
123  * 		)
124  * 	)
125  * </p>
126  *
127  *
128  * <h6 class='topic'>Examples</h6>
129  * <p class='bcode'>
130  * 	<jc>// Serialize a Map</jc>
131  * 	Map m = <jk>new</jk> ObjectMap(<js>"{a:'b',c:1,d:false,e:['f',1,false],g:{h:'i'}}"</js>);
132  *
133  * 	<jc>// Serialize to value equivalent to JSON.</jc>
134  * 	<jc>// Produces "a=b&amp;c=$n(1)&amp;d=$b(false)&amp;e=$a(f,$n(1),$b(false))&amp;g=$o(h=i)"</jc>
135  * 	String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
136  *
137  * 	<jc>// Serialize to simplified value (for when data type is already known by receiver).</jc>
138  * 	<jc>// Produces "a=b&amp;c=1&amp;d=false&amp;e=(f,1,false)&amp;g=(h=i))"</jc>
139  * 	String s = UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
140  *
141  * 	<jc>// Serialize a bean</jc>
142  * 	<jk>public class</jk> Person {
143  * 		<jk>public</jk> Person(String s);
144  * 		<jk>public</jk> String getName();
145  * 		<jk>public int</jk> getAge();
146  * 		<jk>public</jk> Address getAddress();
147  * 		<jk>public boolean</jk> deceased;
148  * 	}
149  *
150  * 	<jk>public class</jk> Address {
151  * 		<jk>public</jk> String getStreet();
152  * 		<jk>public</jk> String getCity();
153  * 		<jk>public</jk> String getState();
154  * 		<jk>public int</jk> getZip();
155  * 	}
156  *
157  * 	Person p = <jk>new</jk> Person(<js>"John Doe"</js>, 23, <js>"123 Main St"</js>, <js>"Anywhere"</js>, <js>"NY"</js>, 12345, <jk>false</jk>);
158  *
159  * 	<jc>// Produces "name=John+Doe&amp;age=23&amp;address=$o(street=123+Main+St,city=Anywhere,state=NY,zip=$n(12345))&amp;deceased=$b(false)"</jc>
160  * 	String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
161  *
162  * 	<jc>// Produces "name=John+Doe&amp;age=23&amp;address=(street=123+Main+St,city=Anywhere,state=NY,zip=12345)&amp;deceased=false)"</jc>
163  * 	String s = UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
164  * </p>
165  *
166  * @author James Bognar (james.bognar@salesforce.com)
167  */
168 @Produces("application/x-www-form-urlencoded")

223 	/**
224 	 * Workhorse method. Determines the type of object, and then calls the
225 	 * appropriate type-specific serialization method.
226 	 */
227 	@SuppressWarnings({ "rawtypes", "unchecked" })
228 	private SerializerWriter serializeAnything(UrlEncodingSerializerSession session, UonWriter out, Object o) throws Exception {
229 		BeanContext bc = session.getBeanContext();
230 
231 		boolean addTypeProperty;		// Add "_type" attribute to element?
232 		ClassMeta<?> aType;			// The actual type
233 		ClassMeta<?> sType;			// The serialized type
234 
235 		aType = session.push("root", o, object());
236 		session.indent--;
237 		if (aType == null)
238 			aType = object();
239 
240 		sType = aType.getSerializedClassMeta();
241 		addTypeProperty = (session.isAddBeanTypeProperties());
242 
243 		// Swap if necessary
244 		PojoSwap swap = aType.getPojoSwap();
245 		if (swap != null) {
246 			o = swap.swap(o, bc);
247 
248 			// If the getSwapClass() method returns Object, we need to figure out
249 			// the actual type now.
250 			if (sType.isObject())
251 				sType = bc.getClassMetaForObject(o);
252 		}
253 
254 		if (sType.isMap()) {
255 			if (o instanceof BeanMap)
256 				serializeBeanMap(session, out, (BeanMap)o, addTypeProperty);
257 			else
258 				serializeMap(session, out, (Map)o, sType);
259 		} else if (sType.hasToObjectMapMethod()) {
260 			serializeMap(session, out, sType.toObjectMap(o), sType);
261 		} else if (sType.isBean()) {
262 			serializeBeanMap(session, out, bc.forBean(o), addTypeProperty);
263 		} else if (sType.isCollection()) {
264 			serializeMap(session, out, getCollectionMap((Collection)o), bc.getMapClassMeta(Map.class, Integer.class, sType.getElementType()));
265 		} else {
266 			// All other types can't be serialized as key/value pairs, so we create a
267 			// mock key/value pair with a "_value" key.
268 			out.append("_value=");
269 			super.serializeAnything(session, out, o, null, null, null, false, true);
270 		}
271 
272 		session.pop();
273 		return out;
274 	}

276 	private Map<Integer,Object> getCollectionMap(Collection<?> c) {
277 		Map<Integer,Object> m = new TreeMap<Integer,Object>();
278 		int i = 0;
279 		for (Object o : c)
280 			m.put(i++, o);
281 		return m;
282 	}

284 	@SuppressWarnings({ "rawtypes", "unchecked" })
285 	private SerializerWriter serializeMap(UrlEncodingSerializerSession session, UonWriter out, Map m, ClassMeta<?> type) throws Exception {
286 
287 		m = session.sort(m);
288 
289 		ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
290 
291 		int depth = session.getIndent();
292 		boolean addAmp = false;
293 
294 		Iterator mapEntries = m.entrySet().iterator();
295 
296 		while (mapEntries.hasNext()) {
297 			Map.Entry e = (Map.Entry) mapEntries.next();
298 			Object value = e.getValue();
299 			Object key = session.generalize(e.getKey(), keyType);
300 
301 
302 			if (session.shouldUseExpandedParams(value)) {
303 				Iterator i = value instanceof Collection ? ((Collection)value).iterator() : ArrayUtils.iterator(value);
304 				while (i.hasNext()) {
305 					if (addAmp)
306 						out.cr(depth).append('&');
307 					out.appendObject(key, false, true, true).append('=');
308 					super.serializeAnything(session, out, i.next(), null, (key == null ? null : key.toString()), null, false, true);
309 					addAmp = true;
310 				}
311 			} else {
312 				if (addAmp)
313 					out.cr(depth).append('&');
314 				out.appendObject(key, false, true, true).append('=');
315 				super.serializeAnything(session, out, value, valueType, (key == null ? null : key.toString()), null, false, true);
316 				addAmp = true;
317 			}
318 		}
319 
320 		return out;
321 	}

323 	@SuppressWarnings({ "rawtypes" })
324 	private SerializerWriter serializeBeanMap(UrlEncodingSerializerSession session, UonWriter out, BeanMap<?> m, boolean addTypeProperty) throws Exception {
325 		int depth = session.getIndent();
326 
327 		boolean addAmp = false;
328 
329 		for (BeanPropertyValue p : m.getValues(session.isTrimNulls(), addTypeProperty ? session.createBeanTypeNameProperty(m) : null)) {
330 			BeanPropertyMeta pMeta = p.getMeta();
331 			ClassMeta<?> cMeta = p.getClassMeta();
332 
333 			String key = p.getName();
334 			Object value = p.getValue();
335 			Throwable t = p.getThrown();
336 			if (t != null)
337 				session.addBeanGetterWarning(pMeta, t);
338 
339 			if (session.canIgnoreValue(cMeta, key, value))
340 				continue;
341 
342 			if (value != null && session.shouldUseExpandedParams(pMeta)) {
343 				// Transformed object array bean properties may be transformed resulting in ArrayLists,
344 				// so we need to check type if we think it's an array.
345 				Iterator i = (cMeta.isCollection() || value instanceof Collection) ? ((Collection)value).iterator() : ArrayUtils.iterator(value);
346 				while (i.hasNext()) {
347 					if (addAmp)
348 						out.cr(depth).append('&');
349 
350 					out.appendObject(key, false, true, true).append('=');
351 
352 					super.serializeAnything(session, out, i.next(), cMeta.getElementType(), key, pMeta, false, true);
353 
354 					addAmp = true;
355 				}
356 			} else {
357 				if (addAmp)
358 					out.cr(depth).append('&');
359 
360 				out.appendObject(key, false, true, true).append('=');
361 
362 				super.serializeAnything(session, out, value, cMeta, key, pMeta, false, true);
363 
364 				addAmp = true;
365 			}
366 
367 		}
368 		return out;
369 	}

375 	/**
376 	 * Converts the specified object to a string using this serializers {@link BeanContext#convertToType(Object, Class)} method
377 	 * 	and runs {@link URLEncoder#encode(String,String)} against the results.
378 	 * Useful for constructing URL parts.
379 	 *
380 	 * @param o The object to serialize.
381 	 * @return The serialized object.
382 	 */
383 	public String serializeUrlPart(Object o) {
384 		try {
385 			// Shortcut for simple types.
386 			ClassMeta<?> cm = getBeanContext().getClassMetaForObject(o);
387 			if (cm != null)
388 				if (cm.isCharSequence() || cm.isNumber() || cm.isBoolean())
389 					return o.toString();
390 
391 			StringWriter w = new StringWriter();
392 			UonSerializerSession s = createSession(w, null, null);
393 			super.doSerialize(s, o);
394 			return w.toString();
395 		} catch (Exception e) {
396 			throw new RuntimeException(e);
397 		}
398 	}

410 	@Override /* Serializer */
411 	protected void doSerialize(SerializerSession session, Object o) throws Exception {
412 		UrlEncodingSerializerSession s = (UrlEncodingSerializerSession)session;
413 		serializeAnything(s, s.getWriter(), o);
414 	}
