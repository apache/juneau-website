13  package org.apache.juneau.urlencoding;
14  
15  import org.apache.juneau.*;
16  import org.apache.juneau.internal.*;
17  import org.apache.juneau.urlencoding.annotation.*;
18  
19  /**
20   * Metadata on classes specific to the URL-Encoding serializers and parsers pulled from the {@link UrlEncoding @UrlEncoding} annotation on the class.
21   *
22   * @author James Bognar (james.bognar@salesforce.com)
23   */
24  public class UrlEncodingClassMeta extends ClassMetaExtended {
25  
26  	private final UrlEncoding urlEncoding;
27  	private final boolean expandedParams;
28  
29  	/**
30  	 * Constructor.
31  	 *
32  	 * @param cm The class that this annotation is defined on.
33  	 */
34  	public UrlEncodingClassMeta(ClassMeta<?> cm) {
35  		super(cm);
36  		this.urlEncoding = ReflectionUtils.getAnnotation(UrlEncoding.class, getInnerClass());
37  		if (urlEncoding != null) {
38  			expandedParams = urlEncoding.expandedParams();
39  		} else {
40  			expandedParams = false;
41  		}
42  	}
43  
44  	/**
45  	 * Returns the {@link UrlEncoding} annotation defined on the class.
46  	 *
47  	 * @return The value of the {@link UrlEncoding} annotation, or <jk>null</jk> if annotation is not specified.
48  	 */
49  	protected UrlEncoding getAnnotation() {
50  		return urlEncoding;
51  	}
52  
53  	/**
54  	 * Returns the {@link UrlEncoding#expandedParams()} annotation defined on the class.
55  	 *
56  	 * @return The value of the {@link UrlEncoding#expandedParams()} annotation.
57  	 */
58  	protected boolean isExpandedParams() {
59  		return expandedParams;
60  	}
61  }
