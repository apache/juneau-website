26  /**
27   * Parses UON (a notation for URL-encoded query parameter values) text into POJO models.
28   *
29   *
30   * <h6 class='topic'>Media types</h6>
31   * <p>
32   * 	Handles <code>Content-Type</code> types: <code>text/uon</code>
33   *
34   *
35   * <h6 class='topic'>Description</h6>
36   * <p>
37   * 	This parser uses a state machine, which makes it very fast and efficient.
38   *
39   *
40   * <h6 class='topic'>Configurable properties</h6>
41   * <p>
42   * 	This class has the following properties associated with it:
43   * <ul>
44   * 	<li>{@link UonParserContext}
45   * 	<li>{@link ParserContext}
46   * 	<li>{@link BeanContext}
47   * </ul>
48   *
49   *
50   * @author James Bognar (james.bognar@salesforce.com)
51   */
52  @SuppressWarnings({ "rawtypes", "unchecked" })
53  @Consumes("text/uon")
54  public class UonParser extends ReaderParser {

80  	/**
81  	 * Workhorse method.
82  	 *
83  	 * @param session The parser context for this parse.
84  	 * @param eType The class type being parsed, or <jk>null</jk> if unknown.
85  	 * @param r The reader being parsed.
86  	 * @param outer The outer object (for constructing nested inner classes).
87  	 * @param isUrlParamValue If <jk>true</jk>, then we're parsing a top-level URL-encoded value which is treated a bit different than the default case.
88  	 * @param pMeta The current bean property being parsed.
89  	 * @return The parsed object.
90  	 * @throws Exception
91  	 */
92  	protected <T> T parseAnything(UonParserSession session, ClassMeta<T> eType, ParserReader r, Object outer, boolean isUrlParamValue, BeanPropertyMeta pMeta) throws Exception {
93  
94  		BeanContext bc = session.getBeanContext();
95  		if (eType == null)
96  			eType = (ClassMeta<T>)object();
97  		PojoSwap<T,Object> transform = (PojoSwap<T,Object>)eType.getPojoSwap();
98  		ClassMeta<?> sType = eType.getSerializedClassMeta();
99  		BeanDictionary bd = (pMeta == null ? bc.getBeanDictionary() : pMeta.getBeanDictionary());
100 
101 		Object o = null;
102 
103 		// Parse type flag '$x'
104 		char flag = readFlag(session, r, (char)0);
105 
106 		int c = r.peek();
107 
108 		if (c == -1 || c == AMP) {
109 			// If parameter is blank and it's an array or collection, return an empty list.
110 			if (sType.isArray() || sType.isCollection())
111 				o = sType.newInstance();
112 			else if (sType.isString() || sType.isObject())
113 				o = "";
114 			else if (sType.isPrimitive())
115 				o = sType.getPrimitiveDefault();
116 			// Otherwise, leave null.
117 		} else if (sType.isObject()) {
118 			if (flag == 0 || flag == 's') {
119 				o = parseString(session, r, isUrlParamValue);
120 			} else if (flag == 'b') {
121 				o = parseBoolean(session, r);
122 			} else if (flag == 'n') {
123 				o = parseNumber(session, r, null);
124 			} else if (flag == 'o') {
125 				ObjectMap m = new ObjectMap(bc);
126 				parseIntoMap(session, r, m, string(), object(), pMeta);
127 				o = bd.cast(m);
128 			} else if (flag == 'a') {
129 				Collection l = new ObjectList(bc);
130 				o = parseIntoCollection(session, r, l, sType.getElementType(), isUrlParamValue, pMeta);
131 			} else {
132 				throw new ParseException(session, "Unexpected flag character ''{0}''.", flag);
133 			}
134 		} else if (sType.isBoolean()) {
135 			o = parseBoolean(session, r);
136 		} else if (sType.isCharSequence()) {
137 			o = parseString(session, r, isUrlParamValue);
138 		} else if (sType.isChar()) {
139 			String s = parseString(session, r, isUrlParamValue);
140 			o = s == null ? null : s.charAt(0);
141 		} else if (sType.isNumber()) {
142 			o = parseNumber(session, r, (Class<? extends Number>)sType.getInnerClass());
143 		} else if (sType.isMap()) {
144 			Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : new ObjectMap(bc));
145 			o = parseIntoMap(session, r, m, sType.getKeyType(), sType.getValueType(), pMeta);
146 		} else if (sType.isCollection()) {
147 			if (flag == 'o') {
148 				ObjectMap m = new ObjectMap(bc);
149 				parseIntoMap(session, r, m, string(), object(), pMeta);
150 				// Handle case where it's a collection, but serialized as a map with a _type or _value key.
151 				if (m.containsKey(bc.getBeanTypePropertyName()))
152 					o = bd.cast(m);
153 				// Handle case where it's a collection, but only a single value was specified.
154 				else {
155 					Collection l = (sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance(outer) : new ObjectList(bc));
156 					l.add(m.cast(sType.getElementType()));
157 					o = l;
158 				}
159 			} else {
160 				Collection l = (sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance(outer) : new ObjectList(bc));
161 				o = parseIntoCollection(session, r, l, sType.getElementType(), isUrlParamValue, pMeta);
162 			}
163 		} else if (sType.canCreateNewInstanceFromObjectMap(outer)) {
164 			ObjectMap m = new ObjectMap(bc);
165 			parseIntoMap(session, r, m, string(), object(), pMeta);
166 			o = sType.newInstanceFromObjectMap(outer, m);
167 		} else if (sType.canCreateNewBean(outer)) {
168 			BeanMap m = bc.newBeanMap(outer, sType.getInnerClass());
169 			m = parseIntoBeanMap(session, r, m);
170 			o = m == null ? null : m.getBean();
171 		} else if (sType.canCreateNewInstanceFromString(outer)) {
172 			String s = parseString(session, r, isUrlParamValue);
173 			if (s != null)
174 				o = sType.newInstanceFromString(outer, s);
175 		} else if (sType.canCreateNewInstanceFromNumber(outer)) {
176 			o = sType.newInstanceFromNumber(outer, parseNumber(session, r, sType.getNewInstanceFromNumberClass()));
177 		} else if (sType.isArray()) {
178 			if (flag == 'o') {
179 				ObjectMap m = new ObjectMap(bc);
180 				parseIntoMap(session, r, m, string(), object(), pMeta);
181 				// Handle case where it's an array, but serialized as a map with a _type or _value key.
182 				if (m.containsKey(bc.getBeanTypePropertyName()))
183 					o = bd.cast(m);
184 				// Handle case where it's an array, but only a single value was specified.
185 				else {
186 					ArrayList l = new ArrayList(1);
187 					l.add(m.cast(sType.getElementType()));
188 					o = bc.toArray(sType, l);
189 				}
190 			} else {
191 				ArrayList l = (ArrayList)parseIntoCollection(session, r, new ArrayList(), sType.getElementType(), isUrlParamValue, pMeta);
192 				o = bc.toArray(sType, l);
193 			}
194 		} else if (flag == 'o') {
195 			// It could be a non-bean with _type attribute.
196 			ObjectMap m = new ObjectMap(bc);
197 			parseIntoMap(session, r, m, string(), object(), pMeta);
198 			if (m.containsKey(bc.getBeanTypePropertyName()))
199 				o = bd.cast(m);
200 			else
201 				throw new ParseException(session, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
202 		} else {
203 			throw new ParseException(session, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
204 		}
205 
206 		if (transform != null && o != null)
207 			o = transform.unswap(o, eType, bc);
208 
209 		if (outer != null)
210 			setParent(eType, o, outer);
211 
212 		return (T)o;
213 	}

215 	private <K,V> Map<K,V> parseIntoMap(UonParserSession session, ParserReader r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType, BeanPropertyMeta pMeta) throws Exception {
216 
217 		if (keyType == null)
218 			keyType = (ClassMeta<K>)string();
219 
220 		int c = r.read();
221 		if (c == -1 || c == NUL || c == AMP)
222 			return null;
223 		if (c != '(')
224 			throw new ParseException(session, "Expected '(' at beginning of object.");
225 
226 		final int S1=1; // Looking for attrName start.
227 		final int S2=2; // Found attrName end, looking for =.
228 		final int S3=3; // Found =, looking for valStart.
229 		final int S4=4; // Looking for , or )
230 		boolean isInEscape = false;
231 
232 		int state = S1;
233 		K currAttr = null;
234 		while (c != -1 && c != AMP) {
235 			c = r.read();
236 			if (! isInEscape) {
237 				if (state == S1) {
238 					if (c == ')')
239 						return m;
240 					if ((c == '\n' || c == '\r') && session.isWhitespaceAware())
241 						skipSpace(r);
242 					else {
243 						r.unread();
244 						Object attr = parseAttr(session, r, session.isDecodeChars());
245 						currAttr = session.trim((attr == null ? null : session.getBeanContext().convertToType(attr, keyType)));
246 						state = S2;
247 						c = 0; // Avoid isInEscape if c was '\'
248 					}
249 				} else if (state == S2) {
250 					if (c == EQ || c == '=')
251 						state = S3;
252 					else if (c == -1 || c == ',' || c == ')' || c == AMP) {
253 						if (currAttr == null) {
254 							// Value was '%00'
255 							r.unread();
256 							return null;
257 						}
258 						m.put(currAttr, null);
259 						if (c == ')' || c == -1 || c == AMP)
260 							return m;
261 						state = S1;
262 					}
263 				} else if (state == S3) {
264 					if (c == -1 || c == ',' || c == ')' || c == AMP) {
265 						V value = convertAttrToType(session, m, "", valueType);
266 						m.put(currAttr, value);
267 						if (c == -1 || c == ')' || c == AMP)
268 							return m;
269 						state = S1;
270 					} else  {
271 						V value = parseAnything(session, valueType, r.unread(), m, false, pMeta);
272 						setName(valueType, value, currAttr);
273 						m.put(currAttr, value);
274 						state = S4;
275 						c = 0; // Avoid isInEscape if c was '\'
276 					}
277 				} else if (state == S4) {
278 					if (c == ',')
279 						state = S1;
280 					else if (c == ')' || c == -1 || c == AMP) {
281 						return m;
282 					}
283 				}
284 			}
285 			isInEscape = isInEscape(c, r, isInEscape);
286 		}
287 		if (state == S1)
288 			throw new ParseException(session, "Could not find attribute name on object.");
289 		if (state == S2)
290 			throw new ParseException(session, "Could not find '=' following attribute name on object.");
291 		if (state == S3)
292 			throw new ParseException(session, "Dangling '=' found in object entry");
293 		if (state == S4)
294 			throw new ParseException(session, "Could not find ')' marking end of object.");
295 
296 		return null; // Unreachable.
297 	}

299 	private <E> Collection<E> parseIntoCollection(UonParserSession session, ParserReader r, Collection<E> l, ClassMeta<E> elementType, boolean isUrlParamValue, BeanPropertyMeta pMeta) throws Exception {
300 
301 		int c = r.read();
302 		if (c == -1 || c == NUL || c == AMP)
303 			return null;
304 
305 		// If we're parsing a top-level parameter, we're allowed to have comma-delimited lists outside parenthesis (e.g. "&foo=1,2,3&bar=a,b,c")
306 		// This is not allowed at lower levels since we use comma's as end delimiters.
307 		boolean isInParens = (c == '(');
308 		if (! isInParens)
309 			if (isUrlParamValue)
310 				r.unread();
311 			else
312 				throw new ParseException(session, "Could not find '(' marking beginning of collection.");
313 
314 		if (isInParens) {
315 			final int S1=1; // Looking for starting of first entry.
316 			final int S2=2; // Looking for starting of subsequent entries.
317 			final int S3=3; // Looking for , or ) after first entry.
318 
319 			int state = S1;
320 			while (c != -1 && c != AMP) {
321 				c = r.read();
322 				if (state == S1 || state == S2) {
323 					if (c == ')') {
324 						if (state == S2) {
325 							l.add(parseAnything(session, elementType, r.unread(), l, false, pMeta));
326 							r.read();
327 						}
328 						return l;
329 					} else if ((c == '\n' || c == '\r') && session.isWhitespaceAware()) {
330 						skipSpace(r);
331 					} else {
332 						l.add(parseAnything(session, elementType, r.unread(), l, false, pMeta));
333 						state = S3;
334 					}
335 				} else if (state == S3) {
336 					if (c == ',') {
337 						state = S2;
338 					} else if (c == ')') {
339 						return l;
340 					}
341 				}
342 			}
343 			if (state == S1 || state == S2)
344 				throw new ParseException(session, "Could not find start of entry in array.");
345 			if (state == S3)
346 				throw new ParseException(session, "Could not find end of entry in array.");
347 
348 		} else {
349 			final int S1=1; // Looking for starting of entry.
350 			final int S2=2; // Looking for , or & or END after first entry.
351 
352 			int state = S1;
353 			while (c != -1 && c != AMP) {
354 				c = r.read();
355 				if (state == S1) {
356 					if ((c == '\n' || c == '\r') && session.isWhitespaceAware()) {
357 						skipSpace(r);
358 					} else {
359 						l.add(parseAnything(session, elementType, r.unread(), l, false, pMeta));
360 						state = S2;
361 					}
362 				} else if (state == S2) {
363 					if (c == ',') {
364 						state = S1;
365 					} else if ((c == '\n' || c == '\r') && session.isWhitespaceAware()) {
366 						skipSpace(r);
367 					} else if (c == AMP || c == -1) {
368 						r.unread();
369 						return l;
370 					}
371 				}
372 			}
373 		}
374 
375 		return null;  // Unreachable.
376 	}

378 	private <T> BeanMap<T> parseIntoBeanMap(UonParserSession session, ParserReader r, BeanMap<T> m) throws Exception {
379 
380 		BeanContext bc = session.getBeanContext();
381 
382 		int c = r.read();
383 		if (c == -1 || c == NUL || c == AMP)
384 			return null;
385 		if (c != '(')
386 			throw new ParseException(session, "Expected '(' at beginning of object.");
387 
388 		final int S1=1; // Looking for attrName start.
389 		final int S2=2; // Found attrName end, looking for =.
390 		final int S3=3; // Found =, looking for valStart.
391 		final int S4=4; // Looking for , or }
392 		boolean isInEscape = false;
393 
394 		int state = S1;
395 		String currAttr = "";
396 		int currAttrLine = -1, currAttrCol = -1;
397 		while (c != -1 && c != AMP) {
398 			c = r.read();
399 			if (! isInEscape) {
400 				if (state == S1) {
401 					if (c == ')' || c == -1 || c == AMP) {
402 						return m;
403 					}
404 					if ((c == '\n' || c == '\r') && session.isWhitespaceAware())
405 						skipSpace(r);
406 					else {
407 						r.unread();
408 						currAttrLine= r.getLine();
409 						currAttrCol = r.getColumn();
410 						currAttr = parseAttrName(session, r, session.isDecodeChars());
411 						if (currAttr == null)  // Value was '%00'
412 							return null;
413 						state = S2;
414 					}
415 				} else if (state == S2) {
416 					if (c == EQ || c == '=')
417 						state = S3;
418 					else if (c == -1 || c == ',' || c == ')' || c == AMP) {
419 						m.put(currAttr, null);
420 						if (c == ')' || c == -1 || c == AMP)
421 							return m;
422 						state = S1;
423 					}
424 				} else if (state == S3) {
425 					if (c == -1 || c == ',' || c == ')' || c == AMP) {
426 						if (! currAttr.equals(bc.getBeanTypePropertyName())) {
427 							BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
428 							if (pMeta == null) {
429 								if (m.getMeta().isSubTyped()) {
430 									m.put(currAttr, "");
431 								} else {
432 									onUnknownProperty(session, currAttr, m, currAttrLine, currAttrCol);
433 								}
434 							} else {
435 								Object value = session.getBeanContext().convertToType("", pMeta.getClassMeta());
436 								pMeta.set(m, value);
437 							}
438 						}
439 						if (c == -1 || c == ')' || c == AMP)
440 							return m;
441 						state = S1;
442 					} else {
443 						if (! currAttr.equals(bc.getBeanTypePropertyName())) {
444 							BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
445 							if (pMeta == null) {
446 								if (m.getMeta().isSubTyped()) {
447 									Object value = parseAnything(session, object(), r.unread(), m.getBean(false), false, null);
448 									m.put(currAttr, value);
449 								} else {
450 									onUnknownProperty(session, currAttr, m, currAttrLine, currAttrCol);
451 									parseAnything(session, object(), r.unread(), m.getBean(false), false, null); // Read content anyway to ignore it
452 								}
453 							} else {
454 								session.setCurrentProperty(pMeta);
455 								ClassMeta<?> cm = pMeta.getClassMeta();
456 								Object value = parseAnything(session, cm, r.unread(), m.getBean(false), false, pMeta);
457 								setName(cm, value, currAttr);
458 								pMeta.set(m, value);
459 								session.setCurrentProperty(null);
460 							}
461 						}
462 						state = S4;
463 					}
464 				} else if (state == S4) {
465 					if (c == ',')
466 						state = S1;
467 					else if (c == ')' || c == -1 || c == AMP) {
468 						return m;
469 					}
470 				}
471 			}
472 			isInEscape = isInEscape(c, r, isInEscape);
473 		}
474 		if (state == S1)
475 			throw new ParseException(session, "Could not find attribute name on object.");
476 		if (state == S2)
477 			throw new ParseException(session, "Could not find '=' following attribute name on object.");
478 		if (state == S3)
479 			throw new ParseException(session, "Could not find value following '=' on object.");
480 		if (state == S4)
481 			throw new ParseException(session, "Could not find ')' marking end of object.");
482 
483 		return null; // Unreachable.
484 	}

486 	Object parseAttr(UonParserSession session, ParserReader r, boolean encoded) throws Exception {
487 		Object attr;
488 		int c = r.peek();
489 		if (c == '$') {
490 			char f = readFlag(session, r, (char)0);
491 			if (f == 'b')
492 				attr = parseBoolean(session, r);
493 			else if (f == 'n')
494 				attr = parseNumber(session, r, null);
495 			else
496 				attr = parseAttrName(session, r, encoded);
497 		} else {
498 			attr = parseAttrName(session, r, encoded);
499 		}
500 		return attr;
501 	}

503 	String parseAttrName(UonParserSession session, ParserReader r, boolean encoded) throws Exception {
504 
505 		// If string is of form '(xxx)', we're looking for ')' at the end.
506 		// Otherwise, we're looking for '&' or '=' or -1 denoting the end of this string.
507 
508 		int c = r.peek();
509 		if (c == '$')
510 			readFlag(session, r, 's');
511 		if (c == '(')
512 			return parsePString(session, r);
513 
514 		r.mark();
515 		boolean isInEscape = false;
516 		if (encoded) {
517 			while (c != -1) {
518 				c = r.read();
519 				if (! isInEscape) {
520 					if (c == AMP || c == EQ || c == -1) {
521 						if (c != -1)
522 							r.unread();
523 						String s = r.getMarked();
524 						return (s.equals("\u0000") ? null : s);
525 					}
526 				}
527 				else if (c == AMP)
528 					r.replace('&');
529 				else if (c == EQ)
530 					r.replace('=');
531 				isInEscape = isInEscape(c, r, isInEscape);
532 			}
533 		} else {
534 			while (c != -1) {
535 				c = r.read();
536 				if (! isInEscape) {
537 					if (c == '=' || c == -1) {
538 						if (c != -1)
539 							r.unread();
540 						String s = r.getMarked();
541 						return (s.equals("\u0000") ? null : session.trim(s));
542 					}
543 				}
544 				isInEscape = isInEscape(c, r, isInEscape);
545 			}
546 		}
547 
548 		// We should never get here.
549 		throw new ParseException(session, "Unexpected condition.");
550 	}

553 	/**
554 	 * Returns true if the next character in the stream is preceeded by an escape '~' character.
555 	 * @param c The current character.
556 	 * @param r The reader.
557 	 * @param prevIsInEscape What the flag was last time.
558 	 */
559 	private static final boolean isInEscape(int c, ParserReader r, boolean prevIsInEscape) throws Exception {
560 		if (c == '~' && ! prevIsInEscape) {
561 			c = r.peek();
562 			if (escapedChars.contains(c)) {
563 				r.delete();
564 				return true;
565 			}
566 		}
567 		return false;
568 	}

570 	String parseString(UonParserSession session, ParserReader r, boolean isUrlParamValue) throws Exception {
571 
572 		// If string is of form '(xxx)', we're looking for ')' at the end.
573 		// Otherwise, we're looking for ',' or ')' or -1 denoting the end of this string.
574 
575 		int c = r.peek();
576 		if (c == '(')
577 			return parsePString(session, r);
578 
579 		r.mark();
580 		boolean isInEscape = false;
581 		String s = null;
582 		AsciiSet endChars = (isUrlParamValue ? endCharsParam : endCharsNormal);
583 		while (c != -1) {
584 			c = r.read();
585 			if (! isInEscape) {
586 				// If this is a URL parameter value, we're looking for:  &
587 				// If not, we're looking for:  &,)
588 				if (endChars.contains(c)) {
589 					r.unread();
590 					c = -1;
591 				}
592 			}
593 			if (c == -1)
594 				s = r.getMarked();
595 			else if (c == EQ)
596 				r.replace('=');
597 			else if ((c == '\n' || c == '\r') && session.isWhitespaceAware()) {
598 				s = r.getMarked(0, -1);
599 				skipSpace(r);
600 				c = -1;
601 			}
602 			isInEscape = isInEscape(c, r, isInEscape);
603 		}
604 
605 		return (s == null || s.equals("\u0000") ? null : session.trim(s));
606 	}

608 	private static final AsciiSet endCharsParam = new AsciiSet(""+AMP), endCharsNormal = new AsciiSet(",)"+AMP);

611 	/**
612 	 * Parses a string of the form "(foo)"
613 	 * All whitespace within parenthesis are preserved.
614 	 */
615 	static String parsePString(UonParserSession session, ParserReader r) throws Exception {
616 
617 		r.read(); // Skip first parenthesis.
618 		r.mark();
619 		int c = 0;
620 
621 		boolean isInEscape = false;
622 		while (c != -1) {
623 			c = r.read();
624 			if (! isInEscape) {
625 				if (c == ')')
626 					return session.trim(r.getMarked(0, -1));
627 			}
628 			if (c == EQ)
629 				r.replace('=');
630 			isInEscape = isInEscape(c, r, isInEscape);
631 		}
632 		throw new ParseException(session, "Unmatched parenthesis");
633 	}

635 	private Boolean parseBoolean(UonParserSession session, ParserReader r) throws Exception {
636 		readFlag(session, r, 'b');
637 		String s = parseString(session, r, false);
638 		if (s == null)
639 			return null;
640 		if (s.equals("true"))
641 			return true;
642 		if (s.equals("false"))
643 			return false;
644 		throw new ParseException(session, "Unrecognized syntax for boolean.  ''{0}''.", s);
645 	}

647 	private Number parseNumber(UonParserSession session, ParserReader r, Class<? extends Number> c) throws Exception {
648 		readFlag(session, r, 'n');
649 		String s = parseString(session, r, false);
650 		if (s == null)
651 			return null;
652 		return StringUtils.parseNumber(s, c);
653 	}

656 	 * Call this method after you've finished a parsing a string to make sure that if there's any
657 	 * remainder in the input, that it consists only of whitespace and comments.
658 	 */
659 	private void validateEnd(UonParserSession session, ParserReader r) throws Exception {
660 		int c = r.read();
661 		if (c != -1)
662 			throw new ParseException(session, "Remainder after parse: ''{0}''.", (char)c);
663 	}

665 	/**
666 	 * Reads flag character from "$x(" construct if flag is present.
667 	 * Returns 0 if no flag is present.
668 	 */
669 	static char readFlag(UonParserSession session, ParserReader r, char expected) throws Exception {
670 		char c = (char)r.peek();
671 		if (c == '$') {
672 			r.read();
673 			char f = (char)r.read();
674 			if (expected != 0 && f != expected)
675 				throw new ParseException(session, "Unexpected flag character: ''{0}''.  Expected ''{1}''.", f, expected);
676 			c = (char)r.peek();
677 			// Type flag must be followed by '('
678 			if (c != '(')
679 				throw new ParseException(session, "Unexpected character following flag: ''{0}''.", c);
680 			return f;
681 		}
682 		return 0;
683 	}

685 	private Object[] parseArgs(UonParserSession session, ParserReader r, ClassMeta<?>[] argTypes) throws Exception {
686 
687 		final int S1=1; // Looking for start of entry
688 		final int S2=2; // Looking for , or )
689 
690 		Object[] o = new Object[argTypes.length];
691 		int i = 0;
692 
693 		int c = r.read();
694 		if (c == -1 || c == AMP)
695 			return null;
696 		if (c != '(')
697 			throw new ParseException(session, "Expected '(' at beginning of args array.");
698 
699 		int state = S1;
700 		while (c != -1 && c != AMP) {
701 			c = r.read();
702 			if (state == S1) {
703 				if (c == ')')
704 					return o;
705 				o[i] = parseAnything(session, argTypes[i], r.unread(), session.getOuter(), false, null);
706 				i++;
707 				state = S2;
708 			} else if (state == S2) {
709 				if (c == ',') {
710 					state = S1;
711 				} else if (c == ')') {
712 					return o;
713 				}
714 			}
715 		}
716 
717 		throw new ParseException(session, "Did not find ')' at the end of args array.");
718 	}

720 	private static void skipSpace(ParserReader r) throws Exception {
721 		int c = 0;
722 		while ((c = r.read()) != -1) {
723 			if (c > ' ' || c <= 2) {
724 				r.unread();
725 				return;
726 			}
727 		}
728 	}
