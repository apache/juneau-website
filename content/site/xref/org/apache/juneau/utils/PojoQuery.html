24  /**
25   * Designed to provide query/view/sort/paging filtering on tabular in-memory POJO models.
26   * <p>
27   * 	It can also perform just view filtering on beans/maps.
28   * <p>
29   * 	Examples of tabular POJO models:
30   * 	<ul>
31   * 		<li><tt>Collection{@code <Map>}</tt>
32   * 		<li><tt>Collection{@code <Bean>}</tt>
33   * 		<li><tt>Map[]</tt>
34   * 		<li><tt>Bean[]</tt>
35   * 	</ul>
36   * <p>
37   * 	Tabular POJO models can be thought of as tables of data.  For example, a list of the following beans...
38   * <p class='bcode'>
39   * 	<jk>public</jk> MyBean {
40   * 		<jk>public int</jk> fi;
41   * 		<jk>public</jk> String fs;
42   * 		<jk>public</jk> Date fd;
43   * 	}
44   * <p>
45   * 	... can be thought of a table containing the following columns...
46   * <p>
47   * 	<table class='styled code'>
48   * 		<tr><th>fi</th><th>fs</th><th>fd</th></tr>
49   * 		<tr><td>123</td><td>'foobar'</td><td>yyyy/MM/dd HH:mm:ss</td></tr>
50   * 		<tr><td colspan=3>...</td></tr>
51   * 	</table>
52   * <p>
53   * 	From this table, you can perform the following functions:
54   * 	<ul class='spaced-list'>
55   * 		<li>Query - Return only rows where a search pattern matches.
56   * 		<li>View - Return only the specified subset of columns in the specified order.
57   * 		<li>Sort - Sort the table by one or more columns.
58   * 		<li>Page - Only return a subset of rows.
59   * 	</ul>
60   *
61   * <h5 class='topic'>Query</h5>
62   * <p>
63   * 	The query capabilites allow you to filter based on query patterns against
64   * 	strings, dates, and numbers.  Queries take the form of a Map with column names
65   * 	as keys, and search patterns as values.  <br>
66   * 	Search patterns can be either {@code Strings} or {@code Maps}.<br>
67   * 	Multiple search patterns are ANDed (i.e. all patterns must match for the row to be returned).
68   *
69   * <h6 class='topic'>Examples</h6>
70   * <ul class='spaced-list'>
71   * 	<li><tt>{fi:'123'}</tt> - Return only rows where the <tt>fi</tt> column is 123.
72   * 	<li><tt>{fs:'foobar'}</tt> - Return only rows where the <tt>fs</tt> column is 'foobar'.
73   * 	<li><tt>{fd:'2001'}</tt> - Return only rows where the <tt>fd</tt> column have dates in the year 2001.
74   * 	<li><tt>{fs:'foobar'}</tt> - Return only rows where the <tt>fs</tt> column is 'foobar'.
75   * 		and the <tt>fs</tt> column starts with <tt>"foo"</tt>.
76   * </ul>
77   * <p>
78   * 	Search patterns can also be applied to lower level fields.  For example, the search term
79   * 	<tt>{f1:{f2:{f3{'foobar'}}}</tt> means only return top level rows where the <tt>f1.getF2().getF3()</tt>
80   * 	property is <tt>'foobar'</tt>.
81   *
82   * <h5 class='topic'>String Patterns</h5>
83   * <p>
84   * 	Any objects can be queried against using string patterns.  If the objects being
85   * 	searched are not strings, then the patterns are matched against whatever is
86   * 	return by the {@code Object#toString()} method.
87   *
88   * <h6 class='topic'>Example string query patterns</h6>
89   * <ul>
90   * 	<li><tt>foo</tt> - The string 'foo'
91   * 	<li><tt>foo bar</tt> - The string 'foo' or the string 'bar'
92   * 	<li><tt>'foo bar'</tt> - The phrase 'foo bar'
93   * 	<li><tt>"foo bar"</tt> - The phrase 'foo bar'
94   * 	<li><tt>foo*</tt> - <tt>*</tt> matches zero-or-more characters.
95   * 	<li><tt>foo?</tt> - <tt>?</tt> matches exactly one character
96   * </ul>
97   *
98   * <h6 class='topic'>Notes</h6>
99   * <ul class='spaced-list'>
100  * 	<li>Whitespace is ignored around search patterns.
101  * 	<li>Prepend <tt>+</tt> to tokens that must match.  (e.g. <tt>+foo* +*bar</tt>)
102  * 	<li>Prepend <tt>-</tt> to tokens that must not match.  (e.g. <tt>+foo* -*bar</tt>)
103  * </ul>
104  *
105  * <h5 class='topic'>Numeric Patterns</h5>
106  * <p>
107  * 	Any object of type {@link Number} (or numeric primitives) can be searched using numeric patterns.
108  *
109  * <h6 class='topic'>Example numeric query patterns</h6>
110  * <ul>
111  * 	<li><tt>123</tt> - The single number 123
112  * 	<li><tt>1 2 3</tt>	- 1, 2, or 3
113  * 	<li><tt>1-100</tt> - Between 1 and 100
114  * 	<li><tt>1 - 100</tt> - Between 1 and 100
115  * 	<li><tt>1 - 100 200-300</tt> - Between 1 and 100 or between 200 and 300
116  * 	<li><tt>&gt; 100</tt> - Greater than 100
117  * 	<li><tt>&gt;= 100</tt> - Greater than or equal to 100
118  * 	<li><tt>!123</tt> - Not 123
119  * </ul>
120  *
121  * <h6 class='topic'>Notes</h6>
122  * <ul class='spaced-list'>
123  * 	<li>Whitespace is ignored in search patterns.
124  * 	<li>Negative numbers are supported.
125  * </ul>
126  *
127  * <h5 class='topic'>Date Patterns</h5>
128  * <p>
129  * 	Any object of type {@link Date} or {@link Calendar} can be searched using date patterns.
130  * <p>
131  * 	The default valid input timestamp formats (which can be overridden via the {@link #setValidTimestampFormats(String...)} method are...
132  *
133  * <ul>
134  * 	<li><tt>yyyy.MM.dd.HH.mm.ss</tt>
135  * 	<li><tt>yyyy.MM.dd.HH.mm</tt>
136  * 	<li><tt>yyyy.MM.dd.HH</tt>
137  * 	<li><tt>yyyy.MM.dd</tt>
138  * 	<li><tt>yyyy.MM</tt>
139  * 	<li><tt>yyyy</tt>
140  * </ul>
141  *
142  * <h6 class='topic'>Example date query patterns</h6>
143  * <ul>
144  * 	<li><tt>2001</tt> - A specific year.
145  * 	<li><tt>2001.01.01.10.50</tt> - A specific time.
146  * 	<li><tt>&gt;2001</tt>	- After a specific year.
147  * 	<li><tt>&gt;=2001</tt> - During or after a specific year.
148  * 	<li><tt>2001 - 2003.06.30</tt>	- A date range.
149  * 	<li><tt>2001 2003 2005</tt>	- Multiple date patterns are ORed.
150  * </ul>
151  *
152  * <h6 class='topic'>Notes</h6>
153  * <ul>
154  * 	<li>Whitespace is ignored in search patterns.
155  * </ul>
156  *
157  * <h5 class='topic'>View</h5>
158  * <p>
159  * 	The view capability allows you to return only the specified subset of columns in the
160  * 	specified order.<br>
161  * 	The view parameter is a list of either <tt>Strings</tt> or <tt>Maps</tt>.
162  *
163  * <h6 class='topic'>Example view parameters</h6>
164  * <ul>
165  * 	<li><tt>['f1']</tt> - Return only column 'f1'.
166  * 	<li><tt>['f2','f1']</tt> - Return only columns 'f2' and 'f1'.
167  * 	<li><tt>['f1',{f2:'f3'}]</tt> - Return only columns 'f1' and 'f2', but for 'f2' objects,
168  * 		only show the 'f3' property.
169  * </ul>
170  *
171  * <h5 class='topic'>Sort</h5>
172  * <p>
173  * 	The sort capability allows you to sort values by the specified rows.<br>
174  * 	The sort parameter is a list of either <tt>Strings</tt> or <tt>Maps</tt>.<br>
175  * 	<tt>Strings</tt> represent column names to sort ascending.  If you want
176  * 	to sort descending, you need to specify a <tt>Map</tt> of the form <tt>{colname:'d'}</tt>
177  *
178  * <h6 class='topic'>Example sort parameters</h6>
179  * <ul>
180  * 	<li><tt>['f1']</tt> - Sort rows by column 'f1' ascending.
181  * 	<li><tt>[{f1:'a'}]</tt> - Sort rows by column 'f1' ascending.
182  * 	<li><tt>[{f1:'d'}]</tt> - Sort rows by column 'f1' descending.
183  * 	<li><tt>[{f1:'a'},{f2:'d'}]</tt> - Sort rows by column 'f1' ascending, then 'f2' descending.
184  * </ul>
185  *
186  * <h5 class='topic'>Paging</h5>
187  * <p>
188  * 	Use the <tt>pos</tt> and <tt>limit</tt> parameters to specify a subset of rows to
189  * 	return.
190  *
191  * <h5 class='topic'>Other Notes</h5>
192  * <ul class='spaced-list'>
193  * 	<li>Calling <tt>filterMap()</tt> or <tt>filterCollection()</tt> always returns a new data
194  * 		structure, so the methods can be called multiple times against the same input.
195  * </ul>
196  *
197  * @author James Bognar (james.bognar@salesforce.com)
198  */

206 	/**
207 	 * Constructor.
208 	 *
209 	 * @param input The POJO we're going to be filtering.
210 	 * @param beanContext The bean context to use to create bean maps for beans.
211 	 */
212 	public PojoQuery(Object input, BeanContext beanContext) {
213 		this.input = input;
214 		this.type = beanContext.getClassMetaForObject(input);
215 		this.beanContext = beanContext;
216 	}

218 	/**
219 	 * Filters the input object as a map.
220 	 *
221 	 * @param view The list and order of properties to return from the map.  Values must be of type {@code String} or {@code Map}.
222 	 * @return The filtered map
223 	 */
224 	public Map filterMap(List view) {
225 
226 		if (input == null)
227 			return null;
228 
229 		if (! (type.isBean() || type.isMap()))
230 			throw new RuntimeException("Cannot call filterMap() on class type " + type);
231 
232 		Map m = (Map)replaceWithMutables(input);
233 		doView(m, view);
234 
235 		return m;
236 	}
237 
238 	/**
239 	 * Filters the input object as a collection of maps.
240 	 *
241 	 * @param query The query attributes.  Keys must be column names and values must be of type {@code String} or {@code Map}.
242 	 * @param view The view attributes.  Values must be of type {@code String} or {@code Map}.
243 	 * @param sort The sort attributes.  Values must be of type {@code String} or {@code Map}.
244 	 * @param pos The index into the list to start returning results from.  Default is {@code 0}.
245 	 * @param limit The number of rows to return.  Default is all rows.
246 	 * @param ignoreCase If <jk>true</jk>, then querying is case insensitive.  Default is <jk>false</jk>.
247 	 * @return The filtered collection.
248 	 */
249 	public List filterCollection(Map query, List view, List sort, int pos, int limit, boolean ignoreCase) {
250 
251 		if (input == null)
252 			return null;
253 
254 		if (! (type.isArray() || type.isCollection()))
255 			throw new RuntimeException("Cannot call filterCollection() on class type " + type);
256 
257 		if (view == null)
258 			view = Collections.EMPTY_LIST;
259 
260 		if (sort == null)
261 			sort = Collections.EMPTY_LIST;
262 
263 		// Create a new ObjectList
264 		ObjectList l = (ObjectList)replaceWithMutables(input);
265 
266 		// Do the search
267 		CollectionFilter filter = new CollectionFilter(query, ignoreCase);
268 		filter.doQuery(l);
269 
270 		// If sort or view isn't empty, then we need to make sure that all entries in the
271 		// list are maps.
272 		if ((! sort.isEmpty()) || (! view.isEmpty())) {
273 
274 			if (! sort.isEmpty())
275 				doSort(l, sort);
276 
277 			if (! view.isEmpty())
278 				doView(l, view);
279 		}
280 
281 		// Do the paging.
282 		if (pos != 0 || limit != 0) {
283 			int end = (limit == 0 || limit+pos >= l.size()) ? l.size() : limit + pos;
284 			ObjectList l2 = new DelegateList(((DelegateList)l).getClassMeta());
285 			l2.addAll(l.subList(pos, end));
286 			l = l2;
287 		}
288 
289 		return l;
290 	}

292 	/*
293 	 * If there are any non-Maps in the specified list, replaces them with BeanMaps.
294 	 */
295 	private Object replaceWithMutables(Object o) {
296 		if (o == null)
297 			return null;
298 		ClassMeta cm = beanContext.getClassMetaForObject(o);
299 		if (cm.isCollection()) {
300 			ObjectList l = new DelegateList(beanContext.getClassMetaForObject(o));
301 			for (Object o2 : (Collection)o)
302 				l.add(replaceWithMutables(o2));
303 			return l;
304 		}
305 		if (cm.isMap() && o instanceof BeanMap) {
306 			BeanMap bm = (BeanMap)o;
307 			DelegateBeanMap dbm = new DelegateBeanMap(bm.getBean(), beanContext);
308 			for (BeanMapEntry e : (Set<BeanMapEntry>)bm.entrySet()) {
309 				ClassMeta ct1 = e.getMeta().getClassMeta();
310 				if (ct1.isArray() || ct1.isBean() || ct1.isCollection() || ct1.isMap() || ct1.isObject())
311 					dbm.put(e.getKey(), replaceWithMutables(e.getValue()));
312 				else
313 					dbm.addKey(e.getKey());
314 			}
315 			return dbm;
316 		}
317 		if (cm.isBean()) {
318 			BeanMap bm = beanContext.forBean(o);
319 			DelegateBeanMap dbm = new DelegateBeanMap(bm.getBean(), beanContext);
320 			for (BeanMapEntry e : (Set<BeanMapEntry>)bm.entrySet()) {
321 				ClassMeta ct1 = e.getMeta().getClassMeta();
322 				if (ct1.isArray() || ct1.isBean() || ct1.isCollection() || ct1.isMap() || ct1.isObject()) {
323 					Object val = null;
324 					try {
325 						val = e.getValue();
326 					} catch (BeanRuntimeException ex) {
327 						// Ignore.
328 					}
329 					dbm.put(e.getKey(), replaceWithMutables(val));
330 				}
331 				else
332 					dbm.addKey(e.getKey());
333 			}
334 			return dbm;
335 		}
336 		if (cm.isMap()) {
337 			Map m = (Map)o;
338 			DelegateMap dm = new DelegateMap(beanContext.getClassMetaForObject(m));
339 			for (Map.Entry e : (Set<Map.Entry>)m.entrySet())
340 				dm.put(e.getKey().toString(), replaceWithMutables(e.getValue()));
341 			return dm;
342 		}
343 		if (cm.isArray()) {
344 			return replaceWithMutables(Arrays.asList((Object[])o));
345 		}
346 		return o;
347 	}

349 	/*
350 	 * Sorts the specified list by the sort list.
351 	 */
352 	private void doSort(List list, List sortList) {
353 
354 		Map sort = new LinkedHashMap();
355 		for (Object s : sortList) {
356 			if (s instanceof String)
357 				sort.put(s, "a");
358 			else if (s instanceof Map) {
359 				Map sm = (Map)s;
360 				for (Map.Entry e : (Set<Map.Entry>)sm.entrySet())
361 					sort.put(e.getKey(), e.getValue().toString().toLowerCase(Locale.ENGLISH));
362 			}
363 		}
364 
365 		// Do the sort.
366 		List<String> columns = new ArrayList<String>(sort.keySet());
367 		Collections.reverse(columns);
368 		for (final String c : columns) {
369 			final boolean isDesc = StringUtils.startsWith(sort.get(c).toString(), 'd');
370 			Comparator comp = new Comparator<Map>() {
371 				@Override /* Comparator */
372 				public int compare(Map m1, Map m2) {
373 					Comparable v1 = (Comparable)m1.get(c), v2 = (Comparable)m2.get(c);
374 					if (v1 == null && v2 == null)
375 						return 0;
376 					if (v1 == null)
377 						return (isDesc ? -1 : 1);
378 					if (v2 == null)
379 						return (isDesc ? 1 : -1);
380 					return (isDesc ? v2.compareTo(v1) : v1.compareTo(v2));
381 				}
382 			};
383 			Collections.sort(list, comp);
384 		}
385 	}

387 	/*
388 	 * Filters all but the specified view columns on all entries in the specified list.
389 	 */
390 	private void doView(List list, List view) {
391 
392 		for (ListIterator i = list.listIterator(); i.hasNext();) {
393 			Object o = i.next();
394 			Map m = (Map)o;
395 			doView(m, view);
396 		}
397 	}
398 
399 	/*
400 	 * Creates a new Map with only the entries specified in the view list.
401 	 */
402 	private void doView(Map m, List view) {
403 		List<String> filterKeys = new LinkedList<String>();
404 		for (Object v : view) {
405 			if (v instanceof String) {
406 				filterKeys.add(v.toString());
407 			} else if (v instanceof ObjectMap) {
408 				ObjectMap vm = (ObjectMap)v;
409 				for (Map.Entry<String,Object> e : vm.entrySet()) {
410 					String vmKey = e.getKey();
411 					Object vmVal = e.getValue();
412 					Object mv = m.get(vmKey);
413 					filterKeys.add(vmKey);
414 					if (vmVal instanceof List) {
415 						List l = (List)vmVal;
416 						if (mv instanceof List)
417 							doView((List)mv, l);
418 						else if (mv instanceof Map)
419 							doView((Map)mv, l);
420 					}
421 				}
422 			}
423 		}
424 		if (m instanceof DelegateMap)
425 			((DelegateMap)m).filterKeys(filterKeys);
426 		else
427 			((DelegateBeanMap)m).filterKeys(filterKeys);
428 	}

431 	/*
432 	 * Returns the appropriate IMatcher for the specified class type.
433 	 */
434 	private IMatcher getObjectMatcherForType(String queryString, boolean ignoreCase, ClassMeta cm) {
435 		if (cm.isDate())
436 			return new DateMatcher(queryString);
437 		if (cm.isNumber())
438 			return new NumberMatcher(queryString);
439 		if (cm.isObject())
440 			return new ObjectMatcher(queryString, ignoreCase);
441 		return new StringMatcher(queryString, ignoreCase);
442 	}

445 	// CollectionFilter
446 	//====================================================================================================
447 	private class CollectionFilter {
448 		IMatcher entryMatcher;
449 
450 		public CollectionFilter(Map query, boolean ignoreCase) {
451 			if (query != null && ! query.isEmpty())
452 				entryMatcher = new MapMatcher(query, ignoreCase);
453 		}
454 
455 		public void doQuery(List in) {
456 			if (in == null || entryMatcher == null)
457 				return;
458 			for (Iterator i = in.iterator(); i.hasNext();) {
459 				Object o = i.next();
460 				if (! entryMatcher.matches(o))
461 					i.remove();
462 			}
463 		}
464 	}

467 	// IMatcher
468 	//====================================================================================================
469 	private interface IMatcher<E> {
470 		public boolean matches(E o);
471 	}
472 
473 	//====================================================================================================
474 	// MapMatcher
475 	//====================================================================================================
476 	/*
477 	 * Matches on a Map only if all specified entry matchers match.
478 	 */
479 	private class MapMatcher implements IMatcher<Map> {
480 
481 		Map<String,IMatcher> entryMatchers = new HashMap<String,IMatcher>();
482 
483 		public MapMatcher(Map query, boolean ignoreCase) {
484 			for (Map.Entry e : (Set<Map.Entry>)query.entrySet()) {
485 				String key = e.getKey().toString();
486 				Object value = e.getValue();
487 				IMatcher matcher = null;
488 				if (value instanceof String)
489 					matcher = getObjectMatcherForType((String)value, ignoreCase, beanContext.object());
490 				else if (value instanceof ObjectMap)
491 					matcher = new MapMatcher((ObjectMap)value, ignoreCase);
492 				else
493 					throw new RuntimeException("Invalid value type: " + value);
494 				entryMatchers.put(key, matcher);
495 			}
496 		}
497 
498 		@Override /* IMatcher */
499 		public boolean matches(Map m) {
500 			if (m == null)
501 				return false;
502 			for (Map.Entry<String,IMatcher> e : entryMatchers.entrySet()) {
503 				String key = e.getKey();
504 				Object val = m.get(key);
505 				if (! e.getValue().matches(val))
506 					return false;
507 			}
508 			return true;
509 		}
510 	}

512 	//====================================================================================================
513 	// ObjectMatcher
514 	//====================================================================================================
515 	/*
516 	 * Matcher that uses the correct matcher based on object type.
517 	 * Used for objects when we can't determine the object type beforehand.
518 	 */
519 	private class ObjectMatcher implements IMatcher<Object> {
520 
521 		String searchPattern;
522 		boolean ignoreCase;
523 		DateMatcher dateMatcher;
524 		NumberMatcher numberMatcher;
525 		StringMatcher stringMatcher;
526 
527 		ObjectMatcher(String searchPattern, boolean ignoreCase) {
528 			this.searchPattern = searchPattern;
529 			this.ignoreCase = ignoreCase;
530 		}
531 
532 		@Override /* IMatcher */
533 		public boolean matches(Object o) {
534 			if (o instanceof Number)
535 				return getNumberMatcher().matches(o);
536 			if (o instanceof Date || o instanceof Calendar)
537 				return getDateMatcher().matches(o);
538 			return getStringMatcher().matches(o);
539 		}
540 
541 		private IMatcher getNumberMatcher() {
542 			if (numberMatcher == null)
543 				numberMatcher = new NumberMatcher(searchPattern);
544 			return numberMatcher;
545 		}
546 
547 		private IMatcher getStringMatcher() {
548 			if (stringMatcher == null)
549 				stringMatcher = new StringMatcher(searchPattern, ignoreCase);
550 			return stringMatcher;
551 		}
552 
553 		private IMatcher getDateMatcher() {
554 			if (dateMatcher == null)
555 				dateMatcher = new DateMatcher(searchPattern);
556 			return dateMatcher;
557 		}
558 	}

560 	//====================================================================================================
561 	// NumberMatcher
562 	//====================================================================================================
563 	private static class NumberMatcher implements IMatcher<Number> {
564 
565 		private NumberPattern[] numberPatterns;
566 
567 		/**
568 		 * Construct a number matcher for the given search pattern.
569 		 * @param searchPattern A date search paattern.  See class usage for a description.
570 		 */
571 		public NumberMatcher(String searchPattern) {
572 			numberPatterns = new NumberPattern[1];

595 		NumberRange[] numberRanges;
596 
597 		public NumberPattern(String searchPattern) {
598 
599 			List<NumberRange> l = new LinkedList<NumberRange>();
600 
601 			for (String s : breakUpTokens(searchPattern)) {
602 				boolean isNot = (s.charAt(0) == '!');
603 				String token = s.substring(1);
604 				Pattern p = Pattern.compile("(([<>]=?)?)(-?\\d+)(-?(-?\\d+)?)");
605 
606 				// Possible patterns:
607 				// 123, >123, <123, >=123, <=123, >-123, >=-123, 123-456, -123--456
608 				// Regular expression used:  (([<>]=?)?)(-?\d+)(-??(-?\d+))
609 				Matcher m = p.matcher(token);
610 
611 				// If a non-numeric value was passed in for a numeric value, just set the value to '0'.
612 				// (I think this might resolve a workaround in custom queries).
613 				if (! m.matches())
614 					throw new RuntimeException("Numeric value didn't match pattern:  ["+token+"]");
615 					//m = numericPattern.matcher("0");
616 
617 				String arg1 = m.group(1);
618 				String start = m.group(3);
619 				String end = m.group(5);
620 
621 				l.add(new NumberRange(arg1, start, end, isNot));
622 			}
623 
624 			numberRanges = l.toArray(new NumberRange[l.size()]);
625 		}

627 		private List<String> breakUpTokens(String s) {
628 			// Get rid of whitespace in "123 - 456"
629 			s = s.replaceAll("(-?\\d+)\\s*-\\s*(-?\\d+)", "$1-$2");
630 			// Get rid of whitespace in ">= 123"
631 			s = s.replaceAll("([<>]=?)\\s+(-?\\d+)", "$1$2");
632 			// Get rid of whitespace in "! 123"
633 			s = s.replaceAll("(!)\\s+(-?\\d+)", "$1$2");
634 
635 			// Replace all commas with whitespace
636 			// Allows for alternate notation of: 123,456...
637 			s = s.replaceAll(",", " ");
638 
639 			String[] s2 = s.split("\\s+");
640 
641 			// Make all tokens 'ORed'.  There is no way to AND numeric tokens.
642 			for (int i = 0; i < s2.length; i++)
643 				if (! StringUtils.startsWith(s2[i], '!'))
644 					s2[i] = "^"+s2[i];
645 
646 			List<String> l = new LinkedList<String>();
647 			l.addAll(Arrays.asList(s2));
648 			return l;
649 		}

653 			for (int i = 0; i < numberRanges.length; i++)
654 				if (numberRanges[i].matches(number))
655 					return true;
656 			return false;
657 		}

765 	 * A construct representing a single search pattern.
766 	 */
767 	private class TimestampPattern {
768 		TimestampRange[] ranges;
769 		List<TimestampRange> l = new LinkedList<TimestampRange>();
770 
771 		public TimestampPattern(String s) {
772 
773 			// Handle special case where timestamp is enclosed in quotes.
774 			// This can occur on hyperlinks created by group-by queries.
775 			// e.g. '2007/01/29 04:17:43 PM'
776 			if (s.charAt(0) == '\'' && s.charAt(s.length()-1) == '\'')
777 				s = s.substring(1, s.length()-1);
778 
779 			// Pattern for finding <,>,<=,>=
780 			Pattern p1 = Pattern.compile("^\\s*([<>](?:=)?)\\s*(\\S+.*)$");
781 			// Pattern for finding range dash (e.g. xxx - yyy)
782 			Pattern p2 = Pattern.compile("^(\\s*-\\s*)(\\S+.*)$");
783 
784 			// States are...
785 			// 1 - Looking for <,>,<=,>=
786 			// 2 - Looking for single date.
787 			// 3 - Looking for start date.
788 			// 4 - Looking for -
789 			// 5 - Looking for end date.
790 			int state = 1;
791 
792 			String op = null;
793 			CalendarP startDate = null;
794 
795 			ParsePosition pp = new ParsePosition(0);
796 			Matcher m = null;
797 			String seg = s;
798 
799 			while (! seg.equals("") || state != 1) {
800 				if (state == 1) {
801 					m = p1.matcher(seg);
802 					if (m.matches()) {
803 						op = m.group(1);
804 						seg = m.group(2);
805 						state = 2;
806 					} else {
807 						state = 3;
808 					}
809 				} else if (state == 2) {
810 					l.add(new TimestampRange(op, parseDate(seg, pp)));
811 					//tokens.add("^"+op + parseTimestamp(seg, pp));
812 					seg = seg.substring(pp.getIndex()).trim();
813 					pp.setIndex(0);
814 					state = 1;
815 				} else if (state == 3) {
816 					startDate = parseDate(seg, pp);
817 					seg = seg.substring(pp.getIndex()).trim();
818 					pp.setIndex(0);
819 					state = 4;
820 				} else if (state == 4) {
821 					// Look for '-'
822 					m = p2.matcher(seg);
823 					if (m.matches()) {
824 						state = 5;
825 						seg = m.group(2);
826 					} else {
827 						// This is a single date (e.g. 2002/01/01)
828 						l.add(new TimestampRange(startDate));
829 						state = 1;
830 					}
831 				} else if (state == 5) {
832 					l.add(new TimestampRange(startDate, parseDate(seg, pp)));
833 					seg = seg.substring(pp.getIndex()).trim();
834 					pp.setIndex(0);
835 					state = 1;
836 				}
837 			}
838 
839 			ranges = l.toArray(new TimestampRange[l.size()]);
840 		}
841 
842 		public boolean matches(Calendar c) {
843 			if (ranges.length == 0) return true;
844 			for (int i = 0; i < ranges.length; i++)
845 				if (ranges[i].matches(c))
846 					return true;
847 			return false;
848 		}
849 	}

909 	 * Parses a timestamp string off the beginning of the string segment 'seg'.
910 	 * Goes through each possible valid timestamp format until it finds a match.
911 	 * The position where the parsing left off is stored in pp.
912 	 * @param seg The string segment being parsed.
913 	 * @param pp Where parsing last left off.
914 	 * @return An object represening a timestamp.
915 	 */
916 	protected CalendarP parseDate(String seg, ParsePosition pp) {
917 
918 		CalendarP cal = null;
919 
920 		for (int i = 0; i < validTimestampFormats.length && cal == null; i++) {
921 			pp.setIndex(0);
922 			SimpleDateFormat f = validTimestampFormats[i];
923 			Date d = f.parse(seg, pp);
924 			int idx = pp.getIndex();
925 			if (idx != 0) {
926 				// it only counts if the next character is '-', 'space', or end-of-string.
927 				char c = (seg.length() == idx ? 0 : seg.charAt(idx));
928 				if (c == 0 || c == '-' || Character.isWhitespace(c))
929 					cal = new CalendarP(d, getPrecisionField(f.toPattern()));
930 			}
931 		}
932 
933 		if (cal == null) throw new RuntimeException("Invalid date encountered:  ["+seg+"]");
934 
935 		return cal;
936 	}

1001 	/**
1002 	 * A construct representing a single search pattern.
1003 	 */
1004 	private static class SearchPattern {
1005 		Pattern[] orPatterns, andPatterns, notPatterns;
1006 
1007 		public SearchPattern(String searchPattern, boolean ignoreCase) {
1008 
1009 			List<Pattern> ors = new LinkedList<Pattern>();
1010 			List<Pattern> ands = new LinkedList<Pattern>();
1011 			List<Pattern> nots = new LinkedList<Pattern>();
1012 
1013 			for (String arg : breakUpTokens(searchPattern)) {
1014 				char prefix = arg.charAt(0);
1015 				String token = arg.substring(1);
1016 
1017 				token = token.replaceAll("([\\?\\*\\+\\\\\\[\\]\\{\\}\\(\\)\\^\\$\\.])", "\\\\$1");
1018 				token = token.replace("\u9997", ".*");
1019 				token = token.replace("\u9996", ".?");
1020 
1021 				if (! token.startsWith(".*"))
1022 					token = "^" + token;
1023 				if (! token.endsWith(".*"))
1024 					token = token + "$";
1025 
1026 				int flags = Pattern.DOTALL;
1027 				if (ignoreCase)
1028 					flags |= Pattern.CASE_INSENSITIVE;
1029 
1030 				Pattern p = Pattern.compile(token, flags);
1031 
1032 				if (prefix == '^')
1033 					ors.add(p);
1034 				else if (prefix == '+')
1035 					ands.add(p);
1036 				else if (prefix == '-')
1037 					nots.add(p);
1038 			}
1039 			orPatterns = ors.toArray(new Pattern[ors.size()]);
1040 			andPatterns = ands.toArray(new Pattern[ands.size()]);
1041 			notPatterns = nots.toArray(new Pattern[nots.size()]);
1042 		}
1043 
1044 		/**
1045 		 * Break up search pattern into separate tokens.
1046 		 */
1047 		private List<String> breakUpTokens(String s) {
1048 
1049 			// If the string is null or all whitespace, return an empty vector.
1050 			if (s == null || s.trim().length() == 0)
1051 				return Collections.emptyList();
1052 
1053 			// Pad with spaces.
1054 			s = " " + s + " ";
1055 
1056 			// Replace instances of [+] and [-] inside single and double quotes with
1057 			// \u2001 and \u2002 for later replacement.
1058 			int escapeCount = 0;
1059 			boolean inSingleQuote = false;
1060 			boolean inDoubleQuote = false;
1061 			char[] ca = s.toCharArray();
1062 			for (int i = 0; i < ca.length; i++) {
1063 				if (ca[i] == '\\') escapeCount++;
1064 				else if (escapeCount % 2 == 0) {
1065 					if (ca[i] == '\'') inSingleQuote = ! inSingleQuote;
1066 					else if (ca[i] == '"') inDoubleQuote = ! inDoubleQuote;
1067 					else if (ca[i] == '+' && (inSingleQuote || inDoubleQuote)) ca[i] = '\u9999';
1068 					else if (ca[i] == '-' && (inSingleQuote || inDoubleQuote)) ca[i] = '\u9998';
1069 				}
1070 				if (ca[i] != '\\') escapeCount = 0;
1071 			}
1072 			s = new String(ca);
1073 
1074 			// Remove spaces between '+' or '-' and the keyword.
1075 			//s = perl5Util.substitute("s/([\\+\\-])\\s+/$1/g", s);
1076 			s = s.replaceAll("([\\+\\-])\\s+", "$1");
1077 
1078 			// Replace:  [*]->[\u3001] as placeholder for '%', ignore escaped.
1079 			s = replace(s, '*', '\u9997', true);
1080 			// Replace:  [?]->[\u3002] as placeholder for '_', ignore escaped.
1081 			s = replace(s, '?', '\u9996', true);
1082 			// Replace:  [\*]->[*], [\?]->[?]
1083 			s = unEscapeChars(s, new char[]{'*','?'});
1084 
1085 			// Remove spaces
1086 			s = s.trim();
1087 
1088 			// Re-replace the [+] and [-] characters inside quotes.
1089 			s = s.replace('\u9999', '+');
1090 			s = s.replace('\u9998', '-');
1091 
1092 			String[] sa = splitQuoted(s, ' ');
1093 			List<String> l = new ArrayList<String>(sa.length);
1094 			int numOrs = 0;
1095 			for (int i = 0; i < sa.length; i++) {
1096 				String token = sa[i];
1097 				int len = token.length();
1098 				if (len > 0) {
1099 					char c = token.charAt(0);
1100 					String s2 = null;
1101 					if ((c == '+' || c == '-') && len > 1)
1102 						s2 = token.substring(1);
1103 					else {
1104 						s2 = token;
1105 						c = '^';
1106 						numOrs++;
1107 					}
1108 					// Trim off leading and trailing single and double quotes.
1109 					if (s2.matches("\".*\"") || s2.matches("'.*'"))
1110 						s2 = s2.substring(1, s2.length()-1);
1111 
1112 					// Replace:  [\"]->["]
1113 					s2 = unEscapeChars(s2, new char[]{'"','\''});
1114 
1115 					// Un-escape remaining escaped backslashes.
1116 					s2 = unEscapeChars(s2, new char[]{'\\'});
1117 
1118 					l.add(c + s2);
1119 				}
1120 			}
1121 
1122 			// If there's a single OR clause, turn it into an AND clause (makes the SQL cleaner).
1123 			if (numOrs == 1) {
1124 				int ii = l.size();
1125 				for (int i = 0; i < ii; i++) {
1126 					String x = l.get(i);
1127 					if (x.charAt(0) == '^')
1128 						l.set(i, '+'+x.substring(1));
1129 				}
1130 			}
1131 			return l;
1132 		}
1133 
1134 		public boolean matches(String input) {
1135 			if (input == null) return false;
1136 			for (int i = 0; i < andPatterns.length; i++)
1137 				if (! andPatterns[i].matcher(input).matches())
1138 					return false;
1139 			for (int i = 0; i < notPatterns.length; i++)
1140 				if (notPatterns[i].matcher(input).matches())
1141 					return false;
1142 			for (int i = 0; i < orPatterns.length; i++)
1143 				if (orPatterns[i].matcher(input).matches())
1144 					return true;
1145 			return orPatterns.length == 0;
1146 		}

1150 	/**
1151 	 * Same as split(String, char), but does not split on characters inside
1152 	 * single quotes.
1153 	 * Does not split on escaped delimiters, and escaped quotes are also ignored.
1154 	 * Example:
1155 	 * split("a,b,c",',') -> {"a","b","c"}
1156 	 * split("a,'b,b,b',c",',') -> {"a","'b,b,b'","c"}
1157 	 */
1158 	private static String[] splitQuoted(String s, char c) {
1159 
1160 		if (s == null || s.matches("\\s*"))
1161 			return new String[0];
1162 
1163 		List<String> l = new LinkedList<String>();
1164 		char[] sArray = s.toCharArray();
1165 		int x1 = 0;
1166 		int escapeCount = 0;
