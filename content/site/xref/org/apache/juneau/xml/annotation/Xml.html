13  package org.apache.juneau.xml.annotation;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  
18  import java.lang.annotation.*;
19  
20  import org.apache.juneau.xml.*;
21  
22  /**
23   * Annotation for specifying various XML options for the XML and RDF/XML serializers.
24   * <p>
25   * 	Can be applied to Java packages, types, fields, and methods.
26   * <p>
27   * 	Can be used for the following:
28   * <ul class='spaced-list'>
29   * 	<li>Override the child element name on the XML representation of collection or array properties.
30   * 	<li>Specify the XML namespace on a package, class, or method.
31   * 	<li>Override the XML format on a POJO.
32   * </ul>
33   *
34   * @author James Bognar (james.bognar@salesforce.com)
35   */
36  @Documented
37  @Target({TYPE,FIELD,METHOD})
38  @Retention(RUNTIME)
39  @Inherited
40  public @interface Xml {
41  
42  	/**
43  	 * Sets the name of the XML child elements for bean properties of type collection and array.
44  	 * <p>
45  	 * 	Applies only to collection and array bean properties.
46  	 *
47  	 * <dl>
48  	 * 	<dt>Example:</dt>
49  	 * 	<dd>
50  	 * 		<p class='bcode'>
51  	 * 	<jk>public class</jk> MyBean {
52  	 * 		<ja>@Xml</ja>(childName=<js>"child"</js>}
53  	 * 		<jk>public</jk> String[] <jf>children</jf> = {<js>"foo"</js>,<js>"bar"</js>};
54  	 * 	}
55  	 * 		</p>
56  	 * 		<p>
57  	 * 			Without the <ja>@Xml</ja> annotation, serializing this bean as XML would have produced the following...
58  	 * 		</p>
59  	 * 		<p class='bcode'>
60  	 * 	<xt>&lt;object&gt;</xt>
61  	 * 		<xt>&lt;children&gt;</xt>
62  	 * 			<xt>&lt;string&gt;</xt>foo<xt>&lt;/string&gt;</xt>
63  	 * 			<xt>&lt;string&gt;</xt>bar<xt>&lt;/string&gt;</xt>
64  	 * 		<xt>&lt;/children&gt;</xt>
65  	 * 	<xt>&lt;/object&gt;</xt>
66  	 * 		</p>
67  	 * 		<p>
68  	 * 			With the annotations, serializing this bean as XML produces the following...
69  	 * 		</p>
70  	 * 		<p class='bcode'>
71  	 * 	<xt>&lt;object&gt;</xt>
72  	 * 		<xt>&lt;children&gt;</xt>
73  	 * 			<xt>&lt;child&gt;</xt>foo<xt>&lt;/child&gt;</xt>
74  	 * 			<xt>&lt;child&gt;</xt>bar<xt>&lt;/child&gt;</xt>
75  	 * 		<xt>&lt;/children&gt;</xt>
76  	 * 	<xt>&lt;/object&gt;</xt>
77  	 * 		</p>
78  	 * 	</dd>
79  	 * </dl>
80  	 */
81  	String childName() default "";
82  
83  	/**
84  	 * Sets the XML prefix of this property or class.
85  	 * <ul class='spaced-list'>
86  	 * 	<li>When applied to a {@link ElementType#TYPE}, namespace is applied to all properties in the class, and all
87  	 * 		subclasses of the class.
88  	 * 	<li>When applied to bean properties on {@link ElementType#METHOD} and {@link ElementType#FIELD}, applies
89  	 * 		to the bean property.
90  	 * </ul>
91  	 * <p>
92  	 * 	Must either be matched to a {@link #namespace()} annotation on the same object, parent object, or a {@link XmlNs} with the same name
93  	 * 	through the {@link XmlSchema#xmlNs()} annotation on the package.
94  	 * </p>
95  	 */
96  	String prefix() default "";
97  
98  	/**
99  	 * Sets the namespace URI of this property or class.
100 	 * <p>
101 	 * 	Must be matched with a {@link #prefix()} annotation on this object, a parent object, or a {@link XmlNs} with the same name
102 	 * 	through the {@link XmlSchema#xmlNs()} annotation on the package.
103 	 */
104 	String namespace() default "";
105 
106 	/**
107 	 * The {@link XmlFormat} to use for serializing this object type.
108 	 *
109 	 * <dl>
110 	 * 	<dt>Example:</dt>
111 	 * 	<dd>
112 	 * 		<p class='bcode'>
113 	 * 	<jk>public class</jk> MyBean {
114 	 *
115 	 * 		<jc>// Normally, bean properties would be rendered as child elements of the bean element.</jc>
116 	 * 		<jc>// Override so that it's rendered as a "f1='123'" attribute on the bean element instead.</jc>
117 	 * 		<ja>@Xml</ja>(format=XmlFormat.<jsf>ATTR</jsf>}
118 	 * 		<jk>public int</jk> f1 = 123;
119 	 *
120 	 * 		<jc>// Normally, bean URL properties would be rendered as XML attributes on the bean element.</jc>
121 	 * 		<jc>// Override so that it's rendered as an &lt;href&gt;http://foo&lt;/href&gt; child element instead.</jc>
122 	 * 		<ja>@BeanProperty</ja>(uri=<jk>true</jk>)
123 	 * 		<ja>@Xml</ja>(format=XmlFormat.<jsf>ELEMENT</jsf>}
124 	 * 		<jk>public</jk> URL <jf>href</jf> = <jk>new</jk> URL(<js>"http://foo"</js>);
125 	 *
126 	 * 		<jc>// Normally, collection properties would be grouped under a single &lt;children&gt; child element on the bean element.</jc>
127 	 * 		<jc>// Override so that entries are directly children of the bean element with each entry having an element name of &lt;child&gt;.</jc>
128 	 * 		<ja>@Xml</ja>(format=XmlFormat.<jsf>COLLAPSED</jsf>, childName=<js>"child"</js>}
129 	 * 		<jk>public</jk> String[] <jf>children</jf> = <js>"foo"</js>,<js>"bar"</js>};
130 	 * 	}
131 	 * 		</p>
132 	 * 		<p>
133 	 * 			Without the <ja>@Xml</ja> annotations, serializing this bean as XML would have produced the following...
134 	 * 		</p>
135 	 * 		<p class='bcode'>
136 	 * 	<xt>&lt;object</xt> <xa>href</xa>=<js>'http://foo'</js><xt>&gt;</xt>;
137 	 * 		<xt>&lt;f1&gt;</xt>123<xt>&lt;/f1&gt;</xt>
138 	 * 		<xt>&lt;children&gt;</xt>
139 	 * 			<xt>&lt;string&gt;</xt>foo<xt>&lt;/string&gt;</xt>
140 	 * 			<xt>&lt;string&gt;</xt>bar<xt>&lt;/string&gt;</xt>
141 	 * 		<xt>&lt;/children&gt;</xt>
142 	 * 	<xt>&lt;/object&gt;</xt>
143 	 * 		</p>
144 	 * 		<p>
145 	 * 			With the annotations, serializing this bean as XML produces the following...
146 	 * 		</p>
147 	 * 		<p class='bcode'>
148 	 * 	<xt>&lt;object</xt> <xa>f1</xa>=<js>'123'</js><xt>&gt;</xt>
149 	 * 		<xt>&lt;href&gt;</xt>http://foo<xt>&lt;/href&gt;</xt>;
150 	 * 		<xt>&lt;child&gt;</xt>foo<xt>&lt;/child&gt;</xt>
151 	 * 		<xt>&lt;child&gt;</xt>bar<xt>&lt;/child&gt;</xt>
152 	 * 	<xt>&lt;/object&gt;</xt>
153 	 * 		</p>
154 	 * 	</dd>
155 	 * </dl>
156 	 */
157 	XmlFormat format() default XmlFormat.NORMAL;
158 
159 	/**
160 	 * Associates a content handler with a bean class or bean property.
161 	 * <p>
162 	 * 	Refer to {@link XmlContentHandler} for more information.
163 	 */
164 	Class<? extends XmlContentHandler<?>> contentHandler() default XmlContentHandler.NULL.class;
165 }
