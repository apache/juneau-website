13  package org.apache.juneau.msgpack;
14  
15  import java.lang.reflect.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.json.*;
19  import org.apache.juneau.serializer.*;
20  
21  /**
22   * Session object that lives for the duration of a single use of {@link MsgPackSerializer}.
23   * <p>
24   * This class is NOT thread safe.  It is meant to be discarded after one-time use.
25   *
26   * @author James Bognar (james.bognar@salesforce.com)
27   */
28  public final class MsgPackSerializerSession extends SerializerSession {
29  
30  	/**
31  	 * Create a new session using properties specified in the context.
32  	 *
33  	 * @param ctx The context creating this session object.
34  	 * 	The context contains all the configuration settings for this object.
35  	 * @param beanContext The bean context being used.
36  	 * @param output The output object.  See {@link JsonSerializerSession#getOutputStream()} for valid class types.
37  	 * @param op The override properties.
38  	 * 	These override any context properties defined in the context.
39  	 * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
40  	 */
41  	protected MsgPackSerializerSession(MsgPackSerializerContext ctx, BeanContext beanContext, Object output, ObjectMap op, Method javaMethod) {
42  		super(ctx, beanContext, output, op, javaMethod);
43  	}
44  
45  	@Override
46  	public MsgPackOutputStream getOutputStream() throws Exception {
47  		Object output = getOutput();
48  		if (output instanceof MsgPackOutputStream)
49  			return (MsgPackOutputStream)output;
50  		return new MsgPackOutputStream(super.getOutputStream());
51  	}
52  }
