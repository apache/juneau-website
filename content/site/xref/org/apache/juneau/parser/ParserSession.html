24  /**
25   * Session object that lives for the duration of a single use of {@link Parser}.
26   * <p>
27   * This class is NOT thread safe.  It is meant to be discarded after one-time use.
28   *
29   * @author James Bognar (james.bognar@salesforce.com)
30   */
31  public class ParserSession extends Session {
32  
33  	private static JuneauLogger logger = JuneauLogger.getLogger(ParserSession.class);
34  
35  	private final boolean debug, trimStrings;
36  	private boolean closed;
37  	private final BeanContext beanContext;
38  	private final List<String> warnings = new LinkedList<String>();
39  
40  	private final ObjectMap properties;
41  	private final Method javaMethod;
42  	private final Object outer;
43  	private final Object input;
44  	private InputStream inputStream;
45  	private Reader reader, noCloseReader;
46  	private BeanPropertyMeta currentProperty;
47  	private ClassMeta<?> currentClass;

49  	/**
50  	 * Create a new session using properties specified in the context.
51  	 *
52  	 * @param ctx The context creating this session object.
53  	 * 	The context contains all the configuration settings for this object.
54  	 * @param beanContext The bean context being used.
55  	 * @param input The input.
56  	 * 	<br>For character-based parsers, this can be any of the following types:
57  	 * 	<ul>
58  	 * 		<li><jk>null</jk>
59  	 * 		<li>{@link Reader}
60  	 * 		<li>{@link CharSequence}
61  	 * 		<li>{@link InputStream} containing UTF-8 encoded text.
62  	 * 		<li>{@link File} containing system encoded text.
63  	 * 	</ul>
64  	 * 	<br>For byte-based parsers, this can be any of the following types:
65  	 * 	<ul>
66  	 * 		<li><jk>null</jk>
67  	 * 		<li>{@link InputStream}
68  	 * 		<li><code><jk>byte</jk>[]</code>
69  	 * 		<li>{@link File}
70  	 * 	</ul>
71  	 * @param op The override properties.
72  	 * 	These override any context properties defined in the context.
73  	 * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
74  	 * @param outer The outer object for instantiating top-level non-static inner classes.
75  	 */
76  	public ParserSession(ParserContext ctx, BeanContext beanContext, Object input, ObjectMap op, Method javaMethod, Object outer) {
77  		super(ctx);
78  		if (op == null || op.isEmpty()) {
79  			debug = ctx.debug;
80  			trimStrings = ctx.trimStrings;
81  		} else {
82  			debug = op.getBoolean(PARSER_debug, ctx.debug);
83  			trimStrings = op.getBoolean(PARSER_trimStrings, ctx.trimStrings);
84  		}
85  		this.beanContext = beanContext;
86  		this.input = input;
87  		this.properties = op;
88  		this.javaMethod = javaMethod;
89  		this.outer = outer;
90  	}
91  
92  	/**
93  	 * Wraps the specified input object inside an input stream.
94  	 * Subclasses can override this method to implement their own input streams.
95  	 *
96  	 * @return The input object wrapped in an input stream, or <jk>null</jk> if the object is null.
97  	 * @throws ParseException If object could not be converted to an input stream.
98  	 */
99  	public InputStream getInputStream() throws ParseException {
100 		if (input == null)
101 			return null;
102 		if (input instanceof InputStream)
103 			return (InputStream)input;
104 		if (input instanceof byte[])
105 			return new ByteArrayInputStream((byte[])input);
106 		if (input instanceof File)
107 			try {
108 				inputStream = new FileInputStream((File)input);
109 				return inputStream;
110 			} catch (FileNotFoundException e) {
111 				throw new ParseException(e);
112 			}
113 		throw new ParseException("Cannot convert object of type {0} to an InputStream.", input.getClass().getName());
114 	}
115 
116 
117 	/**
118 	 * Wraps the specified input object inside a reader.
119 	 * Subclasses can override this method to implement their own readers.
120 	 *
121 	 * @return The input object wrapped in a Reader, or <jk>null</jk> if the object is null.
122 	 * @throws Exception If object could not be converted to a reader.
123 	 */
124 	public Reader getReader() throws Exception {
125 		if (input == null)
126 			return null;
127 		if (input instanceof Reader)
128 			return (Reader)input;
129 		if (input instanceof CharSequence) {
130 			if (reader == null)
131 				reader = new ParserReader((CharSequence)input);
132 			return reader;
133 		}
134 		if (input instanceof InputStream) {
135 			if (noCloseReader == null)
136 				noCloseReader = new InputStreamReader((InputStream)input, IOUtils.UTF8);
137 			return noCloseReader;
138 		}
139 		if (input instanceof File) {
140 			if (reader == null)
141 				reader = new FileReader((File)input);
142 			return reader;
143 		}
144 		throw new ParseException("Cannot convert object of type {0} to a Reader.", input.getClass().getName());
145 	}
146 
147 	/**
148 	 * Returns information used to determine at what location in the parse a failure occurred.
149 	 *
150 	 * @return A map, typically containing something like <code>{line:123,column:456,currentProperty:"foobar"}</code>
151 	 */
152 	public Map<String,Object> getLastLocation() {
153 		Map<String,Object> m = new LinkedHashMap<String,Object>();
154 		if (currentClass != null)
155 			m.put("currentClass", currentClass.toString(true));
156 		if (currentProperty != null)
157 			m.put("currentProperty", currentProperty);
158 		return m;
159 	}

244 	/**
245 	 * Logs a warning message.
246 	 *
247 	 * @param msg The warning message.
248 	 * @param args Optional printf arguments to replace in the error message.
249 	 */
250 	public void addWarning(String msg, Object... args) {
251 		logger.warning(msg, args);
252 		msg = args.length == 0 ? msg : String.format(msg, args);
253 		warnings.add((warnings.size() + 1) + ": " + msg);
254 	}
255 
256 	/**
257 	 * Trims the specified object if it's a <code>String</code> and {@link #isTrimStrings()} returns <jk>true</jk>.
258 	 *
259 	 * @param o The object to trim.
260 	 * @return The trimmmed string if it's a string.
261 	 */
262 	@SuppressWarnings("unchecked")
263 	public final <K> K trim(K o) {
264 		if (trimStrings && o instanceof String)
265 			return (K)o.toString().trim();
266 		return o;
267 
268 	}
269 
270 	/**
271 	 * Trims the specified string if {@link ParserSession#isTrimStrings()} returns <jk>true</jk>.
272 	 *
273 	 * @param s The input string to trim.
274 	 * @return The trimmed string, or <jk>null</jk> if the input was <jk>null</jk>.
275 	 */
276 	public final String trim(String s) {
277 		if (trimStrings && s != null)
278 			return s.trim();
279 		return s;
280 	}
281 
282 	/**
283 	 * Perform cleanup on this context object if necessary.
284 	 *
285 	 * @throws ParseException If called more than once, or in debug mode and warnings occurred.
286 	 */
287 	public void close() throws ParseException {
288 		if (closed)
289 			throw new ParseException("Attempt to close ParserSession more than once.");
290 
291 		try {
292 			if (inputStream != null)
293 				inputStream.close();
294 			if (reader != null)
295 				reader.close();
296 		} catch (IOException e) {
297 			throw new ParseException(e);
298 		}
299 
300 		if (debug && warnings.size() > 0)
301 			throw new ParseException("Warnings occurred during parsing: \n" + StringUtils.join(warnings, "\n"));
302 		closed = true;
303 	}
304 
305 	@Override /* Object */
306 	protected void finalize() throws Throwable {
307 		if (! closed)
308 			throw new RuntimeException("ParserSession was not closed.");
309 	}
