13  package org.apache.juneau.svl;
14  
15  import java.io.*;
16  
17  /**
18   * Abstract superclass of all Simple Var Language variables that resolve to simple returned string values.
19   * <p>
20   * Note the difference between this class and {@link StreamedVar} that streams
21   * 	values to writers.
22   * Unlike the {@link StreamedVar} class, the returned value from this class can
23   * 	contain nested variables that will be recursively resolved by {@link VarResolver}.
24   * <p>
25   * Subclasses must implement the {@link #resolve(VarResolverSession, String)} method.
26   *
27   * @see org.apache.juneau.svl
28   * @author James Bognar (james.bognar@salesforce.com)
29   */
30  public abstract class SimpleVar extends Var {
31  
32  	/**
33  	 * Constructor.
34  	 *
35  	 * @param name The variable name (e.g. <js>"C"</js> for variables of the form <js>"$C{...}"</js>)
36  	 */
37  	protected SimpleVar(String name) {
38  		super(name, false);
39  	}
40  
41  	@Override /* Var */
42  	public void resolveTo(VarResolverSession session, Writer w, String arg) {
43  		throw new UnsupportedOperationException("Cannot call streamTo() on SimpleVar class");
44  	}
45  }
